030package org.ecoinformatics.seek.querybuilder;
031
032import java.io.File;
033import java.util.Enumeration;
034import java.util.Hashtable;
035
036import javax.swing.filechooser.FileFilter;
037
038/**
039 * Adapted from the Swing JFileChooser demo and the ExampleFileFilter file
040 */
041public class DBFileFilter extends FileFilter {
042        private static String TYPE_UNKNOWN = "Type Unknown";
043        private static String HIDDEN_FILE = "Hidden File";
044
045        private Hashtable mFilters = null;
046        private String mDescription = null;
047        private String mFullDescription = null;
048        private boolean mUseExtensionsInDescription = true;
049
050        /**
051         * Creates a file filter. If no mFilters are added, then all files are
052         * accepted.
053         */
054        public DBFileFilter() {
055                this.mFilters = new Hashtable();
056        }
057
058        /**
059         * Creates a file filter that accepts files with the given extension.
060         * Example: new ExampleFileFilter("jpg");
061         */
062        public DBFileFilter(String extension) {
063                this(extension, null);
064        }
065
066        /**
067         * Creates a file filter that accepts the given file type. Example: new
068         * ExampleFileFilter("jpg", "JPEG Image Images");
069         * 
070         * Note that the "." before the extension is not needed. If provided, it
071         * will be ignored.
072         */
073        public DBFileFilter(String extension, String mDescription) {
074                this();
075                if (extension != null)
076                        addExtension(extension);
077
078                if (mDescription != null)
079                        setDescription(mDescription);
080        }
081
082        /**
083         * Creates a file filter from the given string array. Example: new
084         * ExampleFileFilter(String {"gif", "jpg"});
085         * 
086         * Note that the "." before the extension is not needed adn will be ignored.
087         */
088        public DBFileFilter(String[] mFilters) {
089                this(mFilters, null);
090        }
091
092        /**
093         * Creates a file filter from the given string array and mDescription.
094         * Example: new DBFileFilter(String {"gif", "jpg"}, "Gif and JPG Images");
095         * 
096         * Note that the "." before the extension is not needed and will be ignored.
097         */
098        public DBFileFilter(String[] mFilters, String mDescription) {
099                this();
100                for (int i = 0; i < mFilters.length; i++) {
101                        // add mFilters one by one
102                        addExtension(mFilters[i]);
103                }
104                if (mDescription != null)
105                        setDescription(mDescription);
106        }
107
108        /**
109         * Return true if this file should be shown in the directory pane, false if
110         * it shouldn't.
111         * 
112         * Files that begin with "." are ignored.
113         * 
114         */
115        public boolean accept(File f) {
116                if (f != null) {
117                        if (f.isDirectory()) {
118                                return true;
119                        }
120                        String extension = getExtension(f);
121                        if (extension != null && mFilters.get(getExtension(f)) != null) {
122                                return true;
123                        }
124                        ;
125                }
126                return false;
127        }
128
129        /**
130         * Return the extension portion of the file's name .
131         */
132        public String getExtension(File f) {
133                if (f != null) {
134                        String filename = f.getName();
135                        int i = filename.lastIndexOf('.');
136                        if (i > 0 && i < filename.length() - 1) {
137                                return filename.substring(i + 1).toLowerCase();
138                        }
139                        ;
140                }
141                return null;
142        }
143
144        /**
145         * Adds a filetype "dot" extension to filter against.
146         * 
147         * For example: the following code will create a filter that mFilters out
148         * all files except those that end in ".jpg" and ".tif":
149         * 
150         * ExampleFileFilter filter = new ExampleFileFilter();
151         * filter.addExtension("jpg"); filter.addExtension("tif");
152         * 
153         * Note that the "." before the extension is not needed and will be ignored.
154         */
155        public void addExtension(String extension) {
156                if (mFilters == null) {
157                        mFilters = new Hashtable(5);
158                }
159                mFilters.put(extension.toLowerCase(), this);
160                mFullDescription = null;
161        }
162
163        /**
164         * Returns the human readable mDescription of this filter. For example:
165         * "JPEG and GIF Image Files (*.jpg, *.gif)"
166         */
167        public String getDescription() {
168                if (mFullDescription == null) {
169                        if (mDescription == null || isExtensionListInDescription()) {
170                                mFullDescription = mDescription == null ? "(" : mDescription
171                                                + " (";
172                                // build the mDescription from the extension list
173                                Enumeration extensions = mFilters.keys();
174                                if (extensions != null) {
175                                        mFullDescription += "." + (String) extensions.nextElement();
176                                        while (extensions.hasMoreElements()) {
177                                                mFullDescription += ", ."
178                                                                + (String) extensions.nextElement();
179                                        }
180                                }
181                                mFullDescription += ")";
182                        } else {
183                                mFullDescription = mDescription;
184                        }
185                }
186                return mFullDescription;
187        }
188
189        /**
190         * Sets the human readable mDescription of this filter. For example:
191         * filter.setDescription("Gif and JPG Images");
192         */
193        public void setDescription(String mDescription) {
194                this.mDescription = mDescription;
195                mFullDescription = null;
196        }
197
198        /**
199         * Determines whether the extension list (.jpg, .gif, etc) should show up in
200         * the human readable mDescription.
201         * 
202         * Only relevent if a mDescription was provided in the constructor or using
203         * setDescription();
204         */
205        public void setExtensionListInDescription(boolean b) {
206                mUseExtensionsInDescription = b;
207                mFullDescription = null;
208        }
209
210        /**
211         * Returns whether the extension list (.jpg, .gif, etc) should show up in
212         * the human readable mDescription.
213         * 
214         * Only relevent if a description was provided in the constructor or using
215         * setDescription();
216         */
217        public boolean isExtensionListInDescription() {
218                return mUseExtensionsInDescription;
219        }
220}
