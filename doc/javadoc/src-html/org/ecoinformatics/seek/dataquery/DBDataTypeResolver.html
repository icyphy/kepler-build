030package org.ecoinformatics.seek.dataquery;
031
032import java.util.List;
033
034import org.kepler.configuration.ConfigurationManager;
035import org.kepler.configuration.ConfigurationProperty;
036import org.kepler.objectmanager.data.UnresolvableTypeException;
037
038/**
039 * This class will resolve a data type in HSQL for a given data type in
040 * Attribute object
041 * 
042 * @author Jing Tao
043 * 
044 */
045
046public class DBDataTypeResolver {
047
048  private ConfigurationProperty sqlEngineProperty = null;
049        private static final String DBDEFAULTYPEPATH = "//sqlEngine[sqlEngineName=\"hsql\"]/DBdataTypeMapping/defaultDBdataType";
050        private static final String JAVADEFAULTPATH = "//sqlEngine[sqlEngineName=\"hsql\"]/DBdataTypeMapping/defaultJavaDataType";
051        private static final String DATATYPEMAPPARENTPATH = "//sqlEngine[sqlEngineName=\"hsql\"]/DBdataTypeMapping/mapping";
052        private static final String METADATATYPEPATH = "./metadataDataType";
053        private static final String DBDATATYPEPATH = "./DBdataType";
054        private static final String JAVADATATYPEPATH = "./javaDataType";
055
056  private static String DEFAULTTYPE;
057  private static String JAVADEFAULT;
058
059        /**
060         * Constructor get two mapping from lib/config.xml file. Mapping is hash
061         * table, key is attribute data type and value is db data type Another
062         * mapping is : key - attribute data type, value - java data type
063         */
064        public DBDataTypeResolver() {
065                        
066      ConfigurationManager confMan = ConfigurationManager.getInstance();
067      ConfigurationProperty commonProperty = confMan.getProperty(ConfigurationManager.getModule("common"));
068      sqlEngineProperty = (ConfigurationProperty)commonProperty
069        .findProperties("sqlEngineName", "hsql", true).get(0);
070      DEFAULTTYPE = sqlEngineProperty.getProperty("DBdataTypeMapping").getProperty("defaultDBdataType").getValue();
071      JAVADEFAULT = sqlEngineProperty.getProperty("DBdataTypeMapping").getProperty("defaultJavaDataType").getValue();
072        }// DBDataTypeResolver
073
074        /**
075         * Method to find mapping db data type to a given attribute data type
076         * 
077         * @param attributeType
078         *            String
079         * @throws UnresolvableTypeException
080         * @return String
081         */
082        public String resolveDBType(String attributeType)
083                        throws UnresolvableTypeException {
084                String dbType = null;
085                if (attributeType == null || attributeType.trim().equals("")) {
086                        dbType = DEFAULTTYPE;
087                } else {
088      ConfigurationProperty dbDatatypeMappingProp = sqlEngineProperty.getProperty("DBdataTypeMapping");
089      List mappingsList = dbDatatypeMappingProp.findProperties("metadataDataType", attributeType, true);
090      ConfigurationProperty mappingProp = (ConfigurationProperty)mappingsList.get(0);
091      dbType = mappingProp.getProperty("DBdataType").getValue();
092                }
093
094                if (dbType == null) {
095                        throw new UnresolvableTypeException(
096                                        "Unable to resolve db data type "
097                                                        + "for attribute data type " + attributeType);
098                }
099                return dbType;
100        }// resolveDBType
101
102        /**
103         * Method to find a mapping between java type and a given metadata data type
104         * 
105         * @param attributeType
106         *            String
107         * @throws UnresolvableTypeException
108         * @return String
109         */
110        public String resolveJavaType(String attributeType)
111                        throws UnresolvableTypeException {
112                String javaType = null;
113                if (attributeType == null || attributeType.trim().equals("")) {
114                        javaType = JAVADEFAULT;
115                } else {
116      ConfigurationProperty dbDatatypeMappingProp = sqlEngineProperty.getProperty("DBdataTypeMapping");
117      List mappingsList = dbDatatypeMappingProp.findProperties("metadataDataType", attributeType, true);
118      ConfigurationProperty mappingProp = (ConfigurationProperty)mappingsList.get(0);
119      javaType = mappingProp.getProperty("javaDataType").getValue();
120                }
121
122                if (javaType == null) {
123                        throw new UnresolvableTypeException(
124                                        "Unable to resolve java data type "
125                                                        + "for attribute data type " + attributeType);
126                }
127                return javaType;
128
129        }// resolveJavaType
130
131}// DBDataTypeResolver
