398        private void initial_setup() throws IllegalActionException {
399                try {
400                        _downloadCompleted.acquire();
401                        log.debug("Is stop requested? " + getDirector().isStopRequested());
402                } catch (InterruptedException e) {
403                        log.debug("Is stop requested? " + getDirector().isStopRequested());
404                        if (getDirector().isStopRequested()) {
405                                throw new IllegalActionException("Execution interrupted");
406                        }
407                }
408                String sqlStr = "";
409
410                String sqlXMLStr = ((Settable) _sqlAttr).getExpression();
411                if (sqlXMLStr == null || sqlXMLStr.length() == 0) {
412                        sqlStr = "SELECT * FROM "
413                                        + (_tableEntity.getMappedName() != null ? _tableEntity
414                                                        .getMappedName() : _tableEntity.getName());
415
416                } else {
417                        Hashtable mappedNameHash = new Hashtable();
418                        if (_tableEntity.getMappedName() != null) {
419                                mappedNameHash.put(_tableEntity.getName(), _tableEntity
420                                                .getMappedName());
421                        }
422                        DBQueryDef queryDef = DBQueryDefParserEmitter.parseQueryDef(
423                                        _schemaDef, sqlXMLStr, mappedNameHash);
424                        _columns = queryDef.getSelects();
425                        sqlStr = DBQueryDefParserEmitter.createSQL(_schemaDef, queryDef);
426                }
427
428                // excuted query
429                if (sqlStr != null && !sqlStr.trim().equals("")) {
430                        // if table gnerated successfully, we will run query
431                        if (_tableGenerator != null && _tableGenerator.getSuccessStatus()) {
432                                try {
433                                        _icon.setBusy();
434                                        HsqlDataQueryAction queryAction = new HsqlDataQueryAction();
435                                        queryAction.setSQL(sqlStr);
436                                        queryAction.actionPerformed(null);
437                                        ResultSet result = queryAction.getResultSet();
438                                        _dataVectors = transformResultSet(result);
439                                        _icon.setReady();
440                                } catch (Exception e) {
441                                        log.debug("Error to run query is ", e);
442                                        throw new IllegalActionException(e.getMessage());
443                                }
444
445                        }
446                }
447        }

449        /*
450         * Method to transform resultset to a vector array data. A vector is a row
451         * and data type is string.
452         */
453        private Vector[] transformResultSet(ResultSet aResultset)
454                        throws SQLException {
455                if (aResultset == null) {
456                        System.err.println("Resultset is NULL in transformResultSet.");
457                        return null;
458                }
459                ResultSetMetaData metadata = aResultset.getMetaData();
460                if (metadata == null) {
461                        System.err.println("Metadata is NULL for resultset.");
462                        return null;
463                }
464                int columnSize = metadata.getColumnCount();
465
466                Vector data = new Vector();
467                while (aResultset.next()) {
468                        Vector rowData = new Vector();
469                        for (int i = 0; i < columnSize; i++) {
470                                String str = aResultset.getString(i + 1);
471                                rowData.add(str);
472                        }
473                        data.add(rowData);
474                }
475                Vector[] dataArray = null;
476                dataArray = transformVectorToArray(data);
477                return dataArray;
478        }

480        /**
481         * 
482         * @param vector
483         *       */
484        private Vector[] transformVectorToArray(Vector vector) {
485                if (vector == null) {
486                        return null;
487                }
488                int size = vector.size();
489                Vector[] array = new Vector[size];
490                for (int i = 0; i < size; i++) {
491                        array[i] = (Vector) vector.elementAt(i);
492                }
493                return array;
494        }

496        /**
497         * Checks for a valid string for a numerical type, empty is NOT valid if not
498         * valid then return the MAX_VALUE for the number
499         * 
500         * @param aValue
501         *            the value to be checked
502         * @param aType
503         *            the type of number it is
504         * @return the new value
505         */
506        private String getValidStringValueForType(String aValue, String aType) {
507                String valueStr = aValue;
508                Integer iVal = (Integer) mType2IntHash.get(aType);
509                switch (iVal.intValue()) {
510                case 0:
511                        try {
512                                Integer.parseInt(aValue);
513                        } catch (Exception e) {
514                                valueStr = Integer.toString(Integer.MAX_VALUE);
515                        }
516                        break;
517
518                case 1:
519                        try {
520                                Float.parseFloat(aValue);
521                        } catch (Exception e) {
522                                valueStr = Float.toString(Float.MAX_VALUE);
523                        }
524                        break;
525
526                case 2:
527                        try {
528                                Double.parseDouble(aValue);
529                        } catch (Exception e) {
530                                valueStr = Double.toString(Double.MAX_VALUE);
531                        }
532                        break;
533
534                case 3:
535                        try {
536                                Long.parseLong(aValue);
537                        } catch (Exception e) {
538                                valueStr = Long.toString(Long.MAX_VALUE);
539                        }
540                        break;
541
542                case 4:
543                        // no-op
544                        break;
545                }
546                return valueStr;
547        }

574        /**
575         * Creates a string table from the resultset records
576         * 
577         * @param aRS
578         *       */
579        private String createTableFromResultset(String delim,
580                        ResultsetTypeRecord[] records) {
581                StringBuffer tableStr = new StringBuffer();
582                StringBuffer rowOfData = new StringBuffer();
583                DSTableIFace tableSchema = (DSTableIFace) _resultsetSchemaDef
584                                .getTables().elementAt(0);
585                Vector columns = tableSchema.getFields();
586
587                appendHeaderRow(tableStr, columns, delim, ROWDELIM);
588
589                for (int i = 0; i < records.length; i++) {
590                        rowOfData.setLength(0);
591
592                        //
593                        // Note: this code assumes all the columns are returned for every
594                        // record.
595                        // It also assumes they are in the same order as in the metadata.
596                        Enumeration colEnum = columns.elements();
597                        int colInx = 0;
598                        while (colEnum.hasMoreElements()) {
599                                DSTableFieldIFace colDef = (DSTableFieldIFace) colEnum
600                                                .nextElement();
601                                ResultsetTypeRecordReturnField field = records[i]
602                                                .getReturnField(colInx);
603                                String eleStr = field.get_value();
604                                if (colInx > 0) {
605                                        rowOfData.append(delim);
606                                }
607                                rowOfData.append(getValidStringValueForType(eleStr, colDef
608                                                .getDataType()));
609                                colInx++;
610                        }
611                        tableStr.append(rowOfData);
612                        tableStr.append(ROWDELIM);
613                }
614                return tableStr.toString();
615        }

617        /**
618         * Creates a string table from the resultset records
619         * 
620         * @param aRS
621         *       */
622        private String createTableFromSQLResults(boolean aIncludeHeader,
623                        String aDelim) {
624                StringBuffer tableStr = new StringBuffer();
625                StringBuffer rowOfData = new StringBuffer();
626                String colDelim = "\t";
627
628                if (aIncludeHeader) {
629                        appendHeaderRow(tableStr, _columns, colDelim, ROWDELIM);
630                }
631
632                for (int i = 0; i < _dataVectors.length; i++) {
633                        rowOfData.setLength(0);
634
635                        Enumeration colEnum = _columns.elements();
636                        Enumeration dataEnum = _dataVectors[i].elements();
637                        int colInx = 0;
638                        while (colEnum.hasMoreElements()) {
639                                DSTableFieldIFace colDef = (DSTableFieldIFace) colEnum
640                                                .nextElement();
641                                String eleStr = (String) dataEnum.nextElement();
642                                if (colInx > 0) {
643                                        rowOfData.append(colDelim);
644                                }
645
646                                rowOfData.append(getValidStringValueForType(eleStr, colDef
647                                                .getDataType()));
648                                colInx++;
649                        }
650                        tableStr.append(rowOfData);
651                        tableStr.append(ROWDELIM);
652                }
653                return tableStr.toString();
654        }

667        /**
668         * Send an entire tableas a single token over the port on each fire event.
669         * 
670         * @exception IllegalActionException
671         *                If there is no director.
672         */
673        private void fireAsTable() throws IllegalActionException {
674                if (_tableEntity == null) {
675                        System.err.println("_tableEntity is null in fireAsTable");
676                        return;
677                }
678
679                TypedIOPort pp = (TypedIOPort) this.getPort(DATATABLE);
680                pp.send(0, new StringToken(createTableFromSQLResults(false, "\t")));
681
682                pp = (TypedIOPort) this.getPort(DELIMITER);
683                String delim = _tableEntity.getDelimiter();
684                pp.send(0, new StringToken(delim));
685
686                pp = (TypedIOPort) this.getPort(NUMCOLUMNS);
687                pp.send(0, new IntToken(_tableEntity.getAttributes().length));

1429         * actor that exists in the actor library. The default documentation is only
1430         * loaded once since it will likely remain quite static. If the given actor

1432         * attribute is preserved (nothing is changed).
1433         * 
1434         * @param emlActor
1435         *            the instance to which documentation will be added
1436         */
1437        public static void generateDocumentationForInstance(
1438                        DarwinCoreDataSource actor) {

1440                        // only look up the documentation only once
1441                        if (defaultDocumentation == null) {
1442                                Iterator cacheIter = CacheManager.getInstance()
1443                                                .getCacheObjectIterator();
1444                                while (cacheIter.hasNext()) {
1445                                        Object co = cacheIter.next();
1446                                        // is this an actor cache object?
1447                                        if (co instanceof ActorCacheObject) {
1448                                                ActorCacheObject aco = (ActorCacheObject) co;
1449                                                // for this class?
1450                                                if (aco.getClassName().equals(actor.getClassName())) {
1451                                                        // get the metadata
1452                                                        ActorMetadata am = aco.getMetadata();
1453                                                        // get the default documentation from the metadata
1454                                                        defaultDocumentation = am
1455                                                                        .getDocumentationAttribute();
1456                                                        log
1457                                                                        .debug("looked up default KeplerDocumentation contained in "
1458                                                                                        + am.getName());
1459                                                        break;
1460                                                }
1461                                        }
1462                                }
1463                        }
1464
1465                        // add the documentation for this actor if it is not there already
1466                        if (actor.getAttribute("KeplerDocumentation") == null) {
1467                                // make an instance of the documentation attribute for the input
1468                                // actor
1469                                KeplerDocumentationAttribute keplerDocumentation = new KeplerDocumentationAttribute(
1470                                                actor, "KeplerDocumentation");
1471
1472                                // copy the default and set it for this one
1473                                keplerDocumentation
1474                                                .createInstanceFromExisting(defaultDocumentation);
1475                                keplerDocumentation.setContainer(actor);
1476                                log.debug("set the KeplerDocumentation for actor instance: "
1477                                                + actor.getName());
1478
1479                        }
1480                } catch (Exception e) {

1483                                                        + e.getMessage());
1484                        e.printStackTrace();
1485                }
