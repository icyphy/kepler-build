054/**
055 * The Darwin Core Schema instance.
056 */
057public class DarwinCoreSchema implements DataCacheListener {
058        static final String XSD_TYPES[] = { "string", "nonNegativeInteger",
059                        "decimal", "gYear", "dateTime" };
060        static final String KEPLER_TYPES[] = { DataType.STR, DataType.INT,
061                        DataType.FLOAT, DataType.STR, DataType.STR };
062
063        private Hashtable mFieldNameToType = null;
064        private Hashtable mXSD2KeplerHashtable = null;
065        private DSSchemaDef mSchemaDef = null;
066
067        /*
068         * The xpath in Config used to extract url for Darwin Core schema.
069         */
070        private static final String schemaXpath = "//ecogridService/digir/schema";
071
072        /*
073         * Synchronization latch. It's released when the data has been downloaded.
074         */
075        private Latch mFinished;
076
077        /*
078         * Singleton instance.
079         */
080        private static DarwinCoreSchema mGlobalDarwinCoreSchema = null;
081
082        /*
083         * Logger.
084         */
085        private static Log log;
086
087        static {
088                log = LogFactory
089                                .getLog("org.ecoinformatics.seek.datasource.darwincore.DarwinCoreSchema");
090        }
091
092        /**
093         * Returns the singleton
094         * 
095         *       */
096        public synchronized static DarwinCoreSchema getInstance() {
097                if (mGlobalDarwinCoreSchema == null) {
098                        mGlobalDarwinCoreSchema = new DarwinCoreSchema();
099                }
100                return mGlobalDarwinCoreSchema;
101        }
102
103        /**
104         * 
105         * @return the schema object for the DarwinCore
106         */
107        public DSSchemaIFace getSchema() {
108                // TODO - implement timeout here and raise exception.
109                try {
110                        mFinished.acquire();
111                } catch (InterruptedException e) {
112                        log.error("Wait for getSchema interrupted", e);
113                        throw new RuntimeException("DarwinCoreSchema acquire interrupted",
114                                        e);
115                }
116                return mSchemaDef;
117        }
118
119        public String lookupTypeFromName(String aName) {
120                // TODO - implement timeout here and raise exception.
121                try {
122                        mFinished.acquire();
123                } catch (InterruptedException e) {
124                        log.error("Wait for getSchema interrupted", e);
125                        throw new RuntimeException("DarwinCoreSchema acquire interrupted",
126                                        e);
127                }
128                String type = (String) mFieldNameToType.get(aName);
129                if (type == null) {
130                        return "STRING";
131                }
132                return type;
133        }
134
135        /**
136         * Construct s an object to produce a DarwinCoreSchema
137         */
138        protected DarwinCoreSchema() {
139                mFinished = new Latch();
140
141                mXSD2KeplerHashtable = new Hashtable();
142                mFieldNameToType = new Hashtable();
143                for (int i = 0; i < XSD_TYPES.length; i++) {
144                        mXSD2KeplerHashtable.put(XSD_TYPES[i], KEPLER_TYPES[i]);
145                }
146
147    //schemaXpath: //ecogridService/digir/schema
148    String urlStr;
149    //get the configuration for this module
150    ConfigurationManager confMan = ConfigurationManager.getInstance();
151    //get the specific configuration we want
152    ConfigurationProperty ecogridProperty = confMan.getProperty(ConfigurationManager.getModule("ecogrid"));
153    ConfigurationProperty schemaProperty = ecogridProperty.getProperty("digir.schema");
154    urlStr = schemaProperty.getValue();
155    
156                EcogridDataCacheItem darwinCoreDataItem = (EcogridDataCacheItem) DataCacheManager
157                                .getCacheItem(this, "DarwinCoreSchema", urlStr,
158                                                EcogridDataCacheItem.class.getName());
159                darwinCoreDataItem.start();
160        }
161
162        public void complete(DataCacheObject item) {
163                try {
164                        if (!item.isReady()) {
165                                // TODO : proper error handling.
166                        }
167
168                        Document dom = null;
169                        try {
170                                dom = EcogridUtils.readXMLFile2DOM(item.getAbsoluteFileName());
171                                if (dom == null) {
172                                        // TODO: proper error handling
173                                        return;
174                                }
175
176                        } catch (Exception e) {
177                                log.error("Exception occurred while parsing schema", e);
178                                return;
179                        }
180
181                        // EcogridUtils.printNode(dom, 0);
182                        Node schemaNode = null;
183                        StringBuffer nameStrBuf = new StringBuffer();
184                        StringBuffer nameSpaceStrBuf = new StringBuffer();
185                        NodeList list = dom.getChildNodes();
186                        if (list != null) {
187                                for (int i = 0; i < list.getLength(); i++) {
188                                        Node child = list.item(i);
189                                        if (child != null) {
190                                                EcogridUtils.parseForNameAndNamespace(child
191                                                                .getNodeName(), nameStrBuf, nameSpaceStrBuf);
192                                                if (nameStrBuf.toString().equals("schema")) {
193                                                        schemaNode = child;
194                                                        break;
195                                                }
196
197                                        }
198                                }
199                        }
200                        if (schemaNode == null)
201                                return;
202
203                        mSchemaDef = new DSSchemaDef();
204                        DSTableDef tableDef = new DSTableDef("DarwinCode");
205                        mSchemaDef.addTable(tableDef);
206
207                        // XXX should substitute this later with XSLT transform for easier
208                        // processing
209                        String elementName = nameSpaceStrBuf.length() > 0 ? nameSpaceStrBuf
210                                        .toString()
211                                        + ":element" : "element";
212                        EcogridUtilsNamedNodeIterator iter = new EcogridUtilsNamedNodeIterator(
213                                        schemaNode, elementName);
214                        while (iter.hasMoreNodes()) {
215                                Node elementNode = iter.nextNode();
216
217                                String name = EcogridUtils.findAttrValue(elementNode, "name");
218                                StringBuffer nameSB = new StringBuffer();
219                                StringBuffer nsSB = new StringBuffer();
220                                EcogridUtils.parseForNameAndNamespace(name, nameSB, nsSB);
221                                // System.out.println("nameSB: "+nameSB);
222
223                                StringBuffer typeSB = new StringBuffer();
224                                StringBuffer typeNSSB = new StringBuffer();
225
226                                String type = EcogridUtils.findAttrValue(elementNode, "type");
227                                if (type == null) {
228                                        String restrictionName = nameSpaceStrBuf.length() > 0 ? nameSpaceStrBuf
229                                                        .toString()
230                                                        + ":restriction"
231                                                        : "restriction";
232                                        Node restriction = EcogridUtils.findNode(elementNode,
233                                                        restrictionName);
234                                        if (restriction != null) {
235                                                type = EcogridUtils.findAttrValue(restriction, "base");
236                                        }
237                                }
238                                EcogridUtils.parseForNameAndNamespace(type, typeSB, typeNSSB);
239                                // System.out.println("typeSB: "+typeSB);
240                                if (nameSB.length() > 0 && typeSB.length() > 0) {
241                                        String keplerType = (String) mXSD2KeplerHashtable
242                                                        .get(typeSB.toString());
243                                        if (keplerType == null) {
244                                                keplerType = typeSB.toString();
245                                                log.error("Missing a Mapping from XSD ["
246                                                                + typeSB.toString() + "] to Kepler type.");
247                                        }
248                                        tableDef.addField(nameSB.toString(), keplerType, null);
249                                        mFieldNameToType.put(nameSB.toString(), keplerType);
250                                } else {
251                                        // XXX
252                                        // System.err.println("["+nameSB.toString()+"]["+typeSB.toString()+"]");
253                                }
254                        }
255                        // XXX
256                        // System.out.println(DBSchemaParserEmitter.emitXML(mSchemaDef));
257
258                } finally {
259                        mFinished.release();
260                }
261        }
