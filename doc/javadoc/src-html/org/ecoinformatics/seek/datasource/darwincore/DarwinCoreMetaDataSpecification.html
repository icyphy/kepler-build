057/**
058 * This class is a plugin for DarwinCore namespace to transfer query group to
059 * ecogrid query type. It also provide a method to create quick search query
060 * 
061 * @author Rod Spears (adpated from EML2MetadataSpecification.java)
062 */
063
064public class DarwinCoreMetaDataSpecification extends
065                MetadataSpecificationInterface {
066        private static final String REPLACE = "#value#";
067        private static final String QUERYID = "darwincore-quick-search-query";
068
069        private int _numResults = 0;
070        private Hashtable systemProviderHash = new Hashtable();
071        private Hashtable fieldNametoIdHash = new Hashtable();
072
073        /**
074         * Default constructor
075         */
076        public DarwinCoreMetaDataSpecification() {
077
078        } // DarwinCoreMetaDataSpecification
079
080        /**
081         * Method to create a quick query search
082         * 
083         * @param value
084         *            String
085         * @return QueryType
086         */
087        public QueryType getQuickSearchEcoGridQuery(String aValue)
088                        throws InvalidEcogridQueryException {
089                Hashtable replaceMapping = new Hashtable();
090                replaceMapping.put(REPLACE, aValue);
091                SearchQueryGenerator darwinCoreQueryGenerator = new SearchQueryGenerator(
092                                QUERYID, replaceMapping);
093                QueryType ecogridQuery = darwinCoreQueryGenerator.getQuery();
094                return ecogridQuery;
095
096        } // getQuickSearchEcoGridQuery
097
098        /**
099         * Method to create a ecogrid query
100         * 
101         * @return QueryType
102         */
103        public QueryType getEcoGridQuery() {
104                QueryType ecogridQuery = null;
105                return ecogridQuery;
106        } // getEcoGridQuery
107
108        /**
109         * Helper class to return a named field's value (content)
110         * 
111         * @param aRec
112         *            the record
113         * @param aFieldName
114         *            the name of the field
115         * @return the contents
116         */
117        private static String getReturnFieldValue(ResultsetTypeRecord aRec,
118                        String aFieldName) {
119                ResultsetTypeRecordReturnField[] fields = aRec.getReturnField();
120                if (fields != null) {
121                        for (int i = 0; i < fields.length; i++) {
122                                if (fields[i].getId().equals(aFieldName)) {
123                                        return fields[i].get_value();
124                                }
125                        }
126                }
127                return null;
128        }
129
130        private void populateSystemHash(ResultsetTypeResultsetMetadata metaData) {
131                ResultsetTypeResultsetMetadataSystem[] system = metaData.getSystem();
132                if (system != null) {
133                        for (int i = 0; i < system.length; i++) {
134                                systemProviderHash.put(system[i].getId(), system[i].get_value()
135                                                .toString());
136                        }
137                }
138        }
139
140        private void populateFieldHash(ResultsetTypeResultsetMetadata metaData) {
141                ResultsetTypeResultsetMetadataRecordStructure rs = metaData
142                                .getRecordStructure();
143                ResultsetTypeResultsetMetadataRecordStructureReturnField[] fields = rs
144                                .getReturnField();
145                if (fields != null) {
146                        for (int i = 0; i < fields.length; i++) {
147                                String value = fields[i].getName();
148                                String id = fields[i].getId();
149                                fieldNametoIdHash.put(value, id);
150                        }
151                }
152        }
153
154        private String findFieldId(String fieldName,
155                        ResultsetTypeResultsetMetadata metaData) {
156                ResultsetTypeResultsetMetadataRecordStructure rs = metaData
157                                .getRecordStructure();
158                ResultsetTypeResultsetMetadataRecordStructureReturnField[] fields = rs
159                                .getReturnField();
160                if (fields != null) {
161                        for (int i = 0; i < fields.length; i++) {
162                                String value = fields[i].getName();
163                                if (value.endsWith(fieldName)) {
164                                        return fields[i].getId();
165                                }
166                        }
167                }
168
169                return null;
170        }
171
172        /**
173         * Creates a DarwinCore DataSource which extends Source
174         * 
175         * @param aResults
176         * @param aEndPointURLStr
177         * @param aContainer
178         * @param aResultList
179         * @throws SAXException
180         * @throws IOException
181         * @throws EcoGridException
182         * @throws NameDuplicationException
183         * @throws IllegalActionException
184         */
185        public boolean addResultsetRecordsToContainer(ResultsetType aResults,
186                        String aEndPointURLStr, CompositeEntity aContainer,
187                        Vector aResultList) throws NameDuplicationException,
188                        IllegalActionException {
189                _numResults = 0;
190                if (aResults == null) {
191                        return false;
192                }
193
194                Hashtable speciesHash = new Hashtable();
195                String fieldId = null;
196
197                // collect all the providers into a hash table
198                ResultsetTypeResultsetMetadata metaData = aResults
199                                .getResultsetMetadata();
200                if (metaData != null) {
201                        populateSystemHash(metaData);
202                        populateFieldHash(metaData);
203                        fieldId = findFieldId("ScientificName", metaData);
204                }
205
206                ResultsetTypeRecord[] records = aResults.getRecord();
207                if (records == null) {
208                        return false;
209                }
210
211                if (records.length == 0) {
212                        return false;
213                }
214                for (int i = 0; i < records.length; i++) {
215                        ResultsetTypeRecord rec = records[i];
216                        String scientificName = getReturnFieldValue(rec, fieldId);
217                        // Drop records
218                        if (scientificName == null || scientificName.trim().length() == 0) {
219                                continue;
220                        }
221                        scientificName = MetadataSpecificationInterface
222                                        .replaceDotByDash(scientificName);
223                        // convert the scientific name to lowercase -- effictively making
224                        // the hash table case insensitive.
225                        scientificName = scientificName.toLowerCase();
226                        Hashtable providers = (Hashtable) speciesHash.get(scientificName);
227                        if (providers == null) {
228                                providers = new Hashtable();
229                                speciesHash.put(scientificName, providers);
230                                // System.out.println("*** Putting ScientificName["+scientificName+"]");
231                        }
232                        providers.put(rec.getSystem(), rec.getSystem());
233                        // System .out.println("  System["+rec.getSystem()+"]");
234                }
235
236                // Sort the keys alphabetically for presentation.
237                Vector sortedKeys = new Vector(speciesHash.keySet());
238                Collections.sort(sortedKeys);
239                Iterator it = sortedKeys.iterator();
240                while (it.hasNext()) {
241                        String scientificName = (String) it.next();
242                        Hashtable providers = (Hashtable) speciesHash.get(scientificName);
243                        StringBuffer providerList = new StringBuffer();
244                        for (Enumeration ep = providers.keys(); ep.hasMoreElements();) {
245                                String providerId = (String) ep.nextElement();
246                                String provider = (String) systemProviderHash.get(providerId);
247                                if (providerList.length() > 0) {
248                                        providerList.append(",");
249                                }
250                                providerList.append(provider);
251                        }
252
253                        // System.out.println("*** ProviderList["+scientificName+"]["+providerList+"]");
254                        DarwinCoreDataSource darwinDS = new DarwinCoreDataSource(
255                                        aContainer, scientificName);
256                        darwinDS.setSearchData(scientificName);
257                        darwinDS.setEndpoint(aEndPointURLStr);
258                        DarwinCoreDataSource.generateDocumentationForInstance(darwinDS);
259                        aResultList.addElement(darwinDS);
260                        _numResults++;
261
262                }
263
264                return _numResults > 0;
265        }
