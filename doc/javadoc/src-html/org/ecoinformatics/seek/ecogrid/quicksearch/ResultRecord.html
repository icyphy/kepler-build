128        /**
129         * Set the identifier of this record.
130         * 
131         * @param id

134        public void setRecordId(String id) {
135                try {
136                        StringAttribute att = (StringAttribute) this.getAttribute(RECORDID);
137                        if (att == null) {
138                                att = new StringAttribute(this, RECORDID);
139                        }
140                        att.setExpression(id);
141                } catch (IllegalActionException iae) {
142                        System.err.println("Could not add the record id.");
143                } catch (NameDuplicationException nde) {
144                        System.err.println("Could not add the record id.");
145                }
146        }

255        /**
256         * Determine if the recordId and endpoint attributes have valid values for
257         * use in retrieving the record for parsing.
258         * 
259         * @return boolean true if both recordid and endpoint are not null and are
260         *         not the empty string
261         */
262        public boolean hasConnectionValues() {
263                boolean hasValues = false;
264                String recordId = this.getRecordId();
265                String endpoint = this.getEndpoint();
266
267                if (recordId != null && !recordId.equals("") && endpoint != null
268                                && !endpoint.equals("")) {
269                        hasValues = true;
270                }
271                return hasValues;
272        }

274        public Reader getFullRecord() {
275                Reader recordReader = null;
276                EcogridMetaDataCacheItem item = (EcogridMetaDataCacheItem) DataCacheManager
277                                .getCacheItem(null, "MetaData " + this.getRecordId(), this
278                                                .getEndpoint(), EcogridMetaDataCacheItem.class
279                                                .getName());
280                item.setEndPoint(this.getEndpoint());
281                item.setRecordId(this.getRecordId());
282                item.start();
283
284                // make sure the item is finished
285                while (!item.isError() && !item.isReady()) {
286                        // do nothing, just waiting
287                        // System.out.println("Waiting!!!!!!!!!!");
288                }
289                // when it is ready
290                InputStream stream = item.getDataInputStream();
291                if (stream != null) {
292                        recordReader = new InputStreamReader(stream);
293                }
294
295                return recordReader;
296        }

303        /**
304         * Method to transfer a array of ResultRecord to a Vector
305         * 
306         * @param source
307         *            the source array of ResultRecord
308         * @param dest
309         *            the destation vector the element will be added.
310         */
311        public static void transformResultRecordArrayToVector(
312                        ResultRecord[] source, Vector dest) {
313                if (source == null || dest == null) {
314                        return;
315                }
316                int size = source.length;
317                for (int i = 0; i < size; i++) {
318                        ResultRecord record = source[i];
319                        dest.add(record);
320                }
321        }
