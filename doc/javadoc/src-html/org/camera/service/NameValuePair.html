030package org.camera.service;
031
032//////////////////////////////////////////////////////////////////////////
033////NameValuePair
034/**
035<p>The NameValuePair is  a place holder for name value pairs.
036</p>
037 * 
038 * @author Madhu, SDSC
039 * @version $Id: NameValuePair.java 24234 2010-05-06 05:21:26Z welker $
040 */
041public class NameValuePair{
042        
043        private String mName, mValue;
044        
045        /**
046         * @param mName
047         * @param mValue
048         */
049        public NameValuePair(String mName, String mValue){
050                this.mName = mName;
051                this.mValue = mValue;
052        }
053        
054        /**
055         *       */
056        public String getName() {
057                return mName;
058        }
059
060        /**
061         * @param pName
062         */
063        public void setName(String pName) {
064                mName = pName;
065        }
066
067        /**
068         *       */
069        public String getValue() {
070                return mValue;
071        }
072
073        /**
074         * @param pValue
075         */
076        public void setValue(String pValue) {
077                mValue = pValue;
078        }
079}
