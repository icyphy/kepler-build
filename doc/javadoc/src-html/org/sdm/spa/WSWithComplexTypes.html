104/**
105 * <p>
106 * 
107 * This actor executes SOAP web services defined by WSDLs. Given a web service's URL
108 * of a WSDL and an operation name, this actor specializes its input and output
109 * ports to reflect the input and output parameters of the operation. For simple
110 * web service types, e.g. string, int, or double, this actor's ports are
111 * configured to the matching ptolemy type. Otherwise, the ports are set to
112 * XMLTOKEN. When this actor fires, it reads each input port, invokes the web
113 * service operation and sends the input data, and outputs the response to the
114 * output ports.
115 * 
116 * </p>
117 * <p>
118 * 
119 * The <i>inputMechanism</i> and <i>outputMechanism</i> parameters control the
120 * creation of helper actors for complex/nested web service types. (These
121 * parameters have no effect for simple web service types). By setting either to
122 * 'composite', a composite actor is created for each parameter that is
123 * complex/nested. Each composite actor is populated with necessary XML
124 * Assembler or XML Disassembler actors needed to build the nested web service
125 * type, and the composite actor ports are all simple ptolemy types. Changing
126 * the mechanism back to 'simple' <b>deletes</b> the connected helper actors. If
127 * you have made changes to the composite actors and don't want them lost,
128 * disconnect them from this actor before changing the mechanism to 'simple'.
129 * 
130 * </p>
131 * <p>
132 * 
133 * <b>Limitations:</b>
134 * <ul>
135 * <li>Unused input ports on composite actors must have the corresponding
136 * internal links to XML Assembler actors removed. This is because the XML
137 * Assembler actors will read a token from each input port whose width is
138 * greater than 0.
139 * <li>The layout of the composite actors on the canvas is not perfect;
140 * sometimes a composite actor is placed over an existing actor.
141 * <li>If the input to the web service contains an array of a nested type, the
142 * generated composite actors will use SequenceToArray with <i>arrayLength</i>
143 * defaulting to 1. For longer arrays, you must manually create them and send to
144 * the appropriate XML Assembler actor.
145 * <li>Web service responses containing multi-reference values (elements
146 * refering to other elements for their content) are not handled.
147 * <li>If the WSDL doesn't fully define the operation response, then the
148 * corresponding output port is set to XMLTOKEN. You can access the values by
149 * using XML Disassembler actor(s).
150 * <li>A web service parameter with the WSDL type "any" sets the corresponding
151 * actor port type to XMLTOKEN.
152 * <li>Multidimensional arrays not handled and the corresponding port is set to
153 * XMLTOKEN. You can get or set the values by using XML Assembler or
154 * Disassembler actor(s).
155 * </ul>
156 * 
157 * </p>
158 * 
159 * TODO: handle commit vs cancel? multiple created array conversion actors not
160 * moved vertically
161 * 
162 * @author Daniel Crawl
163 * @version $Id: WSWithComplexTypes.java 31346 2013-01-17 22:55:09Z crawl $
164 */

292         *            The changed parameter.
293         * @exception IllegalActionException

296        public void attributeChanged(Attribute attribute)
297                        throws IllegalActionException {
298                try {
299                        if (attribute == wsdl) {
300                                String str = ((StringToken)wsdl.getToken()).stringValue();
301                                // System.out.println("wsdl is now " + str);
302
303                                if (!str.equals(_wsdlStr)) {
304                                        _wsdlStr = str;
305                                        _initWSDL();
306                                }
307                        } else if (attribute == method) {
308                                String str = ((StringToken)method.getToken()).stringValue();
309                                //System.out.println(getName() + "method is now " + str);
310
311                                if (!str.equals(_methodStr)) {
312                                        _methodStr = str;
313                                        _initMethod();
314                                }
315                        } else if (attribute == inputMechanism) {
316                                String str = ((StringToken)inputMechanism.getToken()).stringValue();
317
318                                _inputIO = -1;
319                                for (int i = 0; i < _ioTypes.length; i++) {
320                                        if (_ioTypes[i].equals(str)) {
321                                                _inputIO = i;
322                                                break;
323                                        }
324                                }
325                                if (_inputIO == -1) {
326                                        throw new IllegalActionException(this,
327                                                        "Invalid input mechanism: " + str);
328                                }
329
330                                _genInputPortsAndActors();
331                        } else if (attribute == outputMechanism) {
332                                String str = ((StringToken)outputMechanism.getToken()).stringValue();
333
334                                _outputIO = -1;
335                                for (int i = 0; i < _ioTypes.length; i++) {
336                                        if (_ioTypes[i].equals(str)) {
337                                                _outputIO = i;
338                                                break;
339                                        }
340                                }
341                                if (_outputIO == -1) {
342                                        throw new IllegalActionException(this,
343                                                        "Invalid input mechanism: " + str);
344                                }
345
346                                _genOutputPortsAndActors();
347                        } else if (attribute == outputNil) {
348                                _outputNilVal = ((BooleanToken) outputNil.getToken())
349                                                .booleanValue();
350                                _helper.setOutputNil(_outputNilVal);
351                        }
352                } catch (NameDuplicationException e) {
353                        throw new IllegalActionException(this, "NameDuplicationException: "
354                                        + e.getMessage());
355                }
356
357                super.attributeChanged(attribute);
358        }

437        /** Initialize the WSDL Parser and find the SOAP endpoint. */
438        private void _initWSDL() throws IllegalActionException,
439                        NameDuplicationException {
440                int i;
441
442                // parse the WSDL
443                try {
444                        _wsdlParser = new Parser();
445                        _wsdlParser.run(_wsdlStr);
446                } catch (Exception e) {
447                        throw new IllegalActionException(this, "Could not parse WSDL: "
448                                        + e.getMessage());
449                }
450
451                // now find the entry for the ServiceEntry class in the symbol table.
452                _wsdlService = null;
453
454                HashMap<?,?> map = _wsdlParser.getSymbolTable().getHashMap();
455                Iterator<?> entrySetIter = map.entrySet().iterator();
456                while (entrySetIter.hasNext()) {
457                        Map.Entry<?,?> currentEntry = (Map.Entry<?,?>) entrySetIter.next();
458                        Vector<?> valueVector = (Vector<?>) currentEntry.getValue();
459                        for (i = 0; i < valueVector.size(); i++) {
460                                SymTabEntry symTabEntryObj = (SymTabEntry) valueVector.get(i);
461                                if ((ServiceEntry.class).isInstance(symTabEntryObj)) {
462                                        _wsdlService = ((ServiceEntry) symTabEntryObj).getService();
463                                        break;
464                                }
465                        }
466                        if (_wsdlService != null) {
467                                break;
468                        }
469                }
470
471                if (_wsdlService == null) {
472                        throw new IllegalActionException(this, "Could not find service");
473                }
474
475                // find a port with a SOAPAddress extensibility element.
476                _wsdlPort = null;
477                Map<?,?> ports = _wsdlService.getPorts();
478                Iterator<?> nameIter = ports.keySet().iterator();
479                while (nameIter.hasNext()) {
480                        String portName = (String) nameIter.next();
481                        Port port = (Port) ports.get(portName);
482                        List<?> extElemList = port.getExtensibilityElements();
483                        for (i = 0; extElemList != null && i < extElemList.size(); i++) {
484                                Object extEl = extElemList.get(i);
485                                if (extEl instanceof SOAPAddress) {
486                                        _wsdlPort = port;
487                                        break;
488                                }
489                        }
490                        if (_wsdlPort != null) {
491                                break;
492                        }
493                }
494
495                if (_wsdlPort == null) {
496                        throw new IllegalActionException(this, "Could not find port");
497                }
498
499                // now add the valid choices to the method parameter
500                method.removeAllChoices();
501
502                BindingEntry be = _wsdlParser.getSymbolTable().getBindingEntry(
503                                _wsdlPort.getBinding().getQName());
504                Iterator<?> iter = be.getParameters().keySet().iterator();
505                while (iter.hasNext()) {
506                        Operation oper = (Operation) iter.next();
507                        method.addChoice(oper.getName());
508                }
509        }

511        /** Get the SOAP parameters for a method. */
512        private void _initMethod() throws IllegalActionException,
513                        NameDuplicationException {
514                _wsdlParams = null;
515
516        if(_wsdlParser == null)
517        {
518            throw new IllegalActionException(this, "wsdl parser is null");    
519        }
520        else if(_wsdlPort == null)
521        {
522            throw new IllegalActionException(this, "wsdl port is null");
523        }
524
525                BindingEntry be = _wsdlParser.getSymbolTable().getBindingEntry(
526                                _wsdlPort.getBinding().getQName());
527                Iterator<?> iter = be.getParameters().keySet().iterator();
528                while (iter.hasNext()) {
529                        Operation oper = (Operation) iter.next();
530                        if (oper.getName().equals(_methodStr)) {
531                                _wsdlParams = (Parameters) be.getParameters().get(oper);
532                                break;
533                        }
534                }
535
536                if (_wsdlParams == null) {
537                        throw new IllegalActionException(this,
538                                        "Could not get parameters for method " + _methodStr);
539                }
540
541                // determine the use
542                _methodUse = null;
543                iter = _wsdlPort.getBinding().getBindingOperations().iterator();
544                while (iter.hasNext()) {
545                        BindingOperation bo = (BindingOperation) iter.next();
546                        if (bo.getName().equals(_methodStr)) {
547                                List<?> ext = null;
548                                if ((ext = bo.getBindingInput().getExtensibilityElements()) != null
549                                                && ext.size() > 0) {
550                                        SOAPBody body = (SOAPBody) ext.get(0);
551                                        _methodUse = body.getUse();
552                                } else if ((ext = bo.getBindingOutput()
553                                                .getExtensibilityElements()) != null
554                                                && ext.size() > 0) {
555                                        SOAPBody body = (SOAPBody) ext.get(0);
556                                        _methodUse = body.getUse();
557                                }
558                                break;
559                        }
560                }
561
562                if (_methodUse == null) {
563                        throw new IllegalActionException(this,
564                                        "Could not find binding operation for " + _methodStr);
565                }
566
567                // create the mapping between namespaces and types
568                _nsTypeMap = createNSMapping(_wsdlParams.list);
569
570                // NOTE: we remove unused ports here for both input and
571                // output before creating the new ones since the old
572                // method could have had a input parameter with the same
573                // name as an output parameter in the current method
574                // (or vice-versa).
575                _removeUnusedInputPortsAndActors();
576                _removeUnusedOutputPortsAndActors();
577
578                _genInputPortsAndActors();
579                _genOutputPortsAndActors();
580        }

582        /** Create input ports and actors. */
583        private void _genInputPortsAndActors() throws IllegalActionException,
584                        NameDuplicationException {
585                if (_wsdlParams != null) {
586                        _removeUnusedInputPortsAndActors();
587
588                        _linkedPortsMap = new LinkMap(this, true);
589
590                        // create the input ports
591                        Object[] paramArray = _wsdlParams.list.toArray();
592                        Arrays.sort(paramArray, _paramComp);
593                        for (int i = 0; i < paramArray.length; i++) {
594                                Parameter p = (Parameter) paramArray[i];
595                                _createOnePort(this, null, p.getName(), p.getType(), true);
596                        }
597                }
598        }

689         * For XML data types see:
690         * http://www.w3.org/TR/2004/REC-xmlschema-2-20041028/#built-in-datatypes
691         * 
692         * @param typeStr
693         *            the type string
694         * @param name
695         *            the parameter name
696         * @return the ptolemy type
697         */
698        private static Type _getTypeFromWSDLStr(String typeStr, String name) {
699                Type retval = null;
700
701                if (typeStr.equals("string")) {
702                        retval = BaseType.STRING;
703                } else if (typeStr.equals("boolean")) {
704                        retval = BaseType.BOOLEAN;
705                } else if (typeStr.equals("double") || typeStr.equals("float")) {
706                        retval = BaseType.DOUBLE;
707                } else if (typeStr.equals("long")
708                                || typeStr.equals("unsignedLong")
709                                // according to the specification, these don't have
710                                // a max or min range so set them to be long instead
711                                // of int. maybe basetype.scalar would be more appropriate?
712                                || typeStr.equals("integer")
713                                || typeStr.equals("nonPositiveInteger")
714                                || typeStr.equals("negativeInteger")
715                                || typeStr.equals("nonNegativeInteger")
716                                || typeStr.equals("positiveInteger")) {
717                        retval = BaseType.LONG;
718                } else if (typeStr.equals("int") || typeStr.equals("short")
719                                || typeStr.equals("byte") || typeStr.equals("unsignedInt")
720                                || typeStr.equals("unsignedShort")
721                                || typeStr.equals("unsignedByte")) {
722                        retval = BaseType.INT;
723                } else if (typeStr.equals("unsignedByte")) {
724                        retval = BaseType.UNSIGNED_BYTE;
725                } else {
726                        // throw new IllegalActionException(this, "Unknown type: "
727                        // + typeStr);
728                        _log.warn("WARNING: " + name + " has unknown WSDL type: " + typeStr);
729                }
730                return retval;
731        }

1078                        throws IllegalActionException, NameDuplicationException {
1079                boolean retval = false;

1082

1084                        portName = portName.substring(_outputNamePrepend.length());

1087

1090                        Iterator<?> conPorts = list.iterator();
1091                        while (conPorts.hasNext()) {
1092                                IOPort p = (IOPort) conPorts.next();

1099                        }
1100                }
1101
1102                // if the composite actor was found and the port will be

1104                if (retval
1105                                && (portWillBeDeleted || _getIOSetting(port.isInput()) == IO_SIMPLE)) {

1107                        composite.removeAllEntities();
1108                        composite.removeAllPorts();

1115                    ((TypedIOPort)port).getType() == BaseType.XMLTOKEN) {
1116                    List<?> relations = port.linkedRelationList();
1117                    for(Object object : relations)
1118                    {
1119                        if(object != null)
1120                        {
1121                            ((IORelation)object).setContainer(null);
1122                        }
1123                    }
1124                    //port.unlinkAll();
1125                }
1126
1127                return retval;
1128        }
