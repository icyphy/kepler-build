049///////////////////////////////////////////////////////////////
050////SoaplabChooseResultType
051/**
052 * The following actor is for choosing get_&lt;name&gt; operations for
053 * displaying web service execution results as desired. The input is a soaplab
054 * service client on which the call is made after the user selects the
055 * get_&lt;name&gt; operation.
056 * 
057 * @author Nandita Mangal
058 * @version $Id: SoaplabChooseResultType.java, v 1.0 2005/19/07
059 * @category.name web
060 * @category.name external execution
061 */
062
063public class SoaplabChooseResultType extends TypedAtomicActor {
064
065        /**
066         * Construct a SoaplabChooseResultType actor with given container and name.
067         * 
068         * @param container
069         *            The container.
070         * @param name
071         *            The name of this actor.
072         * @exception NameDuplicationException
073         *                If the container already has an actor with this name
074         */
075
076        public SoaplabChooseResultType(CompositeEntity container, String name)
077                        throws NameDuplicationException, IllegalActionException {
078
079                super(container, name);
080
081                wsdlUrl = new StringParameter(this, "wsdlUrl");
082                selectOutputGetMethod = new StringParameter(this, "outputGetMethods");
083
084                input = new TypedIOPort(this, "input", true, false);
085                input.setTypeEquals(BaseType.OBJECT);
086
087                output = new TypedIOPort(this, "output", false, true);
088                output.setTypeEquals(BaseType.STRING);
089
090                _attachText("_iconDescription", "<svg>\n" + "<rect x=\"0\" y=\"0\" "
091                                + "width=\"60\" height=\"30\" " + "style=\"fill:white\"/>\n"
092                                + "</svg>\n");
093        }
094
095        // ///////////////////////////////////////////////////////////////////
096        // // Ports and Parameters ////
097        /**
098         * The web service URL which is registered at EBI
099         */
100        public StringParameter wsdlUrl = null;
101
102        /**
103         * The "get_&lt;name&gt;" method for the soaplab webservice
104         */
105        public StringParameter selectOutputGetMethod = null;
106
107        /**
108         * output string value from the get_&lt;name&gt; operation
109         */
110        public TypedIOPort output;
111
112        /**
113         * The input is a soaplab service client on which the call is made after the
114         * user selects the get_&lt;name&gt; operation.
115         */
116        public TypedIOPort input;
117
118        // //////////////////////////////////////////////////////////////////////
119        // // public methods ////
120
121        /**
122         * Callback for changes in attribute values Get the WSDL from the given URL.
123         * 
124         * @param at
125         *            The attribute that changed.
126         * @exception IllegalActionException
127         */
128
129        public void attributeChanged(Attribute at) throws IllegalActionException {
130                if (at == wsdlUrl) {
131
132                        if (wsdlUrl.getExpression().equals("")) {
133
134                                selectOutputGetMethod.removeAllChoices();
135
136                        } else {
137
138                                selectOutputGetMethod.removeAllChoices();
139                                try {
140
141                                        // NOTE:- The following client is just created for geting
142                                        // the WSDL output methods
143                                        // beforehand i.e before actual running of the workflow.
144                                        client = new SoaplabServiceClient(wsdlUrl.getExpression());
145                                        client.setJobId();
146                                        client.generateOutputMethods();
147
148                                        Vector outputMethods = client.getOutputMethods();
149
150                                        for (int i = 0; i < outputMethods.size(); i++) {
151                                                selectOutputGetMethod.addChoice("get_"
152                                                                + (String) (outputMethods.elementAt(i)));
153                                        }
154
155                                } catch (Exception ex) {
156                                        _debug("<EXCEPTION> There was an error while parsing the WSDL. "
157                                                        + ex + ". </EXCEPTION>");
158                                        // GraphicalMessageHandler.message(
159                                        _confErrorStr += "\n"
160                                                        + ex.getMessage()
161                                                        + "There was an error while parsing the WSDL in the actor: "
162                                                        + this.getName();// );
163                                }
164
165                                if (!(_confErrorStr.equals(""))) {
166                                        GraphicalMessageHandler.message(_confErrorStr);
167
168                                }
169
170                        }
171
172                }
173
174        } // end of attributeChanged
175
176        /**
177         * Calls the appropriate get_&lt;name&gt; operation on the client and
178         * outputs the result obtained.
179         * 
180         * @exception IllegalActionException
181         *                If there is no director.
182         */
183        public void fire() throws IllegalActionException {
184
185                super.fire();
186
187                try {
188                        List inPortList = this.inputPortList();
189                        Iterator ports = inPortList.iterator();
190
191                        while (ports.hasNext()) {
192
193                                IOPort p = (IOPort) ports.next();
194                                List connections = p.connectedPortList();
195                                if (!(connections.isEmpty())) {
196                                        ObjectToken inputToken = (ObjectToken) (p.get(0));
197                                        client = (SoaplabServiceClient) (inputToken.getValue());
198                                        output.broadcast(new StringToken(selectOutputGetMethod
199                                                        .getExpression()
200                                                        + ":\n"
201                                                        + client.doCall(selectOutputGetMethod
202                                                                        .getExpression(), new Object[] { client
203                                                                        .getJobId() })));
204                                }
205
206                        }
207                } catch (Exception ex) {
208
209                        _debug("<EXCEPTION> There was an error while getting the web service results. "
210                                        + ex + ". </EXCEPTION>");
211                        // GraphicalMessageHandler.message(
212                        _confErrorStr += "\n"
213                                        + ex.getMessage()
214                                        + "\nThere was an error while getting the web service results in the actor: "
215                                        + this.getName();// );
216
217                }
218
219        }// end of fire
220
221        // //////////////////////////////////////////////////////////////////////////////
222        // // private variables ////
223
224        private SoaplabServiceClient client;
225        protected String _confErrorStr = "";
226}
