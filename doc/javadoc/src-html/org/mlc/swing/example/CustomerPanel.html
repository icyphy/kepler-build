099    /** Instantiate the CustomerPanel. */
100    public CustomerPanel() {
101        super();
102        java.io.InputStream is = this.getClass()
103                .getResourceAsStream("customerLayout.xml");
104        if (is == null) {
105            System.err.println("Could not find constraints customerLayout.xml");
106            return;
107        }
108        org.mlc.swing.layout.LayoutConstraintsManager layoutConstraintsManager = org.mlc.swing.layout.LayoutConstraintsManager
109                .getLayoutConstraintsManager(is);
110        setBorder(com.jgoodies.forms.factories.Borders.DIALOG_BORDER);
111        LayoutManager layout = layoutConstraintsManager.createLayout("panel",
112                this);
113        this.setLayout(layout);
114
115        this.add(nameSeparator, "nameSeparator");
116        this.add(surnameLabel, "surnameLabel");
117        this.add(surnameCombo, "surnameCombo");
118        this.add(firstNameLabel, "firstNameLabel");
119        this.add(firstNameText, "firstNameText");
120        this.add(lastNameLabel, "lastNameLabel");
121        this.add(lastNameText, "lastNameText");
122        this.add(suffixLabel, "suffixLabel");
123        this.add(suffixCombo, "suffixCombo");
124        this.add(contactSeparator, "contactSeparator");
125        this.add(workPhoneLabel, "workPhoneLabel");
126        this.add(workPhoneText, "workPhoneText");
127        this.add(homePhoneLabel, "homePhoneLabel");
128        this.add(homePhoneText, "homePhoneText");
129        this.add(faxLabel, "faxLabel");
130        this.add(faxText, "faxText");
131        this.add(emailLabel, "emailLabel");
132        this.add(emailText, "emailText");
133        this.add(ordersSeparator, "ordersSeparator");
134        this.add(orderTable, "orderTable");
135        this.add(buttonBar, "buttonBar");
136
137        LayoutFrame frame = new LayoutFrame(layoutConstraintsManager);
138        frame.setVisible(true);
139    }
140
141    /** Run the CustomerPanel example.
142     *  @param args Not used.
143     */
144    public static void main(String[] args) {
145        UIDefaults defaults = UIManager.getDefaults();
146        defaults.put("Label.font", new javax.swing.plaf.FontUIResource(
147                new java.awt.Font("Arial", java.awt.Font.PLAIN, 12)));
148        defaults.put("ComboBox.background",
149                new javax.swing.plaf.ColorUIResource(255, 255, 255));
150
151        CustomerPanel customerPanel = new CustomerPanel();
152
153        JFrame frame = new JFrame("Edit Customer");
154        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
155        frame.getContentPane().setLayout(new BorderLayout());
156        frame.getContentPane().add(customerPanel, BorderLayout.CENTER);
157        frame.setSize(600, 700);
158        frame.setVisible(true);
159    }
