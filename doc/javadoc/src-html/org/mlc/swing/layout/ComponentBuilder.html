030package org.mlc.swing.layout;
031
032import java.awt.Component;
033import java.util.List;
034import java.util.Map;
035
036/**
037 * All components that want to be included in the list of components that want
038 * to be part of the builder must implement this interface
039 *
040 * @author Michael Connor
041@version $Id$
042@since Ptolemy II 8.0
043 */
044public interface ComponentBuilder {
045    /**
046     * Returns a list of properties that are appropriate for construction
047     */
048    public List<BeanProperty> getProperties();
049
050    /**
051     * Returns the appropriate declaration suitable for this component
052     */
053    public String getDeclaration(String name, Map<String, Object> properties);
054
055    /**
056     * Returns an instance of this object given the specified properties
057     */
058    public Component getInstance(Map<String, Object> properties)
059            throws Exception;
060
061    public boolean isComponentALayoutContainer();
062
063    public ComponentDef getComponentDef(String name,
064            Map<String, Object> properties);
065}
