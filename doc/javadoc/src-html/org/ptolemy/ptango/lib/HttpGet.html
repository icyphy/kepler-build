028package org.ptolemy.ptango.lib;
029
030import java.io.FileReader;
031import java.io.IOException;
032import java.net.URL;
033
034import org.ptolemy.ptango.lib.HttpRequest.Method;
035
036import ptolemy.actor.TypedIOPort;
037import ptolemy.actor.lib.LimitedFiringSource;
038import ptolemy.actor.parameters.PortParameter;
039import ptolemy.data.BooleanToken;
040import ptolemy.data.IntToken;
041import ptolemy.data.StringToken;
042import ptolemy.data.expr.Parameter;
043import ptolemy.data.expr.SingletonParameter;
044import ptolemy.data.expr.StringParameter;
045import ptolemy.data.type.BaseType;
046import ptolemy.kernel.CompositeEntity;
047import ptolemy.kernel.util.IllegalActionException;
048import ptolemy.kernel.util.NameDuplicationException;
049import ptolemy.kernel.util.StringAttribute;
050
051/** Get the contents of the specified URL and output as a string.
052 *  This is similar to the {@link FileReader}, except that it has
053 *  a timeout parameter. If the timeout expires while
054 *  performing the get, and the timeoutResponse parameter has been
055 *  set, then send the specified response. If no timeoutResponse
056 *  has been set, then throw an exception on timeout.
057 *
058 *  @see HttpPost
059 *  @author  Edward A. Lee
060 *  @version $Id$
061 *  @since Ptolemy II 10.0
062 *  @Pt.ProposedRating Yellow (eal)
063 *  @Pt.AcceptedRating Red (eal)
064 */
065public class HttpGet extends LimitedFiringSource {
066    /** Construct an actor with the given container and name.
067     *  @param container The container.
068     *  @param name The name of this actor.
069     *  @exception IllegalActionException If the actor cannot be contained
070     *   by the proposed container.
071     *  @exception NameDuplicationException If the container already has an
072     *   actor with this name.
073     */
074    public HttpGet(CompositeEntity container, String name)
075            throws IllegalActionException, NameDuplicationException {
076        super(container, name);
077
078        url = new PortParameter(this, "url");
079        url.setStringMode(true);
080        url.setExpression("http://localhost");
081        new SingletonParameter(url.getPort(), "_showName")
082                .setToken(BooleanToken.TRUE);
083        StringAttribute cardinal = new StringAttribute(url.getPort(),
084                "_cardinal");
085        cardinal.setExpression("SOUTH");
086
087        timeout = new Parameter(this, "timeout");
088        timeout.setTypeEquals(BaseType.INT);
089        timeout.setExpression("30000");
090        timeout.addChoice("NONE");
091
092        timeoutResponse = new StringParameter(this, "timeoutResponse");
093
094        newline = new Parameter(this, "newline");
095        newline.setExpression("property(\"line.separator\")");
096
097        output.setTypeEquals(BaseType.STRING);
098        new SingletonParameter(output, "_showName").setToken(BooleanToken.TRUE);
099
100        status = new TypedIOPort(this, "status", false, true);
101        status.setTypeEquals(HttpResponse.getStatusType());
102        new SingletonParameter(status, "_showName").setToken(BooleanToken.TRUE);
103    }
104
105    ///////////////////////////////////////////////////////////////////
106    ////                     ports and parameters                  ////
107
108    /** The end of line character(s).  The default value is the value
109     *  of the line.separator property
110     */
111    public Parameter newline;
112
113    /** An output port for transmitting a token containing the status of the
114     * request.  This is a RecordToken comprised of the response code,
115     * response message, a boolean indicating if the request was successful,
116     * and a boolean indicating if further action is expected.
117     */
118    public TypedIOPort status;
119
120    /** The timeout in milliseconds for establishing a connection or reading a value.
121     *  Set to NONE to specify no timeout.
122     *  This is an integer that defaults to 30000, giving a timeout of 30 seconds.
123     */
124    public Parameter timeout;
125
126    /** The response to send upon timeout.
127     *  If this is empty, then this actor will throw an exception rather than
128     *  send a response.  This is a string that defaults to empty.
129     */
130    public StringParameter timeoutResponse;
131
132    /** The URL to send the request to.
133     *  This is a string that defaults to "http://localhost", which
134     *  refers to a web server on the local host.
135     */
136    public PortParameter url;
137
138    ///////////////////////////////////////////////////////////////////
139    ////                         public methods                    ////
140
141    /** If there is an input, then post to the specified URL the
142     *  data on the input record, wait for a response, and output
143     *  the response on the output port.
144     *  @exception IllegalActionException If an IO error occurs.
145     */
146    @Override
147    public void fire() throws IllegalActionException {
148        super.fire();
149        url.update();
150
151        if (_request == null) {
152            _request = new HttpRequest();
153        }
154
155        String urlValue = ((StringToken) url.getToken()).stringValue();
156        if (urlValue == null || urlValue.isEmpty()) {
157            throw new IllegalActionException("No URL provided.");
158        }
159
160        try {
161            _request.setUrl(new URL(urlValue));
162
163            _request.setMethod(Method.GET);
164
165            // If a timeout has been specified, set it.
166            int timeoutValue = ((IntToken) timeout.getToken()).intValue();
167            if (timeoutValue >= 0) {
168                _request.setTimeout(timeoutValue);
169            }
170
171            HttpResponse response = _request.execute();
172
173            // If a timeout occurs, check if an exception should be thrown
174            if (response.timedOut()) {
175                if (_debugging) {
176                    _debug("*** Timeout occurred.");
177                }
178                String timeout = timeoutResponse.stringValue();
179                if (timeout.trim().equals("")) {
180                    throw new IllegalActionException(this,
181                            "HTTP " + _request.getMethod() + " "
182                                    + response.getResponseMessage());
183                }
184            }
185
186            // FIXME: default response upon failure or empty string?
187            output.send(0, new StringToken(response.getBody()));
188            status.send(0, response.getStatus());
189        } catch (IOException e) {
190            throw new IllegalActionException(this, e, "HTTP request failed");
191        }
192    }
193
194    ///////////////////////////////////////////////////////////////////
195    ////                         private variables                 ////
196
197    /** The Http request **/
198    HttpRequest _request;
199}
