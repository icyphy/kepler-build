077/** An actor that handles an HTTP request by producing output
078 *  and waiting for an input that provides a response.
079 *  This actor requires that the model that contains it includes an
080 *  instance of {@link WebServer}, which discovers this actor and
081 *  delegates HTTP requests to a servlet that this actor creates.
082 *  It also requires that the model containing this actor provide
083 *  exactly one input to either the {@link #response} or {@link #setCookies}
084 *  input port (or both) in response to every output request,
085 *  and that such responses are delivered in the same order as
086 *  the requests they are responding to.
087 *  <p>
088 *  The <i>path</i> parameter specifies which HTTP requests will
089 *  be delegated to this actor. If the base URL for the web server
090 *  is "http://localhost:8080", say, then request of the form
091 *  "http://localhost:8080/<i>path</i>" will be delegated to this
092 *  actor.
093 *  <p>
094 *  When this actor receives an HTTP request from the WebServer, it
095 *  issues a request for the director to fire it at the greater of
096 *  the current time of the director or the time elapsed since
097 *  the last invocation of initialize() (in seconds).
098 *  When the actor fires, it produces on its output ports the details
099 *  of the request, time stamped by the elapsed time since the model
100 *  started executing. It expects to be in a model that will send
101 *  it data to either the {@link #response} or {@link #setCookies}
102 *  input port (or both) with the response to HTTP request.
103 *  If that response does not arrive within <i>timeout</i>
104 *  (default 30000) milliseconds, then this actor will a issue
105 *  timeout response and will discard the response when it eventually
106 *  arrives.
107 *  <p>
108 *  This actor is designed to be used in a DE model (with a DEDirector).
109 *  Since a model using this actor must deliver a response for
110 *  every request, the model must put this actor in a feedback
111 *  loop. The requests appear on the outputs of this actor,
112 *  and the responses appear on the inputs. Each such feedback
113 *  loop is required to include an instance of {@link MicrostepDelay}
114 *  (or {@link TimeDelay}, if you wish to model response times) in order
115 *  to break the causality loop that the feedback loop would
116 *  otherwise incur.
117 *  The downstream model should be used to construct a response.
118 *  For example, to simply serve a web page, put a
119 *  {@link FileReader} actor to read the response from a local
120 *  file and a {@link MicrostepDelay} downstream in a feedback
121 *  loop connected back to the response input.
122 *
123 *  @author Elizabeth Latronico and Edward A. Lee
124 *  @version $Id$
125 *  @since Ptolemy II 10.0
126 *  @Pt.ProposedRating Red (ltrnc)
127 *  @Pt.AcceptedRating Red (ltrnc)
128 *  @see org.ptolemy.ptango.lib.WebServer
129 */
130public class HttpActor extends TypedAtomicActor
131        implements HttpService, ExceptionSubscriber {

354    /** React to a change in an attribute.  In this case, check the
355     *  value of the <i>path</i> attribute to make sure it is a valid URI.
356     *  @param attribute The attribute that changed.
357     *  @exception IllegalActionException If the change is not acceptable
358     *   to this container (not thrown in this base class).
359     */
360    @Override
361    public void attributeChanged(Attribute attribute)
362            throws IllegalActionException {
363        if (attribute == path) {
364            String pathValue = ((StringToken) path.getToken()).stringValue();
365            try {
366                // Path should start with a "/", or be "*"
367                if (!pathValue.trim().equals("")) {
368                    if (!pathValue.trim().startsWith("/")) {
369                        _URIpath = URI.create("/" + pathValue);
370                    } else {
371                        _URIpath = URI.create(pathValue);
372                    }
373                } else {
374                    _URIpath = URI.create("/*");
375                }
376            } catch (IllegalArgumentException e2) {
377                throw new IllegalActionException(this,
378                        "Path is not a valid URI: " + pathValue);
379            }
380        } else {
381            super.attributeChanged(attribute);
382        }
383    }

414    /** Generate an HTTP response for the client if an exception occurs.  No
415     *  response will be received on the input port in the event of an
416     *  exception.
417     *
418     *  @param policy The exception handling policy of the exception handler;
419     *   see {@link CatchExceptionAttribute}
420     *  @param exception  The exception that occurred
421     *  @return True since a response is always sent to the client, and there
422     *    there are no operations that throw exceptions
423     */
424
425    @Override
426    public synchronized boolean exceptionOccurred(String policy,
427            Throwable exception) {
428
429        // If there is a pending request,
430        // For "restart" policy, generate an error page with retry
431        // For other policies, generate a Server Error error page
432        // These method calls notifyAll() so that the response will be sent
433        // by the servlet thread
434        if (_request != null) {
435            if (policy.equals(CatchExceptionAttribute.RESTART)) {
436                _respondWithRetryMessage(10);
437            } else {
438                _respondWithServerErrorMessage();
439            }
440        }
441
442        return true;
443    }

472    /** If there are input tokens on the {@link #response} or
473     *  {@link #setCookies} ports, then queue a response to be
474     *  sent by the servelet for a corresponding request.
475     *  If the servlet has received
476     *  an HTTP request, then also produce on the output ports
477     *  the details of the request.
478     *  If there is an HTTP request, but the current superdense time
479     *  matches the time of the previously produced output,
480     *  then request a refiring at the current time (next microstep)
481     *  so that the outputs are produced at the next microstep.
482     *  This actor is designed to normally be fired twice at each superdense
483     *  time, first to produce outputs corresponding to an HTTP
484     *  request, and next to send back the response to that request.
485     *  This strategy avoids creating signals with multiple values
486     *  at the same superdense time instant.
487     *  @exception IllegalActionException If sending the
488     *   outputs fails.
489     */
490    @Override
491    public synchronized void fire() throws IllegalActionException {
492        // The methods of the servlet are invoked in another
493        // thread, so we synchronize on this actor for mutual exclusion.
494        super.fire();
495
496        boolean responseFound = false;
497        HttpResponseItems responseData = new HttpResponseItems();
498        // Check for new cookies on the setCookies port.
499        if (setCookies.getWidth() > 0 && setCookies.hasToken(0)) {
500            RecordToken cookieToken = (RecordToken) setCookies.get(0);
501            // Note that we issue a response if we get cookies only.
502            // If there is no response input, then the response will be an empty string.
503            responseFound = true;
504            responseData.hasNewCookies = true;
505            responseData.cookies = cookieToken;
506            // Default response has empty text.
507            responseData.response = "";
508            if (_debugging) {
509                _debug("Received cookies on the setCookies input port: "
510                        + cookieToken);
511            }
512        }
513
514        for (int i = 0; i < response.getWidth(); i++) {
515            if (response.hasToken(i)) {
516                responseData.response = ((StringToken) response.get(i))
517                        .stringValue();
518                responseFound = true;
519                if (_debugging) {
520                    _debug("Received response on the response input port: "
521                            + responseData.response);
522                }
523            }
524        }
525        if (responseFound) {
526            if (_request == null) {
527                // There is no pending request, so ignore the response.
528                if (_debugging) {
529                    _debug("Discarding the response because there is no pending request.");
530                }
531            } else {
532                _response = responseData;
533                // Indicate that the request has been handled.
534                _request = null;
535                // Notify the servlet thread.
536                notifyAll();
537            }
538        }
539
540        // If there is a pending request, produce outputs for that request,
541        // including any cookies from that request.
542        if (_request != null) {
543
544            // To avoid the risk of producing two outputs at the same superdense time,
545            // check the time of the last output.
546            Director director = getDirector();
547            // Do not do this if the director does not support superdense time.
548            // In that case, there is no issue with multiple outputs.
549            if (director instanceof SuperdenseTimeDirector) {
550                Time currentTime = director.getModelTime();
551                int microstep = ((SuperdenseTimeDirector) director).getIndex();
552                if (_lastOutputTime != null
553                        && _lastOutputTime.equals(currentTime)
554                        && microstep == _lastMicrostep) {
555                    director.fireAtCurrentTime(this);
556                    // Leave the request queue alone and don't produce outputs now.
557                    return;
558                }
559                _lastMicrostep = microstep;
560                _lastOutputTime = currentTime;
561            }
562
563            // Remove the request from the head of the queue so that
564            // each request is handled no more than once.
565            if (_request.requestType == 0) {
566                if (_debugging) {
567                    _debug("Sending get request URI: " + _request.requestURI);
568                    _debug("Sending get request parameters: "
569                            + _request.parameters);
570                }
571                getRequestURI.send(0, new StringToken(_request.requestURI));
572                // Send parameters, but handle type errors locally.
573                try {
574                    getParameters.send(0, _request.parameters);
575                } catch (TypedIOPort.RunTimeTypeCheckException ex) {
576                    // Parameters provided do not match the required type.
577                    // Construct an appropriate response.
578                    _respondWithBadRequestMessage(_request.parameters,
579                            getParameters.getType(), "parameters");
580                }
581                if (_request.cookies != null && _request.cookies.length() > 0) {
582                    if (_debugging) {
583                        _debug("Sending cookies to getCookies port: "
584                                + _request.cookies);
585                    }
586                    try {
587                        getCookies.send(0, _request.cookies);
588                    } catch (TypedIOPort.RunTimeTypeCheckException ex) {
589                        // Parameters provided do not match the required type.
590                        // Construct an appropriate response.
591                        _respondWithBadRequestMessage(_request.parameters,
592                                getParameters.getType(), "cookies");
593                    }
594                }
595            } else if (_request.requestType == 1) {
596                if (_debugging) {
597                    _debug("Sending post request URI: " + _request.requestURI);
598                    _debug("Sending post request parameters: "
599                            + _request.parameters);
600                }
601                postRequestURI.send(0, new StringToken(_request.requestURI));
602                try {
603                    postParameters.send(0, _request.parameters);
604                } catch (TypedIOPort.RunTimeTypeCheckException ex) {
605                    // Parameters provided do not match the required type.
606                    // Construct an appropriate response.
607                    _respondWithBadRequestMessage(_request.parameters,
608                            getParameters.getType(), "parameters");
609                }
610                if (_request.cookies != null && _request.cookies.length() > 0) {
611                    try {
612                        postCookies.send(0, _request.cookies);
613                    } catch (TypedIOPort.RunTimeTypeCheckException ex) {
614                        // Parameters provided do not match the required type.
615                        // Construct an appropriate response.
616                        _respondWithBadRequestMessage(_request.parameters,
617                                getParameters.getType(), "cookies");
618                    }
619                    if (_debugging) {
620                        _debug("Sending cookies to postCookies port: "
621                                + _request.cookies);
622                    }
623                }
624                if (_request.body != null) {
625                    postBody.send(0, _request.body);
626                }
627            } else {
628                if (_debugging) {
629                    _debug("Sending put request URI: " + _request.requestURI);
630                    _debug("Sending put request parameters: "
631                            + _request.parameters);
632                }
633                putRequestURI.send(0, new StringToken(_request.requestURI));
634                try {
635                    putParameters.send(0, _request.parameters);
636                } catch (TypedIOPort.RunTimeTypeCheckException ex) {
637                    // Parameters provided do not match the required type.
638                    // Construct an appropriate response.
639                    _respondWithBadRequestMessage(_request.parameters,
640                            getParameters.getType(), "parameters");
641                }
642                if (_request.cookies != null && _request.cookies.length() > 0) {
643                    try {
644                        putCookies.send(0, _request.cookies);
645                    } catch (TypedIOPort.RunTimeTypeCheckException ex) {
646                        // Parameters provided do not match the required type.
647                        // Construct an appropriate response.
648                        _respondWithBadRequestMessage(_request.parameters,
649                                getParameters.getType(), "cookies");
650                    }
651                    if (_debugging) {
652                        _debug("Sending cookies to putCookies port: "
653                                + _request.cookies);
654                    }
655                }
656                if (_request.body != null) {
657                    putBody.send(0, _request.body);
658                }
659            }
660        }
661        if (_debugging) {
662            _debug("Ending fire.");
663        }
664    }

903    /** A servlet providing implementations of get and post.
904     *  The way this servlet works is that when a get or post
905     *  HTTP request is received, it records the properties of
906     *  the request (the URI and parameters) and requests a
907     *  firing of the enclosing actor at model time equal to
908     *  the time elapsed since the start of execution of the model.
909     *  The thread making the get or post request then suspends,
910     *  giving the model a chance to execute. When the model has
911     *  determined what the response to the request should be,
912     *  it notifies this servlet thread, which then completes
913     *  the handling of the request, sending back the response
914     *  that has been provided by the enclosing actor.
915     *
916     *  See <a href="http://wiki.eclipse.org/Jetty/Tutorial/Embedding_Jetty">http://wiki.eclipse.org/Jetty/Tutorial/Embedding_Jetty</a>
917     */
918    @SuppressWarnings("serial")

982        /** Handle an HTTP get or post request by creating a web page as the HTTP
983         *  response.
984         *  See <a href="http://docs.oracle.com/javaee/6/api/javax/servlet/http/HttpServletRequest.html">http://docs.oracle.com/javaee/6/api/javax/servlet/http/HttpServletRequest.html</a>.
985         *  @param request The HTTP request.
986         *  @param response The HTTP response to write to.
987         *  @param type The type of request. 0 for get, 1 for post.
988         *  @exception ServletException  If there is a problem reading from the
989         *  servlet request or other servlet problem
990         *  @exception IOException  If there is a problem writing to the servlet
991         *  response
992         */
993        private void _handleRequest(HttpServletRequest request,
994                HttpServletResponse response, int type)
995                throws ServletException, IOException {
996            // The following codeblock is synchronized on the enclosing
997            // actor. This lock _is_ released while waiting for the response,
998            // allowing the fire method to execute its own synchronized blocks.
999            synchronized (HttpActor.this) {
1000                _request = new HttpRequestItems();
1001
1002                _request.requestURI = request.getRequestURI();
1003                _request.requestType = type;
1004
1005                response.setContentType("text/html");
1006
1007                // Set up a buffer for the output so we can set the length of
1008                // the response, thereby enabling persistent connections
1009                // http://docstore.mik.ua/orelly/java-ent/servlet/ch05_03.htm
1010                ByteArrayOutputStream bytes = new ByteArrayOutputStream();
1011                PrintWriter writer = new PrintWriter(bytes, true);// true forces
1012                // flushing
1013
1014                if (_debugging) {
1015                    _debug("**** Handling a "
1016                            + ((type == 0) ? "get"
1017                                    : (type == 1) ? "post" : "put")
1018                            + " request to URI " + _request.requestURI);
1019                }
1020
1021                try {
1022                    // Read cookies from the request and store.
1023                    _request.cookies = _readCookies(request);
1024
1025                    // Get the parameters that have been either posted or included
1026                    // as assignments in a get using the URL syntax ...?name=value.
1027                    // Note that each parameter name may have more than one value,
1028                    // hence the array of strings.
1029                    RecordToken content = _readContent(request);
1030                    _request.parameters = (RecordToken) content
1031                            .get("parameters");
1032                    _request.body = content.get("body");
1033
1034                    // Figure out what time to request a firing for.
1035                    long elapsedRealTime = System.currentTimeMillis()
1036                            - _initializeRealTime;
1037
1038                    // Assume model time is in seconds, not milliseconds.
1039                    Time timeOfRequest = _initializeModelTime
1040                            .add(elapsedRealTime / 1000.0);
1041
1042                    if (_debugging) {
1043                        _debug("**** Request firing at time " + timeOfRequest);
1044                    }
1045
1046                    // Note that fireAt() will modify the requested firing time
1047                    // if it is in the past.
1048                    // Note that past firing times might not be modified
1049                    // if ThreadedComposite actors are used (since the request
1050                    // might be at a present time inside the ThreadedComposite,
1051                    // but a past time for the top-level model).
1052                    getDirector().fireAt(HttpActor.this, timeOfRequest);
1053                } catch (IllegalActionException e) {
1054                    _request = null;
1055                    _writeError(response, HttpServletResponse.SC_BAD_REQUEST,
1056                            e.getMessage(), bytes, writer);
1057                    return;
1058                }
1059                //////////////////////////////////////////////////////
1060                // Wait for a response.
1061                // We are assuming every request gets exactly one response in FIFO order.
1062                while (_response == null) {
1063                    if (_debugging) {
1064                        _debug("**** Waiting for a response");
1065                    }
1066                    try {
1067                        // Timeout after time given by timeout parameter.
1068                        // Unfortunately, we can't tell whether the timeout
1069                        // occurred unless we record the current time.
1070                        long startTime = System.currentTimeMillis();
1071                        long timeoutValue = 30000L;
1072                        try {
1073                            timeoutValue = ((LongToken) timeout.getToken())
1074                                    .longValue();
1075                        } catch (IllegalActionException e) {
1076                            // Ignore and use default of 30 seconds.
1077                        }
1078                        HttpActor.this.wait(timeoutValue);
1079                        if (System.currentTimeMillis()
1080                                - startTime >= timeoutValue) {
1081                            if (_debugging) {
1082                                _debug("**** Request timed out.");
1083                            }
1084
1085                            // Set the content length (enables persistent
1086                            // connections) and send the buffer
1087                            writer.println("Request timed out");
1088                            response.setContentLength(bytes.size());
1089                            bytes.writeTo(response.getOutputStream());
1090
1091                            // Indicate that there is no longer a pending request or response.
1092                            _request = null;
1093                            _response = null;
1094
1095                            // Close the PrintWriter
1096                            // Close the ByteArrayOutputStream to avoid a
1097                            // warning, though this is not necessary since
1098                            // ByteArrayOutputStream's close() method has no
1099                            // effect
1100                            // http://docs.oracle.com/javase/6/docs/api/java/io/ByteArrayOutputStream.html#close%28%29
1101                            writer.close();
1102                            bytes.close();
1103                            response.setStatus(
1104                                    HttpServletResponse.SC_REQUEST_TIMEOUT);
1105
1106                            return;
1107                        }
1108                    } catch (InterruptedException e) {
1109                        if (_debugging) {
1110                            _debug("*** Request thread interrupted.");
1111                        }
1112
1113                        // Set the content length (enables persistent
1114                        // connections) and send the buffer
1115                        writer.println("Get request thread interrupted");
1116                        response.setContentLength(bytes.size());
1117                        bytes.writeTo(response.getOutputStream());
1118
1119                        // Indicate that there is no longer a pending request or response.
1120                        _request = null;
1121                        _response = null;
1122
1123                        // Close the PrintWriter
1124                        // Close the ByteArrayOutputStream to avoid a
1125                        // warning, though this is not necessary since
1126                        // ByteArrayOutputStream's close() method has no
1127                        // effect
1128                        // http://docs.oracle.com/javase/6/docs/api/java/io/ByteArrayOutputStream.html#close%28%29
1129                        writer.close();
1130                        bytes.close();
1131                        response.setStatus(
1132                                HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
1133                        return;
1134                    }
1135                }
1136
1137                response.setStatus(_response.statusCode);
1138
1139                // Write all cookies to the response, if there are some new
1140                // cookies to write
1141                if (_response.hasNewCookies) {
1142                    _writeCookies(_response.cookies, response);
1143                }
1144                if (_debugging) {
1145                    _debug("**** Servlet received response: "
1146                            + _response.response);
1147                }
1148
1149                // Set the content length (enables persistent
1150                // connections) and send the buffer
1151                // Use print rather than println to prevent an extra \n on the response.
1152                writer.print(_response.response);
1153                // Printwriter only flushes println() automatically. So force a flush.
1154                writer.flush();
1155                response.setContentLength(bytes.size());
1156                bytes.writeTo(response.getOutputStream());
1157
1158                // Close the PrintWriter
1159                // Close the ByteArrayOutputStream to avoid a
1160                // warning, though this is not necessary since
1161                // ByteArrayOutputStream's close() method has no
1162                // effect
1163                // http://docs.oracle.com/javase/6/docs/api/java/io/ByteArrayOutputStream.html#close%28%29
1164                writer.close();
1165                bytes.close();
1166
1167                // Indicate response has been handled.
1168                _response = null;
1169            }
1170        }
1171
1172        /** Read the Cookies from the HttpServletRequest, construct
1173         *  a record token with one field for each name in the requestedCookies
1174         *  parameter and the value given by the request, or if there is no
1175         *  value, with an empty string as a value.
1176         *  @param request  The HttpServletRequest to read Cookies from.  The
1177         *   HttpServletRequest can be of any type - i.e. both GET and POST
1178         *   requests are allowed to have Cookies.
1179         *  @return A record of cookies.
1180         *  @exception IllegalActionException If construction of the record token fails.
1181         */
1182        private RecordToken _readCookies(HttpServletRequest request)
1183                throws IllegalActionException {
1184            ArrayToken labels = (ArrayToken) requestedCookies.getToken();
1185            if (labels.length() == 0) {
1186                // No cookies requested.
1187                // Return an empty record.
1188                return RecordToken.EMPTY_RECORD;
1189            }
1190            // First, provide default empty string values for requested cookies.
1191            LinkedHashMap<String, Token> map = new LinkedHashMap<String, Token>();
1192            StringToken emptyString = new StringToken("");
1193            for (int i = 0; i < labels.length(); i++) {
1194                String label = ((StringToken) labels.getElement(i))
1195                        .stringValue();
1196                map.put(label, emptyString);
1197            }
1198            // Next, override these default values with any cookies provided in the request.
1199            for (Cookie cookie : request.getCookies()) {
1200                // Cookie must have a name
1201                if (cookie.getName() != null && !cookie.getName().isEmpty()) {
1202                    String label = cookie.getName();
1203                    if (map.containsKey(label)) {
1204                        // Override the default value.
1205                        map.put(label, new StringToken(cookie.getValue()));
1206                    }
1207                }
1208            }
1209            return new RecordToken(map);
1210        }
1211
1212        /** Read the parameters from the HttpServletRequest, construct
1213         *  a record token containing the parameters, and return that record.
1214         *  @param request  The HttpServletRequest to read paramters from.  The
1215         *   HttpServletRequest can be of any type - i.e. both GET and POST
1216         *   requests are allowed to have parameters.
1217         *  @return A record of parameters.
1218         *  @exception IllegalActionException If construction of the record token fails.
1219         */
1220        private RecordToken _readContent(HttpServletRequest request)
1221                throws IllegalActionException {
1222
1223            RecordToken parameters;
1224            Token body;
1225
1226            // Handle the request according to the Content-Type
1227            // Note that request.getParameterMap() and request.
1228            // BOTH read and consume the request body.  getParameterMap()
1229            // additionally reads the URL.  Note that getParameterMap() does
1230            // NOT differentiate between parameters from the URL and those from
1231            // the body
1232
1233            // Use contains instead of equals since the type may include other
1234            // things, e.g. charset=UTF-8
1235            if (request.getContentType() != null && request.getContentType()
1236                    .contains("application/x-www-form-urlencoded")) {
1237
1238                // Form data:  application/x-www-form-urlencoded
1239                // Use request.getParameterMap() which reads parameters from both
1240                // the URL and from the body
1241                // These will be emitted on a "parameters" output port, regardless
1242                // of whether the content was from the URL or the body
1243
1244                parameters = _readParameterMap(request.getParameterMap());
1245                body = new Token();
1246
1247            } else if (request.getContentType() != null
1248                    && request.getContentType().startsWith("image")) {
1249
1250                // Images.  Supported: bmp, gif, jpg, png, and wbmp
1251                // See:  https://docs.oracle.com/javase/tutorial/2d/images/loadimage.html
1252
1253                try {
1254
1255                    BufferedImage image = ImageIO
1256                            .read(request.getInputStream());
1257
1258                    if (image != null) {
1259                        body = new AWTImageToken(image);
1260                    } else {
1261                        body = Token.NIL;
1262                    }
1263
1264                    // Read any remaining URL parameters
1265                    parameters = _readParameterMap(request.getParameterMap());
1266
1267                } catch (IOException e) {
1268                    throw new IllegalActionException(
1269                            "Can't read body of " + " HTTP request");
1270                }
1271            } else {
1272
1273                // Anything else - Read as string
1274                // Includes (among others) application/json, text/*
1275
1276                // TODO:  Parse JSON object from JSON?
1277
1278                StringBuffer content = new StringBuffer();
1279
1280                // Text or JSON - return StringToken
1281                try {
1282                    BufferedReader reader = new BufferedReader(
1283                            new InputStreamReader(request.getInputStream()));
1284
1285                    String line;
1286                    while ((line = reader.readLine()) != null) {
1287                        content.append(line);
1288                    }
1289                } catch (IOException e) {
1290                    throw new IllegalActionException(
1291                            "Can't read body of " + " HTTP request");
1292                }
1293
1294                body = new StringToken(content.toString());
1295
1296                // Read any remaining URL parameters
1297                parameters = _readParameterMap(request.getParameterMap());
1298
1299            }
1300
1301            // TODO:  Audio and video
1302            // Audio capture outputs an array of doubles - use that?
1303            // for images and audio?
1304            // Content-types audio/* or video/*
1305
1306            // TODO:  Handle binary data - what is corresponding Ptolemy type?
1307            // Array of integers?
1308
1309            // TODO:  Handle multi-part requests
1310
1311            String[] labels = new String[2];
1312            labels[0] = "parameters";
1313            labels[1] = "body";
1314
1315            Token[] fields = new Token[2];
1316            fields[0] = parameters;
1317            fields[1] = body;
1318
1319            RecordToken content = new RecordToken(labels, fields);
1320            return content;
1321        }

1323        /** Read the parameters from the parameter map, construct a RecordToken
1324         *  containing the parameters, and return that record token.
1325         *  @param parameterMap  The HttpServletRequest's parameter map.
1326         *   Contains parameters passed as part of the URL, and may contain
1327         *   parameters from the body if the body was not previously processed.
1328         *   HttpServletRequest can be of any type - i.e. both GET and POST
1329         *   requests are allowed to have parameters.
1330         *  @return A RecordToken of parameters.
1331         *  @exception IllegalActionException If construction of the record token fails.
1332         */
1333
1334        private RecordToken _readParameterMap(
1335                Map<String, String[]> parameterMap)
1336                throws IllegalActionException {
1337            // FIXME:  This currently treats all values as StringTokens,
1338            // which they may not be.  How to either 1) correctly determine
1339            // the type of the token, or 2) create a token of unspecified
1340            // type but still be able to set its expression?  Class
1341            // Token does not have the setExpression() method
1342
1343            // Convert these parameters to a record token to send to
1344            // the appropriate output.
1345            Set<String> names = parameterMap.keySet();
1346            String[] fieldNames = new String[names.size()];
1347            Token[] fieldValues = new Token[names.size()];
1348            int i = 0;
1349            for (String name : names) {
1350                fieldNames[i] = name;
1351                String[] values = parameterMap.get(name);
1352                if (values == null || values.length == 0) {
1353                    fieldValues[i] = StringToken.NIL;
1354                } else if (values.length == 1) {
1355                    fieldValues[i] = new StringToken(values[0]);
1356                } else {
1357                    // More than one value. Use an array.
1358                    StringToken[] arrayEntries = new StringToken[values.length];
1359                    for (int j = 0; j < values.length; j++) {
1360                        arrayEntries[j] = new StringToken(values[j]);
1361                    }
1362                    ArrayToken array = new ArrayToken(arrayEntries);
1363                    fieldValues[i] = array;
1364                }
1365                i++;
1366            }
1367            return new RecordToken(fieldNames, fieldValues);
1368        }

1370        /** Write the cookies in the specified RecordToken to the HttpResponse.
1371         *  If the value of a cookie is an empty string, then clear the cookie instead.
1372         *  @param cookies The cookies.
1373         *  @param response The HttpServletResponse to write the cookies to.
1374         */
1375        private void _writeCookies(RecordToken cookies,
1376                HttpServletResponse response) {
1377            // TODO:  Allow permanent cookies.  Current implementation produces
1378            // session cookies.  Session cookies are stored in
1379            // the browser's memory, and are erased when the browser is closed
1380            // http://www.javascriptkit.com/javatutors/cookie.shtml
1381            for (String label : cookies.labelSet()) {
1382                String value;
1383                Token recordValue = cookies.get(label);
1384                if (recordValue instanceof StringToken) {
1385                    value = ((StringToken) recordValue).stringValue();
1386                } else {
1387                    value = recordValue.toString();
1388                }
1389                // Clear the cookie if the value is empty
1390                // TODO:  Should there be an explicit clear cookie port?
1391                // Is there a scenario where we want the cookie to exist,
1392                // but to have an empty string for the value?
1393                if (value.isEmpty()) {
1394                    Cookie cookie = new Cookie(label, "");
1395                    cookie.setMaxAge(0);
1396                    cookie.setPath("/"); // Cookie visible to all web pages
1397                    response.addCookie(cookie);
1398                } else {
1399                    // FIXME:  Special character handling?  Copied from previous
1400                    // code. Should cookies be encoded using URLEncoder.encode()?
1401                    if (value.indexOf("\"") != -1) {
1402                        value = value.replace("\"", "");
1403                    }
1404                    Cookie cookie = new Cookie(label, value);
1405                    cookie.setPath("/");
1406                    response.addCookie(cookie);
1407                }
1408            }
1409        }

1411        /** Write an error message to the given HttpServletResponse.
1412         *  @param response The HttpServletResponse to write the message to.
1413         *  @param responseCode The HTTP response code for the message.  Should be
1414         *   one of HttpServletResponse.X
1415         *  @param message The error message to write.
1416         *  @exception IOException If the write fails.
1417         */
1418        private void _writeError(HttpServletResponse response, int responseCode,
1419                String message, ByteArrayOutputStream bytes, PrintWriter writer)
1420                throws IOException {
1421
1422            response.setContentType("text/html");
1423            response.setStatus(responseCode);
1424
1425            writer.println("<!DOCTYPE html>");
1426            writer.println("<html>");
1427            writer.println("<body>");
1428            writer.println("<h1> Error </h1>");
1429            writer.println(message);
1430            writer.println("</body>");
1431            writer.println("</html>");
1432
1433            response.setContentLength(bytes.size());
1434            bytes.writeTo(response.getOutputStream());
1435
1436            // Close the PrintWriter
1437            // Close the ByteArrayOutputStream to avoid a
1438            // warning, though this is not necessary since
1439            // ByteArrayOutputStream's close() method has no
1440            // effect
1441            // http://docs.oracle.com/javase/6/docs/api/java/io/ByteArrayOutputStream.html#close%28%29
1442            writer.close();
1443            bytes.close();
1444        }
