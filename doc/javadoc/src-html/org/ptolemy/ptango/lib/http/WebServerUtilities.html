172     * @return true if the web server is hosting the given model; false

175    public boolean isHostingModel(String fullModelName) {
176        for (WebApplicationInfo app : _applications) {
177            if (fullModelName != null
178                    && app.getModelName().equalsIgnoreCase(fullModelName)) {
179                return true;
180            }
181        }
182        return false;

196    /** Register the following application on this server.  Start the server
197     * if it has not already been started.  Throw an exception if the new
198     * application conflicts with existing applications, for example, by
199     * requesting the same URL mapping.
200     *
201     * @param appInfo  The application to add to this server
202     * @exception Exception   If the new application conflicts with existing
203     * applications, for example, by requesting the same URL mapping.
204     */
205    public void register(WebApplicationInfo appInfo) throws Exception {
206
207        // FIXME:  Require the application paths to be distinct?  Distinct
208        // application paths are not strictly necessary, but matching
209        // application paths could be confusing since it's unclear which
210        // Ptolemy model is serving the response
211        // If applicationPath is just "/", don't add this as a prefix to
212        // the servlet path, since that would create a path like "//myApp"
213        // instead of "/myApp"
214        String applicationPath = appInfo.getApplicationPath().toString();
215        if (applicationPath.equals("/")) {
216            applicationPath = "";
217        }
218
219        // Ensure that new servlet mappings are distinct from existing
220        // servlet and resource handler mappings.  If not, throw exception
221        for (URI servletPath : appInfo.getServletInfo().keySet()) {
222            for (WebApplicationInfo application : _applications) {
223                if (application
224                        .hasPath(applicationPath + servletPath.toString())) {
225                    throw new Exception("Model " + appInfo.getModelName()
226                            + " requested a conflicting URL mapping, "
227                            + applicationPath + servletPath.toString());
228                }
229            }
230        }
231
232        // If no conflicts, create a handler for the new application
233        // If this is the first application, need to create a Jetty server and
234        // set the server's properties
235        if (_applications.isEmpty()) {
236            startServer();
237        }
238
239        // Add this application to the list of registered applications
240        _applications.add(appInfo);
241
242        // Create a handler for servlet requests
243        _createServletHandler(appInfo);
244
245        // Create / re-use handlers for resource requests
246        // Check if this web app wants to re-use the existing resource handler,
247        // or if a new one is needed
248        _createResourceHandlers(appInfo);
249    }

261    /** Set the maximum amount of time, in milliseconds, that the server will
262     * wait before returning a timeout response page.
263     *
264     * @param maxIdleTime The maximum amount of time, in milliseconds, that the
265     * server will wait before returning a timeout response page
266     * @see #getMaxIdleTime()
267     */
268    public void setMaxIdleTime(int maxIdleTime) {
269        // If the specified max idle time is <=0, use the default
270        if (maxIdleTime <= 0) {
271            _maxIdleTime = DEFAULT_MAX_IDLE_TIME;
272        } else {
273            _maxIdleTime = maxIdleTime;
274        }
275
276        // If there is a server instantiated, change the server's max idle time
277        if (_server != null) {
278            // Get the server's SelectChannelConnector to change the idle time
279            // There should be only one connector, since the current
280            // implementation runs one server per port, so that web applications
281            // can be developed and tested independently.  The idea is that
282            // each developer could be assigned a separate port for his or
283            // her applications.  If there were a single server listening to
284            // multiple ports, then all web applications would listen to all
285            // ports, meaning web applications from other developers could
286            // intercept requests.
287
288            // FIXME:  Need to pause the server if it is running, or OK to
289            // do this while running?
290            _selectChannelConnector.setMaxIdleTime(_maxIdleTime);
291
292        }
293    }

295    /** Unregister the given application from this server.  If this leaves no
296     * applications on the server, stop the server.
297     *
298     * @param appInfo  The application to remove from the server.
299     * @exception Exception if the application is not registered with this
300     * server or if the server cannot be stopped when the last application
301     * is unregistered
302     */
303    public void unregister(WebApplicationInfo appInfo) throws Exception {
304
305        // Check if this application is registered on this server.  The model
306        // name is used as a key - there should be only one instance of a
307        // model registered on any server at one time.  This restriction could
308        // be removed in the future if necessary.  But, running two instances
309        // of the same model simultaneously is confusing and probably
310        // unintentional.
311        boolean found = false;
312
313        if (_applications.contains(appInfo)) {
314            // Find the handler associated with this application, stop it and
315            // remove it from the server
316            ContextHandlerCollection handlers = (ContextHandlerCollection) _server
317                    .getHandler();
318            for (int i = 0; i < handlers.getHandlers().length; i++) {
319                Handler handler = handlers.getHandlers()[i];
320
321                // Only need to check servlet handlers (e.g. not the
322                // DefaultHandler, not resource handlers...)
323                // Check for matching application path
324                if (handler instanceof ServletContextHandler
325                        && ((ServletContextHandler) handler).getContextPath()
326                                .equalsIgnoreCase(appInfo.getApplicationPath()
327                                        .toString())) {
328                    for (URI servletPath : appInfo.getServletInfo().keySet()) {
329                        ServletHandler servletHandler = ((ServletContextHandler) handler)
330                                .getServletHandler();
331                        for (ServletMapping mapping : servletHandler
332                                .getServletMappings()) {
333
334                            // Any matching path means this is the servlet we want
335                            // to stop and remove
336                            for (String path : mapping.getPathSpecs()) {
337                                if (path.equalsIgnoreCase(
338                                        servletPath.toString())) {
339                                    handler.stop();
340                                    handlers.removeHandler(handler);
341                                    found = true;
342                                    break;
343                                }
344                            }
345                        }
346                    }
347                }
348            }
349
350            // Throw an exception if this application was not found
351            if (!found) {
352                throw new Exception("Application " + appInfo.getModelName()
353                        + " requested to be unregistered, but it was not found "
354                        + " on the server.");
355            }
356
357            // What to do about the resource handler?  This might be shared
358            // by other applications.  Need to keep a list.
359            // FIXME:  For now, just leave it running.
360            _applications.remove(appInfo);
361
362        } else {
363            throw new Exception("Application " + appInfo.getModelName()
364                    + " attempted to unregister itself, but this application is "
365                    + " not registered with the server on port " + _portNumber);
366        }
367
368        // If no applications are left, stop the server
369        if (_applications.isEmpty()) {
370            // Explicitly close the port connection so that the port will be
371            // available soon.  Otherwise, it was taking 30 seconds plus for
372            // the port to be relinquished, blocking any new web servers
373            _selectChannelConnector.close();
374            _serverThread.interrupt();
375            _serverThread = null;
376        }
377    }

395    /** Create a ContextHandler to store all of the servlets defined in the
396     *  given application (e.g. a Ptolemy model).  Add this handler to the
397     *  collection of handlers for this web server.
398     *
399     *  @param appInfo Information about the web application.
400     *  @exception Exception If the servlet cannot be started on the server
401     */
402    protected void _createServletHandler(WebApplicationInfo appInfo)
403            throws Exception {
404        // Create a new handler to hold servlets from the actors
405        ServletContextHandler servletHandler = new ServletContextHandler(
406                ServletContextHandler.SESSIONS);
407
408        servletHandler.setContextPath(appInfo.getApplicationPath().toString());
409
410        for (URI servletPath : appInfo.getServletInfo().keySet()) {
411            servletHandler.addServlet(
412                    new ServletHolder(
413                            appInfo.getServletInfo().get(servletPath)),
414                    servletPath.toString());
415        }
416
417        ((ContextHandlerCollection) _server.getHandler())
418                .addHandler(servletHandler);
419
420        // Need to explicitly start the handler since it is added to the
421        // server's handler list AFTER the server has already started
422        servletHandler.start();
423    }

425    /** Create or re-use resource handler(s) to serve files such as images,
426     *  audio, etc.  If the application requests a handler for a path that is
427     *  already used by an existing resource handler, then the existing handler
428     *  will be re-used, and the new search locations will be added to its
429     *  search list.  If a new path is provided, a new resource handler will be
430     *  created, as long as at least one search location is given.  New
431     *  handlers are added to the _handlers list.
432     *
433     *  @param appInfo An object containing information needed to register the
434     *   new application
435     *  @exception Exception If a FileParameter is found that is not a valid URI or
436     *  references a resource that cannot be found, or if the handler cannot
437     *  be started on the server
438     */
439
440    // FIXME:  Throw exception instead of allowing an empty list?  An
441    // application is not required to have a resource handler, so an empty
442    // list is permitted, but how to notify user in case the model has a
443    // mistake e.g. the user forgot to add resource locations?
444
445    protected void _createResourceHandlers(WebApplicationInfo appInfo)
446            throws Exception {
447
448        // Create resource handlers to serve files such as images, audio, ...
449        // See http://restlet-discuss.1400322.n2.nabble.com/Jetty-Webapp-td7313234.html
450        // There will be one resource handler per requested path
451        ArrayList<ContextHandler> handlers = new ArrayList<ContextHandler>();
452
453        for (URI path : appInfo.getResourceInfo().keySet()) {
454            boolean shareHandler = false;
455
456            for (WebApplicationInfo application : _applications) {
457                if (application.hasResourcePath(path.toString())) {
458                    // Share a handler.  Find the handler and add any new
459                    // resource locations.
460                    for (int i = 0; i < ((ContextHandlerCollection) _server
461                            .getHandler()).getHandlers().length; i++) {
462                        ContextHandler handler = (ContextHandler) ((ContextHandlerCollection) _server
463                                .getHandler()).getHandlers()[i];
464
465                        if (handler.getContextPath()
466                                .equalsIgnoreCase(path.toString())) {
467
468                            // Stop this resource handler
469                            handler.stop();
470
471                            // Add the new resource locations (if any)
472                            // FIXME:  Check if this cast is OK.  Otherwise
473                            // will have to manually remember the resources
474                            ResourceCollection resources = (ResourceCollection) handler
475                                    .getBaseResource();
476
477                            for (Resource resource : appInfo.getResourceInfo()
478                                    .get(path)) {
479                                if (!resource.isContainedIn(resources)) {
480                                    // Jetty doesn't seem to offer a method
481                                    // to add a resource to a
482                                    // ResourceCollection?
483                                    // Has addPath(String) but this returns
484                                    // a Resource, but not sure if it
485                                    // creates one...
486                                    ArrayList<Resource> newResources = new ArrayList<Resource>(
487                                            Arrays.asList(
488                                                    resources.getResources()));
489                                    newResources.add(resource);
490                                    resources.setResources(newResources.toArray(
491                                            new Resource[newResources.size()]));
492
493                                }
494                            }
495
496                            // Add the temporary file location (if any), if
497                            // not already included
498
499                            if (appInfo.getTemporaryFileLocation() != null
500                                    && !appInfo.getTemporaryFileLocation()
501                                            .toString().isEmpty()) {
502                                FileResource tempResource = new FileResource(
503                                        appInfo.getTemporaryFileLocation()
504                                                .asURL());
505                                if (resources != null && !tempResource
506                                        .isContainedIn(resources)) {
507                                    ArrayList<Resource> newResources = new ArrayList<Resource>(
508                                            Arrays.asList(
509                                                    resources.getResources()));
510                                    newResources.add(tempResource);
511                                    resources.setResources(newResources.toArray(
512                                            new Resource[newResources.size()]));
513                                }
514                            }
515
516                            // Restart the handler
517                            handler.start();
518
519                            shareHandler = true;
520                            break;
521                        }
522                    }
523                }
524
525                // Create a new handler if this is a new resource path
526                if (!shareHandler) {
527                    ContextHandler fileHandler = new ContextHandler();
528
529                    // Set the path used by other web apps to request files
530                    // Example:
531                    // $PTII/org/ptolemy/ptango/demo/WebServerDE/WebServerDE.xml
532                    fileHandler.setContextPath(path.toString());
533
534                    // Enable aliases so that we can use $TMPDIR under Mac OS X
535                    // because $TMPDIR is in /var, which is a symbolic link.
536                    // FIXME: this opens up a series of security holes.
537                    fileHandler.setAliases(true);
538
539                    // Create a new resource handler
540                    ResourceHandler resourceHandler = new ResourceHandler();
541                    // For security, do not support listing of directories in the local
542                    // resource locations.
543                    resourceHandler.setDirectoriesListed(false);
544
545                    // Tell handler to search requested locations and temporary file
546                    // location (if any) for files
547                    ArrayList<Resource> resources = new ArrayList<Resource>();
548                    resources.addAll(appInfo.getResourceInfo().get(path));
549
550                    if (appInfo.getTemporaryFileLocation() != null
551                            && (appInfo.getTemporaryFileLocation()
552                                    .getExpression() != null)
553                            && (!appInfo.getTemporaryFileLocation()
554                                    .getExpression().isEmpty())) {
555
556                        try {
557                            resources.add(new FileResource(appInfo
558                                    .getTemporaryFileLocation().asURL()));
559                        } catch (MalformedURLException e) {
560                            throw new Exception("Temporary file location "
561                                    + appInfo.getTemporaryFileLocation()
562                                    + " cannot be accessed.");
563                        } catch (URISyntaxException e2) {
564                            throw new Exception("Temporary file location "
565                                    + appInfo.getTemporaryFileLocation()
566                                    + " cannot be accessed.");
567                        }
568                    }
569
570                    // Ensure that at least one resource location has been
571                    // specified. Otherwise, don't add new handler to collection
572
573                    if (resources.size() > 0) {
574                        // Use setBaseResource(ResourceCollection) instead of
575                        // setResouceBase(String) so that we may add multiple
576                        // file locations.
577                        // setResourceBase(String) is a wrapper for
578                        // setBaseResource(ResourceCollection) that only allows
579                        // one location.
580
581                        ResourceCollection collection = new ResourceCollection();
582                        collection.setResources(resources
583                                .toArray(new Resource[resources.size()]));
584
585                        resourceHandler.setBaseResource(collection);
586
587                        fileHandler.setHandler(resourceHandler);
588                        handlers.add(fileHandler);
589                    }
590                }
591            }
592        }
593
594        for (ContextHandler handler : handlers) {
595            ((ContextHandlerCollection) _server.getHandler())
596                    .addHandler(handler);
597
598            // Need to explicitly start each new handler since they are added to
599            // the server's handler list AFTER the server has already started
600            handler.start();
601        }
602    }

607    private void startServer() throws Exception {
608        _server = new Server();
609        _selectChannelConnector = new SelectChannelConnector();
610        _selectChannelConnector.setPort(_portNumber);
611        _selectChannelConnector.setMaxIdleTime(_maxIdleTime);
612
613        // Don't allow other programs to use this port (e.g. another
614        // Jetty instance)
615        // FIXME:  Need to catch exception
616        _selectChannelConnector.setReuseAddress(false);
617        _server.setConnectors(new Connector[] { _selectChannelConnector });
618
619        // Create a ContextHandlerCollection containing only a
620        // DefaultHandler.  Other handlers will be added later for web apps
621        // See "Configuring the Server - Handlers"
622        // http://www.eclipse.org/jetty/documentation/current/quickstart-config-what.html
623
624        ContextHandlerCollection handlers = new ContextHandlerCollection();
625        handlers.addHandler(new DefaultHandler());
626        _server.setHandler(handlers);
627
628        // Start the server in a new thread. Use a custom uncaught exception
629        // handler so the server thread can log information about exceptions
630        // to the _exceptionMessage variable that the main thread can access
631        _startAttempted = false;
632        _exception = null;
633        _serverThread = new Thread(new RunnableServer());
634        _serverThread.start();
635
636        // Wait until the server has attempted to start
637        // The server thread will set _startAttempted to true
638        // Then, see if an exception has occurred
639        synchronized (_lock) {
640            while (!_startAttempted) {
641                try {
642                    _lock.wait();
643                } catch (InterruptedException e) {
644                    // FIXME: Do anything special if thread is interrupted?
645                    break;
646                }
647            }
648        }
649
650        boolean success = false;
651
652        // If an exception occurred, re-throw it
653        // One alternative considered was to use a Callable instead of a
654        // a Runnable, since a Callable can throw an exception
655        // However, the calling thread gets the exception back by calling
656        // Future.get(), which blocks the calling thread
657        // Since in normal operation the server thread runs indefinitely,
658        // Future.get() would block Ptolemy model execution indefinitely
659        if (_exception != null) {
660            if (_exception instanceof BindException) {
661                if (_dynamicPortSelection) {
662                    _portNumber = _initialDynamicPortNumber;
663
664                    while (_portNumber <= _maxDynamicPortNumber) {
665                        _startAttempted = false;
666                        _exception = null;
667                        _serverThread = new Thread(new RunnableServer());
668
669                        _selectChannelConnector.setPort(_portNumber);
670                        _serverThread.start();
671
672                        // Wait until the server has attempted to start
673                        // The server thread will set _startAttempted to true
674                        // Then, see if an exception has occurred
675                        synchronized (_lock) {
676                            while (!_startAttempted) {
677                                try {
678                                    _lock.wait();
679                                } catch (InterruptedException e) {
680                                    // FIXME: Do anything special if thread is interrupted?
681                                    break;
682                                }
683                            }
684                        }
685
686                        if (_exception != null) {
687                            if (_exception instanceof BindException) {
688                                _portNumber++;
689                            } else {
690                                throw new Exception(_exception);
691                            }
692                        } else {
693                            // No exception means server started successfully
694                            success = true;
695                            break;
696                        }
697                    }
698
699                    // Ran out of port numbers to try
700                    if (!success) {
701                        throw new Exception("The web server could not find an "
702                                + "available port between "
703                                + _initialDynamicPortNumber + " and "
704                                + _maxDynamicPortNumber);
705                    }
706                } else {
707                    throw new Exception("The web server attempted to start on"
708                            + " port " + _portNumber + ", but this port is "
709                            + "already in use.  Perhaps another instance of "
710                            + "Ptolemy is running a web server on this port?");
711                }
712            } else {
713                throw new Exception(_exception);
714            }
715        }
716    }

764    /** A Runnable class to run a Jetty web server in a separate thread.
765     */
766    private class RunnableServer implements Runnable {
767
768        /** Run the Jetty web server.  Stop the server if this thread is
769         *  interrupted (for example, when the model is finished executing,
770         *  WebServer's wrapup() will interrupt this thread) or if an
771         *  exception occurs.
772         */
773        @Override
774        public void run() {
775
776            while (!Thread.interrupted()) {
777
778                // Start the server.
779                // This is synchronized with the register() method, so that
780                // register() will wait for the server to either start properly
781                // or throw an exception
782                synchronized (_lock) {
783                    try {
784                        _startAttempted = true;
785                        _server.start();
786                        _lock.notify();
787                    } catch (Exception e) {
788                        // Notify thread users and terminate the server and this
789                        // thread if an exception occurs
790                        try {
791                            _exception = e;
792                            _server.stop();
793                            _lock.notify();
794                        } catch (Exception e2) {
795                            _exception = e;
796                            _lock.notify();
797                            Thread.currentThread().interrupt();
798                            return;
799                        }
800                        _lock.notify();
801                        Thread.currentThread().interrupt();
802                        return;
803                    }
804                    ;
805                }
806
807                // The .join() method blocks the thread until the server
808                // terminates.
809                try {
810                    _server.join();
811
812                } catch (InterruptedException e) {
813                    // Notify thread users and terminate the server and this
814                    // thread if an exception occurs
815                    try {
816                        _server.stop();
817                    } catch (Exception e2) {
818                        Thread.currentThread().interrupt();
819                        return;
820                    }
821                    Thread.currentThread().interrupt();
822                    return;
823                }
824                ;
825            }
826
827            try {
828                _server.stop();
829            } catch (Throwable throwable) {
830                // Nothing to do here since the thread will terminate next
831            }
832            return;
833        }
834    }
