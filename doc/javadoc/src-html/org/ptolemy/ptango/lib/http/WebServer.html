062/** An attribute that runs a Jetty web server and routes incoming
063 *  HTTP requests to objects in the model that implement
064 *  {@link HttpService}. The server is set up during
065 *  {@link #initialize()} and taken down during
066 *  {@link #wrapup()}.  The <i>resourceLocation</i>
067 *  parameter gives a directory or URL relative to which this
068 *  web server should look for resources (like image files and
069 *  the like).
070 *  You can add additional resource bases by adding additional
071 *  parameters of type ptolemy.data.expr.FileParameter to
072 *  this WebServer (select Configure in the context menu).
073 *
074 *  <p><a href="http://wiki.eclipse.org/Jetty/Tutorial">http://wiki.eclipse.org/Jetty/Tutorial</a>
075 *  - The Jetty Tutorial</p>
076 *
077 *   @author Elizabeth Latronico and Edward A. Lee
078 *   @version $Id$
079 *   @since Ptolemy II 10.0
080 *   @Pt.ProposedRating Yellow (eal)
081 *   @Pt.AcceptedRating Red (ltrnc)
082 */
083public class WebServer extends AbstractInitializableAttribute {
084
085    /** Construct an instance of the attribute.
086     * @param container The container.
087     * @param name The name.
088     * @exception IllegalActionException If the superclass throws it.
089     * @exception NameDuplicationException If the superclass throws it.
090     */
091    public WebServer(NamedObj container, String name)
092            throws IllegalActionException, NameDuplicationException {
093        super(container, name);
094
095        preferredPort = new Parameter(this, "preferredPort");
096        preferredPort.setTypeEquals(BaseType.INT);
097        preferredPort.setExpression(
098                Integer.toString(WebServerUtilities.DEFAULT_PORT_NUMBER));
099
100        applicationPath = new StringParameter(this, "applicationPath");
101        applicationPath.setExpression("/");
102
103        resourcePath = new StringParameter(this, "resourcePath");
104        resourcePath.setExpression("/");
105
106        // Set up a parameter to specify the location for reading and writing
107        // resources (files).  This parameter defaults to the directory that
108        // the current model is located in.
109        // FIXME: The full path is encoded. If this file is in the $PTII tree,
110        // then the path should begin with $PTII.
111
112        // The Jetty web server supports searching multiple directories/URLs for
113        // resources (files) to return as part of an HttpResponse.  However, if
114        // there are two different files with the same names in different
115        // directories, it is not clear which file will be served.  Right now,
116        // the directories are searched in alphabetical order by parameter name.
117        // WebServer itself currently only contains a reader
118        // (the resourceHandler in setResourceHandlers()
119        // Other actors (e.g. HttpCompositeServiceActor) are writers, and
120        // will look for a WebServer in the model to determine the directory
121        // to write to
122        resourceLocation = new FileParameter(this, "resourceLocation");
123
124        // Want resourceLocation to correspond to a directory
125        // Add parameters to specify that only directories are allowed
126        Parameter allowDirectories = new Parameter(resourceLocation,
127                "allowDirectories");
128        allowDirectories.setToken("true");
129        allowDirectories.setVisibility(Settable.NONE);
130
131        Parameter allowFiles = new Parameter(resourceLocation, "allowFiles");
132        allowFiles.setToken("false");
133        allowFiles.setVisibility(Settable.NONE);
134
135        URI modelURI = URIAttribute.getModelURI(this);
136        // Get the directory excluding the model's name
137        // This may be null for newly created models that have not been saved
138        // In that case, default to the temporary directory
139        // FIXME:  Register an attributeChanged event for when a model is saved
140        // to update this directory?
141        String path;
142        if (modelURI != null && modelURI.getPath() != null
143                && !modelURI.getPath().isEmpty()) {
144            path = modelURI.getPath();
145            int slash = path.lastIndexOf("/");
146            if (slash != -1) {
147                path = path.substring(0, slash);
148            }
149        } else {
150            path = "$TMPDIR";
151        }
152        resourceLocation.setExpression(path);
153
154        temporaryFileLocation = new FileParameter(this,
155                "temporaryFileLocation");
156        temporaryFileLocation.setExpression("$TMPDIR");
157
158        deployedPort = new Parameter(this, "deployedPort");
159        deployedPort.setExpression("-1");
160        deployedPort.setVisibility(Settable.NOT_EDITABLE);
161        // Don't save deployed port in MoML.  Otherwise, Ptolemy will ask you to
162        // save the model upon exiting every time a new deployed port is chosen
163        deployedPort.setPersistent(false);
164
165        _dynamicPortSelection = false;
166    }

171    /** The URL prefix to map this model to.  A model can be thought of as a
172     *  single application.
173     *
174     *  This defaults to "/", meaning, accept requests for all URLs.
175     *  For example, if the WebServer is handling requests on {@link #preferredPort} 8078
176     *  of localhost, the model will handle all requests that begin with
177     *  <pre>
178     *  http://localhost:8078/
179     *  </pre>
180     *
181     *  If multiple applications are running on the same server, it's typical
182     *  to assign a unique application prefix for each.  For example, two
183     *  applications with application paths of "/calendar" and "/music" would
184     *  receive requests directed to
185     *  <pre>
186     *  http://localhost:8078/calendar
187     *  http://localhost:8078/music
188     *  </pre>
189     *  respectively.
190     *
191     *  Applications may contain multiple servlets (provided by e.g.
192     *  {@link HttpActor}s).
193     *  Each servlet typically defines a servlet path which is appended to
194     *  the application path for request routing.  For example, two servlets
195     *  with servlet paths "/play" and "/upload" in the music application
196     *  would receive requests directed to:
197     *  <pre>
198     *  http://localhost:8078/music/play
199     *  http://localhost:8078/music/upload
200     *  </pre>
201     *  respectively.
202     */
203    public StringParameter applicationPath;
204
205    /** The port number the server is listening to.  Can be different from
206     *  preferredPort if dynamic port selection is enabled.
207     */
208    public Parameter deployedPort;
209
210    /** The preferred port number for the server to listen to. This is a integer
211     *  that defaults to 8078.
212     */
213    public Parameter preferredPort;

215    /** The URL prefix used to request resources (files) from this web service.
216     *
217     *  This defaults to "/", meaning, clients should issue requests for files
218     *  to the base URL.  For example, if the WebServer is handling requests on
219     *  {@link #preferredPort} 8078 of localhost, the client can request file "image.png"
220     *  using:
221     *  <pre>
222     *  http://localhost:8078/image.png
223     *  </pre>
224     *
225     *  The resourcePath is added as a prefix before the file's name in the URL.
226     *  For example, given the resourcePath "/files", the client can request
227     *  file "image.png" using:
228     *  <pre>
229     *  http://localhost:8078/files/image.png
230     *  </pre>
231     *
232     *  Relative URLs are permissible for files requested by pages served by
233     *  this WebServer (common for e.g. custom Javascript libraries).
234     *  For example, given the resourcePath "/files", a file named "custom.js"
235     *  may be retrieved at the relative URL:
236     *  <pre>
237     *  /files/custom.js
238     *  </pre>
239     *
240     *  Note that the name of the directory containing resources does NOT have
241     *  to match the URL used to access these resources.  See
242     *  {@link #resourceLocation} regarding the resource directory.
243     *
244     *  Subdirectory search is supported (here, for subdirectories to the
245     *  directory specified by {@link #resourceLocation}). Add the subdirectory
246     *  path to the URL.  For example, given the resourcePath "/files",
247     *  a file in subdirectory /user/photos named "selfie.png" may be
248     *  retrieved at URL:
249     *
250     *  <pre>
251     *  http://localhost:8078/files/user/photos/selfie.png
252     *  </pre>
253     */
254
255    public StringParameter resourcePath;
256
257    /** A directory or URL where the web server will look for resources
258     *  (like image files and the like).
259     *  This defaults to the current model's directory.
260     *
261     *  You can add additional resource locations by adding additional
262     *  parameters of type ptolemy.data.expr.FileParameter to
263     *  this WebServer (select Configure in the context menu).
264     *
265     *  If multiple resourceLocations are given, they will be searched in the
266     *  order that the parameters were instantiated.  The first file located
267     *  will be returned.
268     *
269     *  Note that the name of the directory containing resources does NOT have
270     *  to match the URL used to access these resources.  See
271     *  {@link #resourcePath} regarding the URL.
272     */
273    public FileParameter resourceLocation;
274
275    /** A directory where the web server will look for resources
276     *  (like image files and the like). This specifies an additional
277     *  resource location after {@link #resourceLocation}, but the
278     *  directory specified here may be used by components implementing
279     *  {@link HttpService} as a place to write temporary files
280     *  that will be found by the web server.
281     *  This defaults to "$TMPDIR", a built-in variable
282     *  that specifies a temporary file location.
283     *  See the explanation of {@link #resourcePath}.
284     */
285    public FileParameter temporaryFileLocation;

290    /** React to a change in an attribute.  If the attribute is an
291     *  instance of FileParameter and the server is running (initialize()
292     *  has been called and wrapup() has not), then update the resource
293     *  handlers in the server.
294     *  @param attribute The attribute that changed.
295     *  @exception IllegalActionException If the change is not acceptable
296     *   to this container (not thrown in this base class).
297     */
298
299    @Override
300    public void attributeChanged(Attribute attribute)
301            throws IllegalActionException {
302
303        // Changes to attributes are not currently propagated to the
304        // web server directly.  They will take effect the next time
305        // initialize() is called.
306
307        // In the future, changes could be propagated immediately to the
308        // web server.
309
310        if (attribute == preferredPort) {
311            if (preferredPort == null
312                    || preferredPort.getExpression().isEmpty()) {
313                _dynamicPortSelection = true;
314            } else {
315                _dynamicPortSelection = false;
316            }
317        } else {
318            super.attributeChanged(attribute);
319        }
320    }

322    /** Clone the attribute.
323     *  @param workspace The workspace in which to place the cloned attribute.
324     *  @exception CloneNotSupportedException Not thrown in this base class.
325     *  @see java.lang.Object#clone()
326     *  @return The cloned attribute.
327     */
328    @Override
329    public Object clone(Workspace workspace) throws CloneNotSupportedException {
330        WebServer newObject = (WebServer) super.clone(workspace);
331        // _appInfo is set in initialize()
332        newObject._appInfo = null;
333        newObject._dynamicPortSelection = false;
334        newObject._serverManager = WebServerManager.getInstance();
335
336        return newObject;
337    }

339    /** Collect servlets from all model objects implementing HttpService
340     *  and start the web server in a new thread.
341     *  <p>
342     * In the current implementation, servlets must be registered before the
343     * Jetty server starts.  Servlets are not allowed to be added to a running
344     * ContextHandler.  Currently, the Jetty server is started once and runs
345     * until the model finishes executing.  It would also be possible to pause
346     * the server, add a servlet, and restart the server, which would allow
347     * a model to dynamically add servlets.  This might cause strange behavior
348     * to an outside observer, however, since some HttpRequests could fail
349     * if a servlet has not been loaded yet.
350     *
351     *  References:
352     *  <ul>
353     *  <li> <a href="http://wiki.eclipse.org/Jetty/Tutorial/Embedding_Jetty">http://wiki.eclipse.org/Jetty/Tutorial/Embedding_Jetty</a></li>
354     *  <li> <a href="http://wiki.eclipse.org/Jetty/Tutorial/Embedding_Jetty">http://wiki.eclipse.org/Jetty/Tutorial/Embedding_Jetty</a></li>
355     *  <li> <a href="http://draconianoverlord.com/2009/01/10/war-less-dev-with-jetty.html">http://draconianoverlord.com/2009/01/10/war-less-dev-with-jetty.html</a></li>
356     *  </ul>
357     *  @exception IllegalActionException Not thrown in this base class.
358     */
359    @Override
360    public void initialize() throws IllegalActionException {
361
362        super.initialize();
363
364        if (_debugging) {
365            _debug("Initializing web server.");
366        }
367
368        int preferredPortValue = WebServerUtilities.DEFAULT_PORT_NUMBER;
369
370        if (preferredPort != null && !preferredPort.getExpression().isEmpty()) {
371            preferredPortValue = Integer
372                    .parseInt(preferredPort.getExpression());
373            _dynamicPortSelection = false;
374        } else {
375            _dynamicPortSelection = true;
376        }
377
378        // Get the web server manager to register this application with
379        if (_serverManager == null) {
380            _serverManager = WebServerManager.getInstance();
381        }
382
383        // Get the model name, application path and temporary file location
384        String modelName = getFullName();
385
386        String applicationPathString = "/";
387        if (applicationPath != null) {
388
389            // Application path must begin with an "/"; add one if missing
390            if (applicationPath.getExpression().startsWith("/")) {
391                applicationPathString = applicationPath.getExpression();
392            } else {
393                applicationPathString = "/" + applicationPath.getExpression();
394            }
395        }
396
397        // Assemble info about this model into a WebApplicationInfo object
398        // Throw an exception if the model does not have a name
399        try {
400            _appInfo = new WebApplicationInfo(modelName, applicationPathString,
401                    temporaryFileLocation);
402        } catch (Exception e) {
403            throw new IllegalActionException(this, e,
404                    "Failed to create WebApplicationInfo");
405        }
406
407        // Collect requested servlet mappings from all model objects
408        // implementing HttpService.  Check for duplicates.
409        // NOTE: This used to use the top level, but it makes more sense to use the container.
410        // Also, now it only looks for entities, and it does not penetrate opaque composites.
411        NamedObj container = getContainer();
412        if (!(container instanceof CompositeEntity)) {
413            throw new IllegalActionException(this,
414                    "Container is required to be a CompositeEntity.");
415        }
416        @SuppressWarnings("unchecked")
417        List<Entity<Port>> entities = ((CompositeEntity) container)
418                .allAtomicEntityList();
419        for (Entity<Port> entity : entities) {
420            if (entity instanceof HttpService) {
421                HttpService service = (HttpService) entity;
422                // Tell the HttpService that this is its WebServer,
423                // so that it can get, for example, critical information such
424                // as resourcePath.
425                service.setWebServer(this);
426
427                if (_debugging) {
428                    _debug("Found web service: " + entity.getFullName());
429                }
430
431                // Add this path to the list of servlet paths
432                URI path = service.getRelativePath();
433
434                try {
435                    _appInfo.addServletInfo(path, service.getServlet());
436                } catch (Exception e) {
437                    throw new IllegalActionException(this, "Actor "
438                            + entity.getName()
439                            + " requested the web service URL " + path
440                            + " , but this URL has already been claimed "
441                            + "by another actor or by a resource in this WebServer."
442                            + "  Please specify a unique URL.");
443                }
444            }
445        }
446
447        // Specify directories or URLs in which to look for resources.
448        // These are given by all instances of FileParameter in this
449        // WebServer. Use a LinkedHashSet to preserve the order.
450        LinkedHashSet<Resource> resourceLocations = new LinkedHashSet<Resource>();
451        List<FileParameter> bases = attributeList(FileParameter.class);
452        // To prevent duplicates, keep track of bases added
453        // This set includes the temporary file location
454        HashSet<URL> seen = new HashSet<URL>();
455        for (FileParameter base : bases) {
456            // If blank, omit
457            if (base.getExpression() != null
458                    && !base.getExpression().isEmpty()) {
459                try {
460
461                    // Use the ClassLoader to obtain the location (vs. specifying
462                    // a directory on the file system), so that the demos
463                    // can also be run from within a .jar file
464
465                    // If expression starts with $PTII/ , strip this
466                    // Also strip out $CLASSPATH (WebStart uses this)
467                    // Assumes .jar file uses $PTII as root location
468                    // Assumes full path is given
469                    // TODO:  What to do about paths relative to model's location?
470                    // Would these work?  None are used so far.
471
472                    String expression = base.getExpression();
473                    if (expression.startsWith("$PTII/")) {
474                        expression = expression.substring(6);
475                    }
476
477                    // Get directory
478                    // Try ClassLoader first to resolve any directories within
479                    // Ptolemy tree.  If the directory is not part of the tree
480                    // (e.g. $TMPDIR), the ClassLoader will not find it, so then
481                    // use the expression directly
482                    URL baseURL;
483
484                    if (expression.startsWith("$CLASSPATH/")) {
485                        baseURL = base.asURL();
486                    } else if (this.getClass().getClassLoader()
487                            .getResource(expression) != null) {
488                        baseURL = new URL(this.getClass().getClassLoader()
489                                .getResource(expression).toExternalForm());
490                    } else {
491                        baseURL = base.asURL();
492                    }
493
494                    if (baseURL != null) {
495                        URL baseAsURL = base.asURL();
496                        if (seen.contains(baseAsURL)) {
497                            continue;
498                        }
499                        seen.add(baseAsURL);
500                        if (_debugging) {
501                            _debug("Adding resource location: " + baseAsURL);
502                        }
503
504                        // Use Resource.newResource() to support WebStart
505                        // http://67-23-9-112.static.slicehost.net/faq?s=960-WebStart&t=Eclipse
506                        String stringURL = baseAsURL.toExternalForm();
507                        if (!stringURL.endsWith("/")) {
508                            stringURL = stringURL + "/";
509                        }
510
511                        resourceLocations.add(Resource.newResource(stringURL));
512                    }
513                } catch (IOException e3) {
514                    throw new IllegalActionException(this,
515                            "Can't access resource base: "
516                                    + base.stringValue());
517                }
518            }
519        }
520
521        // Throw an exception if resource path is not a valid URI, if a
522        // duplicate path is requested or if the directory does not exist
523        try {
524            _appInfo.addResourceInfo(new URI(resourcePath.stringValue()),
525                    resourceLocations);
526        } catch (URISyntaxException e) {
527            throw new IllegalActionException(this,
528                    "Resource path is not a " + "valid URI.");
529        } catch (Exception e2) {
530            throw new IllegalActionException(this, e2,
531                    "Failed to add resource info.");
532        }
533
534        try {
535            int actualPort = _serverManager.register(_appInfo,
536                    preferredPortValue, _dynamicPortSelection);
537            if (actualPort != -1) {
538                deployedPort.setExpression(Integer.toString(actualPort));
539                deployedPort.validate();
540            }
541        } catch (Exception e) {
542            throw new IllegalActionException(this, e,
543                    "Failed to register web server.");
544        }
545    }
546
547    /** Unregister this application with the web server manager.
548     *
549     * @exception IllegalActionException if there is a problem unregistering
550     * the application */
551    @Override
552    public void wrapup() throws IllegalActionException {
553        super.wrapup();
554        if (_debugging) {
555            _debug("Unregistering web application.");
556        }
557
558        if (deployedPort != null && deployedPort.getExpression() != null) {
559            // Only attempt to unregister application if registration was
560            // successful.
561
562            // If we are exporting to JNLP, then initialize might not
563            // have been called.
564
565            int deployedPortValue = Integer
566                    .parseInt(deployedPort.getExpression());
567            if (_serverManager != null && _appInfo != null && _serverManager
568                    .isRegistered(_appInfo.getModelName(), deployedPortValue)) {
569                try {
570                    _serverManager.unregister(_appInfo, deployedPortValue);
571                } catch (Exception e) {
572                    throw new IllegalActionException(this, e,
573                            "Failed to stop web application.");
574                }
575            }
576        }
577    }
