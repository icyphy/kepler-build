045/** Information for a web application running on a web server.  Used by
046 * {@link org.ptolemy.ptango.lib.WebServerUtilities}
047 *
048 *   @author Elizabeth Latronico and Edward A. Lee
049 *   @version $Id$
050 *   @since Ptolemy II 10.0
051 *   @Pt.ProposedRating Red (ltrnc)
052 *   @Pt.AcceptedRating Red (ltrnc)

055public class WebApplicationInfo {
056
057    /** Construct a new instance of this class with the given model name, an
058     * application path, and a temporary file location.
059     *
060     * @param modelName  The full name of the model
061     * @param applicationPath  The base path for the application
062     * @param temporaryFileLocation  The directory where temporary files are
063     * stored
064     * @exception Exception If the model does not yet have a name (for example,
065     *  a new unsaved model)
066     */
067    public WebApplicationInfo(String modelName, String applicationPath,
068            FileParameter temporaryFileLocation) throws Exception {
069
070        // Throw an exception if the model does not have a name (for example,
071        // if a new unnamed model is not saved yet).
072        // WebApplicationInfo is not a Nameable, so we can't use
073        // IllegalActionException here.  The caller can catch this exception
074        // and wrap it in an IllegalActionException
075        if (modelName == null || modelName.isEmpty()) {
076            throw new Exception("A model must have a name to create a web "
077                    + "application.  Perhaps this is a new, unnamed model?  "
078                    + "Please save the model with a name.");
079        }
080
081        _modelName = modelName;
082
083        // Assign a default application path of "/" if none is given
084        if (applicationPath == null || applicationPath.isEmpty()) {
085            _applicationPath = new URI("/");
086        } else {
087            _applicationPath = new URI(applicationPath);
088        }
089
090        // The temporary file location may be null or empty.
091        if (temporaryFileLocation == null) {
092            _temporaryFileLocation = null;
093        } else {
094            _temporaryFileLocation = temporaryFileLocation;
095        }
096
097        // Initialize lists
098        _servletInfo = new HashMap<URI, Servlet>();
099        _resourceInfo = new HashMap<URI, HashSet<Resource>>();
100
101    }
102

106    /** Add a new servlet path to the set of paths of this web application.
107     * Null, empty or duplicate paths are not allowed.  Return true if the path
108     * was added successfully, or false if the path was null, empty or a
109     * duplicate
110     *
111     * @param servletPath  The new servlet path to add
112     * @param servlet The servlet to associated with the servlet path
113     * @exception Exception If the path has already been requested by another servlet
114     */
115    public void addServletInfo(URI servletPath, Servlet servlet)
116            throws Exception {
117        if (_servletInfo == null) {
118            _servletInfo = new HashMap<URI, Servlet>();
119        }
120
121        // Check if this path is null, empty or has already been taken by
122        // another controller or by a resource handler
123
124        // It is generally OK if this path is a prefix of an existing path, or
125        // an existing path is a prefix of the new path.  This is common in
126        // RESTful web design.  The HTPP request will be matched to the most
127        // specific path.  There might be a rare case where the designer
128        // intended a less-specific path to handle all requests - in this
129        // scenario, the new controller with the more-specific path will
130        // erroneously handle some of the requests.
131        if (servletPath == null || servletPath.toString().isEmpty()
132                || _servletInfo.containsKey(servletPath)
133                || _resourceInfo.keySet().contains(servletPath)) {
134            throw new Exception("Duplicate path requested by a servlet, "
135                    + servletPath
136                    + " . Please check other servlet paths and resource "
137                    + "paths for matches.");
138        }
139
140        _servletInfo.put(servletPath, servlet);
141    }
142
143    /** Add a new resource path and location set for this web application.
144     * A null path is not allowed.  An empty path is OK (since apps may share a
145     * resource handler, it can make sense to map the handler to the root path).
146     * If the path is a duplicate of a path already assigned to a resource
147     * handler, the resource location list for that path will be updated. The
148     * path may not be a duplicate of a path already assigned to a context
149     * handler.
150     *
151     * There should be at least one resource location.
152     *
153     * @param resourcePath  The new resource path to add.  May not be null.
154     * @param resourceLocations The non-empty set of resource locations to add
155     * @exception Exception If this path is a duplicate of a path already requested by
156     * a servlet, or if the resource does not exist
157     */
158    public void addResourceInfo(URI resourcePath,
159            Set<Resource> resourceLocations) throws Exception {
160        if (resourcePath == null || _servletInfo.containsKey(resourcePath)
161                || resourceLocations == null || resourceLocations.isEmpty()) {
162            throw new Exception(
163                    "Duplicate path requested by a resource, " + resourcePath
164                            + " . Please check servlet paths for matches.");
165        }
166
167        for (Resource resource : resourceLocations) {
168            if (!resource.exists()) {
169                throw new Exception("Resource " + resource.getName()
170                        + " does not exist. Please check the path and "
171                        + "read permission.");
172            }
173        }
174
175        if (_resourceInfo.containsKey(resourcePath)) {
176            _resourceInfo.get(resourcePath).addAll(resourceLocations);
177        } else {
178            _resourceInfo.put(resourcePath, new HashSet(resourceLocations));
179        }
180    }

230    /** Returns true if the web application has already mapped a handler to the
231     * given URL; returns false otherwise.  The handler can be any handler -
232     * for example, a ContextHandler or a ResourceHandler.  This check is
233     * purposefully NOT case-sensitive.  While URLs are case-sensitive
234     * (excluding the domain name), many users are not aware of this,
235     * and it would be confusing to have two URLs that differ only in case.
236     *
237     * The application path should be included as part of this URL, e.g.
238     * /webapp/start for service /start of application /webapp
239     * @param path  The URL to check for existing mappings
240     * @return True if the web application has already mapped a handler to the
241     * given URL; false otherwise
242     */
243    public boolean hasPath(String path) {
244
245        // Check for a null application path.  This is not allowed, but in case
246        // it was just not set yet, use default "/"
247        String appPath;
248        if (_applicationPath == null) {
249            appPath = "/";
250        } else {
251            appPath = _applicationPath.toString();
252        }
253
254        // Check for a match to the application path
255        if (path.equalsIgnoreCase(appPath)) {
256            return true;
257        }
258
259        // Check for a matching handler path
260        for (URI servletPath : _servletInfo.keySet()) {
261            if (servletPath != null && path
262                    .equalsIgnoreCase(appPath + servletPath.toString())) {
263                return true;
264            }
265        }
266
267        // Check for a matching resource path
268        for (URI resourcePath : _resourceInfo.keySet()) {
269            if (resourcePath != null && path
270                    .equalsIgnoreCase(appPath + resourcePath.toString())) {
271                return true;
272            }
273        }
274
275        return false;
276    }
277
278    /** Returns true if the web application has already mapped a handler to the
279     * given URL; returns false otherwise.  See {@link #hasPath(String path)}
280     *
281     * @param path The URL to check for existing mappings
282     * @return True if the web application has already mapped a handler to the
283     * given URL; false otherwise
284     */
285    public boolean hasPath(URI path) {
286        return hasPath(path.toString());
287    }

289    /** Returns true if the web application has already mapped a resource
290     * handler to the given path.  Used by the web server to determine if a new
291     * resource handler is needed or if an existing one can be re-used.
292     *
293     * @param path The URL to check for existing mappings to a resource handler
294     * @return True if the web application has already mapped a resource handler
295     * to the given URL
296     */
297    public boolean hasResourcePath(String path) {
298        // Check for a null application path.  This is not allowed, but in case
299        // it was just not set yet, use default "/"
300        String appPath;
301        if (_applicationPath == null) {
302            appPath = "/";
303        } else {
304            appPath = _applicationPath.toString();
305        }
306
307        // Check for a matching resource path
308        for (URI resourcePath : _resourceInfo.keySet()) {
309            if (resourcePath != null && path
310                    .equalsIgnoreCase(appPath + resourcePath.toString())) {
311                return true;
312            }
313        }
314
315        return false;
316    }

353    /** The base path of this application.  For example, an application with the
354     *  path /modelName on the host localhost at port 8080 would be accessed at
355     *  URL http://localhost:8080/modelName
356     *  There should be one application path per model, since the application
357     *  path is defined in {@link org.ptolemy.ptango.lib.WebServer} which may
358     *  only have one instance per model.
359     *
360     *  See also comments in {@link org.ptolemy.ptango.lib.WebServer}
361     */
362    private URI _applicationPath;
363
364    /** The full name of the Ptolemy model that this web app was created from.
365     */
366    private String _modelName;
367
368    /** A map associating a resource path with one or more resource locations.
369     *
370     * A web application often hosts static content such as HTML pages and
371     * images.  This content is server by a resource handler.  The resource
372     * handler mapped to a URL, the resource path.  For example, the file
373     * image.gif served by a resource handler with path /files in the
374     * application with path /modelName on host localhost at port 8080 would be
375     * accessed at URL http://localhost:8080/modelName/files/image.gif
376     *
377     * This full URL path must be unique on the server and may not be a prefix
378     * of any other unique path.  This ensures that there is a unique handler
379     * for each HTTP request.  One
380     * {@link org.eclipse.jetty.server.handler.ResourceHandler}
381     * will be allocated per resource path.  Models may share a ResourceHandler
382     * by specifying the same resource path.
383     *
384     * A resource location is a directory where resources (e.g. files) are
385     * stored.  A ResourceHandler may have one or more resource locations.
386     * Upon receiving a request, the ResourceHandler will search the resource
387     * locations for the file requested.
388     *
389     * Note that the order of resource location searching is unspecified by
390     * Jetty.  Therefore, if two models share a ResourceHandler, and serve
391     * files with the same name but that are in different resource locations,
392     * an incorrect file may be returned.  In this scenario, the models should
393     * use different ResourceHandlers by specifying different resource paths.
394     *
395     * May be an empty list, if the application does not serve any static
396     * content or temporary files.
397     *
398     * See also comments in {@link org.ptolemy.ptango.lib.WebServer}
399     */
400    private HashMap<URI, HashSet<Resource>> _resourceInfo;
401
402    /** A map associating servlet paths with servlets.
403     *
404     * A web application is typically composed of multiple servlets that
405     * handle requests submitted to a specific sub-path prefix.  For example,
406     * a servlet that handles requests at the path prefix /start on the host
407     * localhost at port 8080 in the application at path /modelName would be
408     * accessed at URL http://localhost:8080/modelName/start
409     *
410     * This full URL path must be unique on the server and may not be a prefix
411     * of any other unique path.  This ensures that there is a unique handler
412     * for each HTTP request.  One
413     * {@link org.eclipse.jetty.server.handler.ContextHandler}
414     * will be allocated per servlet path.  Since the servlet
415     * functionality is different for each web service actor, two separate
416     * actors may not specify the same servlet paths.
417     *
418     * May be an empty list, perhaps if the application only wants to serve
419     * static content using a ResourceHandler.
420     *
421     * See also comments in {@link org.ptolemy.ptango.lib.WebServer}
422     */
423    private HashMap<URI, Servlet> _servletInfo;
424
425    /** The location, typically the directory $TMPDIR, that temporary files are
426     * stored in.  A temporary file is a file that should be deleted upon model
427     * wrapup, for example, and image created by HTMLModelExporter.  All
428     * ResourceHandlers for this model should also search here for resources.
429     * There is one temporary file location per model, since the temporary
430     * file location is defined in the {@link org.ptolemy.ptango.lib.WebServer},
431     * which may only have one instance per model.
432     *
433     * May be null or empty if the model does not store temporary files.
434     */
435    private FileParameter _temporaryFileLocation;
