061/**
062 * A subclass of ModelReference that produces on its output an HTML page for the contained model.
063 *
064 * FIXME: More
065 *
066 * Running in a new thread is not supported by HTMLModelExporter.
067 *
068 * @author Edward A. Lee and Beth Latronico
069 * @version $Id$
070 * @since Ptolemy II 10.0
071 * @see ptolemy.vergil.basic.export.web.HTMLText
072 * @see ptolemy.vergil.basic.export.web.LinkToOpenTableaux
073
074 * @Pt.ProposedRating Red (ltrnc)
075 * @Pt.AcceptedRating Red (ltrnc)
076 */
077public class HTMLModelExporter extends VisualModelReference {
078
079    /** Create an instance of this parameter.
080     *  @param container The container.
081     *  @param name The name.
082     *  @exception IllegalActionException If the superclass throws it.
083     *  @exception NameDuplicationException If the superclass throws it.
084     */
085    public HTMLModelExporter(CompositeEntity container, String name)
086            throws IllegalActionException, NameDuplicationException {
087        super(container, name);
088
089        _parameters = new ExportParameters();
090
091        directoryToExportTo = new FileParameter(this, "directoryToExportTo");
092        Parameter allowFiles = new Parameter(directoryToExportTo, "allowFiles");
093        allowFiles.setExpression("false");
094        allowFiles.setVisibility(Settable.NONE);
095        Parameter allowDirectories = new Parameter(directoryToExportTo,
096                "allowDirectories");
097        allowDirectories.setExpression("true");
098        allowDirectories.setVisibility(Settable.NONE);
099
100        backgroundColor = new ColorAttribute(this, "backgroundColor");
101
102        openCompositesBeforeExport = new Parameter(this,
103                "openCompositesBeforeExport");
104        openCompositesBeforeExport.setTypeEquals(BaseType.BOOLEAN);
105        openCompositesBeforeExport.setExpression("false");
106
107        runBeforeExport = new Parameter(this, "runBeforeExport");
108        runBeforeExport.setTypeEquals(BaseType.BOOLEAN);
109        runBeforeExport.setExpression("false");
110
111        deleteFilesOnExit = new Parameter(this, "deleteFilesOnExit");
112        deleteFilesOnExit.setTypeEquals(BaseType.BOOLEAN);
113        deleteFilesOnExit.setExpression("true");
114
115        copyJavaScriptFiles = new Parameter(this, "copyJavaScriptFiles");
116        copyJavaScriptFiles.setTypeEquals(BaseType.BOOLEAN);
117        copyJavaScriptFiles.setExpression("false");
118
119        usePtWebsite = new Parameter(this, "usePtWebsite");
120        usePtWebsite.setTypeEquals(BaseType.BOOLEAN);
121        usePtWebsite.setExpression("false");
122        usePtWebsite.setVisibility(Settable.EXPERT);
123
124        webPage = new TypedIOPort(this, "webPage", false, true);
125        webPage.setTypeEquals(BaseType.STRING);
126
127        // Use runBeforeExport rather than the superclass parameter
128        // executionOnFiring, which we hide.
129        executionOnFiring.setExpression("do nothing");
130        executionOnFiring.setVisibility(Settable.EXPERT);
131
132        // Control the opening and closing of the model.
133        // FIXME: The options allow for the submodel to not have been
134        // opened at all, in which case we want to export only the results
135        // of execution.  For now, we just hide the parameters
136        // that controls this, openOnFiring and closeOnPostfire.
137        openOnFiring.setExpression("open in Vergil");
138        closeOnPostfire.setExpression("do nothing");
139        openOnFiring.setVisibility(Settable.EXPERT);
140        closeOnPostfire.setVisibility(Settable.EXPERT);
141
142        postfireAction.setVisibility(Settable.EXPERT);
143
144        // Do not offer the option to show the result in a browser.
145        // It is an output of this actor.
146        _parameters.showInBrowser = false;
147    }

213    /** React to a change in an attribute.
214     *  @param attribute The attribute that changed.
215     *  @exception IllegalActionException If the change is not acceptable
216     *   to this container (not thrown in this base class).
217     */
218    @Override
219    public void attributeChanged(Attribute attribute)
220            throws IllegalActionException {
221        if (attribute == backgroundColor) {
222            if (backgroundColor.getToken() == null) {
223                // No color specified. Use the model's background color,
224                // if there is one, and the default background if not.
225                // First, see whether there is a local preferences object
226                // in the model.
227                NamedObj container = getContainer();
228                if (container != null) {
229                    List list = container
230                            .attributeList(PtolemyPreferences.class);
231                    if (list.size() > 0) {
232                        // Use the last of the preferences if there is more than one.
233                        PtolemyPreferences preferences = (PtolemyPreferences) list
234                                .get(list.size() - 1);
235                        _parameters.backgroundColor = preferences.backgroundColor
236                                .asColor();
237                        return;
238                    }
239                }
240                // There is no local preferences. If we have previously
241                // looked up a default color in the configuration, use that
242                // color.
243                if (_defaultColor == null) {
244                    // Look for default preferences in the configuration.
245                    Effigy effigy = Configuration
246                            .findEffigy(container.toplevel());
247                    if (effigy == null) {
248                        // No effigy. Can't find a configuration.
249                        _defaultColor = Color.white;
250                    } else {
251                        Configuration configuration = (Configuration) effigy
252                                .toplevel();
253                        try {
254                            PtolemyPreferences preferences = PtolemyPreferences
255                                    .getPtolemyPreferencesWithinConfiguration(
256                                            configuration);
257                            if (preferences != null) {
258                                _defaultColor = preferences.backgroundColor
259                                        .asColor();
260                            }
261                        } catch (IllegalActionException ex) {
262                            _defaultColor = Color.white;
263                        }
264                    }
265                }
266                _parameters.backgroundColor = _defaultColor;
267            } else {
268                _parameters.backgroundColor = backgroundColor.asColor();
269            }
270        } else if (attribute == copyJavaScriptFiles) {
271            _parameters.copyJavaScriptFiles = ((BooleanToken) copyJavaScriptFiles
272                    .getToken()).booleanValue();
273        } else if (attribute == deleteFilesOnExit) {
274            _parameters.deleteFilesOnExit = ((BooleanToken) deleteFilesOnExit
275                    .getToken()).booleanValue();
276        } else if (attribute == directoryToExportTo) {
277            _parameters.directoryToExportTo = directoryToExportTo.asFile();
278        } else if (attribute == openCompositesBeforeExport) {
279            _parameters.openCompositesBeforeExport = ((BooleanToken) openCompositesBeforeExport
280                    .getToken()).booleanValue();
281        } else if (attribute == runBeforeExport) {
282            _parameters.runBeforeExport = ((BooleanToken) runBeforeExport
283                    .getToken()).booleanValue();
284            if (_parameters.runBeforeExport) {
285                executionOnFiring.setExpression("run in calling thread");
286            } else {
287                executionOnFiring.setExpression("do nothing");
288            }
289        } else if (attribute == usePtWebsite) {
290            _parameters.usePtWebsite = ((BooleanToken) usePtWebsite.getToken())
291                    .booleanValue();
292        } else if (attribute == executionOnFiring) {
293            String executionOnFiringValue = executionOnFiring.stringValue();
294
295            if (executionOnFiringValue.equals("run in a new thread")) {
296                throw new IllegalActionException(this,
297                        "Running in a new thread is not supported by HTMLModelExporter.");
298            }
299            super.attributeChanged(attribute);
300        } else {
301            super.attributeChanged(attribute);
302        }
303    }

305    /** Clone the attribute.
306     *  @param workspace The workspace in which to place the cloned attribute.
307     *  @exception CloneNotSupportedException Not thrown in this base class.
308     *  @see java.lang.Object#clone()
309     *  @return The cloned attribute.
310     */
311    @Override
312    public Object clone(Workspace workspace) throws CloneNotSupportedException {
313        HTMLModelExporter newObject = (HTMLModelExporter) super.clone(
314                workspace);
315        newObject._defaultColor = null;
316        newObject._parameters = new ExportParameters();
317        return newObject;
318    }

320    @Override
321    public void fire() throws IllegalActionException {
322        super.fire();
323        // NOTE: The opening must occur in the event thread.
324        // Regrettably, we cannot continue with the firing until
325        // everything is complete, so we use the very dangerous
326        // invokeAndWait() method.
327        _exception = null;
328        Runnable doExport = new Runnable() {
329            @Override
330            public void run() {
331                // Since we do not support running in a new thread,
332                // the model has completed execution if it is to execute at all.
333                if (_tableau != null) {
334                    // FIXME: If _parameters.directoryToExportTo == null or
335                    // is not a directory, then
336                    // get the temporary directory from the WebServer.
337                    // See ExportHTMLAction.exportToWeb
338                    // FIXME: Following should not be necessary. The above super.fire() call should shown the tableau.
339                    _tableau.show();
340                    // FIXME: Do we have to wait for the frame to exist?
341                    BasicGraphFrame graphFrame = (BasicGraphFrame) _tableau
342                            .getFrame();
343                    if (!ExportHTMLAction.copyJavaScriptFilesIfNeeded(
344                            graphFrame, _parameters)) {
345                        // Canceled the export.
346                        _exception = new IllegalActionException(
347                                HTMLModelExporter.this, "Export canceled.");
348                        return;
349                    }
350
351                    // Create a writer to write to.
352                    StringWriter writer = new StringWriter();
353
354                    // Now do the export.
355                    try {
356                        // The null third argument ensure that there is no attempt to open
357                        // the exported file in a browser.
358                        ExportHTMLAction.openRunAndWriteHTML(graphFrame,
359                                _parameters, null, writer, true);
360                        writer.flush();
361                        webPage.send(0, new StringToken(writer.toString()));
362                    } catch (IllegalActionException e) {
363                        _exception = e;
364                    }
365                } else {
366                    _exception = new IllegalActionException(
367                            HTMLModelExporter.this, "No model to export.");
368                }
369            }
370        };
371        try {
372            // FIXME: No, this can't work. If the model runs and wants
373            // to perform GUI actions, such as generate plots, in other
374            // threads, such as PN threads, then those GUI actions will block.
375            // We need to run the model outside the GUI thread.
376            SwingUtilities.invokeAndWait(doExport);
377        } catch (Exception ex) {
378            throw new IllegalActionException(this, null, ex, "Open failed.");
379        }
380        if (_exception != null) {
381            throw _exception;
382        }
383    }
