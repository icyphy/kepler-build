029package org.ptolemy.ptango.lib.webSocket;
030
031import java.net.URI;
032
033import ptolemy.actor.TypedAtomicActor;
034import ptolemy.actor.TypedIOPort;
035import ptolemy.data.BooleanToken;
036import ptolemy.data.StringToken;
037import ptolemy.data.expr.Parameter;
038import ptolemy.data.expr.StringParameter;
039import ptolemy.data.type.BaseType;
040import ptolemy.kernel.CompositeEntity;
041import ptolemy.kernel.util.Attribute;
042import ptolemy.kernel.util.IllegalActionException;
043import ptolemy.kernel.util.NameDuplicationException;
044import ptolemy.kernel.util.Settable;
045import ptolemy.kernel.util.Workspace;
046
047///////////////////////////////////////////////////////////////////
048////WebSocketWriter
049
050/** An actor that writes information to a websocket.  Multiple writers to the
051 * same URL path are allowed.
052 *
053 *  @author Elizabeth Latronico
054 *  @version $Id$
055 *  @since Ptolemy II 11.0
056 *  @Pt.ProposedRating Red (ltrnc)
057 *  @Pt.AcceptedRating Red (ltrnc)
058 */
059
060public class WebSocketWriter extends TypedAtomicActor
061        implements WebSocketService {
062
063    /** Create an instance of the actor.
064     *  @param container The container
065     *  @param name The name.
066     *  @exception IllegalActionException If the superclass throws it.
067     *  @exception NameDuplicationException If the superclass throws it.
068     */
069    public WebSocketWriter(CompositeEntity container, String name)
070            throws IllegalActionException, NameDuplicationException {
071        super(container, name);
072
073        path = new StringParameter(this, "path");
074        path.setExpression("/*");
075
076        client = new Parameter(this, "client");
077        client.setToken("true");
078        // Separate menu items are available for clients and servlets.
079        // This parameter is set in the menu.
080        client.setVisibility(Settable.NOT_EDITABLE);
081
082        input = new TypedIOPort(this, "input", true, false);
083        input.setTypeEquals(BaseType.STRING);
084    }
085
086    ///////////////////////////////////////////////////////////////////
087    ////                     ports and parameters                  ////
088
089    /** A flag indicating if the reader acts as a client or as part of
090     * the server.  True for client; false for server.
091     */
092    public Parameter client;
093
094    /** A port that accepts messages to write to the websocket. */
095    public TypedIOPort input;
096
097    /** The URL affiliated with the websocket.  Can refer to a locally
098     * hosted websocket or a remotely hosted websocket.  Locally hosted
099     * websockets have paths of the form /* such as / or /mysocket
100     * or /mysocket/first   .  These will internally be translated to
101     * ws://localhost:port/path e.g. ws://localhost:8078/mysocket    .
102     * Paths for remotely hosted websockets should start with ws:// or wss://
103     */
104    public StringParameter path;
105
106    ///////////////////////////////////////////////////////////////////
107    ////                         public methods                    ////
108
109    /** React to a change in an attribute.  In this case, check the
110     *  value of the <i>path</i> attribute to make sure it is a valid URI.
111     *  @param attribute The attribute that changed.
112     *  @exception IllegalActionException If the change is not acceptable
113     *   to this container (not thrown in this base class).
114     */
115    @Override
116    public void attributeChanged(Attribute attribute)
117            throws IllegalActionException {
118        if (attribute == path) {
119            // Unsubscribe from previous path (if any)
120            if (_URIpath != null && !_URIpath.toString().isEmpty()
121                    && _endpointManager != null) {
122                // TODO: Shared vs. individual
123                _endpointManager.unsubscribe(this, _URIpath.toString());
124            }
125
126            // TODO:  Allow dynamic subscribing here
127            // If the endpoint exists, service should be added as a subscriber
128            // If the endpoint does not exists, a new one should be created
129            // and started ONLY if the model is currently executing
130
131            String pathValue = ((StringToken) path.getToken()).stringValue();
132            _URIpath = WebSocketEndpointManager.pathToURI(pathValue);
133        } else {
134            super.attributeChanged(attribute);
135        }
136    }
137
138    /** Clone the actor.
139     *  @param workspace The workspace in which to place the cloned attribute.
140     *  @return The cloned attribute.
141     *  @exception CloneNotSupportedException Not thrown in this base class.
142     *  @see java.lang.Object#clone()
143     */
144
145    @Override
146    public Object clone(Workspace workspace) throws CloneNotSupportedException {
147        WebSocketWriter newObject = (WebSocketWriter) super.clone(workspace);
148
149        newObject._endpoint = null;
150        newObject._endpointManager = null;
151        newObject._URIpath = null;
152        return newObject;
153    }
154
155    /** Subscribe this service to the appropriate endpoint.
156     *  @exception IllegalActionException If the parent throws it.
157     */
158    @Override
159    public void initialize() throws IllegalActionException {
160        super.initialize();
161
162        if (_endpointManager == null) {
163            _endpointManager = WebSocketEndpointManager.getInstance();
164        }
165
166        // Subscribe to this endpoint.  Creates a new endpoint if needed.
167        // Do not subscribe local clients.  The WebServer handles these, since
168        // it ensures that connections are only opened after it starts.
169        if (!(!WebSocketEndpointManager.isRemoteURI(_URIpath) && isClient())) {
170            _endpointManager.subscribe(this, _URIpath.toString());
171        }
172    }
173
174    /** Write a message to the websocket.
175     * @exception IllegalActionException If the websocket connection cannot be
176     * established or if the message cannot be written.
177     */
178
179    @Override
180    public void fire() throws IllegalActionException {
181        super.fire();
182
183        // Read token on input port, if any.  Write token contents to socket.
184        // Empty string tokens are acceptable
185
186        if (input.hasToken(0)) {
187
188            // Send message through endpoint.
189            // TODO:  Enhance to allow waiting on connection open, buffering,...
190            String message = ((StringToken) input.get(0)).stringValue();
191            if (_endpoint == null) {
192                throw new IllegalActionException(this,
193                        "Cannnot connect to " + "websocket");
194            }
195
196            if (!_endpoint.sendMessage(message)) {
197                throw new IllegalActionException(this,
198                        "Cannot write to " + "WebSocket");
199            }
200        }
201    }
202
203    /** Return the relative path that this WebSocketService is mapped to,
204     *  which is the value of the <i>path</i> parameter.
205     *  @return The relative path that this WebSocketService is mapped to.
206     */
207    @Override
208    public URI getRelativePath() {
209        return _URIpath;
210    }
211
212    /** Return true if this actor acts as a client; false if it acts as a
213     * part of the server (responding to incoming messages).
214     * @return True if this actor acts as a client; false if it acts as a
215     * part of the server (responding to incoming messages).
216     */
217    @Override
218    public boolean isClient() {
219        boolean isClient = true;
220        // Assume client side if no value given
221        try {
222            isClient = ((BooleanToken) client.getToken()).booleanValue();
223        } catch (IllegalActionException e) {
224        }
225        ;
226        return isClient;
227    }
228
229    /** Do nothing upon receipt of a message, since this actor is a writer.
230     * @param message The message that was received from the websocket.
231     */
232    @Override
233    public void onMessage(String message) {
234    }
235
236    /** Set the endpoint responsible for websocket communication.
237     * @param endpoint  The endpoint responsible for websocket communication.
238     */
239    @Override
240    public void setEndpoint(WebSocketEndpoint endpoint) {
241        _endpoint = endpoint;
242    }
243
244    /** Unsubscribe this service from the endpoint manager.  The endpoint
245     * manager will close connections with no subscribers.
246     * @exception IllegalActionException If thrown by the parent.
247     */
248    @Override
249    public void wrapup() throws IllegalActionException {
250        _endpointManager.unsubscribe(this, _URIpath.toString());
251    }
252
253    ///////////////////////////////////////////////////////////////////
254    ////                         private variables                 ////
255
256    /** The endpoint used for obtaining a connection to send messages through.*/
257    private WebSocketEndpoint _endpoint;
258
259    /** A manager responsible for generating and releasing websockets. */
260    private WebSocketEndpointManager _endpointManager;
261
262    /** The URI for the relative path from the "path" parameter.
263     *  A URI is used here to make sure the "path" parameter conforms to
264     *  all of the URI naming conventions.
265     */
266    private URI _URIpath;
267
268}
