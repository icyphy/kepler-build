030package org.ptolemy.ptango.lib.webSocket;
031
032import java.io.IOException;
033import java.util.HashSet;
034
035import org.eclipse.jetty.websocket.WebSocket.OnTextMessage;
036
037///////////////////////////////////////////////////////////////////
038//// WebSocketEndpoint
039
040/** A class that keeps a reference to the WebSocket connection, notifies
041 *  services of incoming messages and allows services to write to the WebSocket.
042 *
043 *  Uses Jetty 8 WebSocket classes.  Note that Jetty 9 has substantial WebSocket
044 *  revisions that are not backward compatible.
045 *
046 *  Based on http://java.dzone.com/articles/creating-websocket-chat
047 *
048 *  @author Elizabeth Latronico
049 *  @version $Id$
050 *  @since Ptolemy II 10.0
051 *  @Pt.ProposedRating Red (ltrnc)
052 *  @Pt.AcceptedRating Red (ltrnc)
053 */
054public class WebSocketEndpoint implements OnTextMessage {
055
056    /** Create a new WebSocketEndpoint.
057    */
058    public WebSocketEndpoint() {
059        _connection = null;
060        _isOpen = false;
061        _subscribers = new HashSet();
062    }
063
064    /** Create a new WebSocketEndpoint with the given parent service.
065     * @param service  The WebSocketService to be notified of messages.
066     */
067    public WebSocketEndpoint(WebSocketService service) {
068        _connection = null;
069        _isOpen = false;
070        _subscribers = new HashSet();
071        _subscribers.add(service);
072    }
073
074    /** Add a subscriber to this endpoint.
075     * @param service  A parent WebSocketService of this endpoint.
076     * @see #removeSubscriber(WebSocketService)
077     */
078    public void addSubscriber(WebSocketService service) {
079        _subscribers.add(service);
080    }
081
082    /** Close the connection. _connection will be assigned to null in onClose(),
083     * which is invoked automatically after a close().
084     */
085    public void close() {
086        if (_connection != null) {
087            _connection.close();
088        }
089    }
090
091    /** Return the number of subscribers this endpoint has.
092     * @return The number of subscribers this endpoint has.
093     */
094    public int getSubscriberCount() {
095        return _subscribers.size();
096    }
097
098    /** Return true if the connection is open; false otherwise.
099     * @return True if the connection is open; false otherwise.
100     */
101    public boolean isOpen() {
102        return _isOpen;
103    }
104
105    /** Upon close, set the connection to null.  Note that the connection might
106     *  be closed unexpectedly and close() is not necessarily called.
107     *  Synchronized so that onOpen() and onClose() can't be interleaved.
108     *  @param statusCode The status code of the closed connection.
109     *  @param statusMessage The status message of the closed connection.
110     *  @see #onClose(int, String)
111     */
112    @Override
113    public synchronized void onClose(int statusCode, String statusMessage) {
114        _connection = null;
115        _isOpen = false;
116    }
117
118    /** Notify subscribers about a new message.
119     * @param message The message that was received.
120     */
121    @Override
122    public void onMessage(String message) {
123        if (_subscribers != null) {
124            for (WebSocketService service : _subscribers) {
125                service.onMessage(message);
126            }
127        }
128    }
129
130    /** Upon opening, save a reference to the connection. Synchronized so that
131     * onOpen() and onClose() can't be interleaved.
132     * @param connection The connection that was opened.
133     * @see #onOpen(Connection)
134     */
135    @Override
136    public synchronized void onOpen(Connection connection) {
137        _connection = connection;
138        _isOpen = true;
139    }
140
141    /** Remove the given subscriber for this endpoint.
142     * @param service  A parent WebSocketService of this endpoint.
143     * @see #addSubscriber(WebSocketService)
144     */
145    public void removeSubscriber(WebSocketService service) {
146        _subscribers.remove(service);
147    }
148
149    /** Send the given message.  If the connection is not open, return false.
150     *  If the connection is not open, there are multiple potential options:
151     *  the sender could block, messages could be buffered, messages could be
152     *  discarded, or some combination.  Currently, messages are discarded.
153     *  Synchronized so that only one sender can write to the websocket at a
154     *  time.  Returns true if the message was successfully sent; false
155     *  otherwise.
156     *  @param message The message to send.
157     *  @return True if the message was successfully sent; false otherwise.
158     */
159    // TODO:  Could enhance by trying to open connection if closed.
160    // TODO:  Enhance this to handle waiting on connection, buffering, etc.
161    // SENT
162    // BUFFERED
163    // WAITING (on connection)
164    // FAILED
165    public synchronized boolean sendMessage(String message) {
166        if (_isOpen && _connection != null) {
167            try {
168                _connection.sendMessage(message);
169            } catch (IOException e) {
170                return false;
171            }
172        } else {
173            return false;
174        }
175        return true;
176    }
177
178    // TODO:  Do something onError?
179
180    ///////////////////////////////////////////////////////////////////
181    ////                         private variables                 ////
182
183    /** The WebSocket connection */
184    private Connection _connection;
185
186    /** An exception that might be thrown when connecting.  */
187    // TODO:  Do something with this exception?
188    Exception _connectionException;
189
190    /** Flag indicating if the connection is open. */
191    private boolean _isOpen;
192
193    /** The parent services to be notified of incoming messages. */
194    private HashSet<WebSocketService> _subscribers;
195}
