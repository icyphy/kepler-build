042/** A singleton class that manages websocket endpoints and their connection
043* openers (clients, servlets).
044*
045*  @author Elizabeth Latronico
046*  @version $Id$
047*  @since Ptolemy II 11.0
048*  @Pt.ProposedRating Red (ltrnc)
049*  @Pt.AcceptedRating Red (ltrnc)
050*/
051
052public class WebSocketEndpointManager {
053
054    ///////////////////////////////////////////////////////////////////
055    ////                         public methods                    ////
056
057    /** Get the factory instance, creating a new one if none created yet.
058    * @return The websocket client factory instance.
059    */
060    public static WebSocketEndpointManager getInstance() {
061        return WebSocketEndpointFactoryHolder.INSTANCE;
062    }
063
064    /** Create a new WebSocketConnectionFactory.  The private
065    *  constructor prevents instantiation from other classes.
066    */
067    private WebSocketEndpointManager() {
068
069        _webSocketClientFactory = new WebSocketClientFactory();
070        _pathToServlet = new Hashtable();
071        _pathToClientEndpoint = new Hashtable();
072        _pathToServletEndpoint = new Hashtable();
073
074        try {
075            _webSocketClientFactory.start();
076            _webSocketClient = _webSocketClientFactory.newWebSocketClient();
077        } catch (Exception e) {
078            // If an exception occurs, _webSocketClient will remain null.
079            // This is checked for later.
080        }
081    }
082
083    /** The Holder is loaded on the first execution of getInstance()
084    * or the first access to INSTANCE, allowing on-demand loading since
085    * not all models use websockets.
086    */
087    private static class WebSocketEndpointFactoryHolder {
088        private static final WebSocketEndpointManager INSTANCE = new WebSocketEndpointManager();
089    }
090
091    /** Close all servlet websocket endpoints.  Called by webserver in wrapup().
092    * Synchronized to avoid interleaving opening and closing endpoints.
093    */
094    public synchronized void closeServlets() {
095        for (String path : _pathToServletEndpoint.keySet()) {
096            if (_pathToServletEndpoint.get(path).isOpen()) {
097                _pathToServletEndpoint.get(path).close();
098            }
099            _pathToServletEndpoint.remove(path);
100            _pathToServlet.remove(path);
101        }
102    }
103
104    /** Get the servlet for the given path.  If the servlet does not exist,
105    *  create it.  Called by the web server.  Synchronized to avoid interleaving
106    *  creating and deleting a servlet.
107    * @param path  The URI the servlet should be mapped to.
108    * @return The servlet mapped to this path.
109    * @exception IllegalActionException Not thrown in this base class.
110    */
111    public synchronized PtolemyWebSocketServlet getServlet(String path)
112            throws IllegalActionException {
113        if (_pathToServlet.containsKey(path)) {
114            return _pathToServlet.get(path);
115        } else {
116
117            WebSocketEndpoint endpoint;
118
119            // If we already have an endpoint, retrieve it
120            if (_pathToServletEndpoint.containsKey(path)) {
121                endpoint = _pathToServletEndpoint.get(path);
122            } else {
123                endpoint = new WebSocketEndpoint();
124                _pathToServletEndpoint.put(path, endpoint);
125            }
126
127            PtolemyWebSocketServlet servlet = new PtolemyWebSocketServlet(
128                    endpoint);
129            _pathToServlet.put(path, servlet);
130            return servlet;
131        }
132    }
133
134    /** Return true if the path refers to a remote resource; false otherwise.
135    * Throw an exception if the path is not a valid URI.
136    * @param path  The candidate URI.
137    * @return True if the path refers to a remote resource; false otherwise.
138    * @exception IllegalActionException If the path is not a valid URI.
139    */
140    public static boolean isRemoteURI(String path)
141            throws IllegalActionException {
142        String uri = pathToURI(path).toString();
143        if (uri.startsWith("ws://") || uri.startsWith("wss://")) {
144            return true;
145        }
146        return false;
147    }
148
149    /** Return true if the path refers to a remote resource; false otherwise.
150    * Throw an exception if the path is not a valid URI.
151    * @param uri The candidate URI.
152    * @return True if the path refers to a remote resource; false otherwise.
153    */
154    public static boolean isRemoteURI(URI uri) {
155        if (uri.toString().startsWith("ws://")
156                || uri.toString().startsWith("wss://")) {
157            return true;
158        }
159        return false;
160    }
161
162    /** Check if the given path is a valid URI.  Return true if so; false
163    * otherwise.
164    * @param path  The candidate URI.
165    * @return True if the path is a valid URI; false otherwise.
166    */
167    public static boolean isValidURI(String path) {
168        try {
169            pathToURI(path);
170        } catch (IllegalActionException e) {
171            return false;
172        }
173        return true;
174    }
175
176    /** Open connections for the given set of services by looking up each
177    * service's affiliated endpoint and opening a connection for that endpoint.
178    * Assumes that these are local services.  The WebServer uses this to open
179    * local services after it has started.  Synchronized so that subscribing and
180    * unsubscribing are not interleaved.
181    * @param services  The set of services to open connections for.
182    * @param portNumber The port number of the local web server.
183    * @exception IllegalActionException If one or more services cannot be opened.
184    */
185    public synchronized void openLocalServices(
186            HashSet<WebSocketService> services, int portNumber)
187            throws IllegalActionException {
188        for (WebSocketService service : services) {
189            String path = service.getRelativePath().toString();
190
191            // Subscribe this service.  This creates a new endpoint if needed.
192            // Called by the WebServer.  Needed as initialize() in WebServer
193            // might be called before initialize() in the reader / writer actors.
194            subscribe(service, path);
195
196            WebSocketEndpoint endpoint;
197            if (service.isClient()) {
198                endpoint = _pathToClientEndpoint.get(path);
199            } else {
200                endpoint = _pathToServletEndpoint.get(path);
201            }
202
203            URI connectPath = URI.create("ws://localhost:" + portNumber
204                    + service.getRelativePath().toString());
205
206            _open(endpoint, connectPath);
207        }
208    }
209
210    /** Check the given path to see if it is a valid websocket URI and return a
211    *  URI.  Throw an exception if the path is invalid.
212    * @param path The string representation of a websocket URI.
213    * @return The URI corresponding to this path string.
214    * @exception IllegalActionException If the path is not a valid URI.
215    */
216    public static URI pathToURI(String path) throws IllegalActionException {
217
218        try {
219            // Paths connecting to remote websockets should start with ws://
220            // or wss:// (for secure websockets)
221            // Paths not starting with these are assumed to be local
222            // For locally hosted websockets should start with a "/"
223            // or be "*"
224            if (!path.trim().equals("")) {
225                // Check for common incorrect protocols
226                if (path.startsWith("http") || path.startsWith("ftp")) {
227                    throw new IllegalActionException("Remote websocket"
228                            + " paths must start with ws:// or wss://");
229                }
230
231                if (path.startsWith("ws://") || path.startsWith("wss://")) {
232                    return URI.create(path);
233                } else if (!path.trim().startsWith("/")) {
234                    return URI.create("/" + path);
235                } else {
236                    return URI.create(path);
237                }
238            } else {
239                return URI.create("/*");
240            }
241        } catch (IllegalArgumentException e2) {
242            throw new IllegalActionException(
243                    "Path is not a valid URI: " + path);
244        }
245    }
246
247    /** Add the given service as a subscriber to the endpoint for this path.
248    * Create a new endpoint if none exists for this path.  Synchronized to avoid
249    * interleaved subscribing and unsubscribing.
250    * Generally, subscribers should subscribe in initialize() and should
251    * unsubscribe in wrapup().
252    * @param service The service to add as a subscriber.
253    * @param path  The URL to subscribe to.
254    * @exception IllegalActionException If the URI is not valid.
255    * @see #unsubscribe(WebSocketService, String)
256    */
257    // TODO:  Allow non-shared websockets.
258    public synchronized void subscribe(WebSocketService service, String path)
259            throws IllegalActionException {
260
261        // Check for existing endpoint
262        WebSocketEndpoint endpoint;
263
264        if (service.isClient()) {
265            if (_pathToClientEndpoint.containsKey(path)) {
266                endpoint = _pathToClientEndpoint.get(path);
267            } else {
268
269                // If no existing endpoint, create a new endpoint
270                // First, check that the URI is valid
271                if (!isValidURI(path)) {
272                    throw new IllegalActionException(
273                            "Path is not a valid URI: " + path);
274                }
275
276                endpoint = new WebSocketEndpoint();
277                _pathToClientEndpoint.put(path, endpoint);
278
279                // Open any new remote endpoints.
280                // Local client-side endpoints will be opened by the webserver.
281                // If a server-side endpoint, create a servlet.  This servlet
282                // will automatically open the socket when an incoming request is
283                // received.
284                if (isRemoteURI(path)) {
285                    _open(endpoint, pathToURI(path));
286                }
287            }
288        } else {
289
290            if (_pathToServletEndpoint.containsKey(path)) {
291                endpoint = _pathToServletEndpoint.get(path);
292            } else {
293                // If no existing endpoint, create a new endpoint
294                // First, check that the URI is valid
295                if (!isValidURI(path)) {
296                    throw new IllegalActionException(
297                            "Path is not a valid URI: " + path);
298                }
299
300                endpoint = new WebSocketEndpoint();
301                _pathToServletEndpoint.put(path, endpoint);
302
303                // Create a new servlet for server-side endpoints
304                // _pathToServlet should not contain an endpoint if
305                // _pathToEndpoint did not contain the endpoint
306                _pathToServlet.put(path, new PtolemyWebSocketServlet(endpoint));
307            }
308        }
309
310        endpoint.addSubscriber(service);
311        service.setEndpoint(endpoint);
312    }
313
314    /** Remove the given service as a subscriber to the endpoint for this path.
315    * Delete the endpoint if no services are associated with it. Synchronized to
316    * avoid conflicting writes to tables.
317    * @param service  The service to remove as a subscriber
318    * @param path  The URL to subscribe to
319    * @see #subscribe(WebSocketService, String)
320    */
321    // TODO:  Allow non-shared websockets.
322    public synchronized void unsubscribe(WebSocketService service,
323            String path) {
324        WebSocketEndpoint endpoint;
325
326        if (_pathToClientEndpoint.containsKey(path)) {
327            endpoint = _pathToClientEndpoint.get(path);
328            endpoint.removeSubscriber(service);
329            service.setEndpoint(null);
330
331            if (endpoint.getSubscriberCount() <= 0) {
332                endpoint.close();
333                _pathToClientEndpoint.remove(path);
334            }
335        } else if (_pathToServletEndpoint.containsKey(path)) {
336            endpoint = _pathToServletEndpoint.get(path);
337            endpoint.removeSubscriber(service);
338            service.setEndpoint(null);
339
340            // Close connection if no subscribers left
341            if (endpoint.getSubscriberCount() <= 0) {
342                endpoint.close();
343                _pathToServletEndpoint.remove(path);
344            }
345        }
346
347        // The webserver closes server-side connections on wrapup
348        // Server-side connections also close automatically if the client closes
349    }
350
351    ///////////////////////////////////////////////////////////////////
352    ////                         private methods                   ////
353
354    /** Open a websocket connection for the given endpoint.
355    * @param endpoint  The endpoint to open a connection for.
356    * @param uri  The URI to connect to.
357    * @exception IllegalActionException If the connection cannot be opened.
358    */
359    private void _open(WebSocketEndpoint endpoint, URI uri)
360            throws IllegalActionException {
361        if (!endpoint.isOpen()) {
362
363            // .open() will call .onOpen() on the endpoint.
364            // TODO:  Add timeout here?
365            try {
366                _webSocketClient.open(uri, endpoint);
367            } catch (IOException e) {
368                throw new IllegalActionException(
369                        "Cannot open " + "websocket to path " + uri.toString());
370            }
371        }
372    }
