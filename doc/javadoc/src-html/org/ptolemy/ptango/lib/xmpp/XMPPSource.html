051//// XMPPSource

053/** An actor that subscribes to an XMPP XEP-0060 node and fires upon an event.
054 *  An XMPP node is created by a user, where data can be published and subscribed.
055 *  An XMPP node has a <i>nodeId</i>, which is set at creation time.
056 *  This actor refers XMPP node via its <i>nodeId</i>.
057 *  FIXME: comments, explain XEP-0060
058 *  @see XMPPGateway
059 *  @author Marten Lohstroh
060 *  @version $Id$
061 *  @since Ptolemy II 10.0
062 *  @Pt.ProposedRating Red (marten)
063 *  @Pt.AcceptedRating Red (marten)
064 */
065public class XMPPSource extends TypedAtomicActor implements XMPPSubscriber {
066    /** Construct an actor with the given container and name.
067     *  @param container The container.
068     *  @param name The name of this actor.
069     *  @exception IllegalActionException If the actor cannot be contained
070     *   by the proposed container.
071     *  @exception NameDuplicationException If the container already has an
072     *   actor with this name.
073     */
074    public XMPPSource(CompositeEntity container, String name)
075            throws NameDuplicationException, IllegalActionException {
076        super(container, name);
077        nodeId = new StringParameter(this, "nodeId");
078
079        output = new TypedIOPort(this, "output", false, true);
080        output.setTypeEquals(BaseType.STRING);
081        new Parameter(output, "_showName").setExpression("true");
082
083        // Set flag indicating that initialize() has not yet been called.
084        // Must be set prior to initialize() being called, so we set it in
085        // the constructor.
086        _hasInitialized = false;
087    }
088
089    ///////////////////////////////////////////////////////////////////
090    ////                     ports and parameters                  ////
091
092    /** An output port for values received from the XMPP node.
093     */
094    public TypedIOPort output;
095
096    ///////////////////////////////////////////////////////////////////
097    ////                         public variables                  ////
098
099    /** XMPPSource subscribes to an XMPP node, the name of which is
100     *  specified by the value of nodeId.
101     *  The nodeId is set when the XMPP node is created by a user.
102     *  The default value is empty string.
103     */
104    public StringParameter nodeId;
105
106    ///////////////////////////////////////////////////////////////////
107    ////                         public methods                    ////
108
109    /** If the argument is the <i>nodeId</i> parameter, then reset the
110     *  state to the specified value.
111     *  @param attribute The attribute that changed.
112     *  @exception IllegalActionException If <i>nodeId</i> cannot be evaluated
113     *   or cannot be converted to the output type, or if the superclass
114     *   throws it.
115     */
116    @Override
117    public void attributeChanged(Attribute attribute)
118            throws IllegalActionException {
119        if (attribute == nodeId) {
120            _nodeId = ((StringToken) nodeId.getToken()).stringValue();
121        } else {
122            super.attributeChanged(attribute);
123        }
124    }
125
126    /** Clone the actor into the specified workspace.
127     *  @param workspace The workspace for the new object.
128     *  @return A new actor.
129     *  @exception CloneNotSupportedException If a derived class contains
130     *   an attribute that cannot be cloned.
131     */
132    @Override
133    public Object clone(Workspace workspace) throws CloneNotSupportedException {
134        XMPPSource newObject = (XMPPSource) super.clone(workspace);
135        newObject.lock = new Object();
136        return newObject;
137    }
138
139    /** Record the current model time and the current real time
140     *  so that output events can be time stamped with the elapsed
141     *  time since model start.
142     *
143     *  Declare as synchronized so the handlePublishedItems() method can check
144     *  if initialize() has been called and, if not, wait for initialize() to
145     *  be called.
146     *  @exception IllegalActionException If the superclass throws it.
147     */
148    @Override
149    public synchronized void initialize() throws IllegalActionException {
150        super.initialize();
151        _hasFired = false;
152        _initializeModelTime = getDirector().getModelTime();
153        _initializeRealTime = System.currentTimeMillis();
154        _hasInitialized = true;
155        _hasFired = false;
156        this.notify();
157    }
158
159    /** Send the current value of the state of this actor to the output.
160     *  Synchronized on a lock Object so that each data value will be handled
161     *  before the next one is accepted, since incoming values call the
162     *  handlePublishedItems() method from a different thread.
163     *
164     *  @exception IllegalActionException If calling send() or super.fire()
165     *  throws it.
166     */
167    @Override
168    public void fire() throws IllegalActionException {
169        synchronized (lock) {
170            super.fire();
171            output.send(0, new StringToken(_currentValue));
172
173            // Notify the _handlePublishedItems() method that the actor has
174            // fired, so that method can continue executing
175            _hasFired = true;
176            lock.notify();
177        }
178    }
179
180    /** Return the nodeId.
181     *  @return the nodeId.
182     */
183    @Override
184    public String getNodeId() {
185        return _nodeId;
186    }
187
188    /** Return the id of this subscriber.
189     * @return the id of this subscriber.
190     * @see #setSubId(String)
191     */
192    @Override
193    public String getSubId() {
194        return _subscriptionId;
195    }
196
197    /** Parse the published item from the XMPP node, save the XML and
198     *  request that the actor be fired.
199     *  This method is declared in the smack ItemEventListener interface.
200     *  This method is called when an item is published.
201     *
202     *  NOTE: This method is synchronized, and the lock is _not_ released
203     *  until the method is finished processing the current item event.  The
204     *  first item event is completely processed before a second item event
205     *  can be started.
206     *
207     *  @param items  The item event from the XMPP node.  Can contain
208     *  multiple pieces of information.
209     */
210    @Override
211    public synchronized void handlePublishedItems(
212            ItemPublishEvent<Item> items) {
213        // Check if initialize() has been called yet.  If not, wait, releasing
214        // the lock so that initialize may be called.  Further invocations of
215        // handlePublishedItems might also be called, but these will also wait
216        // for initialize().
217        while (!_hasInitialized) {
218            try {
219                this.wait(0);
220            } catch (InterruptedException e) {
221                // FIXME:  Do anything special if thread is interrupted?
222                break;
223            }
224        }
225
226        _handlePublishedItems(items);
227    }
228
229    /** Set _hasInitialized to false, so that invocations of
230     *  handlePublishedItems (from a different thread) will wait until the model
231     *  has been initialized on the next execution.
232     */
233    @Override
234    public void wrapup() throws IllegalActionException {
235
236        synchronized (lock) {
237            super.wrapup();
238            _hasInitialized = false;
239
240            // Notify the handlePublishedItems() method to stop waiting to be
241            // fired, since this actor will not be fire again in this iteration
242            _hasFired = true;
243            lock.notify();
244        }
245    }
246
247    ///////////////////////////////////////////////////////////////////
248    ////                         private methods                   ////
249
250    /** Parse the published item from the XMPP node, save the XML and
251     *  request that the actor be fired.  Synchronized on a lock object so
252     *  this method will release the lock on the lock object so the fire()
253     *  method may execute, but retain the lock on XMPPSource.this object
254     *  so that future invocations of handlePublishedItems() must block until
255     *  the current invocation is finished.
256     *
257     *  @param items  The item event from the XMPP node.  Can contain
258     *  multiple pieces of information.
259     */
260    private void _handlePublishedItems(ItemPublishEvent<Item> items) {
261        // The following codeblock is synchronized on the enclosing
262        // actor. This lock _is_ released while waiting for the response,
263        // allowing the fire method to execute its own synchronized blocks.
264        synchronized (lock) {
265            Object o;
266            PayloadItem<?> item = (PayloadItem<?>) items.getItems().get(0);
267
268            _hasFired = false;
269
270            if ((o = item.getPayload()) instanceof SimplePayload) {
271                SimplePayload payload = (SimplePayload) o;
272                _currentValue = payload.toXML();
273            } else {
274                return; // published item has no recognized pay load
275            }
276
277            try {
278                long elapsedRealTime = System.currentTimeMillis()
279                        - _initializeRealTime;
280                Time timeOfRequest = _initializeModelTime.add(elapsedRealTime);
281                // Note that fireAt() will modify the requested firing time if it is in the past.
282                getDirector().fireAt(XMPPSource.this, timeOfRequest);
283            } catch (IllegalActionException e) {
284                throw new InternalErrorException(this, e,
285                        "Failed to fire at the current time.");
286            }
287
288            // Wait until this actor has been fired (thereby producing a token
289            // with the received value on its output port) before this method
290            // can be called again to receive a new value.
291
292            // Note that other implementations are possible.  For example,
293            // instead of blocking the caller until each value is handled,
294            // this actor could implement a bounded queue and accepted up
295            // to n values before blocking or before discarding values.
296
297            // Note that the fire method purposefully does not block waiting for
298            // a new value, so the actor might be fired multiple times with the
299            // current value, for example if a token is received on the trigger
300            // input port.  This ensures that XMPPSource will not block the
301            // execution of the rest of the model - the DEDirector maintains
302            // authority over execution.
303
304            // wrapup() may be called before this actor is fired, meaning this
305            // actor will not be fired.  In this situation, wrapup() sets
306            // _hasFired to true and calls lock.notify() to wake up this thread
307            // and allow the thread to finish
308            while (!_hasFired) {
309                try {
310                    lock.wait(0);
311                } catch (InterruptedException e) {
312                    // FIXME:  Do anything special if thread is interrupted?
313                    break;
314                }
315            }
316
317        }
318    }

333    @Override
334    public void setSubId(String subId) {
335        // XEP 0060 uses the term subId, not subscriberId.
336        _subscriptionId = subId;
337    }

339    ///////////////////////////////////////////////////////////////////
340    ////                         private variables                 ////
341
342    /** The XML payload from XMPP node. */
343    private String _currentValue;
344
345    /** A flag indicating if the actor has fired after a new value was received.
346     */
347    private boolean _hasFired;
348
349    /** A flag indicating if initialize() was called.  handlePublishedItems()
350     *  will wait if initialize() has not been called.
351     */
352    private boolean _hasInitialized = false;
353
354    /** The model time at which this actor was last initialized. */
355    private Time _initializeModelTime;
356
357    /** The real time at which this actor was last initialized, in milliseconds. */
358    private long _initializeRealTime;
359
360    /** An object for the fire() and private _handlePublishedItems() methods to
361     * use as a lock.  We avoid using XMPPSource as the lock object here since
362     * the public handlePublishedItems uses XMPPSource as the lock object to
363     * ensure one method call is completed before the next, and if we use the
364     * same lock object for the private _handlePublishedItems method, this lock
365     * will be released when wait() is called.  The actual value of this
366     * variable is not used.
367     */
368    private Object lock = new Object();
369
370    /** The cached value of nodeId parameter.
371     *  @see #nodeId
372     */
373    private String _nodeId;
374
375    /** The id of the subscriber. */
376    private String _subscriptionId;
