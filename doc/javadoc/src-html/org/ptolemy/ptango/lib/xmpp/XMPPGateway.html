071/**
072 *  This attribute sets up and maintains a connection to a XMPP server. It
073 *  keeps track of actors within the model that implement the XMPPSubscriber
074 *  or XMPPPublisher interface and relays messages between those actors
075 *  and the XMPP server.
076 *
077 *  The Executable interface is implemented merely to make sure that the
078 *  connection maintained to the server is alive every iteration, before
079 *  any firings occur in which actors may attempt to publish something
080 *  by invoking the <code>XMPP.publish()</code> method.
081 *
082 *  This implementation makes use of the
083 *  <a href="http://www.igniterealtime.org/projects/smack/">Smack</a>
084 *  library.
085 *
086 *  @see XMPPSource
087 *  @see XMPPSink
088 *  @see Executable
089 *  @author Marten Lohstroh and Ben Zhang
090 *  @version $Id$
091 *  @since Ptolemy II 10.0
092 *  @Pt.ProposedRating Red (marten)
093 *  @Pt.AcceptedRating Red (marten)
094 */
095public class XMPPGateway extends AbstractInitializableAttribute
096        implements Executable {
097
098    /** Construct an instance of the XMPPGateway attribute.
099     *  @param container The container.
100     *  @param name The name.
101     *  @exception IllegalActionException If the superclass throws it.
102     *  @exception NameDuplicationException If the superclass throws it.
103     */
104    public XMPPGateway(NamedObj container, String name)
105            throws IllegalActionException, NameDuplicationException {
106        super(container, name);
107
108        server = new StringParameter(this, "server");
109        server.setExpression("localhost");
110
111        port = new Parameter(this, "port");
112        port.setTypeEquals(BaseType.INT);
113        port.setExpression("5222");
114
115        userName = new StringParameter(this, "userName");
116        userName.setExpression("ptolemy");
117
118        passwordFile = new FileParameter(this, "passwordFile");
119        passwordFile.setExpression("");
120
121        // Set the Debug Flag.
122        debugEnable = new Parameter(this, "debugEnable", BooleanToken.FALSE);
123        debugEnable.setTypeEquals(BaseType.BOOLEAN);
124    }

164    /** Update the local variable associated with the changed attribute
165     *  and disconnect from the server.
166     *  @param attribute The changed attribute.
167     *  @exception IllegalActionException If thrown while getting the
168     *  value of the port, server or userName token, or if thrown
169     *  by the method in the super class.
170     */
171    @Override
172    public void attributeChanged(Attribute attribute)
173            throws IllegalActionException {
174
175        if (attribute == port) {
176            _portNumber = ((IntToken) port.getToken()).intValue();
177            _disconnect();
178
179        } else if (attribute == server) {
180            _serverName = ((StringToken) server.getToken()).stringValue();
181            _disconnect();
182        } else if (attribute == userName) {
183            _userName = ((StringToken) userName.getToken()).stringValue();
184            _disconnect();
185        } else {
186            super.attributeChanged(attribute);
187        }
188    }

197    /** Attempt to connect to the server and login. Discover subscribers
198     *  and them as listeners. If a subscriber wants to subscribe to a
199     *  non-existent node, create it.
200     *  discover publishers and give them a reference to this attribute.
201     *  This might change in the future, as changes to running models require
202     *  actors to register themselves instead of being discovered.
203     *  @exception IllegalActionException If unable to login, create a node,
204     *  find a node, or subscribe to a node.
205     *
206     */
207    @Override
208    public void initialize() throws IllegalActionException {
209        // Invoke any registered initializables in the super class.
210        super.initialize();
211
212        // FIXME: if the server runs on localhost, it doesn't accept 'localhost',
213        // but requires 127.0.0.1, look into this
214        String jid = _userName + '@' + _serverName;
215
216        boolean debugEnableValue = ((BooleanToken) debugEnable.getToken())
217                .booleanValue();
218
219        System.setProperty("smack.debugEnabled",
220                String.valueOf(debugEnableValue));
221        Connection.DEBUG_ENABLED = debugEnableValue;
222
223        _connectAndLogin();
224
225        _manager = new PubSubManager(_connection);
226        // discover XMPPSubscribers FIXME: how deep is this search?
227        Iterator<?> objects = toplevel().containedObjectsIterator();
228        while (objects.hasNext()) {
229            Object object = objects.next();
230            // XMPPSubscriber found
231            if (object instanceof XMPPSubscriber) {
232                LeafNode node;
233                XMPPSubscriber subscriber = (XMPPSubscriber) object;
234                String nodeId = subscriber.getNodeId();
235                try {
236                    //_manager.deleteNode(nodeId);
237                    node = (LeafNode) _manager.getNode(nodeId); // FIXME: cast problem here
238                    // FIXME: get rid of duplicate subscriptions here
239                } catch (Exception e) {
240                    try {
241                        /* ConfigureForm form = new ConfigureForm(FormType.submit);
242                         // FIXME: figure out configuration options
243                         form.setAccessModel(AccessModel.open);
244                         form.setDeliverPayloads(false);
245                         form.setNotifyRetract(true);
246                         form.setPersistentItems(true);
247                         form.setPublishModel(PublishModel.open); */
248                        node = _manager.createNode(nodeId);
249
250                    } catch (XMPPException ex) {
251                        throw new IllegalActionException(this, ex,
252                                "Unable find or create node: " + nodeId + ".");
253                    }
254                }
255                if (node != null) {
256                    // register listener
257                    node.addItemEventListener(subscriber);
258                    try {
259                        Subscription subscription = null;
260                        for (Subscription s : node.getSubscriptions()) {
261                            if (s.getJid().equals(jid)) {
262                                // check if state is OK
263                                if (s.getState().equals(State.subscribed)) {
264                                    subscription = s;
265                                    break;
266                                } else {
267                                    node.unsubscribe(jid, s.getId());
268                                }
269                            }
270                        }
271                        // subscribe to node if not exists
272                        if (subscription == null) {
273                            subscription = node.subscribe(jid); // FIXME: subscription fails here
274                        }
275                        // FIXME: check subscription.state here too
276                    } catch (XMPPException ex) {
277                        throw new IllegalActionException(this, ex,
278                                "Unable subscribe to node: " + nodeId + ".");
279                    }
280                } else {
281                    throw new IllegalActionException(this,
282                            "Unable find or create node: " + nodeId + ".");
283                }
284            } else if (object instanceof XMPPPublisher) {
285                ((XMPPPublisher) object).setGateway(this);
286            }
287        }
288
289    }

309    /** Check the connection, reconnect and login if required. */
310    @Override
311    public boolean prefire() throws IllegalActionException {
312        if (_connection == null) {
313            initialize();
314        } else if (!_connection.isConnected()
315                || !_connection.isAuthenticated()) {
316            _connectAndLogin();
317        }
318        return true;
319    }

327    /** Publish a value to a node. The value is wrapped into a message stanza.
328     *  @param nodeId The node to publish to.
329     *  @param value The value to publish.
330     *  @exception IllegalActionException If publishing failed.
331     */
332    public void publish(String nodeId, String value)
333            throws IllegalActionException {
334
335        Node n;
336        LeafNode ln;
337
338        try {
339            if ((n = _manager.getNode(nodeId)) instanceof LeafNode) {
340                ln = (LeafNode) n;
341            } else {
342                throw new IllegalActionException(this,
343                        "Unable to publish a node that is not a leaf.");
344            }
345        } catch (XMPPException e) {
346            try {
347                ln = _manager.createNode(nodeId);
348            } catch (XMPPException e1) {
349                throw new IllegalActionException(this,
350                        "Unable to create node with id: " + nodeId + ".");
351            }
352        }
353
354        SimplePayload payload = new SimplePayload("message", "null",
355                "<message>" + value + "</message>");
356        PayloadItem<SimplePayload> item = new PayloadItem<SimplePayload>(null,
357                payload);
358        ln.publish(item);
359    }

405    /** Disconnect and leave old connection for the garbage collector.
406     *  Note that once disconnected, a Connection cannot be
407     *  reused immediately, therefore it is discarded. A new Connection
408     *  object shall be instantiated for reconnecting.
409     *  @see Connection
410     *  @see #_connectAndLogin()
411     */
412    private void _disconnect() {
413        if (_connection != null) {
414            _connection.disconnect();
415        }
416        _connection = null;
417    }
418
419    /** Connect to the server and login using the provided credentials.
420     *  @exception IllegalActionException If connection or authentication
421     *  process fails.
422     */
423    private void _connectAndLogin() throws IllegalActionException {
424
425        // already connected
426        if (_connection != null && _connection.isConnected()
427                && _connection.isAuthenticated()) {
428            return;
429        }
430
431        // no active connection, establish one
432        if (_connection == null) {
433            ConnectionConfiguration config = new ConnectionConfiguration(
434                    _serverName, _portNumber);
435            config.setSecurityMode(
436                    ConnectionConfiguration.SecurityMode.enabled);
437            config.setCompressionEnabled(true);
438            config.setSASLAuthenticationEnabled(true);
439            _connection = new XMPPConnection(config);
440        }
441        try {
442            // connect to the server
443            if (!_connection.isConnected()) {
444                _connection.connect();
445            }
446        } catch (Exception ex) {
447            throw new IllegalActionException(this, ex,
448                    "Unable to connect to XMPP server.");
449        }
450
451        try {
452            // login to the server
453            if (!_connection.isAuthenticated()) {
454                if (_password == null) {
455                    if (passwordFile.stringValue().length() > 0) {
456                        // Read the password from a file
457                        BufferedReader reader = null;
458                        try {
459                            reader = passwordFile.openForReading();
460                            String line = reader.readLine();
461                            if (line != null) {
462                                _password = line.toCharArray();
463                                line = "";
464                            } else {
465                                throw new IllegalActionException(this,
466                                        "Failed to read a line from "
467                                                + passwordFile);
468                            }
469                        } catch (Exception ex) {
470                            throw new IllegalActionException(this, ex,
471                                    "Failed to read "
472                                            + passwordFile.stringValue());
473                        } finally {
474                            if (reader != null) {
475                                try {
476                                    reader.close();
477                                } catch (IOException ex) {
478                                    throw new IllegalActionException(this, ex,
479                                            "Failed to close " + passwordFile
480                                                    .stringValue());
481                                }
482                            }
483                        }
484                    } // if (passwordFile.stringValue().length() > 0)
485                    if (_password == null) {
486                        // Open a dialog to get the password.
487                        Effigy effigy = Configuration.findEffigy(toplevel());
488                        JFrame frame = null;
489                        if (effigy != null) {
490                            Tableau tableau = effigy.showTableaux();
491                            if (tableau != null) {
492                                frame = tableau.getFrame();
493                            }
494                        }
495
496                        // Next construct a query for user name and password.
497                        Query query = new Query();
498                        query.setTextWidth(60);
499                        query.addPassword("password", "Password", "");
500                        ComponentDialog dialog = new ComponentDialog(frame,
501                                "Open Connection", query);
502
503                        if (dialog.buttonPressed().equals("OK")) {
504                            // The password is not stored as a parameter.;
505                            _password = query.getCharArrayValue("password");
506                        } else {
507                            return;
508                        }
509                    }
510                }
511                _connection.login(_userName, new String(_password), "ptolemy");
512                // After use, set password to null for security
513                _password = null;
514            }
515        } catch (XMPPException ex) {
516            throw new IllegalActionException(this, ex,
517                    "Unable to login to XMPP server.");
518        }
519    }
