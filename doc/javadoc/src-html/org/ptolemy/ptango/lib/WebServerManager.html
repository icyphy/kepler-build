028package org.ptolemy.ptango.lib;
029
030import java.util.HashSet;
031
032///////////////////////////////////////////////////////////////////
033////WebServerManager
034
035/** A manager for creating and managing instances of a Jetty web server.
036 *  Models with the {@link org.ptolemy.ptango.lib.WebServer} attribute use this
037 *  manager to register, start, and stop web applications while checking for
038 *  conflicts with other web applications.  This manager creates and starts up
039 *  new web servers, one server per unique port, and shuts down web servers when
040 *  no more models are using them.
041 *
042 *   @author Elizabeth Latronico and Edward A. Lee
043 *   @version $Id$
044 *   @since Ptolemy II 10.0
045 *   @Pt.ProposedRating Red (ltrnc)
046 *   @Pt.AcceptedRating Red (ltrnc)
047 */
048
049// Declare class as final to disallow subclassing.  If subclasses are needed
050// in the future, then the private constructor will need to be changed to a
051// protected implementation.
052// http://www.javaworld.com/javaworld/jw-04-2003/jw-0425-designpatterns.html
053public final class WebServerManager {
054
055    // Note:  The constructor is private.  See private methods section.
056
057    ///////////////////////////////////////////////////////////////////
058    ////                         public methods                    ////
059
060    /** Return the existing web server manager; create and return a new one if
061     * none is present yet.
062     *
063     * @return The web server manager
064     */
065    public static WebServerManager getInstance() {
066        if (_instance == null) {
067            _instance = new WebServerManager();
068        }
069        return _instance;
070    }
071
072    ///////////////////////////////////////////////////////////////////
073    ////                         public methods                    ////
074
075    /** Return true if the given application is registered with the
076     * server associated with the given port; false otherwise. The model name
077     * is used to identify the application - only one instance of a model may
078     * be registered to a server.  (Note that multiple instances of a model are
079     * possible by different servers on different ports.)
080     *
081     * @param fullModelName The full name of the model
082     * @param portNumber The port this application would be hosted on
083     * @return true if the given application is registered with the
084     * server associated with the given port; false otherwise.
085     */
086    public boolean isRegistered(String fullModelName, int portNumber) {
087        // Check if a server is associated with this port
088        WebServerUtilities server = null;
089
090        for (WebServerUtilities theServer : _servers) {
091            if (theServer.getPortNumber() == portNumber) {
092                server = theServer;
093                break;
094            }
095        }
096
097        if (server != null) {
098            // FIXME:  Use just model name, or whole application info object?
099            if (server.isHostingModel(fullModelName)) {
100                return true;
101            }
102        }
103        return false;
104    }
105
106    /** Return true if the given application is registered with the
107     * server associated with the given port and this server is running;
108     * false otherwise.  The model name is used to identify the application -
109     * only one instance of a model may be registered to a server.  (Note that
110     * multiple instances of a model are possible by different servers on
111     * different ports.)
112     *
113     * @param fullModelName The full name of the model
114     * @param portNumber The port this application would be hosted on
115     * @return true if the given application is registered with the
116     * server associated with the given port and this server is running;
117     * false otherwise.
118     */
119    public boolean isRunning(String fullModelName, int portNumber) {
120        // Check if a server is associated with this port
121        WebServerUtilities server = null;
122
123        for (WebServerUtilities theServer : _servers) {
124            if (theServer.getPortNumber() == portNumber) {
125                server = theServer;
126                break;
127            }
128        }
129
130        if (server != null) {
131            // FIXME:  Use just model name, or whole application info object?
132            if (server.isHostingModel(fullModelName) && server.isRunning()) {
133                return true;
134            }
135        }
136
137        return false;
138    }
139
140    /** Register the following application with the manager.  Check if the
141     * model in appInfo is already associated with a server.  If not, assign
142     * the web application to the server, checking for conflicts with
143     * existing applications, creating new handlers where appropriate and
144     * pausing and restarting the server if necessary.  Returns true if the
145     * application was successfully registered; false otherwise.
146     *
147     * @param appInfo Information about the registered web application
148     * @param portNumber The desired port number to host this web application on
149     * @param dynamicPortSelection If true, then select the port dynamically.
150     * @return The actual port number the web application is running on (may be
151     * different from the portNumber parameter if dynamic selection is enabled)
152     * @exception Exception thrown if web server cannot be instantiated or if
153     * application conflicts with an already-registered application
154     */
155    public int register(WebApplicationInfo appInfo, int portNumber,
156            boolean dynamicPortSelection) throws Exception {
157        // Fetch server at this port, if any.  If none, create and start a new
158        // server for this port.  If the port number is <=0, WebServerUtilities
159        // uses a default port
160        WebServerUtilities server = null;
161
162        // If dynamic port selection is enabled, first try to reuse any server
163        // currently hosting models for this Ptolemy instance.
164        // TODO:  Server sharing could be prohibited in the future if desired
165        // for security reasons, depending on the application
166        // Otherwise, retrieve any server at the specified port
167        // Create a new server if no suitable server is found
168
169        if (dynamicPortSelection) {
170            if (!_servers.isEmpty()) {
171                for (WebServerUtilities reuseServer : _servers) {
172                    server = reuseServer;
173                    break;
174                }
175            }
176
177            if (server == null) {
178                server = new WebServerUtilities();
179                server.setDynamicPortSelection(dynamicPortSelection);
180            }
181        } else {
182            for (WebServerUtilities theServer : _servers) {
183                if (theServer.getPortNumber() == portNumber) {
184                    server = theServer;
185                    break;
186                }
187            }
188
189            if (server == null) {
190                server = new WebServerUtilities(portNumber);
191                server.setDynamicPortSelection(dynamicPortSelection);
192            }
193        }
194
195        // Register this application.  This will check for URL conflicts,
196        // create handlers for this app (context and resource), and start the
197        // server if not already started
198        server.register(appInfo);
199
200        // Update information about this server in the server table
201        _servers.add(server);
202
203        return server.getPortNumber();
204    }
205
206    /** Unregister the following application with the manager.  Remove all
207     * context handlers specific to this application and stop the web server
208     * if this is the only application running on it.  Providing a port number
209     * allows other instances of the application on other ports to continue
210     * running.
211     *
212     * @param appInfo Information about the unregistered web application
213     * @param portNumber  The port number that this application is running on
214     * @exception Exception if there is no server associated with this port,
215     * the application is not running on the server associated with this port,
216     * or the application cannot be stopped.
217     */
218    public void unregister(WebApplicationInfo appInfo, int portNumber)
219            throws Exception {
220        // Check if this application has been registered to a server.
221        // First, get the server running on this port.
222        WebServerUtilities server = null;
223        for (WebServerUtilities theServer : _servers) {
224            if (theServer.getPortNumber() == portNumber) {
225                server = theServer;
226                break;
227            }
228        }
229
230        if (server == null) {
231            throw new Exception("Application " + appInfo.getModelName()
232                    + " attempted to unregister itself for port number "
233                    + portNumber
234                    + ", but there is no server associated with this port.");
235        }
236
237        // This will throw an exception if the application is not registered on
238        // this server
239        server.unregister(appInfo);
240    }
241
242    ///////////////////////////////////////////////////////////////////
243    ////                         private methods                   ////
244
245    // WebServerManager is a singleton class; therefore, make the constructor
246    // private so other objects cannot directly construct a new one
247    // http://www.javaworld.com/javaworld/jw-04-2003/jw-0425-designpatterns.html
248    private WebServerManager() {
249
250    }
251
252    ///////////////////////////////////////////////////////////////////
253    ////                         private variables                 ////
254
255    /** The singleton instance of the WebServerManager class */
256    private static WebServerManager _instance = null;
257
258    private HashSet<WebServerUtilities> _servers = new HashSet<WebServerUtilities>();
259}
