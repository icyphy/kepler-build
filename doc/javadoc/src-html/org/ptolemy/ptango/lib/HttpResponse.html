032import java.io.IOException;
033import java.io.InputStreamReader;
034import java.net.HttpURLConnection;
035import java.net.SocketTimeoutException;
036import java.util.ArrayList;
037import java.util.Arrays;
038
039import ptolemy.data.BooleanToken;
040import ptolemy.data.IntToken;
041import ptolemy.data.RecordToken;
042import ptolemy.data.StringToken;
043import ptolemy.data.Token;
044import ptolemy.data.type.Type;

047/** An HTTP response, comprised of selected fields from an HttpURLConnection.
048 *  Useful for classes that want response data but should not have authority
049 *  to open and close connections.
050 *
051 *  @see org.ptolemy.ptango.lib.HttpGet
052 *  @see org.ptolemy.ptango.lib.HttpPost
053 *  @see org.ptolemy.ptango.lib.HttpPut
054 *  @author Marten Lohstroh, Elizabeth Latronico
055 *  @version $Id$
056 *  @since Ptolemy II 10.0
057 *  @Pt.ProposedRating Red (eal)
058 *  @Pt.AcceptedRating Red (eal)
059 */
060public class HttpResponse {
061
062    /** Construct a new, blank HttpResponse object.
063     */
064    public HttpResponse() {
065        _body = "";
066        _contentLength = 0;
067        _contentType = "text/plain";
068        _responseCode = -1;
069        _responseMessage = "No response";
070    }
071
072    /** Construct a new HttpResponse with the given message. Used for creating
073     *  responses when errors occur on the Ptolemy side, such as an IOException.
074     *
075     *  @param message The response message
076     */
077    public HttpResponse(String message) {
078        _body = "";
079        _contentLength = 0;
080        _contentType = "text/plain";
081        _responseCode = -1;
082        _responseMessage = message;
083    }
084
085    /** Construct a new HttpResponse object from an HttpURLConnection object.
086     *
087     * @param connection The connection object
088     */
089    public HttpResponse(HttpURLConnection connection) {
090        // FIXME: HttpURLConnection getContentLengthLong() is not
091        //present in Java 1.6, which we need to compile under Mac OS X.
092        //_contentLength = connection.getContentLengthLong();
093        _contentLength = connection.getContentLength();
094
095        _contentType = connection.getContentType();
096
097        try {
098            _responseCode = connection.getResponseCode();
099            _responseMessage = connection.getResponseMessage();
100        } catch (IOException e) {
101            _responseCode = -1;
102            _responseMessage = "Error connecting to server.";
103        }
104
105        // Store the response body.
106        // Successful requests have data on the connection's input stream.
107        // Erroneous requests have data on the connection's error stream.
108        BufferedReader reader = null;
109        StringBuffer response = new StringBuffer();
110        String line = "";
111        _body = "";
112
113        try {
114            reader = new BufferedReader(
115                    new InputStreamReader(connection.getInputStream()));
116        } catch (SocketTimeoutException e1) {
117            _body = "";
118            _responseCode = 408;
119            _responseMessage = "Request timed out";
120        } catch (IOException e2) {
121            // IOException is thrown by connection.getInputStream() in case of
122            // erroneous requests.  If so, connection.getErrorStream() has body
123            // getErrorStream() does not throw exceptions
124            if (connection.getErrorStream() != null) {
125                reader = new BufferedReader(
126                        new InputStreamReader(connection.getErrorStream()));
127            }
128        }
129
130        // Read response.
131        if (reader != null) {
132            try {
133                while ((line = reader.readLine()) != null) {
134                    response.append(line);
135                }
136
137                _body = response.toString();
138                reader.close();
139            } catch (IOException e) {
140                _body = "";
141                _responseCode = -1;
142                _responseMessage = "Error reading response body.";
143            }
144        }
145    }
146
147    ///////////////////////////////////////////////////////////////////
148    ////                         public methods                    ////
149
150    /** Return the response body.
151     *
152     * @return The response body.
153     */
154    public String getBody() {
155        return _body;
156    }
157
158    /** Return the length of the response body, in bytes, or -1 if length is
159     * unknown.
160     *
161     * @return  The length of the response body, in bytes, or -1 if length is
162     * unknown.
163     */
164    public long getContentLength() {
165        return _contentLength;
166    }
167
168    /** Return the Internet media (MIME) type of the response.
169     *
170     * @return The Internet media (MIME) type of the response.
171     */
172    public String getContentType() {
173        return _contentType;
174    }
175
176    /** Return the source of the error, ErrorSource.CLIENT or
177     *  ErrorSource.SERVER, or ErrorSource.NONE if no error.
178     *
179     * @return ErrorSource.CLIENT, ErrorSource.SERVER or Error.NONE.
180     */
181    public ErrorSource getErrorSource() {
182        if (isSuccessful()) {
183            return ErrorSource.NONE;
184        } else if (_responseCode >= 400 && _responseCode < 500) {
185            return ErrorSource.CLIENT;
186        } else {
187            return ErrorSource.SERVER;
188        }
189    }
190
191    /** Return a code indicating the status of the response.
192     *
193     * @return A code indicating the status of the response.
194     */
195    public int getResponseCode() {
196        return _responseCode;
197    }
198
199    /** A message describing the status of the response.
200     *
201     * @return A message describing the status of the response.
202     */
203    public String getResponseMessage() {
204        return _responseMessage;
205    }
206
207    /** Return all status items as a RecordToken.
208     *
209     * @return All status items as a RecordToken
210     */
211    public RecordToken getStatus() {
212        ArrayList<Token> values = new ArrayList(Arrays.asList(
213                new IntToken(_responseCode), new StringToken(_responseMessage),
214                new BooleanToken(isSuccessful()),
215                new BooleanToken(isFurtherActionExpected())));
216
217        try {
218            return new RecordToken(_labels,
219                    values.toArray(new Token[values.size()]));
220        } catch (IllegalActionException e) {
221            return new RecordToken();
222        }
223    }
224
225    /** Return the Type of the token of getStatus().  Used for setting type
226     *  constraints before the response content is available.
227     *
228     * @return The Type of the token of getStatus()
229     */
230    public static Type getStatusType() {
231        // Define default values here so that a record token can be instantiated
232        // in order to return the Type of that token
233        ArrayList<Token> values = new ArrayList(
234                Arrays.asList(new IntToken(200), new StringToken("OK"),
235                        new BooleanToken(true), new BooleanToken(false)));
236
237        try {
238            return new RecordToken(_labels,
239                    values.toArray(new Token[values.size()])).getType();
240        } catch (IllegalActionException e) {
241            return new RecordToken().getType();
242        }
243    }
244
245    /** Return true if further action is expected of the client, e.g., issuing
246     *  a second request; false otherwise.  Status codes 1xx and 3xx.
247     *
248     * @return True if further action is expected of the client, e.g., issuing
249     *  a second request; false otherwise.  Status codes 1xx and 3xx.
250     */
251    public boolean isFurtherActionExpected() {
252        if ((_responseCode >= 100 && _responseCode < 200)
253                || (_responseCode >= 300 && _responseCode < 400)) {
254            return true;
255        } else {
256            return false;
257        }
258    }
259
260    /** Return true if no errors were encountered; false otherwise.
261     *  Status codes 1xx, 2xx, 3xx.
262     *
263     * @return True if no errors were encountered.  Status codes 1xx, 2xx, 3xx
264     */
265    public boolean isSuccessful() {
266        if (_responseCode >= 100 && _responseCode < 400) {
267            return true;
268        } else {
269            return false;
270        }
271    }
272
273    /** Create a timed-out response.  Used for creating a response in the event
274     * of a send time out.
275     */
276    public void setTimedOut() {
277        _responseCode = 408;
278        _responseMessage = "Request timed out";
279    }
280
281    /** Return true if the request timed out; false otherwise.  Status code 408.
282     *
283     * @return True if the request timed out; false otherwise.  Status code 408.
284     */
285    public boolean timedOut() {
286        if (_responseCode == 408) {
287            return true;
288        } else {
289            return false;
290        }
291    }
292
293    ///////////////////////////////////////////////////////////////////
294    ////                         public  variables                 ////
295
296    /** Enumeration of possible sources of response errors, or none if none.
297     *  CLIENT:  Status codes 4xx
298     *  SERVER:  Status codes 5xx and -1
299     *  NONE:  Any other status code
300     */
301    public enum ErrorSource {
302        // Each of these requires a comment to avoid Javadoc warnings
303        /** NONE:  Any other status code. */
304        NONE,
305        /** CLIENT:  Status codes 4xx. */
306        CLIENT,
307        /** SERVER:  Status codes 5xx and -1. */
308        SERVER
309    }
310
311    ///////////////////////////////////////////////////////////////////
312    ////                         private variables                 ////
313
314    /** The body of the response.  */
315    private String _body;
316
317    /** Length of the response body, in bytes.  -1 if length it unknown. */
318    private long _contentLength;
319
320    /** The Internet media (MIME) type of the response.
321     *  For categories and a list, see:
322     *  http://en.wikipedia.org/wiki/MIME_type
323     *  For mappings to common file extensions, see:
324     *  http://www.sitepoint.com/web-foundations/mime-types-complete-list/
325     */
326    private String _contentType;
327
328    /** Labels for returning the status as a RecordToken */
329    static final String[] _labels = { "responseCode", "responseMessage",
330            "successful", "furtherActionExpected" };
331
332    /** A code indicating the status of the response.
333     *  See http://en.wikipedia.org/wiki/List_of_HTTP_status_codes
334     *  -1  If the response is not valid Http
335     *  1xx Informational:  Request received; intermediate response.  Client
336     *          typically takes further action.  For example, client obtains
337     *          permission to upload a large file.
338     *  2xx Success
339     *  3xx Redirection.  Client typically takes additional action to complete
340     *          the request.
341     *  4xx Client Error
342     *  5xx Server Error
343     *
344     */
345    private int _responseCode;
346
347    /** A description of the response status.  For example, "OK" for code 200.
348     */
349    private String _responseMessage;
