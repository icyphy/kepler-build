030package org.kepler.stratosphere.type;
031
032import java.io.DataInput;
033import java.io.DataOutput;
034import java.io.IOException;
035import java.util.Arrays;
036
037import eu.stratosphere.core.memory.MemorySegment;
038import eu.stratosphere.types.Key;
039import eu.stratosphere.types.NormalizableKey;
040
041/** A PACT Key/Value for an array of bytes.
042 * 
043 * @author Daniel Crawl
044 * @version $Id: ByteArray.java 33068 2014-11-12 23:18:26Z crawl $
045 *
046 */
047public class ByteArray implements NormalizableKey {
048
049    /** Default constructor. */
050    public ByteArray() {
051                hashCode = 0;
052    }
053        
054    /** Construct a new ByteArrayValue with a byte array. */
055    public ByteArray(byte[] bytes) {
056        _bytes = bytes;
057                hashCode = 0; 
058    }
059    
060    /** Get the byte array. */
061    public byte[] getValue() {
062        return _bytes;
063    }
064    
065        public void setValue(byte[] seedbuffer, int i, int length) {
066                _bytes = new byte[length];
067                System.arraycopy(seedbuffer, i, _bytes, 0, length);
068                hashCode = 0;
069        }
070        
071        public void setValue(byte[] seedbuffer) {
072                _bytes = new byte[seedbuffer.length];
073                System.arraycopy(seedbuffer, 0, _bytes, 0, seedbuffer.length);
074                hashCode = 0;
075        }
076    
077    /** Read the byte array from the given data input stream. */
078    @Override
079    public void read(DataInput in) throws IOException {
080        final int length = in.readInt();
081        _bytes = new byte[length];
082        in.readFully(_bytes);
083                hashCode = 0;
084    }
085
086    /** Write the byte array to the given data output stream. */
087    @Override
088    public void write(DataOutput out) throws IOException {
089        out.writeInt(_bytes.length);
090        out.write(_bytes);
091    }
092
093        @Override
094        public int compareTo(Key o) {
095                
096                if (!(o instanceof ByteArray))
097                        throw new ClassCastException("Cannot compare " + o.getClass().getName() + " to ByteArray!");
098                
099                ByteArray b = (ByteArray)o;
100                int n = Math.min(_bytes.length, b._bytes.length);
101                for (int i = 0; i < n; i++)
102                {
103                        int diff = _bytes[i] - b._bytes[i];
104                        if (diff != 0) { return diff; }
105                }
106                return _bytes.length - b._bytes.length;
107        }
108        
109        @Override
110        public boolean equals(final Object obj) {
111                if (obj instanceof ByteArray) {
112                        return Arrays.equals(((ByteArray) obj)._bytes, this._bytes);
113                }
114                return false;
115        }
116        
117        @Override
118        public int hashCode() {
119                int h = this.hashCode;
120                if (h == 0 && this._bytes.length > 0) {
121                        int off = 0;
122                        int len = this._bytes.length;
123                        for (int i = 0; i < len; i++) {
124                                h = 31 * h + this._bytes[off++];
125                        }
126                        this.hashCode = h;
127                }
128                return h;
129        }
130        
131        @Override
132    public String toString() {
133        StringBuilder sb = new StringBuilder();
134        for(byte b : _bytes)
135                  sb.append(String.format("%02x", b & 0xff));
136        return sb.toString();
137    }
138        
139        @Override
140        public int getMaxNormalizedKeyLen() {
141                return Integer.MAX_VALUE;
142        }
143
144        @Override
145        public void copyNormalizedKey(MemorySegment target, int offset, int len) {
146
147                if (_bytes.length >= len) {
148                        for (int i = 0; i < len; ++i) {
149                                target.put(offset + i, (byte) (_bytes[i] + 128));
150                        }
151                } else {
152                        for (int i = 0; i < _bytes.length; ++i) {
153                                target.put(offset + i, (byte) (_bytes[i] + 128));                       
154                        }
155                        final int limit = offset + len;
156                        offset += _bytes.length;
157                        while (offset < limit) {
158                                target.put(offset++, (byte) 0);
159                        }
160                }
161        }
162    
163    /** The byte array. */
164    private byte[] _bytes;
165    
166    /** hash code for the byte array object. */
167    private int hashCode;
168
169}
