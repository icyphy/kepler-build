030package org.kepler.stratosphere.type;
031
032
033import eu.stratosphere.types.DoubleValue;
034import eu.stratosphere.types.IntValue;
035import eu.stratosphere.types.Key;
036import eu.stratosphere.types.LongValue;
037import eu.stratosphere.types.NullValue;
038import eu.stratosphere.types.StringValue;
039import eu.stratosphere.types.Value;
040import ptolemy.actor.TypedIOPort;
041import ptolemy.data.type.ArrayType;
042import ptolemy.data.type.BaseType;
043import ptolemy.data.type.ObjectType;
044import ptolemy.data.type.RecordType;
045import ptolemy.data.type.Type;
046
047/** Utilities to convert between Stratosphere and Ptolemy types.
048 * 
049 * @author Daniel Crawl
050 * @version $Id: TypeUtilities.java 33628 2015-08-24 22:42:20Z crawl $
051 * 
052 */
053public class TypeUtilities {
054
055    /** Get the PACT Key data type for a specific field in port with the key-value type.
056     *  @param actor the actor containing the port
057     *  @param port the port
058     *  @param fieldName the name of the field in the key-value type 
059     */
060    public static Class<? extends Key> getPactKeyTypeForFieldInKeyValuePort(TypedIOPort port, String fieldName) {
061        
062        final Type type = _getPtolemyTypeForFieldInKeyValuePort(port, fieldName);
063        return getPactKeyTypeForPtolemyType(type);
064    }
065
066    /** Get the PACT Value data type for a specific field in port with the key-value type.
067     *  @param actor the actor containing the port
068     *  @param port the port
069     *  @param fieldName the name of the field in the key-value type 
070     */
071    public static Class<? extends Value> getPactValueTypeForFieldInKeyValuePort(TypedIOPort port, String fieldName) {
072        
073        final Type type = _getPtolemyTypeForFieldInKeyValuePort(port, fieldName);
074        return getPactValueTypeForPtolemyType(type);
075    }
076
077    /** Get the PACT Value type for a Ptolemy type. */
078    public static Class<? extends Value> getPactValueTypeForPtolemyType(Type type) {
079        
080        if(type instanceof ObjectType) {
081            
082            final Class<?> clazz = ((ObjectType)type).getValueClass();
083            if(clazz == byte[].class) {
084                return ByteArray.class;
085            } else {
086                throw new RuntimeException("Conversion of Ptolemy Object Type using class " + clazz +
087                    " to PACT type is unsupported.");
088            }
089        } else {
090            return getPactKeyTypeForPtolemyType(type);
091        }
092    }
093
094    /** Get the PACT Key type for a Ptolemy type. */
095    public static Class<? extends Key> getPactKeyTypeForPtolemyType(Type type) {
096        
097        // find the equivalent pact type.
098        if(type == BaseType.STRING) {
099            return StringValue.class;
100        } else if(type == BaseType.INT) {
101            return IntValue.class;
102        } else if(type == BaseType.DOUBLE) {
103            return DoubleValue.class;
104        } else if(type == BaseType.LONG) {
105            return LongValue.class;
106        } else if(type == BaseType.NIL) {
107            return NullValue.class;
108        } else if(type instanceof ObjectType) {
109            if(((ObjectType)type).getValueClass() == byte[].class) {
110                return ByteArray.class;
111            } else {
112                throw new RuntimeException("Conversion of Ptolemy object type " +
113                        ((ObjectType)type).getValueClass() + " to PACT type is unsupported.");
114            }
115        } else if(type instanceof ArrayType) {
116            if(((ArrayType)type).getElementType() == BaseType.UNSIGNED_BYTE) {
117                return ByteArray.class;
118            } else {
119                throw new RuntimeException("Conversion of Ptolemy array of type " +
120                        ((ArrayType)type).getElementType() + " to PACT type is unsupported.");
121            }
122        } else if(type instanceof RecordType) {
123            return RecordValue.class;
124        } else {
125            throw new RuntimeException("Conversion of Ptolemy type " +
126                type.toString() + " to PACT type is unsupported.");
127        }
128    }
129    
130    /** Get the Ptolemy type for a field in a port with a key-value type. */
131    private static Type _getPtolemyTypeForFieldInKeyValuePort(TypedIOPort port, String fieldName) {
132        
133        // get the port type
134        final Type portType = port.getType();
135        
136        final String portName = port.getFullName();
137        
138        if(!(portType instanceof ArrayType)) {
139            throw new IllegalArgumentException("Port \"" + portName + "\" must be an" +
140                " array of records type. Type is " + portType);
141        }
142        
143        final Type elementType = ((ArrayType)portType).getElementType();
144        if(!(elementType instanceof RecordType)) {
145            throw new IllegalArgumentException("Port \"" + portName + "\" must be an" +
146                " array of records type. Type is " + portType);
147        }
148        
149        // get the field type
150        final Type fieldType = ((RecordType)elementType).get(fieldName);
151        if(fieldType == null) {
152            throw new IllegalArgumentException("Port \"" + portName + "\" type does" +
153                " not have field \"" + fieldName + "\".");
154        }
155        
156        return fieldType;
157    }
158
159    /** The index of the key field. */
160    public final static int KEY_FIELD = 0;
161    
162    /** The index of the value field. */
163    public final static int VALUE_FIELD = 1;
164}
