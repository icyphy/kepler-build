029package org.kepler.module.provenance;
030
031import org.kepler.module.ModuleShutdownable;
032import org.kepler.provenance.ProvenanceRecorder;
033import org.kepler.provenance.RecordPlayer;
034
035import ptolemy.kernel.util.IllegalActionException;
036import ptolemy.kernel.util.NameDuplicationException;
037import ptolemy.util.MessageHandler;
038
039/** Perform cleanup for the provenance module.
040 * 
041 * @author Daniel Crawl
042 * @version $Id: Shutdown.java 34190 2015-11-02 19:02:03Z crawl $
043 */
044public class Shutdown implements ModuleShutdownable
045{
046
047    /** Perform any module-specific cleanup. */
048    @Override
049    public void shutdownModule()
050    {
051        try
052        {
053            ProvenanceRecorder.disconnectAll();
054        }
055        catch (IllegalActionException e)
056        {
057            MessageHandler.error("Error cleaning up provenance recorders.", e);
058        }
059        
060        // close any opened effigies for replayed workflows.
061        try {
062            RecordPlayer.closeAllEffigies();
063        } catch (IllegalActionException | NameDuplicationException e) {
064            MessageHandler.error("Error closing effigies from replayed workflows.", e);
065        }
066    }
067}
