064        public void initializeModule() {
065                setupCatalogFile();
066                listenForExternalSemanticChanges();
067                
068        //System.out.println("start of workflow-scheduler-gui module initializing");
069                ConfigurationUtilities.copyAllProperties("uiMenuMappings", "gui", "tagging");
070        }
071
072        private void listenForExternalSemanticChanges() {
073                
074                SMSTagChangeListener listener = new TagsChangedListener();
075                SMSServices.addListener( listener );
076        }
077
078        private void setupCatalogFile() {
079                OntologyConfiguration oc = OntologyConfiguration.instance();
080
081                String ontCatFileStr = "configs/ptolemy/configs/kepler/ontologies/ontology_catalog.xml";
082                File original = FileUtil.getHighestRankedFile(ontCatFileStr);
083
084                // TODO get module name from somewhere
085                File dotKeplerTaggingCacheDir = DotKeplerManager.getInstance()
086                                .getTransientModuleDirectory("tagging");
087
088                // shouldn't be necessary, but shouldn't hurt
089                // noinspection ResultOfMethodCallIgnored
090                dotKeplerTaggingCacheDir.mkdirs();
091
092                File catalogLocation = new File(dotKeplerTaggingCacheDir,
093                                "ontology_catalog.xml");
094                if (!catalogLocation.isFile()) {
095                        // Copy it
096                        FileCopierBase copier = FileCopierFactory.getFileCopier("local");
097
098                        // Whizbang reflection goes here
099                        try {
100                                Method copyMethod = FileCopierBase.class.getDeclaredMethod("copyLocal", String.class, String.class, boolean.class);
101                                copyMethod.setAccessible(true);
102                                Object obj = copyMethod.invoke(copier, original.getAbsolutePath(), catalogLocation.getAbsolutePath(), false);
103                                FileCopierBase.CopyResult cr = (FileCopierBase.CopyResult) obj;
104                        }
105                        catch (NoSuchMethodException ex) {
106                                ex.printStackTrace();
107                        }
108                        catch (InvocationTargetException ex) {
109                                ex.printStackTrace();
110                        }
111                        catch (IllegalAccessException ex) {
112                                ex.printStackTrace();
113                        }
114
115                }
116                oc.setIndexFile(catalogLocation);
117                oc.initialize();
118        }

120        /** 
121         * SMSTagChangeListener
122         */
123        class TagsChangedListener implements SMSTagChangeListener {
124                
125                public void tagsChanged(NamedObj object, List<NamedOntClass> added,
126                                List<NamedOntClass> removed) {
127                        // This has to fire a tag change event iff the object to be
128                        // tagged is a top-level workflow.
129                        if (object.depthInHierarchy() != 0) {
130                                // It's not top-level, so no notification sent
131                                return;
132                        }
133                        // else this is top level, so send notification
134                        Set<WindowIdentifier> identifiers = TaggingContextManager
135                                        .getInstanceKeys();
136                        if (Configuration.findEffigy(object) == null)
137                                return;
138                        
139                        Effigy effigy = Configuration.findEffigy(object).topEffigy();
140                        for (WindowIdentifier identifier : identifiers) {
141                                KeplerGraphFrame kgf = identifier.getKeplerGraphFrame();
142                                if (kgf.getEffigy() == null) {
143                                        log.debug("Skipping notification of " + identifier
144                                                        + " due to non-existent effigy");
145                                        continue;
146                                }
147                                if (kgf.getEffigy().topEffigy() == effigy) {
148                                        // System.out.println("Match found on: " + identifier);
149                                        TaggingContextManager tcm = TaggingContextManager
150                                                        .getInstance(identifier);
151                                        for (TagReloadType change : TagReloadType
152                                                        .createExternalChanges(object, added, removed)) {
153                                                tcm.reloadTags(true, change);
154                                        }
155                                }
156                        }
157                }
158        }
159        
160        private static Log log = LogFactory.getLog(Initialize.class);
