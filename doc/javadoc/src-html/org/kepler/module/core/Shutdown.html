030package org.kepler.module.core;
031
032import org.kepler.module.ModuleShutdownable;
033import org.kepler.objectmanager.cache.CacheManager;
034import org.kepler.objectmanager.lsid.LSIDGenerator;
035import org.kepler.util.sql.HSQL;
036
037/** A class to perform cleanup for the core module.
038 * 
039 * @author Daniel Crawl
040 * @version $Id: Shutdown.java 34502 2016-07-09 01:18:09Z crawl $
041 */
042public class Shutdown implements ModuleShutdownable
043{
044
045    /** Perform any module-specific cleanup. */
046    @Override
047    public void shutdownModule()
048    {
049        // close the lsid database;
050        LSIDGenerator.getInstance().closeDatabase();
051        
052        // shutdown the cache manager
053        CacheManager.shutdown();
054        
055        // shutdown down hsql servers so that database files are
056        // properly closed.
057        HSQL.shutdownServers();
058    }
059}
