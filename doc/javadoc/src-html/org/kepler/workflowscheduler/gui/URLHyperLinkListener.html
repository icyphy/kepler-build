031import java.awt.event.ActionEvent;
032import java.lang.reflect.Method;
033
034import javax.swing.JOptionPane;
035import javax.swing.event.HyperlinkEvent;
036import javax.swing.event.HyperlinkListener;
037
038import org.kepler.workflowscheduler.gui.configurationwizard.OpenWorkflowRunEngineConfigDialogAction;
039import org.kepler.workflowscheduler.gui.configurationwizard.OpenWorkflowSchedulerURLConfigDialogAction;
040
041import ptolemy.actor.gui.TableauFrame;
042
043/**
044 * A class to handle url link. If the url is a special one, it has some special way to 
045 * handle it. If it is normal url, it will open it by a browser
046 * @author tao
047 *
048 */
049public class URLHyperLinkListener implements HyperlinkListener 
050{
051  public static final String WORKFLOWSCHEDULERURL= "http://workflow.scheduler.url";
052  public static final String WORKFLOWRUENGINE = "http://workflow.run.engine";
053  private TableauFrame parent = null;
054  
055  /**
056   * Constructor
057   * @param parent
058   */
059  public URLHyperLinkListener(TableauFrame parent)
060  {
061    this.parent = parent;
062  }
063 
064    @Override
065  public void hyperlinkUpdate (HyperlinkEvent event) 
066  {
067      if (event.getEventType() == HyperlinkEvent.EventType.ACTIVATED) 
068      {
069        if(event.getURL() != null)
070        {
071          //System.out.println("the url is === "+event.getURL().toString());
072          if(event.getURL().toString().equals(WORKFLOWSCHEDULERURL))
073          {
074            OpenWorkflowSchedulerURLConfigDialogAction action = 
075                  new OpenWorkflowSchedulerURLConfigDialogAction(parent);
076            //action.setListener(dialog);
077            ActionEvent e = new ActionEvent(this, 100, "Fire workflow scheduler configuration");
078            action.actionPerformed(e);
079            //System.out.println("==== the action to configure workflow scheduler url being performed");
080          }
081          else if (event.getURL().toString().equals(WORKFLOWRUENGINE))
082          {
083            OpenWorkflowRunEngineConfigDialogAction action = 
084                new  OpenWorkflowRunEngineConfigDialogAction(parent);
085            //action.setListener(dialog);
086            ActionEvent e = new ActionEvent(this, 100, "Fire workflow run engine configuration");
087            action.actionPerformed(e);
088          }
089          else
090          {
091            openURL(event.getURL().toString());
092          }
093          
094        }
095        else
096        {
097         //System.out.println("============event.getURL() is null");
098        }
099        
100      }
101  }
102  
103    /*
104     * open a given url
105     */
106  private  void openURL(String url) 
107  {
108      String osName = System.getProperty("os.name");
109      try 
110      {
111         if (osName.startsWith("Mac OS")) 
112         {
113            Class fileMgr = Class.forName("com.apple.eio.FileManager");
114            Method openURL = fileMgr.getDeclaredMethod("openURL",
115               new Class[] {String.class});
116            openURL.invoke(null, new Object[] {url});
117         }
118         else if (osName.startsWith("Windows"))
119         {
120           Runtime.getRuntime().exec("rundll32 url.dll,FileProtocolHandler " + url);
121         }      
122         else 
123         { //assume Unix or Linux
124            String[] browsers = {
125               "firefox", "opera", "konqueror", "epiphany", "mozilla", "netscape" };
126            String browser = null;
127            for (int count = 0; count < browsers.length && browser == null; count++)
128               if (Runtime.getRuntime().exec(
129                     new String[] {"which", browsers[count]}).waitFor() == 0)
130                  browser = browsers[count];
131            if (browser == null)
132               throw new Exception("Could not find web browser");
133            else
134               Runtime.getRuntime().exec(new String[] {browser, url});
135            }
136         }
137         catch (Exception e) {
138         JOptionPane.showMessageDialog(parent, "Error attempting to launch web browser" + ":\n" + e.getLocalizedMessage());
139         }
140  }
