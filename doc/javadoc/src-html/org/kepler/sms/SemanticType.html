046/**
047 * A data structure to hold a semantic type as a concept id in a MoML file.
048 * 
049 *@author Shawn Bowers
050 *@created May 9, 2005
051 */
052
053/**
054 * This is a subtype of StringParameter. All the data is stored through the
055 * StringParameter.setExpression() call, as a single string. This string is
056 * interpreted as follows:
057 * 
058 * If form is A#B#C, then A is the ontology URI, B is the concept ID, and C is
059 * the concept label
060 * else if form is A#B, then A is the ontology URI, B is the concept ID
061 * else if form is A, then A is the concept ID
062 */
063
064public class SemanticType extends StringParameter {
065        // the main value of the property (the class name)
066
067        public SemanticType() throws IllegalActionException,
068                        NameDuplicationException {
069            // super(null, null);
070            super(new NamedObj(), "semtype");
071        }
072
073        /**
074         * Constructor for the SemanticTypeParam object
075         * 
076         *@param container
077         *            Description of the Parameter
078         *@param name
079         *            Description of the Parameter
080         *@exception IllegalActionException
081         *                Description of the Exception
082         *@exception NameDuplicationException
083         *                Description of the Exception
084         */
085        public SemanticType(NamedObj container, String name)
086                        throws IllegalActionException, NameDuplicationException {
087                super(container, name);
088                setExpression("urn:lsid:localhost:onto:2:1#Workflow");
089        setVisibility(Settable.NONE);
090        }
091        
092        @Override
093        public void setExpression(String string) {
094                super.setExpression(string);
095                // NOTE: It is difficult to have this called directly and maintain
096                // class invariants.
097                log.debug("SemanticType.setExpression() should probably not be called directly. Use the appropriate setXXX() methods.");
098//              writeOutStackTrace();
099                _setFieldsFromRawExpression(string);
100        }
101
102        public void writeOutStackTrace() {
103                try {
104                        tryWriteOutStackTrace();
105                }
106                catch(IOException ex) {
107                        log.error("Error writing out stack trace", ex);
108                }
109        }
110        private void tryWriteOutStackTrace() throws IOException {
111                File file;
112                BufferedWriter writer;
113                try {
114                        throw new Exception();
115                }
116                catch(Exception ex) {
117                        StackTraceElement[] stackTrace = ex.getStackTrace();
118                        if (!isGoodTrace(stackTrace)) {
119                                log.debug("Not writing out this trace");
120                                return;
121                        }
122                        file = File.createTempFile("exception", ".txt");
123                        writer = new BufferedWriter(new FileWriter(file));
124                        for (StackTraceElement element : ex.getStackTrace()) {
125                                writer.write(element.toString());
126                                writer.write("\n");
127                        }
128                }
129                writer.close();
130                log.info("Wrote stack trace to: " + file.getAbsolutePath());
131        }
132
133        private static boolean isGoodTrace(StackTraceElement[] stackTrace) {
134                for (StackTraceElement ste : stackTrace) {
135                        if ("org.kepler.sms.SemanticType".equals(ste.getClassName()) && ste.getLineNumber() == 75) {
136                                return false;
137                        }
138                }
139                return true;
140        }
141
142        private void _setFieldsFromRawExpression(String string) {
143                String[] parts = string.split("#");
144                if (parts.length == 1) {
145                        _conceptName = parts[0];
146                        _namespace = null;
147                        _label = null;
148                }
149                else {
150                        _namespace = parts[0];
151                        _conceptName = parts[1];
152                        try {
153                                _label = parts[2];
154                        }
155                        catch(ArrayIndexOutOfBoundsException ex) {
156                                _label = null;
157                        }
158                }
159        }
160        
161        private void _setExpressionFromFields() {
162                StringBuilder builder = new StringBuilder();
163                if (_namespace != null) {
164                        builder.append(_namespace).append("#");
165                }
166                if (_conceptName != null) {
167                        builder.append(_conceptName).append("#");
168                }
169                if (_label != null) {
170                        builder.append(_label);
171                }
172                super.setExpression(builder.toString());
173        }
174
175        /**
176         * Description of the Method
177         * 
178         *@param obj
179         *            Description of the Parameter
180         *@return Description of the Return Value
181         */
182        public boolean equals(Object obj) {
183                if (!(obj instanceof SemanticType)) {
184                        return false;
185                }
186                SemanticType semtypeId = (SemanticType) obj;
187                String str = semtypeId.getExpression();
188                if (this.getExpression() == null) {
189                        return str == null;
190                }
191                String[] parts = getExpression().split("#");
192                if (parts.length < 3) {
193                        return this.getExpression().equals(semtypeId.getExpression());
194                }
195                String[] otherParts = semtypeId.getExpression().split("#");
196                
197                if (parts.length < 2 || otherParts.length < 2) {
198                        System.out.println("This should not happen!");
199                }
200                
201                // Compare up to but not including the third section of these strings
202                for (int i = 0; i < 2; i++) {
203                        if (!parts[i].equals(otherParts[i])) {
204                                return false;
205                        }
206                }
207                
208                return true;            
209        }

211        /**
212         * return the value of the semantic type
213         * 
214         *@return The concept id
215         */
216        public String getConceptId() {
217                return getExpression();
218        }
219        
220
221        public String getConceptUri() {
222                return getNamespace() + "#" + getConceptName(); 
223        }
224        
225
226        /**
227         * set the semantic type concept value
228         * 
229         *@param expr
230         *            The new conceptId value
231         *@exception ptolemy.kernel.util.IllegalActionException
232         *                Description of the Exception
233         */
234        public void setConceptId(String expr)
235                        throws ptolemy.kernel.util.IllegalActionException {
236                setExpression(expr);
237        }
238
239        public String getLabel() {
240                if (_label != null) {
241                        return _label;
242                }
243                return _conceptName;
244        }
245        
246        /**
247         * Gets the namespace attribute of the SemanticTypeParam object
248         * 
249         *@return The namespace value
250         */
251        public String getNamespace() {
252                return _namespace;
253        }
254
255        /**
256         * Gets the conceptName attribute of the SemanticTypeParam object
257         * 
258         *@return The conceptName value
259         */
260        public String getConceptName() {
261                return _conceptName;
262        }
263        
264        public void setLabel(String label) {
265                _label = label;
266                _setExpressionFromFields();
267        }
268        
269        public void setNamespace(String namespace) {
270                _namespace = namespace;
271                _setExpressionFromFields();
272        }
273        
274        public void setConceptName(String conceptName) {
275                _conceptName = conceptName;
276                _setExpressionFromFields();
277        }
