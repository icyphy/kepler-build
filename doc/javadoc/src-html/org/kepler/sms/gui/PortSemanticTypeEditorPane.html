135        /**
136         * Performs the commit button operation.
137         */
138        public void doCommit() {
139                // save the state of the old selection
140                Object port = null;
141                if (_selectedNode != null) {
142                        port = _selectedNode.getPort();
143                }
144
145                // remove old state
146                _manager.removeTypes(port);
147
148                Iterator<NamedOntClass> typeIter = _classSelector.getNamedOntClasses().iterator();
149                while (typeIter.hasNext())
150                        _manager.addType(port, typeIter.next());
151
152                // do the commit itself
153                Iterator objIter = _manager.getObjects().iterator();
154                while (objIter.hasNext()) {
155                        NamedObj obj = (NamedObj) objIter.next();
156                        
157                        Vector<NamedOntClass> namedOntClasses = _manager.getTypesAsNamedOntClasses(obj);
158                        Iterator<NamedOntClass> nocItr = namedOntClasses.iterator();
159                        Vector<NamedOntClass> nocToAdd = _manager.getTypesAsNamedOntClasses(obj);
160                        
161                        //remove from existing those that are not in namedOntClasses
162                        Iterator<SemanticType> existingSemTypeItr = obj.attributeList(SemanticType.class).iterator();
163                        while (existingSemTypeItr.hasNext()){
164                                SemanticType existingSemType = existingSemTypeItr.next();
165                                boolean remove = true;
166                                nocItr = namedOntClasses.iterator();
167                                while (nocItr.hasNext()){
168                                        NamedOntClass noc = nocItr.next();
169                                        if (noc.getConceptId().equals(existingSemType.getConceptId())){
170                                                
171                                                // remove it from the to-add list, it's already there
172                                                nocToAdd.remove(noc);
173                                                
174                                                // and don't remove it from existing.
175                                                remove = false;
176                                                break;
177                                        }
178                                }
179                                if (remove){
180                                        try {
181                                                // REMOVE
182                                                existingSemType.setContainer(null);
183                                                // update LSID revision
184                                                NamedObj container = obj.getContainer();
185                                                NamedObjId.getIdAttributeFor(container).updateRevision();
186                                        } catch (Exception e) {
187                                                e.printStackTrace();
188                                        }
189                                }
190                        }
191                        
192                        // now add those that aren't already there
193                        _addSemTypes(obj, nocToAdd);
194                }
195        }

197        /**
198         * Adds the semantic types in the dialog to the entity.
199         */
200        private void _addSemTypes(NamedObj obj, Vector semTypes) {
201                Iterator typeIter = semTypes.iterator();
202                while (typeIter.hasNext()) {
203                        try {
204                                SemanticType st = new SemanticType(obj, obj
205                                                .uniqueName("semanticType"));
206                                st.setExpression(((NamedOntClass) typeIter.next())
207                                                .getConceptId());
208                                
209                                // update LSID revision
210                                NamedObj container = obj.getContainer();
211                                NamedObjId.getIdAttributeFor(container).updateRevision();
212                                
213                        } catch (Exception e) {
214                                e.printStackTrace();
215                        }
216                }
217        }

219        /**
220         * Removes the current semtypes on the named object
221         */
222        private void _removeNamedObjSemtypes(NamedObj obj) {
223                if (obj == null)
224                        return;
225
226                Vector<SemanticType> semtypes = new Vector<SemanticType>();
227                Iterator<SemanticType> iter = obj.attributeList(SemanticType.class).iterator();
228                while (iter.hasNext())
229                        semtypes.add(iter.next());
230
231                iter = semtypes.iterator();
232                while (iter.hasNext()) {
233                        SemanticType st = iter.next();
234                        try {
235                                st.setContainer(null);
236                                // update LSID revision
237                                NamedObj container = obj.getContainer();
238                                NamedObjId.getIdAttributeFor(container).updateRevision();
239                        } catch (Exception e) {
240                                e.printStackTrace();
241                        }
242                }
243        }

382        protected DefaultTreeModel _createModel() {
383                _PortTreeNode root = new _PortTreeNode("");
384                _treeModel = new DefaultTreeModel(root);
385
386                // get all the ports (real and virtual)
387                Vector ioPorts = SMSServices.getIOPorts(_namedObj);
388                Vector compositePorts = SMSServices.getPortBundles(_namedObj);
389                Vector refinePortNames = new Vector();
390
391                // get the refinemet port names
392                Iterator refIter = SMSServices.getRefinementPorts(_namedObj).iterator();
393                while (refIter.hasNext())
394                        refinePortNames.add(((KeplerRefinementIOPort) refIter.next())
395                                        .getPointer());
396
397                // add the objects semantic types
398                // Iterator typeIter = _getSemTypes(obj).iterator();
399                // while(typeIter.hasNext())
400                // _manager.addType(obj, (String)typeIter.next());
401
402                // build up the regular ports first
403                Iterator portIter = ioPorts.iterator();
404                while (portIter.hasNext()) {
405                        TypedIOPort port = (TypedIOPort) portIter.next();
406                        if (validPort(port)) {
407                                _PortTreeNode node = new _PortTreeNode(port);
408                                root.addChild(node);
409                                Type t = port.getType();
410                                if (t instanceof RecordType)
411                                        _addRefinementPorts(node, (RecordType) t, refinePortNames);
412                                // else if(t instanceof ArrayType)
413                                // _addRefinementPorts(node, (ArrayType)t, refinePortNames);
414                        }
415                }
416
417                // build up the composite ports if selected
418                if (!_compositeBox.isSelected())
419                        return _treeModel;
420
421                Iterator compPortIter = compositePorts.iterator();
422                while (compPortIter.hasNext()) {
423                        KeplerCompositeIOPort port = (KeplerCompositeIOPort) compPortIter
424                                        .next();
425                        if (validPort(port)) {
426                                _PortTreeNode node = new _CompositePortTreeNode(port);
427                                root.addChild(node);
428                                // get the "children" and add ...
429                                Iterator refs = port.getEncapsulatedPortReferences().iterator();
430                                while (refs.hasNext()) {
431                                        KeplerIOPortReference ref = (KeplerIOPortReference) refs
432                                                        .next();
433                                        _PortTreeNode ref_node = new _ReferencePortTreeNode(ref
434                                                        .getPort());
435                                        node.addChild(ref_node);
436                                }
437                        }
438                }
439
440                return _treeModel;
441        }

452        private void _initSemanticTypeManager(_PortTreeNode node) {
453                Object obj = node.getPort();
454                _manager.addObject(obj);
455
456                Iterator typeIter = _getSemTypes(obj).iterator();
457                while (typeIter.hasNext()) {
458                        SemanticType st = (SemanticType) typeIter.next();
459                        NamedOntClass cls = OntologyCatalog.instance().getNamedOntClass(st);
460                        _manager.addType(obj, cls);
461                }
462
463                for (int i = 0; i < node.getChildCount(); i++)
464                        _initSemanticTypeManager((_PortTreeNode) node.getChildAt(i));
465        }

473                else if (obj instanceof KeplerCompositeIOPort)
474                        return SMSServices

476                return new Vector();
477        }

506     *
507     */
508        private void _addRefinementPorts(_PortTreeNode parent, RecordType t,
509                        Vector refinePortNames) {
510                Iterator iter = t.labelSet().iterator(); // get record component labels
511                while (iter.hasNext()) {
512                        String label = (String) iter.next();
513                        Object obj = parent.getPort();
514                        String portname = null;
515                        if (obj instanceof IOPort)
516                                portname = ((IOPort) obj).getName();
517                        else if (obj instanceof KeplerVirtualIOPort)
518                                portname = ((KeplerVirtualIOPort) obj).getName();
519                        String pointer = portname + "/" + label;
520                        // check if exists already as a port
521                        if (!refinePortNames.contains(pointer)) {
522                                try {
523                                        KeplerRefinementIOPort p = new KeplerRefinementIOPort(
524                                                        _namedObj, pointer);
525                                        _PortTreeNode node = new _PortTreeNode(p);
526                                        parent.addChild(node);
527                                        if (p.getType() instanceof RecordType) {
528                                                _addRefinementPorts(node, (RecordType) p.getType(),
529                                                                refinePortNames);
530                                        }
531                                } catch (Exception e) {
532                                        e.printStackTrace();
533                                }
534                        }
535                }
536        }

617        /**
618         * Inner class for tree nodes
619         */
620        private class _PortTreeNode extends DefaultMutableTreeNode {
621                /** constructor */
622                public _PortTreeNode(Object port) {
623                        _port = port;
624                }
625
626                /** @return The port encapsulated in the node */
627                public Object getPort() {
628                        return _port;
629                }
630
631                /** Add given node as child */
632                public void addChild(_PortTreeNode child) {
633                        if (!_children.contains(child)) {
634                                _children.add(child);
635                                child.removeFromParent();
636                                child.setParent(this);
637                        }
638                }
639
640                /** Set the parent of this node */
641                public void setParent(_PortTreeNode parent) {
642                        _parent = parent;
643                }
644
645                /** Remove the parent of this node */
646                public void removeFromParent() {
647                        _parent = null;
648                }
649
650                /** @return The children of this node */
651                public Enumeration children() {
652                        return _children.elements();
653                }
654
655                /** @return Always true. */
656                public boolean getAllowsChildren() {
657                        return true;
658                }
659
660                /** @return The ith tree node */
661                public TreeNode getChildAt(int index) {
662                        if (index < _children.size() && index >= 0)
663                                return (TreeNode) _children.elementAt(index);
664                        return null;
665                }
666
667                /** @return the index of the child node */
668                public int getIndex(TreeNode node) {
669                        return _children.indexOf(node);
670                }
671
672                /** @return The number of children */
673                public int getChildCount() {
674                        return _children.size();
675                }
676
677                /** @return The parent of this node */
678                public TreeNode getParent() {
679                        return _parent;
680                }
681
682                /** @return True if no children */
683                public boolean isLeaf() {
684                        return _children.size() == 0;
685                }
686
687                public String toString() {
688                        String str = "";
689                        if (_port instanceof IOPort)
690                                str += ((IOPort) _port).getName() + "  (";
691                        else if (_port instanceof KeplerCompositeIOPort)
692                                str += ((KeplerCompositeIOPort) _port).getName() + "  (";
693                        else if (_port instanceof KeplerRefinementIOPort)
694                                str += ((KeplerRefinementIOPort) _port).getName() + "  (";
695
696                        if (_port instanceof TypedIOPort) {
697                                Type type = ((TypedIOPort) _port).getType();
698                                if (type instanceof BaseType)
699                                        str += type;
700                                else if (type instanceof RecordType)
701                                        str += "record";
702                                else if (type instanceof ArrayType)
703                                        str += "array";
704                                else
705                                        str += "other";
706                        } else if (_port instanceof KeplerVirtualIOPort) {
707                                Type type = ((KeplerVirtualIOPort) _port).getType();
708                                if (type instanceof BaseType)
709                                        str += type;
710                                else if (type instanceof RecordType)
711                                        str += "record";
712                                else if (type instanceof ArrayType)
713                                        str += "array";
714                                else
715                                        str += "other";
716                        } else
717                                str += "unknown";
718                        return str + ")";
719                }

805        /**
806         * Add an object to the set of objects being annotated within this panel.
807         * 
808         * @param obj
809         *            The object to add.
810         */
811        private void _addAnnotationObject(NamedObj obj) {
812                if (obj == null)
813                        return;
814
815                for (Iterator iter = _annotationObjects.iterator(); iter.hasNext();) {
816                        Object[] entry = (Object[]) iter.next();
817                        if (entry[0].equals(obj))
818                                return;
819                }
820                Object[] entry = new Object[2];
821                entry[0] = obj;
822                // entry[1] = _initializeNamedObj(obj);
823                _annotationObjects.add(entry);
824        }
825
826        /**
827         * Remove the given annotation object from the editor
828         */
829        private void _removeAnnotationObject(NamedObj obj) {
830                if (obj == null)
831                        return;
832                Object[] objentry = null;
833                for (Iterator iter = _annotationObjects.iterator(); iter.hasNext();) {
834                        Object[] entry = (Object[]) iter.next();
835                        if (entry[0].equals(obj)) {
836                                objentry = entry;
837                                break;
838                        }
839                }
840                if (objentry == null)
841                        return;
842                _annotationObjects.remove(objentry);
843        }

846         * @return The set of annotation objects being annotated with the panel.
847         */
848        private Vector _getAnnotationObjects() {

851                        Object[] entry = (Object[]) iter.next();
852                        result.add(entry[0]);
853                }
854                return result;
