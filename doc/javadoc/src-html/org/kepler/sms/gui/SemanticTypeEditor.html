070         * This is the default constructor
071         */
072        public SemanticTypeEditor(Frame owner, NamedObj namedObj) {
073                super(owner);
074                // set title and close behavior
075                this.setTitle("Semantic Type Annotation");
076                this.setDefaultCloseOperation(JDialog.DO_NOTHING_ON_CLOSE);
077                // create the frame's pane
078                JPanel pane = new JPanel();
079                pane.setLayout(new BoxLayout(pane, BoxLayout.PAGE_AXIS));
080                pane.setBorder(BorderFactory.createEmptyBorder(10, 10, 10, 10));
081
082                // Add our private anonymous window listener so we can
083                // call _doClose on windowClosing
084                this.addWindowListener(_windowListener);
085                
086                // create the actor and port annotations pane
087                _actorPane = new ActorSemanticTypeEditorPane(owner, namedObj);
088                int IN = PortSemanticTypeEditorPane.INPUT;
089                int OUT = PortSemanticTypeEditorPane.OUTPUT;
090                _inputPortPane = new PortSemanticTypeEditorPane(owner, namedObj, IN);
091                _outputPortPane = new PortSemanticTypeEditorPane(owner, namedObj, OUT);
092
093                // create the tabbed pane
094                JTabbedPane tabbedPane = new JTabbedPane();
095                tabbedPane.addTab("Component", _actorPane);
096                tabbedPane.addTab("Input", _inputPortPane);
097                tabbedPane.addTab("Output", _outputPortPane);
098
099                // add the components
100                pane.add(tabbedPane);
101                pane.add(Box.createRigidArea(new Dimension(5, 0)));
102                pane.add(_createButtons());
103                // add the pane
104                setContentPane(pane);
105                setSize(550, 700);
106                // this.setResizable(false);
107        }
108
109        /**
110         * Initialize the bottom buttons (close)
111         */
112        private JPanel _createButtons() {
113                // init buttons
114                _helpBtn.setMnemonic(KeyEvent.VK_H);
115                _commitBtn.setMnemonic(KeyEvent.VK_O);
116                _commitBtn.setActionCommand("commit");
117                _commitBtn.setToolTipText("Save changes to annotations");
118                _commitBtn.addActionListener(_buttonListener);
119                _closeBtn.setMnemonic(KeyEvent.VK_C);
120                _closeBtn.setActionCommand("close");
121                _closeBtn.setToolTipText("Close editor");
122                _closeBtn.addActionListener(_buttonListener);
123
124                JPanel pane = new JPanel();
125                pane.setLayout(new BoxLayout(pane, BoxLayout.X_AXIS));
126                pane.setBorder(BorderFactory.createEmptyBorder(10, 10, 10, 10));
127                pane.add(Box.createHorizontalGlue());
128                pane.add(_helpBtn);
129                pane.add(Box.createRigidArea(new Dimension(10, 0)));
130                pane.add(_closeBtn);
131                pane.add(Box.createRigidArea(new Dimension(10, 0)));
132                pane.add(_commitBtn);
133
134                return pane;
135        }
136
137        /**
138         * anonymous class to handle button events
139         */
140        private ActionListener _buttonListener = new ActionListener() {
141                public void actionPerformed(ActionEvent e) {
142                        if (e.getActionCommand().equals("commit"))
143                                _doCommit();
144                        else if (e.getActionCommand().equals("close"))
145                                _doClose();
146                }
147        };
148
149        /**
150         * Performs the cancel button operation.
151         */
152        private void _doClose() {
153                if (_actorPane.hasModifiedSemTypes()
154                                || _inputPortPane.hasModifiedSemTypes()
155                                || _outputPortPane.hasModifiedSemTypes()) {
156
157                        // do a JDialog
158                        Object[] options = { "Yes", "No", "Cancel" };
159                        String msg = "Would you like to save the changes you made?";
160                        int n = JOptionPane.showOptionDialog(this, msg, "Message",
161                                        JOptionPane.YES_NO_CANCEL_OPTION,
162                                        JOptionPane.QUESTION_MESSAGE, null, options, options[2]);
163                        if (n == 0) {
164                                _doCommit();
165                                return;
166                        } else if (n == 1) {
167                                _close();
168                                return;
169                        }
170                }
171                _close();
172        }
173
174        /**
175         * Performs the commit button operation. Commit, like in Ptolemy, saves the
176         * changes and closes the window.
177         */
178        private void _doCommit() {
179                // ensure well formed semtypes
180                String msg;
181                if ((msg = _actorPane.wellFormedSemTypes()) != null) {
182                        JOptionPane.showMessageDialog(this, msg, "Message",
183                                        JOptionPane.ERROR_MESSAGE);
184                        return; // abort
185                } else if ((msg = _inputPortPane.wellFormedSemTypes()) != null) {
186                        JOptionPane.showMessageDialog(this, msg, "Message",
187                                        JOptionPane.ERROR_MESSAGE);
188                        return; // abort
189                } else if ((msg = _outputPortPane.wellFormedSemTypes()) != null) {
190                        JOptionPane.showMessageDialog(this, msg, "Message",
191                                        JOptionPane.ERROR_MESSAGE);
192                        return; // abort
193                }
194                _actorPane.doCommit();
195                _inputPortPane.doCommit();
196                _outputPortPane.doCommit();
197                _close();
198        }
199
200        /**
201         * Really shut down everything
202         */
203        private void _close() {
204                _actorPane.doClose();
205                _inputPortPane.doClose();
206                _outputPortPane.doClose();
207                dispose();
208        }

309        public static TypedIOPort addInputType(TypedAtomicActor actor, ptolemy.data.type.Type type,
310                        String name) throws NameDuplicationException,
311                        IllegalActionException {
312                TypedIOPort p = (TypedIOPort) actor.newPort(name);
313                p.setTypeEquals(type);
314                p.setInput(true);
315                p.setOutput(false);
316                return p;
317        }
318
319        public static TypedIOPort addOutputType(TypedAtomicActor actor, ptolemy.data.type.Type type,
320                        String name) throws NameDuplicationException,
321                        IllegalActionException {
322                TypedIOPort p = (TypedIOPort) actor.newPort(name);
323                p.setTypeEquals(type);
324                p.setInput(false);
325                p.setOutput(true);
326                return p;
327        }
