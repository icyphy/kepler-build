028package org.kepler.webview.server.app.ro;
029
030import java.net.URI;
031
032import org.apache.commons.httpclient.util.HttpURLConnection;
033import org.kepler.webview.server.WebViewConfiguration;
034import org.kepler.webview.server.WebViewServer;
035import org.kepler.webview.server.app.AbstractApp;
036
037import io.vertx.core.AsyncResult;
038import io.vertx.core.Future;
039import io.vertx.core.Handler;
040import io.vertx.core.MultiMap;
041import io.vertx.core.http.HttpClientRequest;
042import io.vertx.core.json.JsonArray;
043import io.vertx.core.json.JsonObject;
044import io.vertx.ext.auth.User;
045
046/** Create a research object on ROHUB.
047 * 
048 *  @author Daniel Crawl
049 */
050public class CreateRO extends AbstractApp {
051
052    public CreateRO() {
053        _authStr = WebViewConfiguration.getROHubAuthToken();
054        
055        String roHubStr = WebViewConfiguration.getROHubURI();
056        
057        try {
058            _roHubURI = new URI(roHubStr);
059        } catch(Exception e) {
060            System.err.println("Syntax error parsing ROHUB URI: " + roHubStr);
061        }
062        
063        _roHubPort = _roHubURI.getPort();
064        if(_roHubPort == -1) {
065            _roHubPort = _roHubURI.getScheme().startsWith("https") ? 443 : 80;
066        }
067
068    }
069    
070    @Override
071    public void exec(User user, JsonObject inputs, Handler<AsyncResult<JsonArray>> handler)
072        throws Exception {
073                       
074        if(_authStr == null) {
075            handler.handle(Future.failedFuture("Server not configured with ROHUB token."));
076        } else if(_roHubURI == null) {
077            handler.handle(Future.failedFuture("Server configured with invalid ROHUB URI."));
078        } else if(!inputs.containsKey("name")) {
079            handler.handle(Future.failedFuture("Must specify name."));
080        } else if(!inputs.containsKey("title")) {
081            handler.handle(Future.failedFuture("Must specify title."));
082        } else if(!inputs.containsKey("description")) {
083            handler.handle(Future.failedFuture("Must specify description."));
084        } else {
085            String name = inputs.getString("name");
086                                    
087            HttpClientRequest request = WebViewServer.vertx().createHttpClient()
088                .post(_roHubPort, _roHubURI.getHost(), _roHubURI.getPath(), response -> {
089                    int status = response.statusCode();
090                    if(status != HttpURLConnection.HTTP_OK &&
091                        status != HttpURLConnection.HTTP_CREATED) {
092                        handler.handle(Future.failedFuture("Unexpected response from ROHUB: " + status));
093                    } else {
094                        /*
095                        response.headers().forEach(a -> {
096                            System.out.println(a.getKey() + " -> " + a.getValue());
097                        });
098                        */
099                        String location = response.headers().get("Location");
100                        if(location == null) {
101                            handler.handle(Future.failedFuture("No research object URI in response."));
102                        } else {
103                            //System.out.println(location);
104                            _annotateRO(location, inputs, handler);
105                        }
106                    }
107            });
108            
109            MultiMap headers = request.headers();
110            headers.set("Content-type", "text/plain")
111                .set("Accept", "text/turtle")
112                .set("slug", name)
113                .set("Authorization", "Bearer " + _authStr);
114            request.setFollowRedirects(true).end();
115        }
116    }
117    
118    private void _annotateRO(String location, JsonObject inputs, Handler<AsyncResult<JsonArray>> handler) {
119
120        String title = inputs.getString("title");
121        String description = inputs.getString("description");
122        
123        URI roURI;
124        try {
125            roURI = new URI(location);
126        } catch(Exception e) {
127            handler.handle(Future.failedFuture(new Exception("Error parsing RO URI: " + location)));
128            return;
129        }
130        
131            HttpClientRequest request = WebViewServer.vertx()
132                .createHttpClient()
133                .post(roURI.getHost(), roURI.getPath(), response -> {
134                    //int status = response.statusCode();
135                    //System.out.println("annotation status: " + status);
136                    _stringResponse(handler, "location", location);
137                    /*
138                    response.bodyHandler(b -> {
139                       System.out.println(b); 
140                    });
141                    */
142            });
143            
144            String body = "<?xml version=\"1.0\"?>\n" + 
145                    "<rdf:RDF xmlns:rdf=\"http://www.w3.org/1999/02/22-rdf-syntax-ns#\" xmlns:prov=\"http://www.w3.org/ns/prov#\" xmlns:dct=\"http://purl.org/dc/terms/\" xmlns:pav=\"http://purl.org/pav/\" xmlns:foaf=\"http://xmlns.com/foaf/0.1/\" xmlns:skos=\"http://www.w3.org/2004/02/skos/core#\" xmlns:dc=\"http://purl.org/dc/elements/1.1/\" xmlns:odrs=\"http://schema.theodi.org/odrs#\" xmlns:roes=\"http://w3id.org/ro/earth-science#\" xmlns:xsd=\"http://www.w3.org/2001/XMLSchema#\" xmlns:roterms=\"http://purl.org/wf4ever/roterms#\">\n" + 
146                    "  <rdf:Description rdf:about=\"" + location + "\">\n" + 
147                    "    <dct:title>" + title + "</dct:title>\n" + 
148                    "    <dct:description>" + description + "</dct:description>\n" + 
149                    "  </rdf:Description>\n" + 
150                    "</rdf:RDF>\n";
151            
152            //System.out.println(body);
153            
154            request.headers()
155                .set("Content-type", "application/rdf+xml")
156                .set("Content-length", String.valueOf(body.length()))
157                .set("Slug", "title.rdf")
158                .set("Authorization", "Bearer " + _authStr)
159                .set("Link", "<" + location + ">; rel=\"http://purl.org/ao/annotatesResource\"");
160            request.setFollowRedirects(true).write(body).end();
161            
162            //System.out.println(request.headers().get("Link"));
163    }
164    
165                        
166    private String _authStr;
167    private URI _roHubURI;
168    private int _roHubPort;
169}
