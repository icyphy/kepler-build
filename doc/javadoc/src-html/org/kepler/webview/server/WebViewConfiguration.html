081    /** Get the authorization group.
082     *  @param username The user name.
083     *  @param password The password.
084     *  @return Returns null if username not found, or password does not match.
085     */ 
086    public static String getSimpleAuthGroup(String username, String password) {
087        for(ConfigurationProperty property : _getConfigurationProperties("server.auth.entity")) {
088            if(property.getProperty("user").getValue().equals(username)) {
089                if(property.getProperty("password").getValue().equals(password)) {
090                    return property.getProperty("group").getValue();
091                }
092                // found user, but incorrect password.
093                return null;
094            }
095        }
096        // did not find user.
097        return null;
098    }

100    /** Get the class name of an app. */
101    public static String getAppClassName(String appName) {
102        for(ConfigurationProperty property : _getConfigurationProperties("server.apps.app")) {
103            if(property.getProperty("name").getValue().equals(appName)) {
104                return property.getProperty("class").getValue();
105            }
106        }

121    /** Get an authentication provider based on the authentication type
122     *  specified in the configuration.
123     */
124    public static AuthProvider getAuthProvider() {
125        String authType = _getConfigurationString("server.auth.type", null);
126        if(authType == null || authType.equals("simple")) {
127            return new SimpleAuth();
128        } else if(authType.equals("none")) {
129            System.err.println("WARNING: WebView server authentication disabled.");
130            return new NoneAuth();
131        } else if(authType.equals("drupal")) {
132            return new DrupalAuth(_getConfigurationString("server.auth.drupal.host", null),
133                _getConfigurationString("server.auth.drupal.service", null),
134                _getConfigurationString("server.auth.drupal.role", null),
135                _getConfigurationString("server.auth.drupal.groupsField", null),
136                _getConfigurationString("server.auth.drupal.fullNameField", null));
137
138        } else if(authType.equals("class")) {
139            String className = _getConfigurationString("server.auth.class", null);
140            if(className != null) {
141                try {
142                    Class<?> clazz = Class.forName(className);
143                    return (AuthProvider)clazz.newInstance();
144                } catch (InstantiationException | IllegalAccessException | ClassNotFoundException e) {
145                    System.err.println("ERROR: could not instantiate auth class " + className + ": " + e.getMessage());
146                    e.printStackTrace();
147                }
148            }
149        } else {
150            System.err.println("ERROR: unknown type of authentication: " + authType);
151        }
152        return null;
153    }

350    /** Get a configuration property in the web-view config file. */
351    private static ConfigurationProperty _getConfigurationProperty(String path) {
352        return ConfigurationManager.getInstance()
353                .getProperty(ConfigurationManager.getModule("web-view")).getProperty(path);
354    }
355
356    /** Get a list of configuration properties in the web-view config file. */
357    private static List<ConfigurationProperty> _getConfigurationProperties(String path) {
358        return ConfigurationManager.getInstance()
359                .getProperties(ConfigurationManager.getModule("web-view"), path);
360    }
361
362    /** Get a configuration value in the web-view config file. Returns
363     *  the default value if the path does not exist or is not set.
364     */
365    private static String _getConfigurationString(String path, String defaultValue) {
366        ConfigurationProperty property = _getConfigurationProperty(path);
367        if(property != null) {
368            String value = property.getValue();
369            if(value != null) {
370                return value;
371            }
372        }
373        return defaultValue;
374    }
375    
376    /** Get set of strings from multiple configuration values with the same name. */
377    private static Set<String> _getConfigurationStrings(String path) {
378        Set<String> retval = new HashSet<String>();
379        for(ConfigurationProperty property: _getConfigurationProperties(path)) {
380            retval.add(property.getValue());
381        }
382        return retval;
383    }
384        
385    /** Get a boolean configuration value in the web-view config file. */
386    private static boolean _getConfigurationBoolean(String path, boolean defaultValue) {
387        String value = _getConfigurationString(path, null);
388        if(value != null) {
389            return Boolean.valueOf(value);
390        }
391        return defaultValue;
392    }
393    
394    /** Get an integer configuration value in the web-view config file. */
395    private static int _getConfigurationInteger(String path, int defaultValue) {
396        String value = _getConfigurationString(path, null);
397        if(value != null) {
398            return Integer.valueOf(value);
399        }
400        return defaultValue;
401    }
402    
403    /** Get a long configuration value in the web-view config file. */
404    private static long _getConfigurationLong(String path, long defaultValue) {
405        String value = _getConfigurationString(path, null);
406        if(value != null) {
407            return Long.valueOf(value);
408        }
409        return defaultValue;
410    }
411
412    private static Module _getConfigurationModule(String path) {
413        ConfigurationProperty property = _getConfigurationProperty(path);
414        if(property != null) {
415            return property.getModule();
416        }
417        return null;
418    }
