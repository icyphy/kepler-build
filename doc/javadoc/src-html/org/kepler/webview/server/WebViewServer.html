119    /** Execute an app.
120     *  @param json The JSON object with the workflow name and any parameters.
121     *  @param handler Asynchronous handler to receive the a JSON object with any results.
122     */
123    public void executeApp(JsonObject json, User user,
124        Handler<AsyncResult<JsonObject>> handler) {
125
126        String appName = json.getString("app_name");
127        //System.out.println("execute workflow " + wfName);
128        
129        if(appName == null || appName.trim().isEmpty()) {
130            handler.handle(Future.failedFuture("No app_name specified."));
131            return;
132        }
133
134        if(user == null) {
135            handler.handle(Future.failedFuture("User is not authenticated"));
136            return;
137        }
138        
139        for(String key: json.fieldNames()) {
140            if(!key.equals("app_name") &&
141                !key.equals("app_param") &&
142                !key.equals("prov") && 
143                !key.equals("reqid") &&
144                !key.equals("sync")) {
145                System.err.println("WARNING: unknown property in request: " + key);
146            }
147        }
148        
149        // the following will block, e.g., waiting on the workspace lock,
150        // so execute in blocking threads.
151        _vertx.<JsonObject>executeBlocking(future -> {
152
153            //System.out.println("executeBlocking " + wfName);
154            
155            boolean runSynchronously = false;
156            App app = null;
157            boolean running = false;
158            
159            // see if app is already loaded
160            try {
161                app = _getApp(appName);
162                
163                if(app == null) {
164                    throw new Exception("App not found.");
165                }
166                     
167                JsonObject appParams;
168                if(json.containsKey("app_param")) {
169                    try {
170                        appParams = json.getJsonObject("app_param");
171                    } catch(ClassCastException e) {
172                        throw new Exception("app_param must be a json object.");
173                    }                
174                } else {
175                    appParams = new JsonObject();
176                }
177                                        
178                runSynchronously = json.getBoolean("sync", false);
179                if(!runSynchronously) {
180                    throw new Exception("Asynchronous execution not supported for apps.");
181                }
182                
183                if(runSynchronously) {                                                                                                                
184                    // execute the application
185                    running = true;
186                    final App appCopy = app;
187                    app.exec(user, appParams, appHandler -> {
188
189                        if(appHandler.failed()) {
190                            future.fail(appHandler.cause());
191                        } else {
192                            future.complete(new JsonObject().put("responses", appHandler.result()));                            
193                        }
194                        appCopy.close();
195                    });
196                }
197            } catch (Exception e) {
198                e.printStackTrace();
199                future.fail("Error executing app: " + e.getMessage());
200                return;
201            } finally {
202                if(!running) {
203                    app.close();                   
204                }
205            }
206                        
207        // set ordered to false to allow parallel executions,
208        // and use handler for result.    
209        }, false, handler);
210    }

212    /** Execute a workflow.
213     *  @param json The JSON object with the workflow name and any parameters.
214     *  @param handler Asynchronous handler to receive the a JSON object with any results.
215     */
216    public void executeWorkflow(JsonObject json, User user,
217        Handler<AsyncResult<JsonObject>> handler) {
218
219        String wfName = json.getString("wf_name");
220        //System.out.println("execute workflow " + wfName);
221        
222        if(wfName == null || wfName.trim().isEmpty()) {
223            handler.handle(Future.failedFuture("No wf_name specified."));
224            return;
225        }
226
227        if(user == null) {
228            handler.handle(Future.failedFuture("User is not authenticated"));
229            return;
230        }
231        
232        // TODO correct way to convert Buffer to JSON?
233        //System.out.println("recvd: " + json);
234       
235        // the following will block, e.g., waiting on the workspace lock,
236        // so execute in blocking threads.
237        _vertx.<JsonObject>executeBlocking(future -> {
238
239            //System.out.println("executeBlocking " + wfName);
240            
241            boolean runSynchronously = false;
242            ProvenanceRecorder recorder = null;
243            CompositeActor model = null;
244            
245            // see if workflow is already loaded
246            try {
247                model = _getModel(wfName);
248
249                if(model == null) {
250                    throw new Exception("Workflow not found.");
251                }
252                                    
253                try {
254                    _setModelParameters(model, json, user);
255                } catch (IllegalActionException e) {
256                    throw new Exception("Error setting parameters: " + e.getMessage());
257                }
258                
259                boolean recordProvenance = json.getBoolean("prov", true);
260                runSynchronously = json.getBoolean("sync", false);
261                
262                // cannot run async without provenance
263                if(!runSynchronously && !recordProvenance) {
264                    throw new Exception("Cannot execute workflow asynchronously without recording provenance.");
265                }
266                
267                recorder = ProvenanceRecorder.getDefaultProvenanceRecorder(model);
268
269                if(recordProvenance) {
270                    if(recorder == null) {                        
271                        if(!ProvenanceRecorder.addProvenanceRecorder(model, null, null)) {                            
272                            throw new Exception("Error adding provenance recorder to workflow.");
273                        }
274                        recorder = ProvenanceRecorder.getDefaultProvenanceRecorder(model);
275                        if(recorder == null) {
276                            throw new Exception("Cannot find provenance recorder in workflow after adding one.");
277                        }
278                    }                    
279                    
280                    //System.out.println("setting provenance username = " + user.principal().getString("username"));
281                    recorder.username.setToken(user.principal().getString("username"));
282                } else if (recorder != null) {
283                    recorder.setContainer(null);
284                }
285                
286                // create manager and add model
287                final Manager manager = new Manager(model.workspace(), "Manager");
288
289                try {
290                    model.setManager(manager);
291                } catch(IllegalActionException e) {
292                    throw new Exception("Error setting Manager for sub-workflow: " + e.getMessage());
293                }    
294                
295                String[] errorMessage = new String[1];
296
297                ExecutionListener managerListener = new ExecutionListener() {
298                    
299                    @Override
300                    public void executionError(Manager m, Throwable throwable) {
301                        System.err.println("Workflow execution error: " + throwable.getMessage());
302                        throwable.printStackTrace();
303                        errorMessage[0] = throwable.getMessage();
304                    }
305
306                    @Override
307                    public void executionFinished(Manager m) {
308                        //System.out.println("finished");
309                    }
310
311                    @Override
312                    public void managerStateChanged(Manager m) {                    
313                        //System.out.println("manager state changed " + m.getState().getDescription());
314                    }
315                };
316                
317                manager.addExecutionListener(managerListener);
318                
319                if(runSynchronously) {
320                    
321                    final String[] runLSIDStr = new String[1];
322                    if(recordProvenance) {
323                        recorder.addPiggyback(new Recording() {
324                            @Override
325                            public void executionStart(KeplerLSID lsid) {
326                                runLSIDStr[0] = lsid.toString();
327                            }
328                        });
329                    }
330
331                    final boolean[] timeout = new boolean[1];
332                    timeout[0] = false;
333                    long timerId = vertx.setTimer(_workflowTimeout, id -> {
334                        timeout[0] = true;
335                        manager.stop();
336                        future.fail("Execution timeout.");
337                    });
338                    
339
340                    // call execute() instead of run, otherwise exceptions
341                    // go to execution listener asynchronously.
342                    manager.execute();
343                    
344                    // see if we timed-out. if so, we already sent the response,
345                    // so exit.
346                    if(timeout[0]) {
347                        return;
348                    } else if(!vertx.cancelTimer(timerId)) {
349                        System.err.println("Workflow timeout Timer does not exist.");
350                    } else { System.out.println("cancelled timer."); }
351                    
352                    // see if there is a workflow exception
353                    if(errorMessage[0] != null) {
354                        future.fail(errorMessage[0]);
355                        return;
356                    }
357                                                                        
358                    JsonObject responseJson = new JsonObject();
359
360                    if(recordProvenance) {
361                        responseJson.put("id", runLSIDStr[0]);  
362                    }
363                    
364                    // build the response JSON object from the client buffers
365                    // for this workflow.
366                    JsonArray arrayJson = new JsonArray();
367                    List<JsonObject> outputs = getClientBuffer(model);
368                    if(outputs != null) {
369                        for(JsonObject outputJson : outputs) {
370                            if(outputJson.containsKey("actor")) {
371                                JsonObject actorObject = outputJson.getJsonObject("actor");
372                                for(String idStr : actorObject.fieldNames()) {
373                                    JsonObject idObject = actorObject.getJsonObject(idStr);
374                                    if(idObject.containsKey("data")) {
375                                        arrayJson.add(idObject.getJsonObject("data"));
376                                    }
377                                }
378                            }
379                        }
380                    }
381                    // send the successful response
382                    //System.out.println(arrayJson.encodePrettily());
383                    responseJson.put("responses", arrayJson);
384                    future.complete(responseJson);
385                } else { // asynchronous             
386                                        
387                    final CompositeActor finalModel = model;
388                    recorder.addPiggyback(new Recording() {
389                        @Override
390                        public void executionStart(KeplerLSID lsid) {
391                            //System.out.println("execution lsid is " + lsid);
392                            future.complete(new JsonObject().put("id", lsid.toString()));
393                        }
394                                                              
395                        @Override
396                        public void executionStop() {
397                            removeModel(finalModel);
398                        }
399                    });
400                    
401                    manager.startRun();
402                }
403            } catch (Exception e) {
404                future.fail("Error executing workflow: " + e.getMessage());
405                return;
406            } finally {
407                if(model != null && runSynchronously) {
408                    removeModel(model);
409                    model = null;
410                }
411            }
412            
413        // set ordered to false to allow parallel executions,
414        // and use handler for result.    
415        }, false, handler);
416    }

418    /** Search for a file. The directories searched are the root directory (if specified),
419     *  and <module>/resources/html/.
420     *  @param name The name of the file to search for.
421     *  @return If the path is found, the absolute path of the file.
422     *  Otherwise, returns null.
423     */
424    public static String findFile(String name) {
425        
426        //System.out.println("findFile: " + name);
427        
428        if(_appendIndexHtml && name.endsWith("/")) {
429            name = name.concat("index.html");
430        }
431        
432        // search root dir first if specified
433        if(_rootDir != null) {
434            String fsPath = new StringBuilder(_rootDir)
435                .append(File.separator)
436                .append(name)
437                .toString();
438            // TODO check path is not outside _rootDir
439            if(_vertx.fileSystem().existsBlocking(fsPath)) {
440                return fsPath;
441            }
442        }
443        
444        // not found in root dir, so search module tree.
445        for(Module module : _moduleTree) {
446            String fsPath = new StringBuilder(module.getResourcesDir().getAbsolutePath())
447                .append(File.separator)
448                .append("html")
449                .append(File.separator)
450                .append(name).toString();
451            // TODO check path is not outside of resources/html dir
452            //System.out.println("checking " + fsPath);
453            if(_vertx.fileSystem().existsBlocking(fsPath)) {
454                return fsPath;
455            }
456        }
457        
458        // not found anywhere
459        return null;
460    }

504                            try {
505                                while((message = _loggingQueue.take()) != STOP_LOGGING_THREAD_MESSAGE) {

601    /** Log an http server request.
602     *  @param request The request
603     *  @param user The user
604     *  @param status The status
605     *  @param timestamp The timestamp when the request occurred.
606     *  @param length The length of the file successfully sent to the client.
607     */
608    public void log(HttpServerRequest request, User user, int status, long timestamp, long length) {
609        
610        String versionStr;
611        switch(request.version()) {
612        case HTTP_1_1:
613            versionStr = "HTTP/1.1";
614            break;
615        case HTTP_1_0:
616            versionStr = "HTTP/1.0";
617            break;
618        default:
619            versionStr = "-";
620            break;
621        }
622        
623        String hostString;
624        SocketAddress address = request.remoteAddress();
625        if(address == null) {
626            hostString = "-";
627        } else {
628            hostString = address.host();
629        }
630        
631        String referrer = request.headers().get("referrer");
632        if(referrer == null) {
633            referrer = "-";
634        }
635        
636        String userAgent = request.headers().get("user-agent");
637        if(userAgent == null) {
638            userAgent = "-";
639        }
640        
641        String userNameStr = "-";
642        if(user != null) {
643            userNameStr = user.principal().getString("username");
644        }
645                    
646        // TODO how to get user-identifier and userid?
647        String message = String.format("%s - %s [%s] \"%s %s %s\" %d %s \"%s\" \"%s\"",
648            hostString,
649            userNameStr,
650            _logTimeFormat.format(new Date(timestamp)),
651            request.method(),
652            request.uri(),
653            versionStr,
654            status,
655            length < 0 ? "-" : String.valueOf(length),
656            referrer,
657            userAgent);
658
659        try {
660            _loggingQueue.put(message);
661        } catch (InterruptedException e) {
662            MessageHandler.error("Interuppted adding to logging thread.", e);
663        }
664    }

714        super.start();
715        
716        _auth = WebViewConfiguration.getAuthProvider();
717        
718        _queryable = ProvenanceRecorder.getDefaultQueryable(null);
719        if(_queryable == null) {
720            throw new Exception("Unable to get provenance queryable.");
721        }
722        
723        Router router = Router.router(vertx);
724
725        if(WebViewConfiguration.getHttpServerCorsEnabled()) {
726            String allowOriginPatternStr = WebViewConfiguration.getHttpServerCorsAllowOriginPattern();
727            if(allowOriginPatternStr == null || allowOriginPatternStr.trim().isEmpty()) {
728                throw new Exception("Must specify allow origini pattern for CORS.");
729            }
730            router.route().handler(CorsHandler.create(allowOriginPatternStr)
731                    .allowedMethod(HttpMethod.GET)
732                    .allowedMethod(HttpMethod.POST)
733                    .allowedMethod(HttpMethod.OPTIONS)
734                    .allowCredentials(true)
735                    .allowedHeader("Access-Control-Allow-Credentials")
736                    .allowedHeader("X-PINGARUNER")
737                    .allowedHeader("Content-Type")
738                    .allowedHeader("authorization"));
739            System.out.println("CORS enabled for " + allowOriginPatternStr);
740        }
741                
742        
743        // create cookie and session handlers
744        // authenticated users are cached in the session so authentication
745        // does not need to be performed for each request.
746        router.route().handler(CookieHandler.create());
747        
748        BodyHandler bodyHandler = BodyHandler.create();
749        
750        if(_rootDir != null) {
751            File uploadsDir = new File(_rootDir, "file-uploads");
752            if(!uploadsDir.exists()) {
753                if(!uploadsDir.mkdirs()) {
754                    throw new Exception("Unable to create file uploads directory " + uploadsDir);

756            }

758        }
759        
760        // NOTE: need to install the BodyHandler before any handlers that

762        router.route().handler(bodyHandler);
763        
764        SessionStore sessionStore = LocalSessionStore.create(_vertx);
765        SessionHandler sessionHandler = SessionHandler.create(sessionStore)
766            .setSessionCookieName("WebViewSession")
767            .setSessionTimeout(WebViewConfiguration.getHttpServerSessionTimeout());
768        router.route().handler(sessionHandler);
769        router.route().handler(UserSessionHandler.create(_auth));
770        
771        //System.out.println("session timeout = " + WebViewConfiguration.getHttpServerSessionTimeout());
772        
773        // use custom auth handler with custom auth scheme to prevent
774        // browsers (e.g., chrome) from opening auth dialog.
775        // FIXME read realm from conf file.        
776        AuthHandler authenticationHandler = new WebViewAuthHandlerImpl(_auth, "WebView");
777        
778        Handler<RoutingContext> authorizationHandler = new Handler<RoutingContext>() {
779            @Override
780            public void handle(RoutingContext c) {
781                if(c.user() == null) {
782                    c.next();
783                } else {
784                    c.user().isAuthorized("login", result -> {
785                        if(result.failed()) {
786                            c.response()
787                                .putHeader("Content-type", "text/plain")
788                                .setStatusCode(HttpURLConnection.HTTP_FORBIDDEN)
789                                .end(result.cause().getMessage());
790                        } else {
791                            c.next();
792                        }
793                    });
794                }
795            }            
796        };
797        
798        // everything under /kepler and /app needs to be
799        // authenticated and authorized.
800        router.route("/kepler/*")
801            .handler(authenticationHandler)
802            .handler(authorizationHandler);
803        router.route("/app")
804            .handler(authenticationHandler)
805            .handler(authorizationHandler);
806        
807        // TODO should these be under /kepler?

810

820        router.getWithRegex("/kepler/runs/" + runIdRegexStr + "/(\\w+)").blockingHandler(runIdHandler::handleBinary);
821        router.getWithRegex("/kepler/runs/" + runIdRegexStr).handler(runIdHandler);
822        
823        if(WebViewConfiguration.getHttpServerMetadataFileName() != null) {
824            System.out.println("Metadata file set; requests at /login");
825            
826            router.route("/login")
827                .handler(authenticationHandler)            
828                .handler(authorizationHandler);
829            
830            LoginHandler loginHandler = new LoginHandler(this);
831            router.route("/login").handler(loginHandler);
832                        
833            // login session handler to check if session cookie is valid.
834            /*
835            router.route("/loginSession")
836                .handler(authorizationHandler)
837                .handler(loginSessionContext -> {
838                
839                System.out.println("session " + loginSessionContext.session().id());
840                
841                // if not valid, return 400
842                if(loginSessionContext.user() == null) {
843                    System.out.println("user is null");
844                    loginSessionContext.response()
845                        .putHeader("Content-Type", "text/plain")
846                        .setStatusCode(HttpURLConnection.HTTP_BAD_REQUEST)
847                        .end();
848                } else {
849                    // otherwise call login handler to return metadata.
850                    loginSessionContext.next();
851                }
852            });
853            router.route("/loginSession").handler(loginHandler);
854            */
855            
856            // logout handler to remove session and user.
857            router.route("/logout").handler(authenticationHandler);
858            // FIXME what about authorization handler?
859            router.route("/logout").handler(logoutContext -> {
860                //System.out.println("destroying session " + logoutContext.session().id());
861                logoutContext.session().destroy();
862                logoutContext.clearUser();
863                logoutContext.response()
864                    .putHeader("Content-Type", "text/plain")
865                    .end();
866            });
867        }
868
869        if(WebViewConfiguration.enableHttpServerTableOfContents()) {
870            String path = WebViewConfiguration.getHttpServerTableOfContentsPath();
871            if(path == null) {
872                path = "^(?:/kepler/toc){0,1}/*$";
873            }
874            System.out.println("Enabling http server table of contents at " + path);
875            router.getWithRegex(path).handler(new TableOfContentsHandler(this));
876        }
877        
878        // add route that handles anything unmatched
879        router.route().handler(new NoMatchHandler(this));
880                
881        if(WebViewConfiguration.enableHttps()) {
882            
883            sessionHandler.setCookieSecureFlag(true);
884            
885            String pemKeyStr = WebViewConfiguration.getHttpsPEMKeyPath();
886            if(pemKeyStr == null) {
887                throw new Exception("Must specify PEM key file for HTTPS server.");
888            }
889            
890            String pemCertStr = WebViewConfiguration.getHttpsPEMCertPath();
891            if(pemCertStr == null) {
892                throw new Exception("Must specify PEM certificate file for HTTPS server.");                
893            }
894            
895            int securePort = WebViewConfiguration.getHttpsServerPort();
896            
897            // FIXME what if pemCert or pemKey do not exist?
898            
899            HttpServerOptions options = new HttpServerOptions()
900                .setSsl(true)
901                .setPemKeyCertOptions(new PemKeyCertOptions()
902                    .setCertPath(pemCertStr)
903                    .setKeyPath(pemKeyStr))
904                .setPort(securePort);
905
906            System.out.println("Starting secure web server on port " + securePort);
907            
908            _secureServer = vertx.createHttpServer(options)
909                    .websocketHandler(new WorkflowWebSocketHandler(this))

911                    .listen();
912           
913            if(WebViewConfiguration.shouldHttpServerRedirect()) {
914
915                int status = WebViewConfiguration.getHttpServerRedirectStatus();
916                String hostname = WebViewConfiguration.getHttpServerRedirectHostname();
917                int redirectFromPort = WebViewConfiguration.getHttpServerPort();
918                int redirectToPort = WebViewConfiguration.getHttpServerRedirectPort();
919                
920                String redirectUrl = "https://" + hostname + ":" + redirectToPort;
921                
922                System.out.println("Redirecting web view server port " + redirectFromPort +
923                    " to " + redirectUrl + " with status " + status);
924                
925                _server = vertx.createHttpServer()
926                    .requestHandler(req -> {
927                        req.response().headers().set("Location", redirectUrl + req.path());                        
928                        req.response()
929                            .setStatusCode(status)
930                            .end();
931                    })
932                    .listen(redirectFromPort);
933            }
934
935        } else {
936        
937            int port = WebViewConfiguration.getHttpServerPort();
938            
939            System.out.println("Starting web view server on port " + port);
940
941            _server = vertx.createHttpServer()
942                .websocketHandler(new WorkflowWebSocketHandler(this))

944                .listen(port);

977    /** Called during shutdown. */
978    public static void shutdown() {
979                
980        KeplerGraphFrame.removeUpdater(_updater);
981            
982        for(CompositeActor model : _models.values()) {
983            removeModel(model);
984        }
985        
986        _models.clear();
987        
988        for(App app : _apps.values()) {
989            app.close();
990        }
991        
992        _apps.clear();
993        
994        // stop the logging thread
995        _loggingQueue.add(STOP_LOGGING_THREAD_MESSAGE);
996        try {
997            _loggingThread.join(5000);
998        } catch (InterruptedException e) {
999            MessageHandler.error("Interrupted while waiting for logging thread to stop.", e);
1000        }
1001        _loggingThread = null;
1002    }

1114    /** Convert from a JSON object value to a string representation of a Ptolemy token. */
1115    private static String _convertJSONToTokenString(Object value) {
1116        if(value instanceof JsonArray) {
1117            StringBuilder buf = new StringBuilder("{");
1118            Iterator<Object> iter = ((JsonArray)value).iterator();
1119            while(iter.hasNext()) {
1120                Object item = iter.next();
1121                // strings need to be surrounded by quotes in arrays
1122                if(item instanceof String) {
1123                    buf.append("\"")
1124                        .append(item)
1125                        .append("\"");
1126                } else {
1127                    buf.append(item);
1128                }
1129                if(iter.hasNext()) {
1130                    buf.append(",");
1131                }
1132            }
1133            return buf.append("}").toString();
1134        } else {
1135            // FIXME if value is a string and parameter is not in string mode,
1136            // will this work?
1137            return value.toString();
1138        }
1139    }

1141    @SuppressWarnings("resource")
1142    private App _getApp(String name) throws Exception {
1143        
1144        App masterApp = null;
1145        synchronized(_apps) {            
1146            // see if we've already instantiated this app
1147            masterApp = _apps.get(name);
1148            if(masterApp == null) {
1149            
1150                String className = WebViewConfiguration.getAppClassName(name);
1151                
1152                if(className == null || className.trim().isEmpty()) {
1153                    throw new Exception("Missing class name for app " + name);
1154                }
1155                
1156                try {
1157                    Class<?> appClass = Class.forName(className);
1158                    masterApp = (App) appClass.newInstance();
1159                    _apps.put(name, masterApp);
1160                } catch(ClassNotFoundException | InstantiationException
1161                    | IllegalAccessException e) {
1162                    throw new Exception("Error instantiating app " + name + ": " + e.getMessage());
1163                }    
1164            }
1165        }        
1166        return (App) masterApp.clone();
1167    }

1169    /** Get the model for a specific name. If the model is not already
1170     *  loaded, then use findFile() to find the model file with the
1171     *  given name, name.kar and name.xml. Otherwise, return a copy
1172     *  of the already-loaded model.
1173     *  @return If a model with the specified name is loaded or can be
1174     *  found and parsed, returns the model. Otherwise, null.
1175     */
1176    private static CompositeActor _getModel(String name) throws Exception {
1177                
1178        //System.out.println("size = " + _models.size());
1179        
1180        synchronized(_models) {
1181            
1182            // see if the model is already loaded.
1183            CompositeActor masterModel = _models.get(name);
1184            if(masterModel != null) {
1185                //System.out.println("found master model.");
1186                return _cloneModel(masterModel);
1187            }
1188                    
1189            // if not, find workflow file and load
1190            String wfFileStr = findFile(name);
1191            if(wfFileStr == null) {
1192                wfFileStr = findFile(name + ".kar");
1193                if(wfFileStr == null) {
1194                    wfFileStr = findFile(name + ".xml");
1195                    if(wfFileStr == null) {
1196                        return null;
1197                    }
1198                }
1199            }
1200            
1201            masterModel = (CompositeActor) ParseWorkflow.parseKAR(new File(wfFileStr), true);
1202            _addModel(masterModel);
1203            _models.put(name, masterModel);
1204            //System.out.println("adding master model " + name);
1205            
1206            // see if there is a configuration file for parameters
1207            File wfFile = new File(wfFileStr);
1208            File wfDir = wfFile.getParentFile();
1209            File wfConfFile = new File(wfDir, 
1210                FilenameUtils.getBaseName(wfFileStr) + ".conf");
1211            if(wfConfFile.exists()) {
1212                ParseWorkflowUtil.setParametersFromFile(masterModel, wfConfFile.getAbsolutePath());
1213            }
1214            
1215            return _cloneModel(masterModel);
1216        }
1217    }

1219    /** Set any parameters for a model from a JSON object. */
1220    private static void _setModelParameters(NamedObj model, JsonObject json,
1221        User user) throws IllegalActionException {
1222        
1223        // get any parameters
1224        JsonObject params = null;
1225        if(json.containsKey("wf_param")) {
1226            try {
1227                params = json.getJsonObject("wf_param");
1228            } catch(ClassCastException e) {
1229                throw new IllegalActionException("wf_param must be a json object.");
1230            }
1231        
1232            for(Map.Entry<String, Object> entry: params) {
1233                final String paramName = entry.getKey();
1234                //System.out.println("attempting to set parameter " + paramName);
1235                Attribute attribute = model.getAttribute(paramName);
1236                if(attribute == null || !(attribute instanceof Settable)) {
1237                    throw new IllegalActionException("Workflow does not have settable parameter " + paramName);
1238                }
1239    
1240                try {
1241                    String valueStr = _convertJSONToTokenString(entry.getValue());
1242                    ((Settable)attribute).setExpression(valueStr);
1243                } catch (IllegalActionException e) {
1244                    throw new IllegalActionException("Error settings parameter " +
1245                                paramName + ": " + e.getMessage());
1246                }
1247            }
1248        }
1249                
1250
1251        // set user full name if present
1252        Attribute attribute = model.getAttribute("WebView_FullName");
1253        if(attribute != null) {
1254            if(!(attribute instanceof Settable)) {
1255                System.err.println("WARNING: WebView_FullName parameter is not settable.");
1256            } else {
1257                ((Settable)attribute).setExpression(user.principal().getString("fullname"));
1258            }
1259            attribute = null;
1260        }
1261        
1262        // set the authorization groups parameter if present        
1263        attribute = model.getAttribute("WebView_groups");
1264        if(attribute != null) {
1265            if(!(attribute instanceof Settable)) {
1266                System.err.println("WARNING: WebView_group parameter is not settable.");
1267            } else {
1268                // add quotes around each string since we use this to set the
1269                // WebView_groups parameter in the workflow.
1270                String authGroupsStr = AuthUtilities.getGroups(user).stream()
1271                    .map(s -> "\"" + s + "\"").collect(Collectors.joining(","));
1272                //System.out.println(authGroupsStr);
1273                ((Settable)attribute).setExpression("{" + authGroupsStr + "}");
1274            }
1275        }
1276
1277    }
