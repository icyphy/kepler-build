017package org.kepler.webview.server.auth;
018
019import java.util.Base64;
020
021import io.vertx.core.http.HttpHeaders;
022import io.vertx.core.http.HttpServerRequest;
023import io.vertx.core.json.JsonObject;
024import io.vertx.ext.auth.AuthProvider;
025import io.vertx.ext.auth.User;
026import io.vertx.ext.web.RoutingContext;
027import io.vertx.ext.web.Session;
028import io.vertx.ext.web.handler.impl.AuthHandlerImpl;
029
030/** Basic authentication handler that uses "WebView" as the scheme
031 * instead of "Basic" so that 401 responses are not caught by the browser. 
032 * 
033 * Copied from vertx BasicAuthHandlerImpl.
034 * 
035 * @author <a href="http://pmlopes@gmail.com">Paulo Lopes</a>
036 * @author <a href="http://tfox.org">Tim Fox</a>
037 */
038public class WebViewAuthHandlerImpl extends AuthHandlerImpl {
039
040  private final String realm;
041
042  public WebViewAuthHandlerImpl(AuthProvider authProvider, String realm) {
043    super(authProvider);
044    this.realm = realm;
045  }
046
047  @Override
048  public void handle(RoutingContext context) {
049    User user = context.user();
050    if (user != null) {
051      // Already authenticated in, just authorise
052      authorise(user, context);
053    } else {
054      HttpServerRequest request = context.request();
055      String authorization = request.headers().get(HttpHeaders.AUTHORIZATION);
056
057      //System.out.println(authorization);
058      
059      if (authorization == null) {
060        handle401(context, "Basic");
061      } else {
062        String suser;
063        String spass;
064        String sscheme;
065
066        try {
067          String[] parts = authorization.split(" ");
068          sscheme = parts[0];
069          String decoded = new String(Base64.getDecoder().decode(parts[1]));
070          int colonIdx = decoded.indexOf(":");
071          if(colonIdx!=-1) {
072              suser = decoded.substring(0,colonIdx);
073              spass = decoded.substring(colonIdx+1);
074          } else {
075              suser = decoded;
076              spass = null;
077          }
078        } catch (ArrayIndexOutOfBoundsException e) {
079          handle401(context, "Basic");
080          return;
081        } catch (IllegalArgumentException | NullPointerException e) {
082          // IllegalArgumentException includes PatternSyntaxException
083          context.fail(e);
084          return;
085        }
086
087        // handle either "WebView" or "Basic" scheme
088        if (!"WebView".equals(sscheme) && !"Basic".equals(sscheme)) {
089          context.fail(400);
090        } else {
091          JsonObject authInfo = new JsonObject().put("username", suser).put("password", spass);
092          authProvider.authenticate(authInfo, res -> {
093            if (res.succeeded()) {
094              User authenticated = res.result();
095              context.setUser(authenticated);
096              Session session = context.session();
097              if (session != null) {
098                // the user has upgraded from unauthenticated to authenticated
099                // session should be upgraded as recommended by owasp
100                session.regenerateId();
101              }
102              authorise(authenticated, context);
103            } else {
104              // if not able to authenticate, send 401 with same scheme as in request
105              handle401(context, sscheme);
106            }
107          });
108        }
109      }
110    }
111  }
112
113  private void handle401(RoutingContext context, String scheme) {
114    context.response().putHeader("WWW-Authenticate", scheme + " realm=\"" + realm + "\"")
115        .putHeader("Content-type", "text/plain");
116    context.fail(401);
117  }
118}
