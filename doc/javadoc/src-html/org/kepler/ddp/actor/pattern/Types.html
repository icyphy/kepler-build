030package org.kepler.ddp.actor.pattern;
031
032import java.util.regex.Matcher;
033import java.util.regex.Pattern;
034
035import ptolemy.data.expr.Parameter;
036import ptolemy.data.type.ArrayType;
037import ptolemy.data.type.BaseType;
038import ptolemy.data.type.ObjectType;
039import ptolemy.data.type.RecordType;
040import ptolemy.data.type.Type;
041import ptolemy.kernel.util.IllegalActionException;
042
043/** A class providing the Ptolemy types used between the pattern actors.
044 * 
045 *  @author Daniel Crawl
046 *  @version $Id: Types.java 32711 2014-05-08 23:03:56Z crawl $
047 */
048public class Types {
049
050   
051    /** Create the Ptolemy type for array of key values for specific key and value types. */
052    public static Type createKeyValueArrayType(Type keyType, Type valueType) {
053        return new ArrayType(new RecordType(new String[] {"key","value"},
054                new Type[] {keyType, valueType}));    
055    }
056    
057    /** Get the Ptolemy type for a key-value type string.
058     *  @param the parameter containing the type string.
059     *  @param typesStr the type string, e.g., "int string", "string long",
060     *  where the first type is the key and the second type is the value.
061     *  @return a Type for an array of records that have key and value fields
062     *  whose types are specified by the typeStr parameter.  
063     */
064    public static Type getKeyValueType(Parameter parameter, String typesStr) throws IllegalActionException {
065        
066        Type[] types = getKeyValueTypes(parameter, typesStr);
067        
068        return Types.createKeyValueArrayType(types[0], types[1]);
069    }
070    
071    /** Get the Ptolemy types for a key-value type string.
072     *  @param the parameter containing the type string.
073     *  @param typesStr the type string, e.g., "int string", "string long",
074     *  where the first type is the key and the second type is the value. 
075     *  @return an array of Type, where the first element is the key type,
076     *  and the second element is the value type. 
077     */
078    public static Type[] getKeyValueTypes(Parameter parameter, String typesStr) throws IllegalActionException {
079        
080        final String typesCleanedStr = typesStr.replaceAll(",", "").trim();
081        if(typesCleanedStr.isEmpty()) {
082            throw new IllegalActionException(parameter, "Must provide non-empty value for types.");
083        }
084        
085        final String[] typesArray = typesCleanedStr.split("\\s+");
086        if(typesArray.length != 2) {
087            throw new IllegalActionException(parameter, "Must provide type for both key and value.");
088        }
089        
090        final Type keyType = getTypeFromString(typesArray[0]);
091        if(keyType == null) {
092            throw new IllegalActionException(parameter, "Unknown key type " + typesArray[0]);
093        }
094
095        final Type valueType = getTypeFromString(typesArray[1]);
096        if(valueType == null) {
097            throw new IllegalActionException(parameter, "Unknown value type " + typesArray[1]);
098        }
099        
100        return new Type[] {keyType, valueType};
101    }
102
103    
104    /** Get the Ptolemy type from a string. */
105    public static Type getTypeFromString(String typeStr) {
106        
107        Type type;
108        
109        // FIXME this is fragile.
110        final Matcher keyMatcher = ARRAY_TYPE_PATTERN.matcher(typeStr);
111        final Matcher keyMatcher4Obj = OBJECT_TYPE_PATTERN.matcher(typeStr);
112        if(keyMatcher.matches()) {
113            Type elementType = BaseType.forName(keyMatcher.group(1));
114            type = new ArrayType(elementType);
115        } else if (keyMatcher4Obj.matches()) {
116                Type elementType;
117                if (keyMatcher4Obj.groupCount() > 1) {
118                        elementType = BaseType.forName(keyMatcher4Obj.group(1));
119                        type = new ObjectType(elementType.getTokenClass());
120                } else
121                        type = new ObjectType(byte[].class);            
122        } else {
123            type = BaseType.forName(typeStr);
124        }
125
126        return type;
127    }
128
129
130    
131    /** The Ptolemy type for top-level ports of DDP pattern actors. */
132    public final static Type keyValueArrayType = createKeyValueArrayType(BaseType.GENERAL, BaseType.GENERAL);
133    
134    /** Regular expression to match the array type and capture the
135     *  element type.
136     */
137    private final static Pattern ARRAY_TYPE_PATTERN = Pattern.compile("arrayType\\((.*)\\)");
138    
139    /** Regular expression to match the object type and capture the
140     *  element type.
141     */   
142    private final static Pattern OBJECT_TYPE_PATTERN = Pattern.compile("^object$|object\\((.*)\\)");
143
144}
