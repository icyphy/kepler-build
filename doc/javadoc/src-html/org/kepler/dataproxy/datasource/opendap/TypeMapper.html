084        /**
085         * Maps a DAP2 DDS object to a list of kepler data types.
086         * 
087         * @param dds
088         *            The DDS to map to kepler.
089         * @param addMetadataRecord
090         *            controls whether or not dap Attribute metadata is added to the
091         *            returned kepler records.
092         * @return A collection (<code>Vector</code>) of kepler data types that
093         *         represent the top level variables in the passed DDS.
094         * @throws IllegalActionException
095         *             When bad things happen.
096         */
097        public static Vector mapDDSToTypeList(DDS dds, boolean addMetadataRecord)
098                        throws IllegalActionException {
099
100                Vector<Type> types = new Vector<Type>();
101
102                Enumeration e = dds.getVariables();
103                // Map each top level variable in the DDS to a kepler data type.
104                while (e.hasMoreElements()) {
105                        types.add(mapDapObjectToType(
106                                        (opendap.dap.BaseType) e.nextElement(), addMetadataRecord));
107                }
108
109                return types;
110        }

112        /**
113         * Maps a DAP2 variable (which is represented as an
114         * <code>opendap.dap.BaseType
115     * </code> to a kepler data Type.
116         * 
117         * @param dapBT
118         *            The DAP2 variable to map.
119         * @param addMetadataRecord
120         *            controls whether or not dap Attribute metadata is added to the
121         *            returned kepler records.
122         * @return The corresponding Kepler Type
123         * @throws IllegalActionException
124         *             When bad things happen.
125         */
126        public static Type mapDapObjectToType(opendap.dap.BaseType dapBT,
127                        boolean addMetadataRecord) throws IllegalActionException {
128
129                Type valueType = null;
130                Type type;
131                Type attType;
132
133                if (dapBT instanceof DConstructor) {
134                        type = typemapDConstructorToRecord((DConstructor) dapBT,
135                                        addMetadataRecord);
136
137                } else {
138                        if (dapBT instanceof DByte) {
139                                valueType = ptolemy.data.type.BaseType.UNSIGNED_BYTE;
140
141                        } else if (dapBT instanceof DUInt16 || dapBT instanceof DInt16) {
142                                valueType = ptolemy.data.type.BaseType.INT;
143
144                        } else if (dapBT instanceof DUInt32 || dapBT instanceof DInt32) {
145                                valueType = ptolemy.data.type.BaseType.LONG;
146
147                        } else if (dapBT instanceof DFloat32 || dapBT instanceof DFloat64) {
148                                valueType = ptolemy.data.type.BaseType.DOUBLE;
149
150                        } else if (dapBT instanceof DURL || dapBT instanceof DString) {
151                                valueType = ptolemy.data.type.BaseType.STRING;
152
153                        }
154                        // Arrays are special...
155                        else if (dapBT instanceof DArray) {
156
157                                DArray a = (DArray) dapBT;
158
159                                int dimensions = a.numDimensions();
160
161                                switch (dimensions) {
162
163                                case 1: // Map 1D array to 1xN matrix
164                                        valueType = typemapDArrayToMatrix((DArray) dapBT);
165                                        break;
166
167                                case 2: // Map 2D array to MxN matrix
168                                        valueType = typemapDArrayToMatrix((DArray) dapBT);
169                                        break;
170
171                                default: // punt. Just map it to a Kepler Array (very
172                                                        // inefficient)
173                                        valueType = typemapDArrayToArray((DArray) dapBT);
174
175                                }
176
177                        }
178                        if (valueType == null)
179                                throw new IllegalActionException("Unrecognized DAP2 type. "
180                                                + "TypeName=" + dapBT.getTypeName() + "  VariableName="
181                                                + dapBT.getName());
182
183                        if (addMetadataRecord) {
184                                attType = AttTypeMapper.convertAttributeToType(dapBT
185                                                .getAttribute());
186                                type = new RecordType(new String[] { metadataName, valueName },
187                                                new Type[] { attType, valueType });
188                        } else
189                                type = valueType;
190                }
191
192                if (type == null)
193                        throw new IllegalActionException("Unrecognized DAP2 type. "
194                                        + "TypeName=" + dapBT.getTypeName() + "  VariableName="
195                                        + dapBT.getName());
196
197                return type;
198
199        }

201        /**
202         * Maps a DAP2 DConstructor to a Kepler Record Token. All the DAP2 types
203         * that inherit from DConstructor are composite types. I.e. they are
204         * composed of a collection of other types. Specifically:
205         * <ul>
206         * <li>DStructure</li>
207         * <li>DSequence</li>
208         * <li>DGrid</li>
209         * </ul>
210         * All contain one or more DAP2 types.
211         * 
212         * @param dc
213         *            The DConstructor to map.
214         * @param addMetadataRecord
215         *            controls whether or not dap Attribute metadata is added to the
216         *            returned kepler records.
217         * @return The Kepler Record Token.
218         * @throws IllegalActionException
219         *             When bad things happen.
220         */
221        private static Type typemapDConstructorToRecord(DConstructor dc,
222                        boolean addMetadataRecord) throws IllegalActionException {
223
224                // Make a handy Vector of the members of the collection.
225                Vector<BaseType> v = new Vector<BaseType>();
226                Enumeration e = dc.getVariables();
227                while (e.hasMoreElements())
228                        v.add((BaseType) e.nextElement());
229
230                Type[] types = new Type[v.size()];
231                String[] labels = new String[v.size()];
232                Type type, attType, valueType;
233
234                int i = 0;
235
236                // Map the collection.
237                for (BaseType bt : v) {
238
239                        // Map this DAP2 Type.
240                        types[i] = mapDapObjectToType(bt, addMetadataRecord);
241                        labels[i] = replacePeriods(bt.getName());
242                        i++;
243                }
244
245                // Sequences are special - essential arrays of structures, so they map
246                // to Arrays of Records.
247                if (dc instanceof DSequence)
248                        valueType = new ArrayType(new RecordType(labels, types));
249                else
250                        valueType = new RecordType(labels, types);
251
252                if (addMetadataRecord) {
253                        attType = AttTypeMapper.convertAttributeToType(dc.getAttribute());
254                        type = new RecordType(new String[] { metadataName, valueName },
255                                        new Type[] { attType, valueType });
256                } else
257                        type = valueType;
258
259                return type;
260        }

262        /**
263         * Attempts to map a 1 or2 dimensional DAP2 Array to MatrixType. Not all
264         * types can be mapped. Those that can't get mapped to an ArrayType.
265         * 
266         * @param a
267         *            The array to map.
268         * @return The Matrix type to which it's mapped.
269         * @throws IllegalActionException
270         *             When bad things happen.
271         */
272        private static Type typemapDArrayToMatrix(opendap.dap.DArray a)
273                        throws IllegalActionException {
274
275                if (a.numDimensions() != 2 && a.numDimensions() != 1) {
276                        throw new IllegalActionException("Cannot map a "
277                                        + a.numDimensions() + " array to a matrix.");
278                }
279
280                opendap.dap.BaseType dapBT = a.getPrimitiveVector().getTemplate();
281                if (dapBT instanceof DByte || dapBT instanceof DUInt16
282                                || dapBT instanceof DInt16) {
283
284                        return ptolemy.data.type.BaseType.INT_MATRIX;
285
286                } else if (dapBT instanceof DUInt32 || dapBT instanceof DInt32) {
287                        return ptolemy.data.type.BaseType.LONG_MATRIX;
288
289                } else if (dapBT instanceof DFloat32 || dapBT instanceof DFloat64) {
290                        return ptolemy.data.type.BaseType.DOUBLE_MATRIX;
291
292                } else if (dapBT instanceof DURL || dapBT instanceof DString) {
293                        return new ArrayType(ptolemy.data.type.BaseType.STRING);
294
295                } else if (dapBT instanceof DStructure || dapBT instanceof DGrid
296                                || dapBT instanceof DSequence) {
297
298                        return new ArrayType(typemapDConstructorToRecord(
299                                        (DConstructor) dapBT, false));
300
301                }
302
303                return ptolemy.data.type.BaseType.UNKNOWN;
304        }
305
306        /**
307         * Maps a DAP2 Array to Kepler ArrayType.
308         * 
309         * @param a
310         *            The array to map.
311         * @return The Matrix type to which it's mapped.
312         * @throws IllegalActionException
313         *             When bad things happen.
314         */
315        private static ArrayType typemapDArrayToArray(opendap.dap.DArray a)
316                        throws IllegalActionException {
317
318                Type primitiveType = ptolemy.data.type.BaseType.UNKNOWN;
319
320                opendap.dap.BaseType dapBT = a.getPrimitiveVector().getTemplate();
321                if (dapBT instanceof DByte) {
322                        primitiveType = ptolemy.data.type.BaseType.UNSIGNED_BYTE;
323
324                } else if (dapBT instanceof DUInt16 || dapBT instanceof DInt16) {
325                        primitiveType = ptolemy.data.type.BaseType.INT;
326
327                } else if (dapBT instanceof DUInt32 || dapBT instanceof DInt32) {
328                        primitiveType = ptolemy.data.type.BaseType.LONG;
329
330                } else if (dapBT instanceof DFloat32 || dapBT instanceof DFloat64) {
331                        primitiveType = ptolemy.data.type.BaseType.DOUBLE;
332
333                } else if (dapBT instanceof DURL || dapBT instanceof DString) {
334                        primitiveType = ptolemy.data.type.BaseType.STRING;
335
336                } else if (dapBT instanceof DStructure || dapBT instanceof DGrid
337                                || dapBT instanceof DSequence) {
338
339                        primitiveType = typemapDConstructorToRecord((DConstructor) dapBT,
340                                        false);
341
342                }
343
344                ArrayType at = new ArrayType(primitiveType);
345                for (int i = 1; i < a.numDimensions(); i++) {
346                        at = new ArrayType(at);
347
348                }
349
350                return at;
351        }
352
353        public static String replacePeriods(String str) {
354                return str.replaceAll("\\.", "_");
355        }
