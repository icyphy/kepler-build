030package org.kepler.actor;
031
032import java.util.LinkedList;
033import java.util.List;
034
035import ptolemy.actor.TypedAtomicActor;
036import ptolemy.actor.TypedIOPort;
037import ptolemy.actor.parameters.PortParameter;
038import ptolemy.data.ArrayToken;
039import ptolemy.data.BooleanToken;
040import ptolemy.data.IntToken;
041import ptolemy.data.ScalarToken;
042import ptolemy.data.Token;
043import ptolemy.data.expr.Parameter;
044import ptolemy.data.type.ArrayType;
045import ptolemy.data.type.BaseType;
046import ptolemy.kernel.CompositeEntity;
047import ptolemy.kernel.util.IllegalActionException;
048import ptolemy.kernel.util.NameDuplicationException;
049
050/**
051
052   Filter elements from an array.
053
054   @author Daniel Crawl
055   @version $Id: ArrayFilter.java 24234 2010-05-06 05:21:26Z welker $
056
057 */
058
059public class ArrayFilter extends TypedAtomicActor 
060{
061
062   /** Construct an ArrayFilter with the given container and name. */
063    public ArrayFilter(CompositeEntity container, String name)
064        throws NameDuplicationException, IllegalActionException  
065    {
066        super(container, name);
067    
068        input = new TypedIOPort(this, "input", true, false);
069        input.setTypeAtLeast(ArrayType.ARRAY_UNSIZED_BOTTOM);
070
071        // XXX i'd prefer to have min or max be optional and
072        // not have defaults. however, this would require more
073        // complicated methods to set the type.
074
075        min = new PortParameter(this, "min");
076        min.setTypeAtMost(BaseType.SCALAR);
077        min.getPort().setTypeAtMost(BaseType.SCALAR);
078        min.setExpression("-9999999");
079
080        max = new PortParameter(this, "max");
081        max.setTypeAtMost(BaseType.SCALAR);
082        max.getPort().setTypeAtMost(BaseType.SCALAR);
083        max.setExpression("9999999");
084
085        getIndices = new Parameter(this, "getIndices");
086        getIndices.setToken(BooleanToken.FALSE);
087        getIndices.setTypeEquals(BaseType.BOOLEAN);
088
089        output = new TypedIOPort(this, "output", false, true);
090
091        _attachText("_iconDescription", "<svg>\n" +
092                "<rect x=\"0\" y=\"0\" "
093                + "width=\"60\" height=\"20\" "
094                + "style=\"fill:white\"/>\n" +
095                "</svg>\n");
096    }
097
098    ///////////////////////////////////////////////////////////////////
099    ////                     ports and parameters                  ////
100
101    /** The array to filter. */
102    public TypedIOPort input;
103
104    /** Remove elements below this value. */
105    public PortParameter max;
106    
107    /** Remove elements above this value. */
108    public PortParameter min;
109
110    /** If true, return the <b>indices</b> of the array values within
111     *  (or equals) min and max. If false (default), return the
112     *  <b>values</b> of the array within (or equals) min and max.
113     */
114    public Parameter getIndices;
115
116    /** The filtered array. */
117    public TypedIOPort output;
118
119    ///////////////////////////////////////////////////////////////////
120    ////                         public methods                    ////
121
122    /** Read getIndices and set the output type. */
123    public void preinitialize() throws IllegalActionException
124    {
125        super.preinitialize();
126
127        _wantIndices = ((BooleanToken)getIndices.getToken()).booleanValue();
128
129        // if we output indices, output type is array of integers.
130        if(_wantIndices)
131        {
132            output.setTypeEquals(new ArrayType(BaseType.INT));
133        }
134        // otherwise output type is array of input type.
135        else
136        {
137            output.setTypeSameAs(input);
138        }
139    }
140
141    /** Filter elements in the input array.  */
142    public void fire() throws IllegalActionException 
143    {
144        super.fire();
145   
146        ArrayToken array = (ArrayToken)input.get(0);
147        
148        max.update();
149        Token token = max.getToken();
150
151        if(! (token instanceof ScalarToken))
152        {
153            throw new IllegalActionException(this, "max must have scalar " +
154                "value; given value: " + token); 
155        }
156
157        ScalarToken maxToken = (ScalarToken)token;
158        
159        min.update();
160        token = min.getToken();
161
162        if(! (token instanceof ScalarToken))
163        {
164            throw new IllegalActionException(this, "min must have scalar " +
165                "value; given value: " + token);
166        }
167
168        ScalarToken minToken = (ScalarToken)min.getToken();
169        
170        List<Token> outputList = new LinkedList<Token>();
171
172        for(int i = 0; i < array.length(); i++)
173        {
174            ScalarToken curToken = (ScalarToken)array.getElement(i);
175           
176            // see if current element is within thresholds
177            if((curToken.isLessThan(maxToken) == BooleanToken.TRUE ||
178                curToken.isEqualTo(maxToken) == BooleanToken.TRUE) &&
179                (curToken.isGreaterThan(minToken) == BooleanToken.TRUE ||
180                curToken.isEqualTo(minToken) == BooleanToken.TRUE))
181            {
182                // see if output should be values or indices
183                if(_wantIndices)
184                {
185                    outputList.add(new IntToken(i));
186                }
187                else
188                {
189                    outputList.add(curToken); 
190                }
191            }
192        }
193
194        if(outputList.size() == 0)
195        {
196            output.broadcast(new ArrayToken(input.getType())); 
197        }
198        else
199        {
200            output.broadcast(new ArrayToken(outputList.toArray(new Token[0])));
201        }
202    }
203
204    /** The value of the getIndices parameter. */
205    private boolean _wantIndices;
206}
