062/**
063 * <p>
064 * Actor for directory listing.
065 * <ul>
066 * <li>local or remote dir (using ssh)</li>
067 * <li>get new files from previous call</li>
068 * </ul>
069 * 
070 * </p>
071 * <p>
072 * This actor uses org.kepler.io.DirectoryListing class to get file listing
073 * 
074 * </p>
075 * <p>
076 * The initial input should be:
077 * <ul>
078 * <li>the target machine, either null, "" or "local" to denote the local
079 * machine to be used by Java I/O commands, OR "[user@]host[:port]" to denote a
080 * remote machine to be used by an ssh connection.</li>
081 * <li>the directory to be listed as String</li>
082 * <li>the file mask for listing files according to the pattern only, "" means
083 * all files</li>
084 * </ul>
085 * 
086 * </p>
087 * <p>
088 * The actor takes the target and directory arguments only once! After that it
089 * works with that setting at each fire. The mask can be reset to something else
090 * for each firing but in such a case, the previous listings are deleted (as if
091 * starting fresh).
092 * 
093 * </p>
094 * <p>
095 * Depending on the boolean parameter 'newFilesOnly' there are two different
096 * behavior: <br/>
097 * If 'newFilesOnly' is set (default), for each trigger, the file list is
098 * updated, and the array of 'new' files will be the output. 'New' means the
099 * difference between the current and the previous listings. <br/>
100 * If 'newFilesOnly' is not set, for each trigger, the whole list is returned.
101 * This can be used for watching a specific file and look for changes in size or
102 * access time of the file.
103 * 
104 * </p>
105 * <p>
106 * Depending on the boolean parameter 'checkSizeAndDate' the meaning 'new' is
107 * handled as following: <br/>
108 * If 'checkSizeAndDate' is set, modified files are also listed besides brand
109 * new files. I.e., a new file is that is not found in previous listing OR the
110 * size or date of the file has changed between the previous and current
111 * listings. <br/>
112 * If 'checkSizeAndDate' is not set (default), only brand new files are listed.
113 * I.e. only the file names are checked, and file modifications have no effect.
114 * 
115 * </p>
116 * <p>
117 * The files are listed in the following format: Each element in the array will
118 * be a RecordToken: {name=String, size=long, date=long}, where the size is
119 * given in bytes, the date is given in UTC seconds. The resolution depends on
120 * the 'ls -l' output, i.e. at most a minute, and for old files a day. In the
121 * local case, it depends on the resolution of Java and the local OS, usually
122 * less than a millisecond.
123 * 
124 * </p>
125 * <p>
126 * The underlying command is 'ls -l' to list a directory for all masks on a remote machine. 
127 * In case of a parallel file system 
128 * and a directory of thousands of files, this can be a very costly operation. If you do 
129 * not need the date and size information, check 'useLsOnly'. This will use the 'ls' command
130 * to list a directory and both size and date will be set to -1. If 'checkSizeAndDate' is 
131 * set, then do not set this flag because size and date values cannot be compared with
132 * previous listing.
133 * This operation should be much less costly than listing with 'ls -l'.
134 * Note, that if you are looking for modified files (checkSizeAndDate) and you have a 
135 * ? or * in the mask, the 'ls -l &lt;mask&gt;' command will ls the whole directory and
136 * thus can be costly. 
137 * 
138 * </p>
139 * <p>
140 * On error, an empty string is returned.
141 * 
142 * </p>
143 * <p>
144 * Note: the trigger port is needed only if you do not connect any of the other
145 * ports.
146 * </p>
147 * 
148 * @author Norbert Podhorszki
149 * @version $Id: SshDirectoryList_v1_1.java 14369 2008-03-12 22:11:35Z podhorsz$
150 * @since Ptolemy II 5.0.1
151 */
152public class SshDirectoryList_v1_1 extends TypedAtomicActor {
153        /**
154         * Construct an actor with the given container and name.
155         * 
156         * @param container
157         *            The container.
158         * @param name
159         *            The name of this actor.
160         * @exception IllegalActionException
161         *                If the actor cannot be contained by the proposed
162         *                container.
163         * @exception NameDuplicationException
164         *                If the container already has an actor with this name.
165         */
166        public SshDirectoryList_v1_1(CompositeEntity container, String name)
167                        throws NameDuplicationException, IllegalActionException {
168                super(container, name);
169
170                // Uncomment the next line to see debugging statements
171                // addDebugListener(new ptolemy.kernel.util.StreamListener());
172
173                // target selects the machine where the directory is to be accessed
174                target = new PortParameter(this, "target", new StringToken(
175                                "[ local | [user@]host[:port] ]"));
176                new Parameter(target.getPort(), "_showName", BooleanToken.TRUE);
177
178                // dir is the path to the directory to be listed on the target machine
179                dir = new PortParameter(this, "dir", new StringToken("/path/to/dir"));
180                new Parameter(dir.getPort(), "_showName", BooleanToken.TRUE);
181
182                // masks are the file masks for the files to be listed
183                masks = new PortParameter(this, "masks", new ArrayToken("{'*'}"));
184                new Parameter(masks.getPort(), "_showName", BooleanToken.TRUE);
185
186                // a trigger to read it again
187                trigger = new TypedIOPort(this, "trigger", true, false);
188                trigger.setTypeEquals(BaseType.UNKNOWN);
189                new Parameter(trigger, "_showName", BooleanToken.FALSE);
190
191                // new files only or all?
192                newFilesOnly = new Parameter(this, "newFilesOnly", new BooleanToken(true));
193                newFilesOnly.setTypeEquals(BaseType.BOOLEAN);
194
195                // check file changes in size and date? Or just look for brand new
196                // files?
197                checkSizeAndDate = new Parameter(this, "checkSizeAndDate",
198                                new BooleanToken(false));
199                checkSizeAndDate.setTypeEquals(BaseType.BOOLEAN);
200
201                // output empty array if nothing found?
202                sendEmpty = new Parameter(this, "sendEmpty", new BooleanToken(true));
203                sendEmpty.setTypeEquals(BaseType.BOOLEAN);
204
205                // Delay the listing with one step?
206                delay = new Parameter(this, "delay", new BooleanToken(false));
207                delay.setTypeEquals(BaseType.BOOLEAN);
208
209                // use 'ls -l' or just ls to be faster
210                useLsOnly = new Parameter(this, "useLsOnly", new BooleanToken(false));
211                useLsOnly.setTypeEquals(BaseType.BOOLEAN);
212
213                // Frequency of checking
214                frequencySec = new Parameter(this, "frequencySec", new IntToken(0));
215                frequencySec.setTypeEquals(BaseType.INT);
216
217                // stop mask is the file masks for which this actor should stop when
218                // found
219                stopmask = new PortParameter(this, "stopmask", new StringToken(""));
220                new Parameter(stopmask.getPort(), "_showName", BooleanToken.TRUE);
221
222                // the output: an array of filenames
223                newFiles = new TypedIOPort(this, "newFiles", false, true);
224                String[] labels = { "name", "size", "date" };
225                Type[] ctypes = { BaseType.STRING, BaseType.LONG, BaseType.LONG };
226                _etype = new RecordType(labels, ctypes);
227                newFiles.setTypeEquals(new ArrayType(_etype));
228                // newFiles.setTypeAtMost(new RecordType(new String[0], new Type[0]));
229                // newFiles.setTypeEquals(new ArrayType(BaseType.UNKNOWN));
230                // newFiles.setTypeEquals(new BaseType.UNKNOWN);
231                new Parameter(newFiles, "_showName", BooleanToken.FALSE);
232        }

266        public Parameter newFilesOnly;
267

319        /**
320         * initialize() runs once before first exec
321         * 
322         * @exception IllegalActionException
323         *                If the parent class throws it.
324         */
325        public void initialize() throws IllegalActionException {
326                super.initialize();
327                _firstFire = true;
328        }
329
330        /**
331         * fire
332         * 
333         * @exception IllegalActionException
334         */
335        public void fire() throws IllegalActionException {
336                super.fire();
337
338                // consume the trigger token
339                if (trigger.getWidth() > 0) {
340                        // if (!((BooleanToken) trigger.get(0)).equals(null)) {
341                        trigger.get(0);
342                        // log.debug("consume the token at the trigger port.");
343                        // }
344                }
345
346                // save previous values
347                _prevtarget = _target;
348                _prevdir = _dir;
349                _prevmasks = _masks;
350                _prevstopmask = _stopmask;
351
352                // update PortParameters
353                target.update();
354                dir.update();
355                masks.update();
356                stopmask.update();
357                _target = ((StringToken) target.getToken()).stringValue().trim();
358                _dir = ((StringToken) dir.getToken()).stringValue().trim();
359                _stopmask = ((StringToken) stopmask.getToken()).stringValue().trim();
360
361                // process array of masks
362                ArrayToken maskTokens = (ArrayToken) masks.getToken();
363                if (maskTokens.length() >= 1) {
364                        _masks = new String[maskTokens.length()];
365                        for (int i = 0; i < maskTokens.length(); i++) {
366                                _masks[i] = ((StringToken) maskTokens.getElement(i))
367                                                .stringValue().trim();
368                                log.debug(this.getDisplayName() + ": Mask added: " + _masks[i]);
369                        }
370                }
371
372                FileInfo[] _newFiles = null;
373
374                // if target or dir changes, we have to create a new DirectoryListing
375                // object
376                if (_firstFire || !_prevtarget.equals(_target)
377                                || !_prevdir.equals(_dir)) {
378
379                        _newOnly = ((BooleanToken) newFilesOnly.getToken()).booleanValue();
380                        _checkModifications = ((BooleanToken) checkSizeAndDate.getToken())
381                                        .booleanValue();
382                        _freq = ((IntToken) frequencySec.getToken()).intValue();
383                        _sendEmpty = ((BooleanToken) sendEmpty.getToken()).booleanValue();
384                        _useLsOnly = ((BooleanToken) useLsOnly.getToken()).booleanValue();
385                        _delay = ((BooleanToken) delay.getToken()).booleanValue();
386
387                        // constraint: delay & sendEmpty = False
388                        if (_delay && _sendEmpty) {
389                                log
390                                                .error(this.getFullName()
391                                                                + ": Both sendEmpty and delay is set to true, which is not allowed. Delay will be considered as true, sendEmpty as false.");
392                                _sendEmpty = false;
393                        }
394
395                        // set initial values
396                        _delayedList = null;
397
398                        if (isDebugging)
399                                log.debug(this.getDisplayName()
400                                                + ": Create DirectoryListing object: " + "target = "
401                                                + _target + "; dir = " + _dir + "; filemasks = "
402                                                + _masks + "; newFilesOnly = " + _newOnly
403                                                + "; checkSizeAndDate = " + _checkModifications);
404
405                        _dl = new DirectoryListing(_target, _dir, _masks);
406                        if (_stopmask != null && !_stopmask.equals("")) {
407                                String[] sa = { _stopmask };
408                                _dl_stop = new DirectoryListing(_target, _dir, sa);
409                        }
410                        _firstFire = false;
411
412                } else {
413                        if (!Arrays.equals(_masks, _prevmasks)) { // masks can be reset
414                                _dl.setMask(_masks);
415                                _prevmasks = _masks;
416                        }
417                        if (!_stopmask.equals(_prevstopmask)) { // stopmask can be reset
418                                String[] sa = { _stopmask };
419                                _dl_stop.setMask(sa);
420                                _prevstopmask = _stopmask;
421                        }
422                }
423
424                // List directory in a loop if frequency is set, until a stop file is
425                // found
426                boolean doloop = true;
427                while (doloop) {
428
429                        try {
430                                // list directory for the given masks
431                                int n = _dl.list(_useLsOnly);
432                                if (isDebugging)
433                                        log.debug(this.getDisplayName()
434                                                        + ": Number of total files = " + n);
435                                if (n > 0) {
436                                        // get new files
437                                        if (_newOnly)
438                                                _newFiles = _dl.getNewFiles(_checkModifications);
439                                        else
440                                                _newFiles = _dl.getList();
441
442                                        if (isDebugging || _newFiles.length > 0)
443                                                log.info(this.getDisplayName() + ": Number of "
444                                                                + (_newOnly ? "NEW" : "") + " files = "
445                                                                + _newFiles.length);
446                                }
447
448                        } catch (ExecException ex) {
449                                log
450                                                .error(this.getFullName()
451                                                                + ": SshDirectoryList error at remote directory reading. "
452                                                                + ex);
453                        }
454
455                        // create the result and send
456                        if (_newFiles != null && _newFiles.length > 0) {
457                                RecordToken[] rt = new RecordToken[_newFiles.length];
458                                for (int i = 0; i < _newFiles.length; i++) {
459                                        rt[i] = createRecordToken(_newFiles[i]);
460                                }
461                                if (_delay) { // postpone the current list and send the previous
462                                                                // if there is such
463                                        if (_delayedList != null) {
464                                                newFiles.send(0, new ArrayToken(_etype, _delayedList));
465                                                log.debug(this.getDisplayName()
466                                                                + ": Send delayed list: " + _delayedList);
467                                        }
468                                        _delayedList = rt;
469                                } else { // send the new list
470                                        newFiles.send(0, new ArrayToken(_etype, rt));
471                                        log.debug(this.getDisplayName() + ": Found new files: "
472                                                        + rt);
473                                }
474                        } else if (_sendEmpty) { // send an empty array of RecordTokens
475                                newFiles.send(0, new ArrayToken(_etype, new RecordToken[0]));
476                                log.debug(this.getDisplayName()
477                                                + ": Found no files at this time.");
478                        }
479
480                        if (_freq == 0)
481                                doloop = false; // exit from loop if we need to run once
482                        else { // check for the stop files and then sleep if not found
483                                        // anything
484                                try {
485                                        // list directory for stop files
486                                        int n = _dl_stop.list();
487                                        if (isDebugging)
488                                                log.debug(this.getDisplayName()
489                                                                + ": Number of total stop files = " + n);
490                                        if (n > 0) {
491                                                doloop = false; // exit from loop if found any stop file
492                                                _newFiles = _dl_stop.getList(); // we have to send them
493                                                                                                                // out
494                                                RecordToken[] rt = new RecordToken[_newFiles.length];
495                                                for (int i = 0; i < _newFiles.length; i++) {
496                                                        rt[i] = createRecordToken(_newFiles[i]);
497                                                }
498                                                if (_delayedList != null) { // we have to send out the
499                                                                                                        // stored list in delayed
500                                                                                                        // mode
501                                                        newFiles.send(0, new ArrayToken(_etype,
502                                                                        _delayedList));
503                                                        log.debug(this.getDisplayName()
504                                                                        + ": Send delayed list: " + _delayedList);
505                                                }
506                                                newFiles.send(0, new ArrayToken(_etype, rt));
507                                                log.debug(this.getDisplayName()
508                                                                + ": Found stop files: " + _newFiles);
509                                        }
510                                } catch (ExecException ex) {
511                                        log
512                                                        .error(this.getFullName()
513                                                                        + ": SshDirectoryList error at remote directory reading. "
514                                                                        + ex);
515                                }
516
517                                if (doloop) // if we are still looping, let sleep a bit
518                                        try {
519                                                java.lang.Thread.sleep(_freq * 1000L);
520                                        } catch (InterruptedException ex) {
521                                        }
522                        }
523
524                } // end while loop
525
526        }
527
528        /*
529         * Create one RecordToken of format {name=String, size=long, date=long} from
530         * the FileInfo struct.
531         */
532        private RecordToken createRecordToken(FileInfo fi) {
533                String[] labels = { "name", "size", "date" };
534                Token[] values = new Token[3];
535                values[0] = new StringToken(fi.getName());
536                values[1] = new LongToken(fi.getSize());
537                values[2] = new LongToken(fi.getDate());
538                RecordToken rt = null;
539                try {
540                        rt = new RecordToken(labels, values);
541                } catch (IllegalActionException ex) {
542                        log
543                                        .error(this.getFullName()
544                                                        + ": SshDirectoryList: Error at creating a record token for fileinfo: "
545                                                        + fi + "\nlabels = " + labels + "\nvalues = "
546                                                        + values);
547                }
548                return rt;
549        }

555        private String _stopmask;

558        private String[] _prevmasks;
559        private String _prevstopmask;
560        private boolean _newOnly;
561        private boolean _checkModifications;
562        private int _freq;
563        private boolean _sendEmpty;
564        private boolean _useLsOnly;
565        private boolean _delay;
566        private RecordToken[] _delayedList;
567        private DirectoryListing _dl; // the major directory listing object
568        private DirectoryListing _dl_stop; // separate listing for the stop mask
569        private Type _etype;
