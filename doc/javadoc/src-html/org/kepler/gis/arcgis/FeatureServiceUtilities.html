029package org.kepler.gis.arcgis;
030
031import org.kepler.webview.server.WebViewServer;
032
033import io.vertx.core.AsyncResult;
034import io.vertx.core.Future;
035import io.vertx.core.Handler;
036import io.vertx.core.MultiMap;
037import io.vertx.core.json.JsonObject;
038import io.vertx.ext.web.client.WebClient;
039
040/** Utilities for working with ArcGIS FeatureService.
041 * 
042 *  @author Daniel Crawl
043 *  @version $Id: FeatureServiceUtilities.java 34711 2018-10-01 17:31:07Z crawl $ 
044 */
045public class FeatureServiceUtilities {
046
047    /** This class cannot be instantiated. */
048    private FeatureServiceUtilities() {}
049    
050    /** Add features to a feature service.
051     *  @param jsonStr The features to add in esri json format.
052     *  @param featureServiceUrlStr The url of the layer in the feature service.
053     *  @param cliendIdStr The client id of the oath app.
054     *  @param clientSecretStr The client secret of the oath app.
055     *  @param handler Asynchronous handler for the result. 
056     */
057    public static void addFeatures(String jsonStr,
058        String featureServiceUrlStr, String clientIdStr, String clientSecretStr,
059        Handler<AsyncResult<String>> handler) {
060                
061        FeatureServiceUtilities.getAccessToken(clientIdStr, clientSecretStr,
062            tokenResult -> {
063                if(tokenResult.failed()) {
064                    handler.handle(Future.failedFuture(tokenResult.cause()));
065                } else {
066                    _addFeatures(tokenResult.result(), jsonStr, featureServiceUrlStr, handler);                   
067                }
068            });                                
069    }
070    
071    /** Get an access token for an oath app. */
072    public static void getAccessToken(String clientIdStr, String clientSecretStr,
073        Handler<AsyncResult<String>> handler) {
074               
075        MultiMap form = MultiMap.caseInsensitiveMultiMap()
076            .add("client_id", clientIdStr)
077            .add("client_secret", clientSecretStr)
078            .add("grant_type", "client_credentials");
079
080        WebClient client = WebClient.create(WebViewServer.vertx());
081        client.postAbs(_oathUrlStr)
082            .sendForm(form, result -> {
083                //System.out.println(result.result().bodyAsString());
084                if(result.failed()) {
085                    handler.handle(Future.failedFuture(result.cause()));                   
086                } else {
087                    JsonObject json = result.result().bodyAsJsonObject();
088                    if(!json.containsKey("access_token")) {
089                        handler.handle(Future.failedFuture("Missing access_token in response."));
090                    } else {
091                        handler.handle(Future.succeededFuture(json.getString("access_token")));
092                    }
093                }
094            });        
095    }
096
097    /** Add features to a service. */
098    private static void _addFeatures(String tokenStr, String jsonStr,
099            String featureServiceUrlStr, Handler<AsyncResult<String>> handler) {
100        
101        MultiMap form = MultiMap.caseInsensitiveMultiMap()
102            // get response in json
103            .add("f", "json")
104            .add("adds", jsonStr)
105            .add("token", tokenStr);
106        
107        System.out.println(featureServiceUrlStr);
108        System.out.println(jsonStr);
109        
110        WebClient client = WebClient.create(WebViewServer.vertx());
111        client.postAbs(featureServiceUrlStr + "/applyEdits")
112            .sendForm(form, response -> {
113                System.out.println(response.result().bodyAsString());
114                JsonObject responseJson = response.result().bodyAsJsonObject();
115                if(response.failed() || responseJson.containsKey("error")) {
116                    handler.handle(Future.failedFuture(response.cause()));                   
117                } else {
118                    handler.handle(Future.succeededFuture());
119                }
120            });        
121    }
122    
123    /** URL for arcgis online oath tokens. */
124    private final static String _oathUrlStr = "https://www.arcgis.com/sharing/rest/oauth2/token";
125
126}
