070    /** Get a coverage for a raster file.
071     *  @return Returns null if the file is an unknown format.
072     */
073    public static GridCoverage getCoverage(File file, CoordinateReferenceSystem defaultCRS) throws IllegalArgumentException, IOException {
074        AbstractGridCoverage2DReader reader = getReader(file, defaultCRS);
075        if(reader != null) {
076            return reader.read(null);
077        }
078        return null;
079    }
080    
081    /** Get the grid size of a raster.
082     * @return The width and height.
083     */
084    public static Integer[] getGridSize(RasterToken raster) throws IllegalActionException {
085        
086        Integer[] sizes = new Integer[2];
087        
088        File rasterFile = raster.rasterFile();
089        if(rasterFile != null) {
090            throw new IllegalActionException("Getting cell size from file not supported. Use RasterReader to first read the raster.");
091        }
092        
093        GridEnvelope grid = raster.reader().getOriginalGridRange();
094        sizes[0] = grid.getSpan(0);
095        sizes[1] = grid.getSpan(1);
096        
097        return sizes;
098    }

100    /** Get a reader for a raster file.
101     *  @return Returns null if file is unknown format.
102     */
103    public static AbstractGridCoverage2DReader getReader(File file, CoordinateReferenceSystem defaultCRS) {
104        AbstractGridFormat format = GridFormatFinder.findFormat(file);
105        if(!(format instanceof UnknownFormat)) {
106
107            AbstractGridCoverage2DReader reader = format.getReader(file);
108            if(defaultCRS == null) {            
109                reader = format.getReader(file);
110            } else {
111                Hints hints = new Hints();
112                hints.put(Hints.DEFAULT_COORDINATE_REFERENCE_SYSTEM, defaultCRS);
113                reader = format.getReader(file, hints);
114            }
115            
116            if(reader != null) {
117                _readers.add(reader);
118            } else {
119                System.err.println("ERROR: null reader for " + file);
120            }
121            return reader;
122        }
123        return null;
124    }
125    
126    /** Close all readers opened by getReader() and getCoverage(). */
127    public static void closeReaders() {
128        
129        synchronized(_readers) {
130            for(AbstractGridCoverage2DReader reader : _readers) {
131                reader.dispose();
132            }   
133            _readers.clear();
134        }        
135    }

137    /** Get a RasterInfo object for a raster file. This method calls "gdalinfo"
138     *  to find the CRS and bounding box.
139     */
140    public static RasterInfo getInfo(File file) throws IllegalActionException {
141        
142        if(!file.exists()) {
143            throw new IllegalActionException("Raster file does not exist: " + file);
144        }
145        
146        try {
147            ProcessBuilder builder = new ProcessBuilder("gdalinfo", file.getAbsolutePath());
148            builder.redirectErrorStream(true);
149            Process process = builder.start();
150            try(InputStream stdout = process.getInputStream();
151                BufferedReader reader = new BufferedReader(new InputStreamReader(stdout));) {
152                String line;
153                boolean crs = false;
154                boolean upperLeft = false;
155                boolean lowerRight = false;
156                StringBuilder buf = null;
157                double[] minCoords = new double[2];
158                double[] maxCoords = new double[2];
159                RasterInfo info = new RasterInfo();
160                while((line = reader.readLine()) != null) {
161                    
162                    if(line.startsWith("Coordinate System is:")) {
163                        crs = true;
164                        buf = new StringBuilder();
165                    } else if(line.startsWith("Origin = ")) {
166                        crs = false;
167                        if(buf == null) {
168                            System.err.println("WARNING: missing WKT for " + file.getAbsolutePath() +
169                                ". Does .prj file exist?");
170                        } else {
171                            try {
172                                // FIXME buf can be null?
173                                info._crs = CRS.parseWKT(buf.toString());
174                            } catch (FactoryException e) {
175                                System.err.println("WARNING: unable to parse WKT: " + e.getMessage());
176                            }
177                        }
178                    } else if(crs) {
179                        buf.append(line);
180                    } else {
181                        Matcher matcher = UPPER_LEFT_PATTERN.matcher(line);
182                        if(matcher.find()) {
183                            upperLeft = true;
184                            minCoords[0] = Double.parseDouble(matcher.group(1));
185                            maxCoords[1] = Double.parseDouble(matcher.group(2));
186                            continue;
187                        }
188                        matcher = LOWER_RIGHT_PATTERN.matcher(line);
189                        if(matcher.find()) {
190                            lowerRight = true;
191                            maxCoords[0] = Double.parseDouble(matcher.group(1));
192                            minCoords[1] = Double.parseDouble(matcher.group(2));
193                            continue;
194                        }
195                    }
196                }
197                
198                if(upperLeft && lowerRight) {
199                    info._envelope = ReferencedEnvelope.create(
200                            new GeneralEnvelope(minCoords, maxCoords),
201                            info._crs);
202                    //System.out.println("envelope is = " + info._envelope);
203                }
204
205                return info;
206
207            } finally {
208                process.waitFor();
209            }
210            
211        } catch (IOException e) {
212            throw new IllegalActionException("Error executing gdalinfo: " + e.getMessage());
213        } catch (InterruptedException e) {
214            throw new IllegalActionException("Error waiting for gdalinfo to finish: " + e.getMessage());
215        }
216    }

218    /** Reproject a raster file.
219     *  @param rasterFile the raster file.
220     *  @param destCRS the CRS to reproject to.
221     */
222    public static File reproject(File rasterFile, CoordinateReferenceSystem destCRS) throws IllegalActionException {
223        
224        // make sure projection is different
225        RasterInfo info = getInfo(rasterFile);
226        if(info.getCRS() == null) {
227            throw new IllegalActionException("Unknown projection for source raster file.");
228        } else if(info.getCRS().equals(destCRS)) {
229            return rasterFile;
230        } else {
231
232            int projectionNum;
233            try {
234                projectionNum = CRS.lookupEpsgCode(destCRS, false);
235            } catch (FactoryException e) {
236                throw new IllegalActionException("Error looking up EPSG code for destination projectin.");
237            }
238            
239            String srcName = FilenameUtils.getBaseName(rasterFile.getAbsolutePath());
240            
241            // FIXME this assumes input raster is arc ascii
242            
243            String destFileNameVRT = rasterFile.getParent() +
244                File.separator +
245                srcName +
246                "-" +
247                projectionNum +
248                ".vrt";
249            
250            ProcessBuilder builder = new ProcessBuilder("gdalwarp",
251                "-t_srs",
252                "EPSG:" + projectionNum,
253                "-of",
254                "VRT",
255                rasterFile.getAbsolutePath(),
256                destFileNameVRT);
257            
258            builder.redirectErrorStream(true);
259            try {
260                Process process = builder.start();
261                /*
262                try {
263                    try(InputStream stdout = process.getInputStream();
264                        BufferedReader reader = new BufferedReader(new InputStreamReader(stdout));) {
265                        String line;
266                        while((line = reader.readLine()) != null) {
267                            System.out.println(line);
268                        }
269                    }
270                } finally {
271                */
272                    if(process.waitFor() != 0) {
273                        throw new IllegalActionException("gdalwarp did not return 0.");
274                    }
275                //}
276            } catch(IOException | InterruptedException e) {
277                throw new IllegalActionException("Error running gdalwarp: " +
278                        e.getMessage());
279            }
280                            
281            String destFileName = rasterFile.getParent() +
282                File.separator +
283                srcName +
284                "-" +
285                projectionNum +
286                "." +
287                FilenameUtils.getExtension(rasterFile.getAbsolutePath());
288            
289            builder = new ProcessBuilder("gdal_translate",
290                "-of",
291                "AAIGrid",
292                destFileNameVRT,
293                destFileName);
294            
295            builder.redirectErrorStream(true);
296            try {
297                Process process = builder.start();
298                /*
299                try {
300                    try(InputStream stdout = process.getInputStream();
301                        BufferedReader reader = new BufferedReader(new InputStreamReader(stdout));) {
302                        String line;
303                        while((line = reader.readLine()) != null) {
304                            System.out.println(line);
305                        }
306                    }
307                } finally {
308                */
309                    if(process.waitFor() != 0) {
310                        throw new IllegalActionException("gdalwarp did not return 0.");
311                    }
312                //}
313            } catch(IOException | InterruptedException e) {
314                throw new IllegalActionException("Error running gdalwarp: " +
315                        e.getMessage());
316            }
317            return new File(destFileName);
318        }
319    }

321    /** Utility class that contains information about a raster. */
322    public static class RasterInfo {
323        
324        private RasterInfo() {
325        }
326        
327        public CoordinateReferenceSystem getCRS() {
328            return _crs;
329        }
330        
331        public ReferencedEnvelope getEnvelope() {
332            return _envelope;
333        }

335        private CoordinateReferenceSystem _crs;
336        private ReferencedEnvelope _envelope;
337    }

339    /** Regex for upper left of bounding box printed by gdalinfo. */
340    private final static Pattern UPPER_LEFT_PATTERN =
341        Pattern.compile("Upper Left\\s+\\(\\s*(\\-?\\d+\\.\\d+),\\s*(\\-?\\d+\\.\\d+)");
342
343    /** Regex for lower right of bounding box printed by gdalinfo. */
344    private final static Pattern LOWER_RIGHT_PATTERN =
345        Pattern.compile("Lower Right\\s+\\(\\s*(\\-?\\d+\\.\\d+),\\s*(\\-?\\d+\\.\\d+)");
346    
347    /** Set of opened readers. */
348    private static Set<AbstractGridCoverage2DReader> _readers = Collections.synchronizedSet(new HashSet<AbstractGridCoverage2DReader>());
