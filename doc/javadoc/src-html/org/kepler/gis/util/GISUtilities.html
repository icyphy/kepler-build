029package org.kepler.gis.util;
030
031import java.util.Collections;
032import java.util.HashMap;
033import java.util.Map;
034
035import org.geotools.referencing.CRS;
036import org.geotools.referencing.crs.DefaultGeographicCRS;
037import org.opengis.referencing.FactoryException;
038import org.opengis.referencing.crs.CoordinateReferenceSystem;
039
040import ptolemy.kernel.util.IllegalActionException;
041
042/** Misc. GIS utilities.
043 * 
044 *  @author Daniel Crawl
045 *  @version $Id: GISUtilities.java 34525 2016-08-28 02:21:26Z crawl $
046 * 
047 */
048public class GISUtilities {
049
050    /** This class cannot be instantiated. */
051    private GISUtilities() {
052    }
053
054    /** Get the EPSG code from a WKT. */
055    public static Integer getEPSGCode(String wktStr) throws IllegalActionException {
056        
057        CoordinateReferenceSystem crs;
058        try {
059             crs = CRS.parseWKT(wktStr);
060        } catch (FactoryException e) {
061            throw new IllegalActionException("Error parsing WKT: " + e.getMessage());
062        }
063                
064        Integer codeVal = _codeCache.get(wktStr);
065        if(codeVal == null) {
066            try {
067                codeVal = CRS.lookupEpsgCode(crs, true);
068                _codeCache.put(wktStr, codeVal);
069            } catch(FactoryException e) {
070                throw new IllegalActionException("Error looking up EPSG code: " + e.getMessage());
071            }
072        }
073        return codeVal;
074    }
075    
076    /** Check if a projection is WGS84 / EPSG 4326. */
077    public static boolean isCRSWGS84(CoordinateReferenceSystem crs) {
078        return CRS.equalsIgnoreMetadata(crs, DefaultGeographicCRS.WGS84) ||
079                crs.getName().toString().equals("EPSG:WGS 84");
080    }
081    
082    /** A cache for the WKT. */
083    private static final Map<String,Integer> _codeCache =
084        Collections.synchronizedMap(new HashMap<String,Integer>());
085
086}
