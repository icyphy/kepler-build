094/** An actor that writes a vector/feature data set to a KML file.
095 *  An input port must be added for each data set to read. The features
096 *  will be placed in the KML inside a folder with the same name as
097 *  the input port.
098 *  <p>
099 *  If a feature contains a property called "name", then the value
100 *  will be used to specify the name of the KML placemark for that feature.
101 *  </p>
102 *  <p>
103 *  If a feature contains a property called "color", then the value
104 *  will be used to specify the color of the KML placemark for that feature.
105 *  </p>
106 *  <p>
107 *  If a feature contains a property called "timestamp" and a timestamp is not
108 *  supplied to the <i>date</i> port, then the value in the property will be
109 *  used to specify the timestamp of the KML placemark for that feature. The
110 *  format for KML timestamps is: yyyy-MM-dd'T'HH:mmXXX.
111 *  </p>
112 *  
113 *  @author Daniel Crawl
114 *  @version $Id: KMLWriter.java 34503 2016-07-12 06:04:25Z crawl $
115 */
116public class KMLWriter extends TypedAtomicActor {

124        outputType.addChoice("file");
125        outputType.addChoice("text");
126        outputType.addChoice("binary");

134        filename.setToken("output.kml");
135        
136        date = new TypedIOPort(this, "date", true, false);
137        date.setTypeEquals(BaseType.DATE);
138        new SingletonAttribute(date, "_showName");
139
140        folderName = new PortParameter(this, "folderName");
141        folderName.setTypeEquals(BaseType.STRING);
142        folderName.getPort().setTypeEquals(BaseType.STRING);
143        folderName.setStringMode(true);
144        new SingletonAttribute(folderName.getPort(), "_showName");
145
146        fillPolygons = new Parameter(this, "fillPolygons");
147        fillPolygons.setTypeEquals(BaseType.BOOLEAN);
148        fillPolygons.setToken(BooleanToken.TRUE);
149        
150        placemarkNameProperty = new PortParameter(this, "placemarkNameProperty");
151        placemarkNameProperty.setStringMode(true);
152        placemarkNameProperty.setTypeEquals(BaseType.STRING);
153        placemarkNameProperty.getPort().setTypeEquals(BaseType.STRING);
154        new SingletonAttribute(placemarkNameProperty.getPort(), "_showName");

156        output = new TypedIOPort(this, "output", false, true);

159    @Override
160    public void attributeChanged(Attribute attribute) throws IllegalActionException {
161        if(attribute == outputType) {
162            StringToken token = (StringToken) outputType.getToken();
163            if(token != null) {
164                String val = ((StringToken)token).stringValue();
165                if(val.trim().isEmpty()) {
166                    _outputTypeStr = "file";
167                } else if(_outputTypeStr != null && 
168                    !_outputTypeStr.equals("file") &&
169                    !_outputTypeStr.equals("text") &&
170                    !_outputTypeStr.equals("binary")) {
171                    throw new IllegalActionException(this, "Unsupported type of output: " + val);
172                } else {
173                    _outputTypeStr = val;
174                }
175            }            
176        } else {
177            super.attributeChanged(attribute);
178        }
179    }

181    @Override
182    public void fire() throws IllegalActionException {
183        
184        super.fire();
185        
186        filename.update();
187                
188        String filenameStr = null;
189        if(_outputTypeStr.equals("file")) {        
190            filenameStr = ((StringToken)filename.getToken()).stringValue();
191            if(filenameStr.trim().isEmpty()) {
192                throw new IllegalActionException(this, "Must specify output file.");
193            }
194        }
195        
196        Writer writer = null;
197        try {
198            if(_outputTypeStr.equals("file")) {
199                writer = new FileWriter(filenameStr);
200            } else if(_outputTypeStr.equals("text") || _outputTypeStr.equals("binary")) {
201                writer = new StringWriter();
202            } else {
203                throw new IllegalActionException(this, "Unsupported type of output: " + _outputTypeStr);
204            }
205    
206            
207            folderName.update();
208            String folderNameStr = null;
209            StringToken token = (StringToken) folderName.getToken();
210            if(token != null) {
211                folderNameStr = token.stringValue();
212            }
213            
214            String formattedDateStr = null;
215            DateToken dateToken = null;
216            if(date.numberOfSources() > 0) {
217                dateToken = (DateToken)date.get(0);
218    
219                if(dateToken == null) {
220                    //System.out.println("WARNING: no timestamp.");
221                } else {
222                    //String whenStr = String.format("%02d-%02d-%02d", year, mon, day);
223                    
224                    Date dateVal = new Date(dateToken.getValue());
225                    synchronized(WriteTimestampToKML.KML_TIMESTAMP_FORMAT) {
226                        formattedDateStr = WriteTimestampToKML.KML_TIMESTAMP_FORMAT.format(dateVal);   
227                    }
228                }
229            }
230    
231            final Kml kml = new Kml();
232            final Document doc = kml.createAndSetDocument();
233            
234            // only create a top level folder if a name is specified.
235            Folder topFolder = null;
236            if(folderNameStr != null && !folderNameStr.trim().isEmpty()) {
237                topFolder = doc.createAndAddFolder();
238                topFolder.setName(folderNameStr);
239            }
240            
241            // add the features read by each input port into a separate
242            // folder
243            for(TypedIOPort port : inputPortList()) {
244                // do not use the class-specific input ports
245                if(port != date && !(port instanceof ParameterPort)) {
246    
247                    SimpleFeatureCollection features = ((VectorToken)port.get(0)).getVectors();
248                    
249                    // create the folder with the port name.
250                    Folder portFolder;
251                    if(topFolder != null) {
252                        portFolder = topFolder.createAndAddFolder();
253                    } else {
254                        portFolder = doc.createAndAddFolder();
255                    }
256                    portFolder.setName(port.getName());
257                    _addToFolder(portFolder, features, formattedDateStr);                
258                }
259            }
260            
261            //kml.marshal(System.out);
262            kml.marshal(writer);
263                    
264            if(_outputTypeStr.equals("file")) {
265                output.broadcast(new StringToken(filenameStr));
266            } else if(_outputTypeStr.equals("text")) {
267                output.broadcast(new StringToken(writer.toString()));                
268            } else if(_outputTypeStr.equals("binary")) {
269                byte[] bytes = writer.toString().getBytes("UTF-8");
270                Token[] tokens = new Token[bytes.length];
271                for(int i = 0; i < bytes.length; i++) {
272                    tokens[i] = new UnsignedByteToken(bytes[i]);
273                }
274                output.broadcast(new ArrayToken(tokens));
275            }
276            
277        } catch(IOException e) {
278            throw new IllegalActionException(this, e, "Error writing KML.");
279        } finally {
280            if(writer != null) {
281                try {
282                    writer.close();
283                } catch(IOException e) {
284                    throw new IllegalActionException(this, e, "Error writing KML.");
285                }
286            }
287        }
288    }

291    @Override
292    public void preinitialize() throws IllegalActionException {
293        super.preinitialize();
294        
295        for(TypedIOPort port: inputPortList()) {
296            if(port != date && !(port instanceof ParameterPort)) {
297                port.setTypeEquals(VectorToken.VECTOR);
298            }
299        }
300        
301        if(_outputTypeStr.equals("file") || _outputTypeStr.equals("text")) {
302            output.setTypeEquals(BaseType.STRING);
303        } else if(_outputTypeStr.equals("binary")) {
304            output.setTypeEquals(new ArrayType(BaseType.UNSIGNED_BYTE));
305        }
306        
307        Token token = fillPolygons.getToken();
308        if(token == null) {
309            _fillPolygons = true;
310        } else {
311            _fillPolygons = ((BooleanToken)token).booleanValue();
312        }
313        
314        _placemarkNameProperty = null;
315        token = placemarkNameProperty.getToken();
316        if(token != null) {
317            _placemarkNameProperty = ((StringToken)token).stringValue().trim();    
318        }
319    }

358    /** Add a set of features to a folder. */
359    private void _addToFolder(Folder folder, SimpleFeatureCollection features,
360        String formattedDateStr) throws IllegalActionException {
361        
362        SimpleFeatureType currentSchema = features.getSchema();
363        CoordinateReferenceSystem srcCRS = currentSchema.getCoordinateReferenceSystem();       
364        
365        if(srcCRS == null) {
366            //throw new IllegalActionException(this, "Unknown input CRS.");
367            System.out.println("WARNING: Unknown input CRS; assuming WGS84.");
368            srcCRS = DefaultGeographicCRS.WGS84;
369        }
370                
371        MathTransform transform = null;
372        if(!GISUtilities.isCRSWGS84(srcCRS)) {
373            System.out.println(getName() + ": source projection is not WGS84");
374            try {
375                transform = CRS.findMathTransform(srcCRS, DefaultGeographicCRS.WGS84, true);
376            } catch (FactoryException e) {
377                throw new IllegalActionException(this, e,
378                        "Unable to find math transform.");
379            }
380        }
381
382        try(SimpleFeatureIterator iterator = features.features();) {
383        
384            while(iterator.hasNext()) {
385               
386                SimpleFeature feature = iterator.next();
387                
388                Geometry geometry = (Geometry) feature.getDefaultGeometry();
389                Geometry newGeometry = geometry;
390                if(transform != null) {
391                    try {
392                        newGeometry = JTS.transform(geometry, transform);
393                    } catch (MismatchedDimensionException | TransformException e) {
394                        throw new IllegalActionException(this, e,
395                                "Error transforming geometry.");
396                    }
397                }
398
399                Placemark placemark = folder.createAndAddPlacemark();
400                
401                if(newGeometry instanceof MultiPolygon) {
402                
403                    MultiGeometry multigeometry = placemark.createAndSetMultiGeometry();
404                   
405                    for(int i = 0; i < newGeometry.getNumGeometries(); i++) {
406                        Geometry subGeometry = newGeometry.getGeometryN(i);
407                        
408                        if(subGeometry instanceof Polygon) {
409                            _addPolygon((Polygon)subGeometry,
410                                    multigeometry.createAndAddPolygon());
411                        } else if(subGeometry instanceof Point) {
412                            _addPoint((Point)subGeometry,
413                                    multigeometry.createAndAddPoint());
414                        } else if(subGeometry instanceof LineString) {
415                            _addLineString((LineString)subGeometry,
416                                    multigeometry.createAndAddLineString());
417                        } else {
418                            throw new IllegalActionException(this,
419                                    "Unsupported type of geometry: " +
420                                            subGeometry.getClass());
421                        }                       
422                    }
423                } else if(newGeometry instanceof Polygon) {
424                    _addPolygon((Polygon)newGeometry, placemark.createAndSetPolygon());
425                } else if(newGeometry instanceof Point) {
426                    _addPoint((Point)newGeometry, placemark.createAndSetPoint());
427                } else if(newGeometry instanceof LineString) {
428                    _addLineString((LineString)newGeometry,
429                            placemark.createAndSetLineString());
430                } else {
431                    throw new IllegalActionException(this,
432                            "Unsupported type of geometry: " + newGeometry.getClass());
433                }
434
435                String colorStr = "ff0000ff";
436                String curTimeStampStr = null;
437
438                // TODO sort names
439                List<Data> dataList = new LinkedList<Data>();
440                
441                for(Property property : feature.getProperties()) {
442                    String propertyNameStr = property.getName().toString();
443                    Object value = property.getValue();
444                    String valueStr = value.toString();
445                    if(!(value instanceof Geometry) && value != null) {
446                        if(propertyNameStr.toLowerCase().equals("name")) {
447                            placemark.setName(valueStr);
448                        } else {
449                            dataList.add(KmlFactory.createData(valueStr)
450                                    .withName(propertyNameStr));                            
451                        }
452                    }
453                   
454                    if(propertyNameStr.toLowerCase().equals("name") ||
455                        (_placemarkNameProperty != null &&
456                        _placemarkNameProperty.equals(propertyNameStr))) {
457                        placemark.setName(valueStr);
458                    }
459
460                    if(propertyNameStr.equals("color")) {
461                        colorStr = _getColorHex(valueStr);
462                    }
463                    
464                    if(formattedDateStr == null && propertyNameStr.equals("timestamp")) {
465                        curTimeStampStr = valueStr;
466                    }
467
468                }
469                placemark.createAndSetExtendedData().setData(dataList);
470                
471                // TODO choose time
472                /*
473                int year = _parseDateObject(feature.getAttribute("YEAR_"));
474                int mon = _parseDateObject(feature.getAttribute("MONTH_"));
475                if(mon == 0) {
476                    mon = 1;
477                }
478                int day = _parseDateObject(feature.getAttribute("DAY_"));
479                if(day == 0) {
480                    day = 1;
481                }
482                */
483
484                if(formattedDateStr != null) {
485                    placemark.createAndSetTimeStamp().setWhen(formattedDateStr);                    
486                } else if(curTimeStampStr != null) {
487                    placemark.createAndSetTimeStamp().setWhen(curTimeStampStr);
488                }
489                                
490                Style style = placemark.createAndAddStyle();
491                if(_fillPolygons) {
492                    style.createAndSetPolyStyle()
493                        .withColorMode(ColorMode.NORMAL).setColor(colorStr);
494                } else {
495                    style.createAndSetPolyStyle().setFill(false);
496                    style.createAndSetLineStyle().withWidth(5)
497                        .withColorMode(ColorMode.NORMAL).setColor(colorStr);
498                }
499
500            }
501        }
502    }

510    private void _addPolygon(Polygon polygon, de.micromata.opengis.kml.v_2_2_0.Polygon kmlPolygon) {
511                
512        //System.out.println("add polygon");
513        
514        // convert exterior boundary
515        final LineString exterior = polygon.getExteriorRing();
516        
517        final LinearRing kmlExterior = KmlFactory.createLinearRing();
518        
519        for(Coordinate coordinate : exterior.getCoordinates()) {
520            kmlExterior.addToCoordinates(coordinate.x, coordinate.y, 0);
521        }
522                                    
523        kmlPolygon.setOuterBoundaryIs(KmlFactory.createBoundary().withLinearRing(kmlExterior));
524        
525        // convert any interior boundaries
526        for(int j = 0; j < polygon.getNumInteriorRing(); j++) {
527            
528            final LineString interior = polygon.getInteriorRingN(j);
529            
530            final LinearRing kmlInterior = KmlFactory.createLinearRing();
531            
532            for(Coordinate coordinate : interior.getCoordinates()) {
533                kmlInterior.addToCoordinates(coordinate.x, coordinate.y, 0);
534            }
535                                        
536            kmlPolygon.addToInnerBoundaryIs(KmlFactory.createBoundary().withLinearRing(kmlInterior));
537            
538        }
539    }

562    /** Get the KML color string from a color name. */
563    private static String _getColorHex(String name) {        
564        int[] rgb = _colorMap.get(name);
565        if(rgb == null) {
566            return "ffffffff";
567        }
568        return String.format("ff%02x%02x%02x",
569            rgb[ColorIndex.Blue.ordinal()],
570            rgb[ColorIndex.Green.ordinal()],
571            rgb[ColorIndex.Red.ordinal()]).toLowerCase();
572    }
