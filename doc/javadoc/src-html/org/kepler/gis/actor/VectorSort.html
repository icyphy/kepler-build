029package org.kepler.gis.actor;
030
031import org.geotools.data.Query;
032import org.geotools.data.collection.CollectionFeatureSource;
033import org.geotools.data.simple.SimpleFeatureCollection;
034import org.geotools.factory.CommonFactoryFinder;
035import org.kepler.gis.data.VectorToken;
036import org.opengis.filter.FilterFactory2;
037import org.opengis.filter.sort.SortBy;
038import org.opengis.filter.sort.SortOrder;
039
040import ptolemy.actor.lib.Transformer;
041import ptolemy.data.Token;
042import ptolemy.data.StringToken;
043import ptolemy.data.expr.StringParameter;
044import ptolemy.kernel.CompositeEntity;
045import ptolemy.kernel.util.Attribute;
046import ptolemy.kernel.util.IllegalActionException;
047import ptolemy.kernel.util.NameDuplicationException;
048import ptolemy.kernel.util.Workspace;
049
050/** Sort a vector/feature set on a specific property.
051 * 
052 *  @author Daniel Crawl
053 *  @version $Id: VectorSort.java 34447 2016-03-03 00:37:24Z crawl $
054 */
055public class VectorSort extends Transformer {
056
057    /** Create a new VectorSort in the container with the specified name. */
058    public VectorSort(CompositeEntity container, String name) throws NameDuplicationException, IllegalActionException {
059        
060        super(container, name);
061        
062        input.setTypeEquals(VectorToken.VECTOR);
063        output.setTypeEquals(VectorToken.VECTOR);
064        
065        propertyName = new StringParameter(this, "propertyName");
066        
067        order = new StringParameter(this, "order");
068        order.addChoice("ascending");
069        order.addChoice("descending");
070        
071        _filterFactory = CommonFactoryFinder.getFilterFactory2(null);
072    }
073    
074    @Override
075    public void attributeChanged(Attribute attribute) throws IllegalActionException {
076    
077        if(attribute == propertyName) {
078            Token token = propertyName.getToken();
079            if(token != null) {
080                _propertyNameStr = ((StringToken)token).stringValue().trim();
081            }
082        } else if(attribute == order) {
083            Token token = order.getToken();
084            if(token != null) {
085                String val = ((StringToken)token).stringValue();
086                if(val.equals("ascending")) {
087                    _order = SortOrder.ASCENDING;
088                } else if(val.equals("descending")) {
089                    _order = SortOrder.DESCENDING;
090                } else {
091                    _order = null;
092                }
093            }
094        } else {
095            super.attributeChanged(attribute);
096        }
097    }
098    
099    /** Clone this object into the specified workspace. */
100    @Override
101    public Object clone(Workspace workspace) throws CloneNotSupportedException {
102        VectorSort newObject = (VectorSort) super.clone(workspace);
103        newObject._filterFactory = CommonFactoryFinder.getFilterFactory2(null);
104        newObject._order = SortOrder.ASCENDING;
105        newObject._sortBy = null;
106        return newObject;
107    }
108    
109    /** Sort the features based on the property name. */
110    @Override
111    public void fire() throws IllegalActionException {
112        
113        super.fire();
114        
115        VectorToken token = (VectorToken)input.get(0);
116        SimpleFeatureCollection features = token.getVectors();
117
118        // use a Query and CollectionFeatureSource to sort features
119        // since many implementations of SimpleFeatureCollection do
120        // nothing in the sort() method, e.g., DefaultFeatureCollection.
121        
122        Query subQuery = new Query( features.getSchema().getTypeName() );
123        subQuery.setSortBy( new SortBy[] { _sortBy } );
124       
125        CollectionFeatureSource temp = new CollectionFeatureSource( features );
126        SimpleFeatureCollection sortedFeatures = temp.getFeatures(subQuery);
127
128        output.broadcast(new VectorToken(sortedFeatures));
129    }
130    
131    /** Check that propertyName and order are specified. */
132    @Override
133    public void preinitialize() throws IllegalActionException {
134        
135        super.preinitialize();
136        
137        if(_propertyNameStr == null || _propertyNameStr.trim().isEmpty()) {
138            throw new IllegalActionException(this, "Must specify property to sort on.");
139        }
140        
141        _sortBy = _filterFactory.sort(_propertyNameStr, _order);
142    }
143
144    /** The name of the property to sort on. */
145    public StringParameter propertyName;
146    
147    /** The order to sort. */
148    public StringParameter order;
149    
150    /** The name of the property. */
151    private String _propertyNameStr = null;
152    
153    /** The sort order. */
154    private SortOrder _order = SortOrder.ASCENDING;
155    
156    /** The sorting object. */
157    private SortBy _sortBy = null;
158    
159    /** Factory to create sort object. */
160    private FilterFactory2 _filterFactory; 
161}
