040
041public class TagTransition extends Transition {
042        
043        public TagTransition(Vector<Object> oldTags, Vector<Object> newTags, int start, int end, boolean preTransition, boolean verbose) {
044                this(oldTags, newTags, new Range(start, end), preTransition, verbose);
045        }
046
047        /**
048         * If preTransition is true, it means that the range we are giving is of
049         * the changes given from the perspective of the original tags.
050         * 
051         * On the other hand, if preTransition is false, it means that the range
052         * we are getting is of the changes given from the perspective of the
053         * changed tags.
054         */
055        public TagTransition(Vector<Object> oldTags, Vector<Object> newTags, Range range, boolean preTransition, boolean verbose) {
056                if (preTransition) {
057                        if (verbose) System.out.println("");
058                        if (verbose) System.out.println("pre transition mode");
059                        this.setOldRange(range);
060                        if (verbose) System.out.println("old range set as specified: " + range);                        
061                        this.setNewRange(new Range(range.getStart(), newTags.size() - (oldTags.size() - range.getEnd())));
062                        if (verbose) System.out.println("of course the new range started where the old range started: " + range.getStart());
063                }
064                else {
065                        if (verbose) System.out.println("post transition mode");
066                        this.setNewRange(range);
067                        if (verbose) System.out.println("new range set as specified: " + range);                        
068                        this.setOldRange(new Range(range.getStart(), oldTags.size() - (newTags.size() - range.getEnd())));
069                        if (verbose) System.out.println("of course the old range started where the new range started: " + range.getStart());                    
070                }
071        }
072}
