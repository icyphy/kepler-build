052public class TagReloadType {
053        
054        private TagReloadType() {
055                targets = new HashSet<NamedObj>();
056        }
057
058        public Type getType() {
059                return type;
060        }
061
062        public NamedOntClass getTag() {
063                return tag;
064        }
065        
066        public NamedObj getTarget() {
067                if (targets.size() != 1) {
068                        throw new IllegalStateException();
069                }
070                return targets.iterator().next();
071        }
072        
073        public Set<NamedObj> getTargets() {
074                return targets;
075        }
076
077        public static enum Type {
078                SEARCH_STRING,
079                TAG_TARGET,
080                COMPLETE,
081                TAG_ADDED,
082                TAG_REMOVED,
083                TAG_CACHE_UPDATE,
084                TAG_CREATED,
085                TAG_DELETED,
086                TAG_ADDED_AND_CLEAR_SEARCH
087        }
088
089        /**
090         * TAG_ADDED removes the tag from the auto-completions and adds it to the
091         * tag bar.
092         * 
093         * TAG_REMOVED does the reverse.
094         * 
095         * COMPLETE regenerates the tag bar and auto-completions from scratch.
096         * 
097         * TAG_TARGET indicates that a new target of tagging has been selected,
098         * perhaps by clicking the canvas or a workflow run; this necessitates an
099         * update of both the auto-completions and the tag bar.
100         * 
101         * TAG_ADDED_AND_CLEAR_SEARCH adds the tag to the tag bar and regenerates
102         * the auto-completions as if nothing is typed, resulting in all possible
103         * auto-completions, minus whatever tags are currently applied to the tag target.
104         * 
105         * SEARCH_STRING updates the auto-completions, removing those that no
106         * longer match the search string.
107         */
108

110                private TagCacheChange() {
111                        super();
112                        type = Type.TAG_CACHE_UPDATE;

117                }

120                        this.model = model;

124
125                public boolean getCheckboxState() {
126                        return checkboxState;
127                }
128

130                        this.checkboxState = checkboxState;
131                }
132
133                private boolean checkboxState = false;
134        }

170        public static SearchStringChange createSearchStringChange(String oldString, String newString) {
171                SearchStringChange tagReloadType = new SearchStringChange();
172                tagReloadType.type = Type.SEARCH_STRING;
173                tagReloadType.setOldString(oldString);
174                tagReloadType.setNewString(newString);
175                return tagReloadType;
176        }

178        public static TagReloadType createTagTargetChange() {
179                TagReloadType tagReloadType = new TagReloadType();
180                tagReloadType.type = Type.TAG_TARGET;
181                return tagReloadType;
182        }
183        
184        public static TagReloadType createTagTargetChange(NamedObj target) {
185                TagReloadType tagReloadType = createTagTargetChange();
186                tagReloadType.targets.add(target);
187                return tagReloadType;
188        }

190        public static TagReloadType createTagTargetChange(Collection<NamedObj> targets) {

192                tagReloadType.targets.addAll(targets);

194        }

196        public static TagReloadType createTagAddedChange(NamedOntClass cls, NamedObj target) {
197                TagReloadType tagReloadType = new TagReloadType();
198                tagReloadType.type = Type.TAG_ADDED;
199                tagReloadType.tag = cls;
200                tagReloadType.targets.add(target);
201                return tagReloadType;
202        }

204        public static List<TagReloadType> createExternalChanges(NamedObj target, List<NamedOntClass> added, List<NamedOntClass> removed) {
205                
206                if (target == null) {
207                        log.warn("target is null");
208                        return Collections.emptyList();
209                }
210                
211                List<TagReloadType> changes = new ArrayList<TagReloadType>();
212                List<TagReloadType> additions = new ArrayList<TagReloadType>();
213                List<TagReloadType> removals = new ArrayList<TagReloadType>();
214                
215                for (NamedOntClass ontClass : added) {
216                        TagReloadType tagReloadType = new TagReloadType();
217                        tagReloadType.type = Type.TAG_ADDED;
218                        tagReloadType.tag = ontClass;
219                        tagReloadType.targets = Collections.singleton(target);
220                        additions.add(tagReloadType);
221                }
222
223                for (NamedOntClass ontClass : removed) {
224                        TagReloadType tagReloadType = new TagReloadType();
225                        tagReloadType.type = Type.TAG_REMOVED;
226                        tagReloadType.tag = ontClass;
227                        tagReloadType.targets = Collections.singleton(target);
228                        removals.add(tagReloadType);
229                }
230                
231                // Put all items from additions and removals into changes, except for
232                // pairs addition A and removal R where A.tag.equals(R.tag). Not
233                // quite sure why this is explicitly necessary.
234                // Really, really naive algorithm:
235                for (TagReloadType addition : additions) {
236                        boolean unique = true;
237                        for (TagReloadType removal : removals) {
238                                if (addition.getTag().equals(removal.getTag())) {
239                                        unique = false;
240                                        break;
241                                }
242                        }
243                        if (unique) {
244                                changes.add(addition);
245                        }
246                }
247
248                for (TagReloadType removal : removals) {
249                        boolean unique = true;
250                        for (TagReloadType addition : additions) {
251                                if (addition.getTag().equals(removal.getTag())) {
252                                        unique = false;
253                                        break;
254                                }
255                        }
256                        if (unique) {
257                                changes.add(removal);
258                        }
259                }
260                
261                return changes;
262        }

264        public static TagReloadType createInteractiveTagAddedChange(NamedOntClass cls, NamedObj target) {
265                TagReloadType tagReloadType = new TagReloadType();
266                tagReloadType.type = Type.TAG_ADDED_AND_CLEAR_SEARCH;
267                tagReloadType.tag = cls;
268                tagReloadType.targets.add(target);
269                return tagReloadType;
270        }
271        
272        public static TagReloadType createTagRemovedChange(NamedOntClass cls, NamedObj target) {
273                TagReloadType tagReloadType = new TagReloadType();
274                tagReloadType.type = Type.TAG_REMOVED;
275                tagReloadType.tag = cls;
276                tagReloadType.targets.add(target);
277                return tagReloadType;
278        }
279        
280        public static TagReloadType createTagCacheChange(NamedOntModel ontModel, boolean checkboxState) {
281                TagCacheChange change = new TagCacheChange();
282                change.type = Type.TAG_CACHE_UPDATE;
283                change.setModel(ontModel);
284                change.setCheckboxState(checkboxState);
285                return change;
286        }
287                
288        public static TagReloadType createNewOrDeleteChange(NamedOntClass newOntClass, boolean newTag) {
289                TagReloadType tagReloadType = new TagReloadType();
290                if (newTag) {
291                        tagReloadType.type = Type.TAG_CREATED;
292                }
293                else {
294                        tagReloadType.type = Type.TAG_DELETED;
295                }
296                tagReloadType.tag = newOntClass;
297                tagReloadType.targets = Collections.emptySet();
298                return tagReloadType;
299        }
300        
301        public boolean needsReloading(TaggingContextManager tcm) {
302                if (type == Type.TAG_ADDED || type == Type.TAG_ADDED_AND_CLEAR_SEARCH) {
303                        Set<NamedObj> currentTargets = new HashSet<NamedObj>(tcm.getTagTargets());
304                        currentTargets.retainAll(targets);
305                        return !currentTargets.isEmpty();
306                }
307                return false;
308        }
309        
310        protected Set<NamedObj> targets;
311        protected Type type;
312        protected NamedOntClass tag = null;
313        
314        private static Log log = LogFactory.getLog(TagReloadType.class);
