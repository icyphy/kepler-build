066        public static Set<NamedOntClass> getTags(Collection<NamedObj> objs) {
067                if (objs == null || objs.isEmpty()) return Collections.emptySet();
068                boolean firstTime = true;
069                Set<NamedOntClass> classes = null;
070                for (NamedObj obj : objs) {
071                        if (firstTime) {
072                                classes = getTags(obj);
073                                firstTime = false;
074                                continue;
075                        }
076                        classes.retainAll(getTags(obj));
077                }
078                return classes;
079        }
080    /**
081     * Get the tags for a given named object
082     * @param obj the named object
083     * @return the set of (possibly empty) semantic-type tags
084     */
085    public static Set<NamedOntClass> getTags(NamedObj obj) {
086        Set<NamedOntClass> result = new HashSet<NamedOntClass>();
087        if(obj == null)
088            return result;
089        OntologyCatalog catalog = OntologyCatalog.instance();
090                Set<RemovableSemanticType> types = new HashSet<RemovableSemanticType>();
091        for(Object a : obj.attributeList(SemanticType.class))
092            types.add(new RemovableSemanticType((SemanticType)a, true));
093        
094        /// TODO
095        // special case for WorkflowRun, since intersection code in ExistingTagsPanel._getTags wants 
096        // 1 set of tags per object.
097        // hardcoded WorkflowRun.WORKFLOWTAGS_ATTRIBUTE_NAME since don't have access to WorkflowRun:
098        Attribute workflowTags = obj.getAttribute("workflowTags");
099        if (workflowTags != null){
100            for(Object a : workflowTags.attributeList(SemanticType.class))
101                                types.add(new RemovableSemanticType((SemanticType)a, false));
102        }
103        /// end special case
104        
105        for(RemovableSemanticType rtype : types) {
106                        SemanticType type = rtype.getSemanticType();
107            NamedOntClass ontClass = catalog.getNamedOntClass(type);
108            if(ontClass != null ) {
109                                ontClass.setRemovable(rtype.isRemovable());
110                result.add(ontClass);
111                        }
112                        else {
113                                // This is an orphaned concept
114
115                                result.add(new NamedOntClass(type.getConceptUri(), type.getLabel()));
116                        }
117        }
118        return result;
119    }

121        /**
122         * If an object is tagged that is NOT an instanceof any of the classes
123         * that have been registered as atomically taggable, then the change is in
124         * limbo, and the effigy of the Kepler graph is explicitly set as changed
125         * to force a save dialog on closing. For example, tagging a workflow adds
126         * a semantic type that is rendered to MOML when the workflow is saved.
127         * This is not an atomic change. On the other hand, tagging a workflow run
128         * immediately adds the tag to the REAP database, so this is atomic.
129         * @param cls Class to register
130         */
131        public static void registerAtomicallyTaggableClass(Class cls) {
132                atomicallyTaggableClasses.add(cls);
133        }
134        
135        public static boolean isAtomicallyTaggable(Object object) {
136                for (Class atomicallyTaggableClass : atomicallyTaggableClasses) {
137                        try {
138                                atomicallyTaggableClass.cast(object);
139                                return true;
140                        }
141                        catch(ClassCastException ignored) {}
142                }
143                return false;
144        }

148    /**
149     * Add a tag to the named object
150     * @param obj the named object
151     * @param tag the ontology class
152         * @param ptolemyFrame The PtolemyFrame of the current workflow we are
153     * tagging, so it can be set as modified.
154     * @return true iff a new semantic type was added successfully. false
155         * otherwise (if one of the parameters are invalid, or if obj is already
156         * tagged with tag
157     */
158        public static boolean addTag(NamedObj obj, NamedOntClass tag, PtolemyFrame ptolemyFrame) {
159        if ((ptolemyFrame == null) && (obj instanceof ptolemy.actor.TypedCompositeActor)) {
160            log.warn("Could not explicit set effigy as modified");
161        }
162                if (obj == null || tag == null)
163                        return false;
164                Set<NamedOntClass> existingTags = Tagging.getTags(obj);
165                if (existingTags.contains(tag)) {
166                        return false;
167                }
168                try {
169                        SemanticType type = new SemanticType(obj, obj.uniqueName("semanticType"));
170                        
171                        String namespace = tag.getNameSpace();
172                        if (namespace.endsWith("#")) {
173                                namespace = namespace.substring(0, namespace.length() - 1);
174                        }
175                        type.setNamespace(namespace);
176                        type.setConceptName(tag.getLocalName());
177                        type.setLabel(tag.getName());
178                        
179                        // update LSID revision
180                        // FIXME hardcode. let SQLRecordingV8.tagAdded roll revision on runs for now.
181                        // otherwise ordering of events is wrong for reporting suite
182                        if (obj.getName() != "WorkflowRun"){
183                                NamedObjId.getIdAttributeFor(obj).updateRevision();
184                        }
185                        
186//                      type.setConceptId(tag.getConceptIdWithLabel());
187                        //TODO is this necessary now that updateRevision is called?
188            if (ptolemyFrame != null) {
189                ptolemyFrame.getEffigy().setModified(true);
190            }
191                        TagReloadType tagReloadType = TagReloadType.createInteractiveTagAddedChange(tag, obj);
192                        for (TaggingContextManager tcm : TaggingContextManager.getInstances()) {
193                                if (!tagReloadType.needsReloading(tcm)) {
194                                        continue;
195                                }
196                                log.debug("Reloading tags (" + tcm.getWindowId() + ")");
197                                if (tcm.reloadTags(true, tagReloadType)) {
198                                        log.debug("Tags reloaded successfully (" + tcm.getWindowId() + ")");
199                                }
200                                else {
201                                        log.info("Tags reloaded UNSUCCESSFULLY (" + tcm.getWindowId() + ")");
202                                }
203                        }
204                        return true;
205                }
206                catch(Exception e) {
207                        e.printStackTrace();
208                }
209                return false;
210        }

212        public static void removeTag(NamedObj obj, NamedOntClass tag, List<TagEventListener> listeners, boolean lastEvent) {
213                for (TagEventListener listener : listeners) {
214                        listener.tagRemoved(new TagEvent(obj, tag, lastEvent));
215                }
216                removeTag(obj, tag, lastEvent);
217        }

219    /**
220     * Remove a tag from the named object object
221     * @param obj the name object
222     * @param tag the tag
223     */
224    public static void removeTag(NamedObj obj, NamedOntClass tag, boolean lastEvent) {
225        if(obj == null || tag == null)
226            return;
227                if (tag.isOrphaned()) {
228                        removeOrphanTag(obj, tag);
229                        try {
230                                // update LSID revision
231                                // FIXME hardcode. let SQLRecordingV8.tagRemoved roll revision on runs for now.
232                                // otherwise ordering of events is wrong for reporting suite
233                                if (obj.getName() != "WorkflowRun"){
234                                        NamedObjId.getIdAttributeFor(obj).updateRevision();
235                                }
236                        } catch (Exception e) {
237                                // TODO Auto-generated catch block
238                                e.printStackTrace();
239                        }
240                        return;
241                }
242        OntologyCatalog catalog = OntologyCatalog.instance();
243        Set<SemanticType> types = new HashSet<SemanticType>();
244        for(Object a : obj.attributeList(SemanticType.class))
245            types.add((SemanticType)a);
246        for(SemanticType type : types) {
247            NamedOntClass ontClass = catalog.getNamedOntClass(type);
248            if(ontClass == null) {
249                                // TODO: Make sure we remove orphan tag, right?
250                                continue;                               
251                        }
252            String tagId = tag.getConceptId();
253            String clsId = ontClass.getConceptId();
254            if(tagId.equals(clsId)) {
255                try{ 
256                    type.setContainer(null);
257                    // update LSID revision
258                    // FIXME hardcode. let SQLRecordingV8.tagRemoved roll revision on runs for now.
259                    // otherwise ordering of events is wrong for reporting suite
260                                if (obj.getName() != "WorkflowRun"){
261                                        NamedObjId.getIdAttributeFor(obj).updateRevision();
262                                }
263                                        TagReloadType tagReloadType = TagReloadType.createTagRemovedChange(tag, obj);
264                                        for (TaggingContextManager tcm : TaggingContextManager.getInstances()) {
265                                                log.debug("Reloading tags (" + tcm.getWindowId() + ")");
266                                                if (tcm.reloadTags(true, tagReloadType)) {                                              
267                                                        log.debug("Tags reloaded successfully (" + tcm.getWindowId() + ")");
268                                                }
269                                                else {
270                                                        log.info("Tags reloaded UNSUCCESSFULLY (" + tcm.getWindowId() + ")");
271                                                }
272                                        }                                       
273                    return;
274                } catch(Exception e) {
275                    e.printStackTrace();
276                }
277            }
278        }
279    }

281        private static void removeOrphanTag(NamedObj obj, NamedOntClass tag) {
282                assert(tag.isOrphaned());
283                Set<SemanticType> types = new HashSet<SemanticType>();
284                for (Object o : obj.attributeList(SemanticType.class)) {
285                        types.add((SemanticType) o);
286                }
287                for (SemanticType type : types) {
288                        // Changed this so tagId contains the saved label just like clsId, so
289                        // they're comparable as intended. This was flagged in bug 5260.
290                        String tagId = tag.getConceptIdWithLabel(); 
291                        String clsId = type.getConceptId();
292                        if (tagId.equals(clsId)) {
293                                try {
294                                        type.setContainer(null);
295                                }
296                                catch(IllegalActionException e) {
297                                        e.printStackTrace();
298                                }
299                                catch(NameDuplicationException e) {
300                                        e.printStackTrace();
301                                }
302                        }
303                }
304        }

306    /**
307     * Search for and return named objects with one or more of the given types
308     * TODO: extend to support subclass checking
309     * @param types the set of types to search for
310     * @return the set of matching objects
311     */
312    public static Set<NamedObj> findNamedObjects(Set<NamedOntClass> types) {
313        Set<NamedObj> result = new HashSet<NamedObj>();
314        try {
315                CacheManager manager = CacheManager.getInstance();
316                Vector<KeplerLSID> cachedLsids = manager.getCachedLsids();
317            for (KeplerLSID lsid : cachedLsids) {
318                ActorCacheObject aco = 
319                    (ActorCacheObject) manager.getObject(lsid);
320                ActorMetadata am = aco.getMetadata();
321                NamedObj obj = am.getActorAsNamedObj(null);
322                for(NamedOntClass tag : getTags(obj)) {
323                    // TODO: Extend to subclasses
324                    if(types.contains(tag)) {
325                        result.add(obj);
326                        break;
327                    }
328                }
329            }
330        } catch (Exception e) {
331            e.printStackTrace();
332        }
333        return result;
334    }

336    /**
337     * Search for and return named objects with the given type
338     * TODO: extend to support subclass checking
339     * @param type the type to search for
340     * @return the set of matching objects
341     */
342    public static Set<NamedObj> findNamedObjects(NamedOntClass type) {
343        Set<NamedObj> result = new HashSet<NamedObj>();
344        try {
345                CacheManager manager = CacheManager.getInstance();
346                Vector<KeplerLSID> cachedLsids = manager.getCachedLsids();
347            for (KeplerLSID lsid : cachedLsids) {
348                ActorCacheObject aco = 
349                    (ActorCacheObject) manager.getObject(lsid);
350                ActorMetadata am = aco.getMetadata();
351                NamedObj obj = am.getActorAsNamedObj(null);
352                for(NamedOntClass tag : getTags(obj)) {
353                    // TODO: Extend to subclasses
354                    if(type.equals(tag)) {
355                        result.add(obj);
356                        break;
357                    }
358                }
359            }
360        } catch (Exception e) {
361            e.printStackTrace();
362        }
363        return result;
364    }

367    /**
368     * Just for testing ... the findNamedObjects throws exceptions due
369     * to the object manager
370     * @param args Command line arguments.
371     */
372    public static void main(String [] args) {
373        NamedObj myObj = new NamedObj();
374        OntologyCatalog catalog = OntologyCatalog.standaloneInstance();
375        int count = 2;
376        System.out.println("\n\nTAGGING: ");
377        Iterator<NamedOntModel> models = catalog.getNamedOntModels();
378        for(int i = 0; i < count && models.hasNext(); i++) {
379            NamedOntModel model = models.next();
380            Iterator<NamedOntClass> classes = model.getRootClasses(true);
381            while(classes.hasNext()) {
382                NamedOntClass cls = classes.next();
383                if(cls != null)
384                    Tagging.addTag(myObj, cls, null);
385                if(Tagging.findNamedObjects(cls).size() > 0)
386                    System.out.println(">>> FOUND AN OBJECT");
387            }
388        }
389        System.out.println(myObj.exportMoML());
390
391        System.out.println("\n\nTAGS ADDED:");
392        for(NamedOntClass cls : Tagging.getTags(myObj)) 
393            System.out.println("added: " + cls);
394
395        System.out.println("\n\nUNTAGGING:");
396        models = catalog.getNamedOntModels();
397        for(int i = 0; i < count && models.hasNext(); i++) {
398            NamedOntModel model = models.next();
399            Iterator<NamedOntClass> classes = model.getRootClasses(true);
400            while(classes.hasNext()) {
401                NamedOntClass cls = classes.next();
402                if(cls != null){
403                        if (classes.hasNext()){
404                                Tagging.removeTag(myObj, cls, new ArrayList<TagEventListener>(), false);
405                        }
406                        else{
407                                Tagging.removeTag(myObj, cls, new ArrayList<TagEventListener>(), true);
408                        }
409                }
410            }
411        }
412        System.out.println(myObj.exportMoML());
413    }
