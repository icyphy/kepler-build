259        public void valueChanged(ListSelectionEvent e) {
260                // get the selected NamedOntModel and send to listeners
261                int rowIndex = table.getSelectedRow();
262                if (rowIndex == -1) {
263                        return;
264                }
265                if (previousRowSelection == rowIndex)
266                        return;
267                previousRowSelection = rowIndex;
268                OntologyTableModel t = (OntologyTableModel) table.getModel();
269                NamedOntModel m = t.getNamedOntModel(rowIndex);
270                notifySelectionListeners(new OntologyDisplaySelectionEvent(this, m));
271        }

296        private TableModel getOntologyData() {
297                final OntologyTableModel tableModel = new OntologyTableModel(OntologyCatalog.instance(), table);
298                tableModel.addTableModelListener(new TableModelListener() {
299                        public void tableChanged(TableModelEvent e) {
300                                if (e.getColumn() == Columns.TAG_BAR) {
301                                        boolean value = (Boolean) tableModel.getValueAt(e.getFirstRow(), Columns.TAG_BAR);
302                                        OntologyCatalog.instance().setInTagBar(tableModel.getNamedOntModel(e.getFirstRow()), value);
303                                        TagCache.instance().forceRefilter();
304                                        Object nameCell = tableModel.getValueAt(e.getFirstRow(), Columns.NAME);
305                                        TagReloadType tagReloadEvent = TagReloadType.createTagCacheChange((NamedOntModel) nameCell, false);
306                                        for (TaggingContextManager tcm : TaggingContextManager.getInstances()) {
307                                                if (!tcm.reloadTags(true, tagReloadEvent)) {
308                                                        log.warn("Failed to reload tags in TCM " + tcm.getWindowId());
309                                                }
310                                        }
311                                }
312                        }
313                });
314                return tableModel;
315        }

320                if ("new".equals(e.getActionCommand()))
321                        handleNewEvent();
322                else if ("register".equals(e.getActionCommand()))
323                        handleRegisterEvent();

326                else if ("publish".equals(e.getActionCommand()))

330        public void handleNewEvent() {
331                Iterator<NamedOntModel> iterator = OntologyCatalog.instance().getNamedOntModels();
332                List<Object> models = new ArrayList<Object>();
333                while (iterator.hasNext()) {
334                        NamedOntModel model = iterator.next();
335                        models.add(model);
336                }
337                models.add("(no parent)");              // Sentinel
338                Collections.sort(models, new Comparator<Object>() {
339                        public int compare(Object o1, Object o2) {
340                                if (o1 == null || o1 instanceof String) return -1;
341                                if (o2 == null || o2 instanceof String) return 1;
342                                if (o1 instanceof Integer) return -1;
343                                try {
344                                        return ((NamedOntModel) o1).getName().compareToIgnoreCase(((NamedOntModel)o2).getName());
345                                }
346                                catch(ClassCastException ex) {
347                                        return -1;
348                                }
349                        }
350                });
351                Object[] modelsArray = models.toArray(new Object[models.size()]);
352                
353                OntologyDialogResponse selectedValue = newOntologyDialog(modelsArray);
354                if (selectedValue == null) {
355                        log.info("Nothing selected from dialog");
356                        return;
357                }
358                
359                boolean hasAParent = !(selectedValue.getParentOntology() instanceof String);
360                boolean isInLibrary = true;
361                String label = selectedValue.getLabel();
362                NamedOntModel model;
363                if (hasAParent) {
364                        model = _addNewOntology((NamedOntModel) (selectedValue.getParentOntology()), label, isInLibrary, true);
365                }
366                else {
367                        model = _addNewOntology(label, isInLibrary, true);
368                }
369                
370                ((OntologyTableModel) table.getModel()).poke();
371                table.tableChanged(new TableModelEvent(table.getModel()));
372                refresh(model);
373        }

515        private NamedOntModel _addNewOntology(String label, boolean isInLibrary, boolean editable) {
516                OntologyCatalog catalog = OntologyCatalog.instance();
517                NamedOntModel model = catalog.addModel(isInLibrary, label, editable);
518                catalog.saveCatalog();
519                return model;
520        }
521        
522        private NamedOntModel _addNewOntology(NamedOntModel model, String label, boolean isInLibrary, boolean editable) {
523                OWLOntologyManager manager = OntologyCatalog.getManager();
524                NamedOntModel newModel = _addNewOntology(label, isInLibrary, editable);
525                OWLDataFactory factory = manager.getOWLDataFactory();
526                try {
527                        manager.applyChange(new AddAxiom(newModel.getOntology(), factory.getOWLImportsDeclarationAxiom(newModel.getOntology(), model.getOntology().getURI())));
528                        newModel.write();
529                }
530                catch(OWLOntologyChangeException ex) {
531                        ex.printStackTrace();
532                }
533                return newModel;
534        }

575                this.catalog = catalog;
576                this.parent = parent;

580        public void poke() {
581                poke(-1, -1);
582        }
583        
584        public void poke(int rowIndex, int columnIndex) {
585                NamedOntModel highlightedModel = null;
586                if (rowIndex != -1) {
587                        highlightedModel = rows.get(rowIndex).getModel();
588                }
589                Set<NamedOntModel> libraryModels = settifyIterator(catalog.getNamedOntModels(true));
590                Set<NamedOntModel> allModels = settifyIterator(catalog.getNamedOntModels(false));
591                Set<NamedOntModel> tagbarModels = settifyIterator(catalog.getTagBarOntModels());
592                log.debug(libraryModels.size() + " models are displayed of " + allModels.size() + " total");
593
594                Vector<OntologyRow> rows = new Vector<OntologyRow>();
595                for (NamedOntModel model : allModels) {
596                        model.initialize();
597                        rows.add(new OntologyRow(model, libraryModels.contains(model), tagbarModels.contains(model), model.getColor(), model.isLocal()));
598                }
599                Collections.sort(rows, new Comparator<OntologyRow>() {
600
601                        public int compare(OntologyRow o1, OntologyRow o2) {
602                                Matcher m1 = untitledOntology.matcher(o1.getModel().getName());
603                                Matcher m2 = untitledOntology.matcher(o2.getModel().getName());
604                                if (m1.matches() && m2.matches()) {
605                                        return Integer.valueOf(m1.group(1)) - Integer.valueOf(m2.group(1));
606                                }
607                                return o1.getModel().getName().compareToIgnoreCase(o2.getModel().getName());
608                        }
609                });
610                
611                if (rowIndex != -1) {
612                        for (int i = 0; i < rows.size(); i++) {
613                                if (highlightedModel == rows.get(i).getModel()) {
614                                        // Highlight at i, columnIndex
615                                        parent.addRowSelectionInterval(i, i);
616                                        parent.addColumnSelectionInterval(columnIndex, columnIndex);
617                                        break;
618                                }
619                        }
620                }
621                this.rows = rows;
622        }

624        private static Pattern untitledOntology = Pattern.compile("Untitled ([0-9]+)");

626        private <T> Set<T> settifyIterator(Iterator<T> iterator) {
627                Set<T> set = new HashSet<T>();
628                while(iterator.hasNext()) {
629                        set.add(iterator.next());
630                }
631                return set;
632        }

650        public boolean isCellEditable(int rowIndex, int columnIndex) {
651                if (cannotSetCatalog()) {
652                        return false;
653                }
654                if (columnIndex == Columns.LOCAL) {
655                        return false; // This column is immutable
656                }
657                
658                // If this is an external ontology, don't allow changing the name, as
659                // this entails changing the .OWL file.         
660                return columnIndex != Columns.NAME || rows.get(rowIndex).isLocal();
661        }

663        public boolean cannotSetCatalog() {
664                File indexFile = OntologyConfiguration.instance().getIndexFile();
665                File directory = ModuleTree.instance().getModuleByStemName("common").getDir();
666                return indexFile.getAbsolutePath().startsWith(directory.getAbsolutePath());
667        }

673        public Object getValueAt(int rowIndex, int columnIndex) {
674                OntologyRow row = rows.get(rowIndex);
675                switch(columnIndex) {
676                        case Columns.NAME:
677                                return row.getModel();
678                        case Columns.LIBRARY:
679                                return row.isInLibrary();
680                        case Columns.TAG_BAR:
681                                return row.isInTagBar();
682                        case Columns.COLOR:
683                                return row.getColor();
684                        case Columns.LOCAL:
685                                return row.isLocal();
686                }
687                return null;
688        }
689
690        public void setValueAt(Object aValue, int rowIndex, int columnIndex) {
691                OntologyRow row = rows.get(rowIndex);
692                switch(columnIndex) {
693                        case Columns.NAME:
694                                if (row.isLocal()) {
695                                        row.getModel().removeTopLevelLabel();
696                                        row.getModel().setTopLevelLabel((String) aValue);
697                                        row.getModel().write();
698                                }
699                                break;
700                        case Columns.LIBRARY:
701                                row.setInLibrary((Boolean) aValue);
702                                catalog.saveCatalog(rows);
703                                break;
704                        case Columns.TAG_BAR:
705                                row.setInTagBar((Boolean) aValue);
706                                catalog.saveCatalog(rows);
707                                break;
708                        case Columns.COLOR:
709                                org.kepler.sms.Color smsColor = (org.kepler.sms.Color) aValue;
710                                row.setColor(smsColor);
711                                for (WeakReference<TagBarDisplayUI> tagBarDisplayUIReference : TagManagerUI.getTagBarDisplayUIs()) {
712                                        TagBarDisplayUI tagBarDisplayUI = tagBarDisplayUIReference.get();
713                                        if (tagBarDisplayUI == null) {
714                                                continue;
715                                        }
716                                        tagBarDisplayUI.changeColor(rows.get(rowIndex).getModel(), smsColor);
717                                }
718                                catalog.saveCatalog(rows);
719                                break;
720                        case Columns.LOCAL:
721                                row.setLocal((Boolean) aValue);
722                                catalog.saveCatalog(rows);
723                                break;
724                }
725                poke(rowIndex, columnIndex);
726                TableModelEvent tableModelEvent = new TableModelEvent(this, rowIndex, rowIndex, columnIndex);
727                parent.tableChanged(tableModelEvent);
728                for (TableModelListener listener : listeners) {
729                        listener.tableChanged(tableModelEvent);
730                }
731        }
