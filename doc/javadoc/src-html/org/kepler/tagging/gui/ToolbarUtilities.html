030package org.kepler.tagging.gui;
031
032import java.awt.Insets;
033
034import javax.swing.Action;
035import javax.swing.Icon;
036import javax.swing.JButton;
037import javax.swing.JComponent;
038
039/**
040 * Utility functions for adding items to the toolbar, taken from
041 * diva.gui.GUIUtilities and modified to add 'toolbar buttons' to non-JToolBar
042 * components (eg., for if there are containers in the toolbar).
043 */
044
045public class ToolbarUtilities {
046
047    /** This key is used in an action to specify an icon used in toolbars.
048     */
049    public static final String LARGE_ICON = "LargeIcon";
050
051    /** This key is used in an action to specify a rollover icon
052     *  used in toolbars.
053     */
054    public static final String ROLLOVER_ICON = "rolloverIcon";
055
056    /** This key is used in an action to specify a rollover selected icon
057     *  used in toolbars.
058     */
059    public static final String ROLLOVER_SELECTED_ICON = "rolloverSelectedIcon";
060
061    /** This key is used in an action to specify a selected icon
062     *  used in toolbars.
063     */
064    public static final String SELECTED_ICON = "selectedIcon";
065
066    /** Add the action to the given toolbar.
067     *  If the LARGE_ICON property is specified in the Action, then use it
068     *  for the button.  We use this instead of SMALL_ICON, because
069     *  SMALL_ICON shows up when the action is added to a menu, and in
070     *  most cases we don't actually want an icon there.
071     *  If no icon is specified, then the button will just
072     *  have the name of the action.  If the "tooltip" property is specified
073     *  in the action, then create a tooltip for the button with the string.
074     *  The new button is added to the action as the "toolButton" property.
075     *  The button is enabled by default.
076     */
077    public static JButton addToolBarButton(JComponent toolbar, Action action) {
078        Icon icon = (Icon) action.getValue(LARGE_ICON);
079        String label = null;
080
081        if (icon == null) {
082            label = (String) action.getValue(Action.NAME);
083        }
084
085        return addToolBarButton(toolbar, action, null, icon, label, true);
086    }
087
088    /** Add an action to the toolbar.  If the tool tip is null, use
089     * the "tooltip" property already in the action, otherwise add the
090     * property to the action. The new button is added to the action
091     * as the "toolButton" property.  The button represented by an icon
092     * (no text) and is enabled by default.
093     */
094    public static JButton addToolBarButton(JComponent toolbar, Action action,
095            String tooltip, Icon icon) {
096        return addToolBarButton(toolbar, action, tooltip, icon, null, true);
097    }
098
099    /** Add an action to the toolbar.  If the tool tip is null, use
100     * the "tooltip" property already in the action, otherwise add the
101     * property to the action. The new button is added to the action
102     * as the "toolButton" property.  The button represented by an icon
103     * (no text) and is enabled by default.
104     */
105    public static JButton addToolBarButton(JComponent toolbar, Action action,
106            String tooltip, Icon icon, boolean isEnabled) {
107        return addToolBarButton(toolbar, action, tooltip, icon, null, isEnabled);
108    }
109
110    /** Add an action to the toolbar.  If the tool tip is null, use
111     * the "tooltip" property already in the action, otherwise add the
112     * property to the action. The new button is added to the action
113     * as the "toolButton" property.  The button represented by text
114     * (no icon) and is enabled by default.
115     */
116    public static JButton addToolBarButton(JComponent toolbar, Action action,
117            String tooltip, String lbl) {
118        return addToolBarButton(toolbar, action, tooltip, null, lbl, true);
119    }
120
121    /** Add an action to the toolbar.  If either an icon or a text string
122     * are specified (non-null), they are added.  The button is enabled by
123     * default.
124     */
125    public static JButton addToolBarButton(JComponent toolbar, Action action,
126            String tooltip, Icon icon, String lbl) {
127        return addToolBarButton(toolbar, action, tooltip, icon, lbl, true);
128    }
129
130    /** Add an action to the toolbar.  If the tool tip is null, use
131     * the "tooltip" property already in the action, otherwise add the
132     * property to the action.  The new button is added to the action
133     * as the "toolButton" property.  If either an icon or a text string
134     * are specified (non-null), they are added.
135     */
136    public static JButton addToolBarButton(JComponent toolbar, Action action,
137            String tooltip, Icon icon, String lbl, boolean isEnabled) {
138        if (tooltip == null) {
139            tooltip = (String) action.getValue("tooltip");
140        } else {
141            action.putValue("tooltip", tooltip);
142        }
143
144                JButton button = new JButton();         
145//        JButton button = toolbar.add(action);
146                button.setAction(action);
147        button.setToolTipText(tooltip);
148        button.setText(null);
149        button.setRequestFocusEnabled(false);
150                toolbar.add(button);
151
152        if (icon != null) {
153            button.setIcon(icon);
154        }
155
156        if (lbl != null) {
157            button.setText(lbl);
158        }
159
160        Icon rolloverIcon = (Icon) action.getValue(ROLLOVER_ICON);
161        if (rolloverIcon != null) {
162            button.setRolloverIcon(rolloverIcon);
163        }
164        Icon rolloverSelectedIcon = (Icon) action
165                .getValue(ROLLOVER_SELECTED_ICON);
166        if (rolloverSelectedIcon != null) {
167            button.setRolloverSelectedIcon(rolloverSelectedIcon);
168        }
169        Icon selectedIcon = (Icon) action.getValue(SELECTED_ICON);
170        if (selectedIcon != null) {
171            button.setSelectedIcon(selectedIcon);
172        }
173
174        button.setMargin(new Insets(0, 0, 0, 0));
175
176        //        button.setBorderPainted(false);
177        button.setBorderPainted(true);
178        button.setEnabled(isEnabled);
179        action.putValue("toolBarButton", button);
180        return button;
181    }
182}
