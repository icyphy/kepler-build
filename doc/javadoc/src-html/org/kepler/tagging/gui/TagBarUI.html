127        public void addTagEventListener(TagEventListener listener) {
128                listeners.add(listener);
129        }
130
131    /** Add a TagEventListener for a specific frame. The listeners are
132     *  added in the TagBarUI constructor.
133         * @param listener The listener to add to the list of listeners
134         * @param frame The graph frame that this tag UI is associated with
135         */
136        public static void addTagEventListener(TagEventListener listener, KeplerGraphFrame frame) {
137         
138            List<TagEventListener> listenerList = frameListenersMap.get(frame);
139            if(listenerList == null)
140            {
141                listenerList = new LinkedList<TagEventListener>();
142                frameListenersMap.put(frame, listenerList);
143            }
144            listenerList.add(listener);
145        }
146        
147        public void removeTagEventListener(TagEventListener listener) {
148                listeners.remove(listener);
149        }
150        
151        public List<TagEventListener> getTagEventListeners() {
152                return listeners;
153        }
154                
155        private List<TagEventListener> listeners;
156
157    /** A mapping of frames to a list of tag event listeners. */
158        private static Map<KeplerGraphFrame, List<TagEventListener>> frameListenersMap = 
159            new WeakHashMap<KeplerGraphFrame, List<TagEventListener>>();
160
161        public int getId() {

163        }
