104        class AutoCompletingKeyListener implements KeyListener {
105                public AutoCompletingKeyListener(JComboBox combobox) {
106                        super();
107                        this.combobox = combobox;
108                }
109
110                public void keyTyped(KeyEvent e) {}
111                public void keyPressed(KeyEvent e) {}
112
113                public void keyReleased(KeyEvent e) {
114                        if (e.getKeyCode() == KeyEvent.VK_UP || e.getKeyCode() == KeyEvent.VK_DOWN) {
115                                combobox.showPopup();
116                                return; // We just don't care.
117                        }
118                        if (e.getKeyCode() == KeyEvent.VK_ENTER) {
119                                addCurrentTag();
120                        }
121                        else if (KeyEvent.CHAR_UNDEFINED != e.getKeyChar()) {
122                                valid = false;
123                        }
124                }
125

127        }

130                if (instructionsDisplayed || "".equals(((JTextField) getEditor().getEditorComponent()).getText())) {
131                        if (instructionsDisplayed) {
132                                log.debug("Not adding tag because instructions are currently displayed");
133                        }
134                        return;
135                }
136                                
137                if (getEditor().getItem() == null || !valid) {
138                        String input = ((JTextField) getEditor().getEditorComponent()).getText();
139                        tag = new NamedOntClass(input);
140                }
141                                
142                // Commit tags
143                existingTagsPanel.addTag(tag);          
144        }

147                tag = null;
148                valid = true;
149                this.getEditor().setItem(null);
150        }

226        private static class ChangeProcessingFilter extends DocumentFilter {
227                
228                public ChangeProcessingFilter(Document document, int windowId) {
229                        this.document = document;
230                        this.windowId = windowId;
231                }
232                
233                @Override
234                public void replace(FilterBypass fb, int offset, int length, String text, AttributeSet attrs) throws BadLocationException {
235                        synchronized(document) {
236                                String originalText = document.getText(0, document.getLength());
237                                super.replace(fb, offset, length, text, attrs);
238                                String newText = document.getText(0, document.getLength());
239                                boolean insertionOnly = length == 0;
240                                TagReloadType.SearchStringChange change = TagReloadType.createSearchStringChange(originalText, newText);
241                                change.setInsertionOnly(insertionOnly);
242                                getTCM().reloadTags(false, change);
243                        }
244                }
245
246                private TaggingContextManager getTCM() {
247                        if (tcm == null) {
248                                WindowIdentifier wi = new WindowIdentifier(windowId, null);
249                                tcm = TaggingContextManager.getInstance(wi);
250                        }
251                        return tcm;
252                }
253
254                @Override
255                public void remove(FilterBypass fb, int offset, int length) throws BadLocationException {
256                        synchronized(document) {
257                                String originalText = document.getText(0, document.getLength());
258                                super.remove(fb, offset, length);       //To change body of overridden methods use File | Settings | File Templates.
259                                String newText = document.getText(0, document.getLength());
260                                TagReloadType.SearchStringChange change = TagReloadType.createSearchStringChange(originalText, newText);
261                                change.setInsertionOnly(false);
262                                getTCM().reloadTags(false, change);
263                        }
264                }
265
266                private void compare(String originalText, String newText) {
267                        if (tcm == null) {
268                                WindowIdentifier wi = new WindowIdentifier(windowId, null);
269                                tcm = TaggingContextManager.getInstance(wi);
270                        }
271                        tcm.reloadTags(false, TagReloadType.createSearchStringChange(originalText, newText));
272                }

372        /**
373         * Based on the new partial tag we have, regenerate autocompletions.
374         * @param rawTags All the tags in the combo box
375         */
376        private void regenerateAutoCompletions(String rawTags) {
377                synchronized(autoCompletions) {
378                        _regenerateAutoCompletions(rawTags);
379                }
380        }
381        
382        private void _regenerateAutoCompletions(String rawTags) {
383                ExistingTagsPanel etp;
384                try {
385                        etp = this.getDisplayUI().getTagBarUI().getInsertionUI().getExistingTagsPanel();
386                }
387                catch(NullPointerException ex) {
388                        log.info("Bootstrapping from known EPT");
389                        etp = this.existingTagsPanel;
390                }
391//              etp = this.getDisplayUI().getTagBarUI().getInsertionUI().getExistingTagsPanel();
392                autoCompletions.clear();
393                Set<NamedOntClass> classes = new HashSet<NamedOntClass>();
394                Iterator<NamedOntModel> iterator = catalog.getNamedOntModels();
395                while(iterator.hasNext()) {
396                        NamedOntModel model = iterator.next();
397                        OWLOntology ontology = model.getOntology();
398                        for (OWLClass owlClass : ontology.getReferencedClasses()) {
399                                classes.add(new NamedOntClass(owlClass, ontology));
400                        }
401                }
402                
403                // Grab the fragment we're autocompleting on and parse all the other
404                // tags as well so we don't try to autocomplete on them.
405                List<String> currentTags = Util.splitTags(rawTags);
406                String fragment = currentTags.get(currentTags.size() - 1).toLowerCase();
407                
408                for (NamedOntClass cls : classes) {
409                        String tag = catalog.getLabelNameWithConcept(cls.getOntology(), cls.ontClass()).toLowerCase();
410                        boolean fragmentIsInThisTag = tag.indexOf(fragment) != -1;                      
411                        boolean tagAlreadyUsed = etp.hasTag(cls);
412                        
413                        if (fragmentIsInThisTag && !tagAlreadyUsed) {
414                                autoCompletions.add(new TruncatedExistingConceptCBItem(catalog, cls));
415                        }
416                }
417                
418//              sort(autoCompletions, fragment);
419        }

421        private List<ExistingConceptCBItem> sort(Set<ExistingConceptCBItem> autoCompletions, String fragment) {
422                List<ExistingConceptCBItem> acCopy = new ArrayList<ExistingConceptCBItem>(autoCompletions);
423                final String lowercaseFragment = fragment.toLowerCase();
424                Collections.sort(acCopy, new Comparator<ExistingConceptCBItem>() {
425                        public int compare(ExistingConceptCBItem o1, ExistingConceptCBItem o2) {
426                                String name1 = o1.asTag().getName().toLowerCase();
427                                String name2 = o2.asTag().getName().toLowerCase();
428                                int name1Position = name1.indexOf(lowercaseFragment);
429                                int name2Position = name2.indexOf(lowercaseFragment);
430                                if (name1Position == 0 && name2Position != 0) {
431                                        return -1;
432                                }
433                                if (name2Position == 0 && name1Position != 0) {
434                                        return 1;
435                                }
436                                return name1.compareTo(name2);
437                        }
438                });
439                return acCopy;
440        }

458                String fragment = currentTags.get(currentTags.size() - 1).toLowerCase();
