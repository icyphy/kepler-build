063        public void loadTagsFromTarget(boolean notifyListeners) {
064                Set<NamedOntClass> tags;
065                synchronized(this.objects) {
066                        tags = _getTags(this.objects);
067                }
068                for (NamedOntClass tag : tags) {
069                        this.saveTag(tag, notifyListeners);
070                }
071        }
072
073        private static Set<NamedOntClass> _getTags(Collection<NamedObj> targets) {
074                if (targets.isEmpty()) return Collections.emptySet();
075                Iterator<NamedObj> iterator = targets.iterator();
076                Set<NamedOntClass> tags = new HashSet<NamedOntClass>(Tagging.getTags(iterator.next()));
077                while (iterator.hasNext()) {
078                        tags.retainAll(Tagging.getTags(iterator.next()));
079                }
080                return tags;
081        }

126        public void saveTag(NamedOntClass cls, boolean notifyListeners) {
127                if (classes.contains(cls)) return;
128                
129                Collection<NamedObj> tmpObjects = new ArrayList<NamedObj>(objects);
130                
131                synchronized(tmpObjects) {
132                        Iterator<NamedObj> itr = tmpObjects.iterator();
133                        while (itr.hasNext()){
134                                NamedObj object = itr.next();
135                                if (itr.hasNext()){
136                                        informListeners(cls, object, notifyListeners, false);
137                                }
138                                else{
139                                        informListeners(cls, object, notifyListeners, true);
140                                }
141                        }
142                }
143                addLabelToBar(cls);
144        }

146        public void addLabelToBar(NamedOntClass cls) {
147                if (classes.contains(cls)) {
148                        return;
149                }
150                classes.add(cls);
151                DeletableColorIconLabel label;
152                boolean nonDeletable = !cls.isRemovable();
153                if (cls.isOrphaned()) {
154                        label = new DeletableColorIconLabel(cls, true, nonDeletable);
155                }
156                else {
157                        label = new DeletableColorIconLabel(cls, cls.getModel().getColor().getAwtColor(), nonDeletable);
158                }
159                label.addListener(this);
160                this.add(label, null, getIndex(this.getComponents(), label));
161        }
162
163        private int getIndex(Component[] components, DeletableColorIconLabel label) {
164                String b = label.getObject().toString();
165                for (int i = 0; i < components.length; i++) {
166                        String a = ((DeletableColorIconLabel) components[i]).getObject().toString();
167                        if (a.compareToIgnoreCase(b) > -1) {
168                                return i;
169                        }
170                }
171                return components.length;
172        }

181        public void informListeners(NamedOntClass cls, NamedObj object, boolean notifyListeners, boolean lastEvent) {
182                if (tagBarUI == null) {
183                        System.out.println("Skipping tag add listener; our listeners are null");
184                        return;
185                }
186                if (!notifyListeners) {
187                        return;
188                }
189                for (TagEventListener listener : this.tagBarUI.getTagEventListeners()) {
190                        listener.tagAdded(new TagEvent(object, cls, lastEvent));
191                }
192        }

195                
196                //TODO need to not loop on objects itself here for now. WRM currently 
197                // resets tag context during add and remove tag events, thus changing
198                // objects, and causing a ConcurrentModificationException here
199                Collection<NamedObj> tmpObjects = new ArrayList<NamedObj>(objects);
200                synchronized(tmpObjects) {
201                        for (NamedObj object : tmpObjects) {
202                                addTag(cls, object);
203                        }
204                }
205        }

243        private void removeLabel(DeletableColorIconLabel label) {
244                DeletableColorIconLabel thisLabel = null;
245                for (Component component : this.getComponents()) {
246                        DeletableColorIconLabel l = (DeletableColorIconLabel) component;
247                        if (l.getObject().equals(label.getObject())) {
248                                thisLabel = l;
249                                break;
250                        }
251                }
252                if (thisLabel == null) {
253                        log.warn("Unable to delete label: " + ((NamedOntClass) label.getObject()).getName());
254                        return;
255                }
256                this.remove(thisLabel);
257                this.validate();
258                this.repaint();
259                //noinspection SuspiciousMethodCalls
260                this.classes.remove(label.getObject());         
261        }
262        
263        public void labelDeleted(DeletableColorIconLabelEvent event, boolean informListeners) {
264                
265                //TODO need to not loop on objects itself here for now. WRM currently 
266                // resets tag context during add and remove tag events, thus changing
267                // objects, and causing a ConcurrentModificationException here
268                Collection<NamedObj> tmpObjects = new ArrayList<NamedObj>(objects);
269                
270                removeLabel(event.getLabel());
271                NamedOntClass cls = (NamedOntClass) event.getLabel().getObject();
272                List<TagEventListener> listeners = null;
273                if (tagBarUI != null) {
274                        listeners = tagBarUI.getTagEventListeners();
275                }
276                synchronized(tmpObjects) {
277                        Iterator<NamedObj> itr = tmpObjects.iterator();
278                        while (itr.hasNext()){
279                                NamedObj object = itr.next();
280                                if (itr.hasNext()){
281                                        Tagging.removeTag(object, cls, false);
282                                }
283                                else{
284                                        Tagging.removeTag(object, cls, true);
285                                }
286                                if (listeners == null || !informListeners) continue;
287                                for (TagEventListener listener : listeners) {
288                                        if (itr.hasNext()){
289                                                listener.tagRemoved(new TagEvent(object, cls, false));
290                                        }
291                                        else{
292                                                listener.tagRemoved(new TagEvent(object, cls, true));
293                                        }
294                                }
295                        }
296                }
297        }

313        public void changeColor(NamedOntModel model, Color color) {
314                for (Component component : this.getComponents()) {
315                        if (!(component instanceof DeletableColorIconLabel)) {
316                                continue;
317                        }
318                        DeletableColorIconLabel label = (DeletableColorIconLabel) component;
319                        if (((NamedOntClass) label.getObject()).getOntology().equals(model.getOntology())) {
320                                label.setColor(color.getAwtColor());
321                                label.resetColor();
322                        }
323                }
324        }
