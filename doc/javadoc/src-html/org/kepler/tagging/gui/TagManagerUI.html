030package org.kepler.tagging.gui;
031
032
033import java.awt.Dimension;
034import java.awt.event.ActionEvent;
035import java.awt.event.ActionListener;
036import java.lang.ref.WeakReference;
037import java.util.ArrayList;
038import java.util.Collections;
039import java.util.List;
040
041import javax.swing.Box;
042import javax.swing.BoxLayout;
043import javax.swing.JButton;
044import javax.swing.JFrame;
045import javax.swing.JPanel;
046import javax.swing.border.EmptyBorder;
047
048/**
049 * @created April, 2009
050 */
051public class TagManagerUI extends JPanel implements ActionListener {
052
053    private JFrame parent; 
054    private JButton okButton = new JButton("OK");
055
056    /**
057     * @param owner the parent frame
058     */
059        public TagManagerUI(JFrame parent) {
060        super();
061        this.parent = parent;
062        initialize();
063    }
064    
065    /**
066     * Initialize the layout of the pane
067     */
068    public void initialize() { 
069        int width = 400;
070        int height = 275;
071        // ontology display
072        OntologyDisplayUI ontoDisplay = new OntologyDisplayUI();
073        ontoDisplay.setPreferredSize(new Dimension(width, height));
074        OntologyEditorUI ontoEditor = new OntologyEditorUI();
075        ontoEditor.setPreferredSize(new Dimension(width, height));
076        ontoDisplay.addSelectionListener(ontoEditor);
077        // buttons
078        JPanel botPane = new JPanel();
079        botPane.setLayout(new BoxLayout(botPane, BoxLayout.X_AXIS));
080        botPane.add(Box.createHorizontalGlue());
081        botPane.add(okButton);
082        okButton.setActionCommand("ok");        
083        okButton.addActionListener(this);
084        // overall pane
085        this.setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
086        this.setBorder(new EmptyBorder(10,10,10,10));
087        this.add(ontoDisplay);
088        this.add(Box.createRigidArea(new Dimension(0, 10)));
089        this.add(ontoEditor);
090        this.add(Box.createRigidArea(new Dimension(0, 10)));
091        this.add(botPane);
092    }
093
094
095    /**
096     */
097    public void actionPerformed(ActionEvent e) {
098        if ("ok".equals(e.getActionCommand())) 
099            dispose(); 
100    } 
101
102
103    /**
104     * Dispose of the manager (requires access to the parent frame)
105     */
106    public void dispose() {
107        if(parent != null)
108            parent.dispose();
109    }
110
111        public static void showTagManagerFrame() {
112                getTagManagerFrame().setVisible(true);
113        }
114        
115        public static void hideTagManagerFrame() {
116                getTagManagerFrame().setVisible(false);
117        }
118        
119        public static void addListener(WeakReference<TagBarDisplayUI> tagBarDisplayUI) {
120                tagBarDisplayUIs.add(tagBarDisplayUI);
121        }
122        
123        public static void removeListener(WeakReference<TagBarDisplayUI> tagBarDisplayUI) {
124                tagBarDisplayUIs.remove(tagBarDisplayUI);
125        }
126        
127        public static List<WeakReference<TagBarDisplayUI>> getTagBarDisplayUIs() {
128                return tagBarDisplayUIs;
129        }
130        
131        private synchronized static JFrame getTagManagerFrame() {
132                if (frame == null) {
133                        frame = new JFrame();
134                        frame.add(new TagManagerUI(frame));
135                        frame.pack();                   
136                }
137                return frame;
138        }
139        
140        private static JFrame frame = null;
141        private static final List<WeakReference<TagBarDisplayUI>> tagBarDisplayUIs = Collections.synchronizedList(new ArrayList<WeakReference<TagBarDisplayUI>>());
142
143} // TagManagerUI
