112        /**
113         * See if the type label has a size. This is figured internally through
114         * complex interactions between the component and layout managers, so it
115         * is difficult to know for sure when this will be the case. If it has a
116         * size, save it.
117         * @return Were the label dimensions saved successfully?
118         */
119        private boolean saveLabelDimension() {
120                if (largeLabelDimension != null) {
121                        return true;
122                }
123                Dimension size = typeLabel.getSize();
124                double zero = 0;
125                if (Double.compare(size.getHeight(), zero) == 0 && Double.compare(size.getWidth(), zero) == 0) {
126                        return false;
127                }
128                largeLabelDimension = size;
129                return true;
130        }
131
132        /**
133         * Make sure the label always stays the size that it currently is. We
134         * don't want the label to make other components to jump around as it
135         * shrinks and grows with different text contents. The only reason that
136         * the dimension can be saved straight off (as soon as it's initialized,
137         * that is) is because we happen to be initializing it with the longer of
138         * the two text strings it will have in it. If another string is
139         * introduced that is longer, special measures will have to be taken to
140         * ensure that largeLabelDimension is initialized to the size of the
141         * label set to this largest string.
142         * @return Was the label dimension set successfully?
143         */
144        private boolean setLabelSize() {
145                if (saveLabelDimension()) {
146                        typeLabel.setMinimumSize(largeLabelDimension);
147                        typeLabel.setMaximumSize(largeLabelDimension);
148                        typeLabel.setPreferredSize(largeLabelDimension);
149                        return true;
150                }
151                else {
152                        return false;
153                }
154        }
155
156        /**
157         * Create a thread that tries every so often to freeze the size of
158         * typeLabel. This is to make sure we freeze it once it is fully
159         * initialized and laid out, so we can tell it accurately what dimensions
160         * to be.
161         */
162        private void spinOffThread() {
163                class LabelCheckerRunnable implements Runnable {
164                        public void run() {
165                                while (!setLabelSize()) {
166                                        try {
167                                                Thread.sleep(1000);
168                                        }
169                                        catch(InterruptedException ignored) {}
170                                }
171                        }
172                }
173                
174                (new Thread(new LabelCheckerRunnable())).start();
175        }
