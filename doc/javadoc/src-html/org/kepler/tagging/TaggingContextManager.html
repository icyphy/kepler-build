074        private class ComponentFinder<T> {
075                public ComponentFinder(Component[] componentsArray, Class<T> cls) {
076                        for (Component comp : componentsArray) {
077                                try {
078                                        component = cls.cast(comp);
079                                        match = true;
080                                }
081                                catch(ClassCastException ignored) {}
082                        }
083                }
084
085                public boolean matchFound() {
086                        return match;
087                }
088
089                public T getComponent() {
090                        return component;
091                }
092
093                private boolean match = false;
094                private T component;
095        }

097        public TagBarInsertionUI getTagBarInsertionUI() {
098                KeplerGraphFrame kgf = this.windowIdentifier.getKeplerGraphFrame();
099                JToolBar toolbar = kgf.getToolBar();
100                ComponentFinder<TaggingPanel> finder1 = new ComponentFinder<TaggingPanel>(toolbar.getComponents(), TaggingPanel.class);
101                if (!finder1.matchFound()) {
102                        log.error("TaggingPanel not found in toolbar");
103                        return null;
104                }
105                ComponentFinder<TagBarInsertionUI> finder2 = new ComponentFinder<TagBarInsertionUI>(finder1.getComponent().getComponents(), TagBarInsertionUI.class);
106                return finder2.getComponent();
107        }

124        public boolean reloadTags(boolean verbose, TagReloadType tagReloadType) {
125                if (this.getTagBarInsertionUI() == null) {
126                        log.info("Unable to reload tags");
127                        return false;
128                }
129                
130                Set<CBItems.ExistingConceptCBItem> autoCompletions = this.getTagBarInsertionUI().getComboBox().getAutoCompletions();
131                
132                ExistingTagsPanel existingTagsPanel = this.getTagBarInsertionUI().getExistingTagsPanel();
133                NamedOntClass tag = tagReloadType.getTag();
134                
135                if (tagReloadType.getType() == TagReloadType.Type.TAG_ADDED) {
136//                      System.out.println("reloadTags type=TAG_ADDED");
137                        // A tag was added, so we need to remove that from the auto-completions.
138                        CBItems.ExistingConceptCBItem item = new CBItems.ExistingConceptCBItem(OntologyCatalog.instanceNoInitialize(), tag);
139                        boolean removed = autoCompletions.remove(item);
140                        if (!removed) {
141                                log.warn("Could not remove tag '" + item.asTag().getName() + "' from auto-completions");
142                        }
143                        existingTagsPanel.saveTag(tag, verbose);
144                        TagCache.instance().addToCache(tag);
145                }
146                else if (tagReloadType.getType() == TagReloadType.Type.TAG_ADDED_AND_CLEAR_SEARCH) {
147//                      System.out.println("reloadTags type=TAG_ADDED_AND_CLEAR_SEARCH");
148                        existingTagsPanel.saveTag(tag, verbose);
149                        TagCache.instance().addToCache(tag);
150                        autoCompletions.clear();
151                        autoCompletions.addAll(TagCache.instance().getOntClasses());
152                        for (NamedOntClass cls : Tagging.getTags(tagReloadType.getTarget())) {
153                                boolean removed = autoCompletions.remove(new CBItems.ExistingConceptCBItem(OntologyCatalog.instance(), cls));
154                                if (!removed) {
155                                        log.warn("Could not remove tag '" + cls.getName() + "' from auto-completions");
156                                }
157                        }
158                        this.getTagBarInsertionUI().getComboBox().emptyTagField();
159                }
160                else if (tagReloadType.getType() == TagReloadType.Type.TAG_REMOVED) {
161//                      System.out.println("reloadTags type=TAG_REMOVED");
162                        OntologyCatalog catalog = OntologyCatalog.instanceNoInitialize();
163                        autoCompletions.add(new CBItems.TruncatedExistingConceptCBItem(catalog, tagReloadType.getTag()));
164                        existingTagsPanel.removeTag(tag, verbose);
165                }
166                else if (tagReloadType.getType() == TagReloadType.Type.COMPLETE) {
167//                      System.out.println("reloadTags type=COMPLETE");
168                        this.getTagBarInsertionUI().getComboBox().regenerateAutoCompletions();
169                        existingTagsPanel.clear();
170                        existingTagsPanel.revalidate();
171                        existingTagsPanel.loadTagsFromTarget(verbose);
172                }
173                else if (tagReloadType.getType() == TagReloadType.Type.TAG_CACHE_UPDATE) {
174                        TagCache.instance().forceRefilter();
175                        String text = ((JTextField) this.getTagBarInsertionUI().getComboBox().getEditor().getEditorComponent()).getText();
176                        updateAutocompletions(text, autoCompletions, false);
177                        
178                }
179                else if (tagReloadType.getType() == TagReloadType.Type.TAG_CREATED) {
180                        String text = ((JTextField) this.getTagBarInsertionUI().getComboBox().getEditor().getEditorComponent()).getText();
181                        NamedOntClass noc = tagReloadType.getTag();
182                        if (noc.getName().toLowerCase().contains(text.toLowerCase()) && shouldShowModel(noc.getModel())) {
183                                System.out.println("Adding tag: " + noc.getName());
184                                System.out.println("Orig: " + autoCompletions.size());
185                                autoCompletions.add(new CBItems.ExistingConceptCBItem(OntologyCatalog.instance(), noc));
186                                System.out.println("Now: " + autoCompletions.size());
187                        }
188                }
189                else if (tagReloadType.getType() == TagReloadType.Type.TAG_DELETED) {
190                        String text = ((JTextField) this.getTagBarInsertionUI().getComboBox().getEditor().getEditorComponent()).getText();
191                        NamedOntClass noc = tagReloadType.getTag();
192                        if (noc.getName().toLowerCase().contains(text.toLowerCase())) {
193                                System.out.println("Removing tag: " + noc.getName());
194                                System.out.println("Orig: " + autoCompletions.size());
195                                CBItems.ExistingConceptCBItem item = new CBItems.ExistingConceptCBItem(OntologyCatalog.instance(), noc);
196                                autoCompletions.remove(item);
197                                System.out.println("Now: " + autoCompletions.size());
198                        }
199                }
200                else if (tagReloadType.getType() == TagReloadType.Type.SEARCH_STRING) {
201//                      System.out.println("reloadTags type=SEARCH_STRING");
202                        // NOTE: Change auto-completions accordingly, but no need to change ETP
203                        TagReloadType.SearchStringChange change = (TagReloadType.SearchStringChange)tagReloadType;
204                        updateAutocompletions(change.getNewString(), autoCompletions, change.isInsertionOnly());
205                }
206                else if (tagReloadType.getType() == TagReloadType.Type.TAG_TARGET) {
207                        // NOTE: Change auto-completions, and also ETP as needed
208                        Set<NamedOntClass> oldClasses = existingTagsPanel.getClasses();
209                        Set<NamedOntClass> newClasses = Tagging.getTags(tagReloadType.getTargets());
210                        transitionBetweenTagStates(oldClasses, newClasses, existingTagsPanel, autoCompletions);
211//                      System.out.println("reloadTags type=TAG_TARGET");
212                }
213                
214                this.getTagBarInsertionUI().getComboBox().setDropDown();
215                // Have to do this after to make sure this actually gets done in a timely manner
216                existingTagsPanel.revalidate();
217                return true;            
218        }

220        private static boolean shouldShowModel(NamedOntModel model) {
221                Iterator<NamedOntModel> iterator = OntologyCatalog.instance().getTagBarOntModels();
222                Set<NamedOntModel> models = new HashSet<NamedOntModel>();
223                while (iterator.hasNext()) {
224                        models.add(iterator.next());
225                }
226                return models.contains(model);
227        }

229        private void updateAutocompletions(String newString, Set<CBItems.ExistingConceptCBItem> autoCompletions, boolean insertionOnly) {
230                
231                Set<CBItems.ExistingConceptCBItem> compareTo = new TreeSet<CBItems.ExistingConceptCBItem>(new Comparator<CBItems.ExistingConceptCBItem>() {
232                        public int compare(CBItems.ExistingConceptCBItem o1, CBItems.ExistingConceptCBItem o2) {
233                                return o1.getConceptName().compareToIgnoreCase(o2.getConceptName());
234                        }
235                });
236                
237                // Put all the elements in this new collection for reference
238                if (insertionOnly) {
239                        // We only need to compare against the existing auto-completions,
240                        // since by definition we are *more* constrained than we were a
241                        // moment ago.
242                        compareTo.addAll(autoCompletions);
243                }
244                else {
245                        compareTo.addAll(TagCache.instance().getOntClasses());
246                }
247                // Then clear the old collection
248                autoCompletions.clear();
249                // Then iterate through the copy, adding back those that still belong
250                // in the auto-completions
251                for (CBItems.ExistingConceptCBItem item : compareTo) {
252                        if (item.getCaseInsensitiveName().contains(newString.toLowerCase())) {
253                                autoCompletions.add(item);
254                        }
255                }
256                
257        }

259        private static void transitionBetweenTagStates(Set<NamedOntClass> oldClasses, Set<NamedOntClass> newClasses, ExistingTagsPanel existingTagsPanel, Set<CBItems.ExistingConceptCBItem> autoCompletions) {
260                // classesAdded = newClasses - oldClasses
261                // classesRemoved = oldClasses - newClasses
262                // Add classesAdded to the tagbar and remove classesRemoved
263                // Remove classesAdded from the auto-completions and add back classesRemoved
264                SetChange<NamedOntClass> changes = new SetChange<NamedOntClass>(oldClasses, newClasses);
265                
266                Set<NamedOntClass> classesRemoved = changes.getRemoved();
267                Set<NamedOntClass> classesAdded = changes.getAdded();
268
269                existingTagsPanel.clear();
270                existingTagsPanel.revalidate();
271                existingTagsPanel.loadTagsFromTarget(false);
272                
273                for (NamedOntClass newClass : classesAdded) {
274                        autoCompletions.remove(new CBItems.ExistingConceptCBItem(OntologyCatalog.instance(), newClass));
275                }
276                for (NamedOntClass oldClass : classesRemoved) {
277                        autoCompletions.add(new CBItems.ExistingConceptCBItem(OntologyCatalog.instance(), oldClass));
278                }
279        }

307        public static TaggingContextManager getInstance(WindowIdentifier windowIdentifier) {
308                synchronized(instances) {
309                        if (!instances.containsKey(windowIdentifier)) {
310                                instances.put(windowIdentifier, new TaggingContextManager(windowIdentifier));
311                        }
312                }
313                return instances.get(windowIdentifier);
314        }
315        
316        public static Collection<TaggingContextManager> getInstances() {
317                synchronized(instances) {
318                        return new ArrayList<TaggingContextManager>(instances.values());
319                }
320        }

328        public static void dispose(KeplerGraphFrame parent){
329                log.debug("instances's size before removing:" + instances.size());
330                 Object[] keyArray = instances.keySet().toArray();      
331                        for (int i = 0; i < keyArray.length; i++) {
332                                if (keyArray[i] == null)
333                                        log.debug("keyArray[i] == null");
334                                else {
335                                        WindowIdentifier winId = (WindowIdentifier)keyArray[i];
336                                         if (winId.getKeplerGraphFrame() == parent)
337                                         {
338                                                 instances.remove(winId);
339                                                 log.debug(winId + " is removed from instances");
340                                         }
341                                }       
342                        }
343                        log.debug("instances's size after removing:" + instances.size());
344        }
