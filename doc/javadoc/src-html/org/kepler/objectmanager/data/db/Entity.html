101                this(id, name, description, null);

105                }
106                if (orientation != null) {
107                        this.orientation = orientation;
108                }
109                this.numRecords = numRecords;
110        }

294        /**
295         * Serialize the data item in XML format.
296         */
297        public String toXml() {
298                StringBuffer x = new StringBuffer();
299                x.append("<table-entity id=\"");
300                x.append(getId());
301                x.append("\"");
302                if (multiple == true) {
303                        x.append(" multiple=\"true\"");
304                }
305                x.append(">\n");
306                appendElement(x, "entityName", getName());
307                appendElement(x, "entityType", getDataType());
308                appendElement(x, "entityDescription", getDefinition());
309                Attribute[] atts = getAttributes();
310                for (int i = 0; i < atts.length; i++) {
311                        x.append(atts[i].toXml());
312                }
313                x.append("</table-entity>\n");
314
315                return x.toString();
316        }
