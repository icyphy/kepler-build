030package org.kepler.objectmanager.data;
031
032public class DataType {
033        /**
034         * the following constants are definitions for the dataType type as seen in
035         * the config.xml file
036         */
037        /** represents numbers that are not necessarily whole */
038        public static final int REAL = 0;
039        /** represents an integer {...,-3,-2,-1,0,1,2,3,...} */
040        public static final int INTEGER = 1;
041        /** text type. In the config.xml file, this is represented as 'nan' */
042        public static final int STRING = 2;
043
044        // number type definitionation
045        public static final String REALTYPE = "real";
046        public static final String INTEGERTYPE = "integer";
047        public static final String STRINGTYPE = "nan";
048        public static final String NATURALTYPE = "natural";
049        public static final String WHOLETYPE = "whole";
050        public static final String DATETIMETYPE = "datetime";
051        private static final String[] NUMBERTYPEARRAY = { REALTYPE, INTEGERTYPE,
052                        STRINGTYPE, NATURALTYPE, WHOLETYPE, DATETIMETYPE };
053
054        /**
055         * The following definitions are definitions of the types that should be
056         * used in the class Attribute for defining what kind of object an attribute
057         * uses.(DataType name)
058         */
059        public static final String INT = "INTEGER";
060        public static final String LONG = "LONG";
061        public static final String FLOAT = "FLOAT";
062        public static final String DOUBLE = "DOUBLE";
063        public static final String STR = "STRING";
064        public static final String DATETIME = "DATETIME";
065
066        // private variables
067        private double min;
068        private double max;
069        private String numberType; // this should be one of the predefined
070                                                                // numberType in configure.xml
071        private String[] aliases;
072        private String name;
073
074        /**
075         * Constructor. Note that the type should be taken from one of the public
076         * static variables in this class.
077         * 
078         * @param min
079         *            the minimum value for this data type
080         * @param max
081         *            the maximum value for this data type
082         * @param precision
083         *            the precision for this data type
084         * @param type
085         *            the type of number represented by this data type. This value
086         *            should be taken from the public static variables defined in
087         *            this class.
088         * @param aliases
089         *            one or more names given to this type.
090         * @param name
091         *            the proper monarch name of the DataType. This should equate to
092         *            one of (INT, LONG, FLOAT, DOUBLE, STRING) from the above
093         *            static definitions.
094         */
095        public DataType(String name, double min, double max, String type,
096                        String[] aliases) throws Exception {
097                this.min = min;
098                this.max = max;
099                setupNumberType(type);
100                this.aliases = aliases;
101                this.name = name;
102        }
103
104        /**
105         * return the minimum in the range for the type
106         */
107        public String getName() {
108                return name;
109        }
110
111        /**
112         * return the minimum in the range for the type
113         */
114        public double getMin() {
115                return min;
116        }
117
118        /**
119         * return the maximum in the range for the type
120         */
121        public double getMax() {
122                return max;
123        }
124
125        /**
126         * return the type of the data type. This shoud be one of the constants
127         * (REAL, INTEGER, STRING) defined above.
128         */
129        public String getNumberType() {
130                return numberType;
131        }
132
133        public String[] getNames() {
134                return aliases;
135        }
136
137        /**
138         * returns a string representation of this DataType
139         */
140        public String toString() {
141                StringBuilder buf = new StringBuilder("name: ");
142        buf.append(name);
143                buf.append("\naliases: ");
144                for (int i = 0; i < aliases.length; i++) {
145                        buf.append(aliases[i]);
146                        if (i != aliases.length - 1) {
147                                buf.append(", ");
148                        }
149                }
150                buf.append("\nmin: ");
151        buf.append(min);
152                buf.append("\nmax: ");
153        buf.append(max);
154                buf.append("\ntype: ");
155        buf.append(numberType);
156
157                return buf.toString();
158        }
159
160        /*
161         * This method will lookup the array and found the pre-defined number type
162         * If couldn't find it, throw a exception
163         */
164        private void setupNumberType(String numType)
165                        throws UnresolvableTypeException {
166                if (numType == null || numType.equals("")) {
167                        throw new UnresolvableTypeException("Couldn't find numberType for "
168                                        + numType);
169                }
170                boolean findIt = false;
171                int length = NUMBERTYPEARRAY.length;
172                for (int i = 0; i < length; i++) {
173                        String valueInArray = NUMBERTYPEARRAY[i];
174                        if (valueInArray != null && valueInArray.equals(numType)) {
175                                this.numberType = numType;
176                                findIt = true;
177                                break;
178                        }
179                }
180                // if findIt is false, there is no numbertype for the given numType,
181                // throw a exception
182                if (!findIt) {
183                        throw new UnresolvableTypeException("Couldn't find numberType for "
184                                        + numType);
185                }
186        }
187}
