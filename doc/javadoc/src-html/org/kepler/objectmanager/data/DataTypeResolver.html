030package org.kepler.objectmanager.data;
031
032import java.util.List;
033
034import org.kepler.configuration.ConfigurationManager;
035import org.kepler.configuration.ConfigurationProperty;
036
037/**
038 * This class takes in information from the metadata parsers and tries to
039 * intelligently guess as to the data type of an attribute based on provided
040 * metadata. since different metadata standards provide differing levels of
041 * complexity in their attribute level descriptions, some guesses may be better
042 * than others, but this class, should at least be able to distinguish between,
043 * whole (integer), real (float) and text (string) types.
044 */
045public class DataTypeResolver
046{
047  private static DataTypeResolver resolver = null; // singleton variable
048  private static String DEFAULT_TYPE = "STRING";
049
050  private DataType[] dataTypes;
051
052  /**
053   * constructor
054   */
055  private DataTypeResolver()
056  {
057    try
058    {
059      ConfigurationManager confMan = ConfigurationManager.getInstance();
060      ConfigurationProperty commonProperty = confMan
061          .getProperty(ConfigurationManager.getModule("common"));
062      ConfigurationProperty dtdProperty = commonProperty
063          .getProperty("dataTypeDictionary");
064      List types = dtdProperty.getProperties();
065
066      dataTypes = new DataType[types.size()];
067      for (int i = 0; i < types.size(); i++)
068      {
069        ConfigurationProperty type = (ConfigurationProperty) types.get(i);
070        ConfigurationProperty nameProp = type.getProperty("name");
071        ConfigurationProperty numberTypeProp = type.getProperty("numberType");
072        if (nameProp == null || numberTypeProp == null)
073        {
074          throw new RuntimeException("Expecting a name and numberType "
075              + "child of dataType in " + "configuration.xml.");
076        }
077
078        String name = nameProp.getValue();
079        String numberType = numberTypeProp.getValue();
080        List aliasList = type.getProperties("alias");
081        String[] aliases = new String[aliasList.size()];
082        for (int j = 0; j < aliasList.size(); j++)
083        {
084          aliases[j] = ((ConfigurationProperty) aliasList.get(j)).getValue();
085        }
086        ConfigurationProperty numericTypeProp = type.getProperty("numericType");
087        String maxval = null;
088        String minval = null;
089        String textencoding = null;
090        if (numericTypeProp != null)
091        {
092          ConfigurationProperty minValProp = numericTypeProp
093              .getProperty("minValue");
094          ConfigurationProperty maxValProp = numericTypeProp
095              .getProperty("maxValue");
096          if (minValProp == null || maxValProp == null)
097          {
098            throw new RuntimeException("Expecting a minValue and "
099                + "maxValue children of numericType in configuration.xml.");
100          }
101          minval = minValProp.getValue();
102          maxval = maxValProp.getValue();
103        }
104        else
105        {
106          ConfigurationProperty textTypeProp = type.getProperty("textType");
107          if (textTypeProp == null)
108          {
109            throw new RuntimeException(
110                "Each dataType defined in the "
111                    + "config.xml file must have either a textType or a numericType "
112                    + "defined to be valid.");
113          }
114          ConfigurationProperty textEncodingProp = textTypeProp
115              .getProperty("encoding");
116          if (textEncodingProp == null)
117          {
118            throw new RuntimeException("The textType node must have an "
119                + "encoding node as a child in config.xml.");
120          }
121
122          textencoding = textEncodingProp.getValue();
123        }
124
125        double min = -1.0;
126        double max = -1.0;
127        if (minval != null && maxval != null)
128        {
129          min = Double.parseDouble(minval);
130          max = Double.parseDouble(maxval);
131        }
132        //System.out.println("name: " + name + " min: " + min + " max: " + max + " numberType: " + numberType + " aliases: " + aliases);
133        DataType dt = new DataType(name, min, max, numberType, aliases);
134        dataTypes[i] = dt;
135
136      }
137    }
138    catch (javax.xml.transform.TransformerException te)
139    {
140      throw new RuntimeException("Error executing xpath expression in "
141          + "DataTypeResolver.");
142    }
143    catch (Exception e)
144    {
145      e.printStackTrace();
146      throw new RuntimeException("Exception in DataTypeResolver.");
147    }
148  }
149
150  /**
151   * returns a singleton instance of the DataTypeResolver
152   */
153  public static DataTypeResolver instanceOf()
154  {
155    if (resolver == null)
156    {
157      resolver = new DataTypeResolver();
158    }
159    return resolver;
160  }
161
162  /**
163   * returns an array of the data types defined in the config.xml file
164   */
165  public DataType[] getDataTypes()
166  {
167    return dataTypes;
168  }
169
170  /**
171   * this method will look up the numberType parameter which come from
172   * metadata to find the type defined in configure.xml
173   * 
174   * @param min
175   *          the minimum value for the attribute
176   * @param max
177   *          the maximum value for the attribute
178   * @param typeName
179   *          the name of the numberType declared in the metadata
180   * @throws UnresolvableTypeException
181   * @return one of the DataType constants (INT, LONG, FLOAT, DOUBLE, STRING)
182   *         declared in DataType.java
183   */
184  public DataType resolveDataType(String numberType, Double min, Double max)
185      throws UnresolvableTypeException
186  {
187    DataType type = numberTypeLookup(numberType, min, max);
188    return type;
189  }
190
191  /**
192   * resolves the type based solely on the name. This does an alias lookup on
193   * the types in config.xml.
194   * 
195   * @param name
196   *          the name of the type to resolve.
197   */
198  public DataType resolveTypeByAlias(String name, Double min, Double max)
199      throws UnresolvableTypeException
200  {
201    DataType type = aliasLookup(name, min, max);
202    return type;
203  }
204
205  /*
206   * finad a DataType based on a numberType and range
207   */
208  private DataType numberTypeLookup(String type, Double min, Double max)
209      throws UnresolvableTypeException
210  {
211    if (type == null)
212    {
213      throw new UnresolvableTypeException("The given number type is null");
214    }
215    if (min == null || max == null)
216    {
217      for (int i = 0; i < dataTypes.length; i++)
218      {
219        String numberType = dataTypes[i].getNumberType();
220        if (numberType != null && type.trim().equals(numberType))
221        {
222          return dataTypes[i];
223        }
224
225      }
226    }
227    else
228    {
229      double minNum = min.doubleValue();
230      double maxNum = max.doubleValue();
231      for (int i = 0; i < dataTypes.length; i++)
232      {
233        String numberType = dataTypes[i].getNumberType();
234        if (numberType != null && type.trim().equals(numberType)
235            && rangeLookup(minNum, maxNum, dataTypes[i]))
236        {
237          return dataTypes[i];
238        }
239
240      }
241
242    }
243    throw new UnresolvableTypeException(
244        "The number type '"
245            + type.trim()
246            + "' cannot be "
247            + "resolved with any type in the config.xml file.  Please add a numberType "
248            + "for this type to one of the DataType elements in config.xml.");
249  }
250
251  /**
252   * find a DataType based on an alias name and range
253   */
254  private DataType aliasLookup(String typeName, Double min, Double max)
255      throws UnresolvableTypeException
256  {
257    if (typeName == null)
258    {
259      throw new UnresolvableTypeException("The given alias type is null");
260    }
261    if (min == null || max == null)
262    {
263      for (int i = 0; i < dataTypes.length; i++)
264      {
265        String aliases[] = dataTypes[i].getNames();
266        String name = dataTypes[i].getName();
267        for (int j = 0; j < aliases.length; j++)
268        {
269          if (typeName.trim().equals(aliases[j].trim())
270              || typeName.trim().equals(name))
271          {
272
273            return dataTypes[i];
274          }
275        }
276      }
277    }
278    else
279    {
280      double minNum = min.doubleValue();
281      double maxNum = max.doubleValue();
282      for (int i = 0; i < dataTypes.length; i++)
283      {
284        String aliases[] = dataTypes[i].getNames();
285        String name = dataTypes[i].getName();
286        for (int j = 0; j < aliases.length; j++)
287        {
288          if ((typeName.trim().equals(aliases[j].trim()) || typeName.trim()
289              .equals(name))
290              && rangeLookup(minNum, maxNum, dataTypes[i]))
291          {
292
293            return dataTypes[i];
294          }
295        }
296      }
297    }
298    throw new UnresolvableTypeException(
299        "The type '"
300            + typeName.trim()
301            + "' cannot be "
302            + "resolved with any type in the config.xml file.  Please add an alias "
303            + "for this type to one of the DataType elements in config.xml.");
304  }
305
306  /**
307   * If the given range fits the type.
308   */
309  private boolean rangeLookup(double min, double max, DataType type)
310  {
311    boolean fit = false;
312    if (min > max)
313    {
314      return fit;
315    }
316
317    double dtmin = type.getMin();
318    double dtmax = type.getMax();
319    if (dtmin < min && dtmax > max)
320    {
321      fit = true;
322    }
323    return fit;
324
325  }
