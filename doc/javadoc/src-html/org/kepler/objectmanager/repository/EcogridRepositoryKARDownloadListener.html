082        /**
083         * called when a drag is dropped over the canvas
084         */
085        public void drop(DropTargetDropEvent dtde) { 
086                if (!org.kepler.gui.SearchUIJPanel.SEARCHREPOS) {
087                        // if we're not searching the repository, don't do this
088                        return;
089                }
090                try {
091                        // get the iterator of objects being dragged
092                        Iterator iterator = null;
093                        try {
094                                iterator = (Iterator) dtde.getTransferable().getTransferData(
095                                                PtolemyTransferable.namedObjFlavor);
096                        } catch (Exception e) {
097                                System.out.println("e: " + e.getMessage());
098                        }
099
100                        NamedObj dropObj = null;
101                        while (iterator.hasNext()) { // get the namedobj from the dragged
102                                                                                        // obj
103                                dropObj = (NamedObj) iterator.next();
104                                // System.out.println("namedObj: " + dropObj.getName());
105                        }
106
107                        // check to see if the obj has a karid attribute
108                        StringAttribute karIdAtt = (StringAttribute) dropObj
109                                        .getAttribute("karId");
110                        if (karIdAtt == null) { 
111                                // if there is no karId then ignore this actor. 
112                                // all remote actors should have the karId attribute
113                                return;
114                        }
115
116                        KeplerLSID karLSID = null;
117                        karLSID = new KeplerLSID(karIdAtt.getExpression());
118
119                        System.out.println("karId: " + karLSID);
120
121                        // if it does, check to see if that kar is already cached
122                        KARCacheManager kcm = KARCacheManager.getInstance();
123                        if (kcm.isCached(karLSID)) {
124                                // we've already got it
125                        } else {
126                                // if it isn't, download it and cache it.
127                                Repository repository = getRepository();                                
128                                InputStream karIs = repository.get(karLSID, ComponentLibraryTab.AUTHENTICATE);
129                                File tmpFile = CacheManager.getInstance().getTempFile();
130                                FileOutputStream fos = new FileOutputStream(tmpFile);
131                                // write the kar file to a temp file
132                                CacheUtil.writeInputStreamToOutputStream(karIs, fos);
133                                // put the kar file into the cache.
134                                KARFile karf = new KARFile(tmpFile);
135                                LocalRepositoryManager.getInstance().getSaveRepository();
136                                
137                                // TODO redo this whole method!  it's crap
138                                
139                                // done
140                        }
141                } catch (Exception e) {
142                        System.out.println("Error getting kar file on actor drop: "
143                                        + e.getMessage());
144                        e.printStackTrace();
145                        return;
146                }
147        }
148

154        }
155
156        /**
157         * get the ecogrid repository
158         */
159        private static Repository getRepository() throws Exception {
160                RepositoryManager rmanager = RepositoryManager.getInstance();
161                Iterator repoList = rmanager.repositoryList();
162                // HACKALERT: get the first repository
163                Repository repository = (Repository) repoList.next();
164                if (!(repository instanceof EcogridRepository)) {
165                        throw new Exception("Right now, we can only sync ecogrid "
166                                        + "repositories.  Sorry.");
167                }
168                return repository;
169        }
