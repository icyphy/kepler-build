038import org.kepler.hadoop.util.StubUtilities;
039
040import ptolemy.data.Token;
041import ptolemy.kernel.util.IllegalActionException;
042import ptolemy.kernel.util.KernelException;
043
044//////////////////////////////////////////////////////////////////////////
045////KeplerApp4CoGroup
046
047/**
048* This class will parse CoGroup sub-workflow, execute it according to its input, and return execution output. 
049* and send it out through fire().
050* 
051* @author Jianwu Wang (jianwu@sdsc.edu)
052* @version $Id: KeplerApp4CoGroup.java 33070 2014-11-12 23:21:09Z crawl $
053*/
054public class KeplerApp4CoGroup extends KeplerAppBase {
055        
056        static Logger logger = Logger.getLogger(KeplerApp4CoGroup.class.getName());
057
058        public KeplerApp4CoGroup(Configuration hadoopConf) {
059        super(hadoopConf, "CoGroupInput", "CoGroupOutput");        
060        }
061        
062        public List<KeyValuePair> executeWf(WritableComparable key, Iterator<Writable> values1, Iterator<Writable> values2)                                     
063            throws IOException {
064                try {
065                
066                        if (_sourceActor != null)
067                        {
068                            ((CoGroupInput) _sourceActor).setInput(StubUtilities.convertToToken(key),
069                                    StubUtilities.convertToArrayToken(values1), StubUtilities.convertToArrayToken(values2));                    
070                        }
071                        
072            if(_runWorkflowLifecyclePerInput) {
073                _manager.execute();
074            }
075
076            List<KeyValuePair> exeResult =  new LinkedList<KeyValuePair>();
077                        if (_sinkActor != null)
078                        {
079                                final List<Token> tokenList = _sinkActor.getOutput();
080                                if(tokenList != null) {
081                                    StubUtilities.convertTokenToList(tokenList, exeResult);
082                                } else if(_keplerManagerException != null) {
083                                    throw _keplerManagerException;
084                                }
085                        }
086                        return exeResult;       
087                } catch (IllegalActionException e) {
088                        throw new RuntimeException("Error getting output for " + _sinkActor.getName() + ".", e);
089                } catch (KernelException e) {
090                    throw new RuntimeException("Error executing workflow.", e);
091        }
092        }
093}
