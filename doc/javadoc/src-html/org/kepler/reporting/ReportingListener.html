140        public void writeFilesToProvenance() {
141                KeplerLSID reportLayoutLsid = null;
142                
143                KeplerLSID lsid = NamedObjId.getIdFor(this.toplevel());
144                
145                // increment report layout LSID
146                
147                try {
148                        // get the recorder for the workflow
149                        ProvenanceRecorder recorder = ProvenanceRecorder.getDefaultProvenanceRecorder(this.toplevel());
150
151                        // Lookup the layout
152                        ReportLayout reportLayout = WorkflowManager.getInstance().getWorkflow(_tableauFrame, lsid).
153                                getReportLayout(_tableauFrame);
154                        if (reportLayout == null || reportLayout.isBlank()) {
155                                return;
156                        }
157                        
158                        // get report layout LSID
159                        reportLayoutLsid = reportLayout.getLsid();
160                        if (reportLayoutLsid == null){
161                                //shouldn't happen
162                                System.out.println("ERROR ReportingListener reportLayoutLsid == null");
163                        }
164                                
165                        // synch with latest wf lsid revision
166                        reportLayout.setWorkflowLSID(lsid);
167                        
168                        ///reportLayoutLsid = reportLayout.getLsid();
169                        
170                        
171                        File layoutFile = File.createTempFile("roml", ".xml");
172                        layoutFile.deleteOnExit();
173                        ReportRenderer.convertReport2XML(reportLayout, layoutFile);
174                        // Add layout to the provenance records
175                        Map<String, String> metadataMap = new HashMap<String, String>();
176                        metadataMap.put("type", ReportLayout.class.getName());
177                        recorder.getRecording().addFileForLastExecution(metadataMap, layoutFile);
178                        
179                        // Now generate instance doc                    
180                        NamedObj model = this.toplevel();
181                        Queryable queryable = ProvenanceRecorder.getDefaultQueryable(model);
182                        ReportInstance reportInstance = ReportAssembler.populateReportWithLatest(queryable, reportLayout, lsid, _tableauFrame);
183                        File instanceFile = File.createTempFile("rio", ".xml");
184                        instanceFile.deleteOnExit();
185                        ReportRenderer.convertReport2XML(reportInstance, instanceFile);
186                        // Add instance to the provenance records
187                        Map<String, String> instanceMap = new HashMap<String, String>();
188                        instanceMap.put("type", ReportInstance.class.getName());
189                        recorder.getRecording().addFileForLastExecution(instanceMap, instanceFile);
190                        
191                        // now the PDF here
192                        File pdfFile = File.createTempFile("rio", ".pdf");
193                        pdfFile.deleteOnExit();
194                        ReportRenderer.convertReport2PDF(reportInstance, pdfFile, null);
195                        // Add instance to the provenance records
196                        Map<String, String> pdfMap = new HashMap<String, String>();
197                        pdfMap.put("type", pdfFile.getClass().getName());
198                        recorder.getRecording().addFileForLastExecution(pdfMap, pdfFile);
199                        
200                        //fire event to trigger report rendering
201                        JLabel dummyComponent = new JLabel();
202                        ReportingStateChangeEvent event = 
203                                new ReportingStateChangeEvent(
204                                                dummyComponent,
205                                                ReportingStateChangeEvent.WORKFLOW_EXECUTION_COMPLETE,
206                                                this.toplevel());
207                        StateChangeMonitor.getInstance().notifyStateChange(event);
208                        dummyComponent = null;
209                        
210                } catch (Exception e) {
211                        e.printStackTrace();
212                        return;
213                }
214        }

217        public File writeFilesToKAR(){
218                KeplerLSID lsid = NamedObjId.getIdFor(this.toplevel());
219                String workflowName = this.toplevel().getName();
220                File karFile = null;
221
222                //create the KAR file locally
223                try {
224
225                        Queryable queryable = ProvenanceUtil.getQueryable(lsid, _tableauFrame);
226                        KeplerLSID runLSID = queryable
227                                        .getLastExecutionLSIDForWorkflow(lsid);
228                        int execID = queryable.getExecutionForExecutionLSID(runLSID);
229
230                        List<KeplerLSID> runLSIDs = new ArrayList<KeplerLSID>();
231                        runLSIDs.add(runLSID);
232
233                        DotKeplerManager dkm = DotKeplerManager.getInstance();
234
235                        ConfigurationManager cman = ConfigurationManager.getInstance();
236                        ConfigurationProperty cprop = cman.getProperty(ConfigurationManager
237                                        .getModule("reporting"));
238                        String karDirStr = cprop.getProperty("resultantRunKarDir")
239                                        .getValue();
240                        String directory = dkm.getPersistentDirString()
241                                        + DEFAULT_WORKFLOW_RUNS_DIR + File.separator;
242                        
243                        if (karDirStr != null && !karDirStr.equals("")) {
244                                File dir = new File(karDirStr);
245                                if (dir.exists()) {
246                                        directory = dir.toString();
247                                } else if (dir.mkdir()) {
248                                                directory = dir.toString();
249                                }
250                        }
251
252                        File dir = new File(directory);
253                        if (!dir.exists()) {
254                                dir.mkdir();
255                        }
256                        // export
257                        String pattern = "yyyyMMdd-HHmmssSz";
258                    SimpleDateFormat format = new SimpleDateFormat(pattern);
259                    String time = format.format(new Date());
260                    karFile = new File(dir, workflowName + "-execId"
261                                        + execID + "_" + time + ".kar");
262                        ExportRunsToKAR exporter = new ExportRunsToKAR(queryable, lsid);
263                        exporter.exportRunsToKAR(runLSIDs, karFile, true, false);
264                } catch (Exception e) {
265                        e.printStackTrace();
266                        return null;
267                }
268                
269                return karFile;
270        }

293        public void wrapup() throws IllegalActionException {
294                                
295                // check if we are recording, stop if not
296                try {
297                        ProvenanceRecorder recorder = ProvenanceRecorder.getDefaultProvenanceRecorder(this.toplevel());
298                        if (!recorder.isEnabled()) {
299                                return;
300                        }
301                } catch (Exception e) {
302                        e.printStackTrace();
303                        return;
304                }
305                
306                // write the files after provenance is done
307                writeFilesToProvenance();
308                
309                // if we are running "automatically" on the server then package and upload them
310                // TODO: drive this off parameters passed to Kepler instead of config file.
311                ConfigurationManager cman = ConfigurationManager.getInstance();
312                ConfigurationProperty cprop = cman.getProperty(ConfigurationManager.getModule("reporting"));
313                String keepKarStr = cprop.getProperty("keepResultantRunKars").getValue();
314                boolean keepKar = false;
315                File karFile = null;
316                boolean doUpload = false;
317                String doUploadStr = cprop.getProperty("uploadToServer").getValue();
318                
319                if (keepKarStr != null){
320                        keepKar = Boolean.parseBoolean(keepKarStr);
321                }
322                if (doUploadStr != null){
323                        doUpload = Boolean.parseBoolean(doUploadStr);
324                }
325                
326                if (keepKar || doUpload){
327                        karFile = writeFilesToKAR();
328                        if (karFile == null) {
329                                return;
330                        }
331                
332                        if (doUpload){
333                                Configuration _configuration = (Configuration) Configuration.configurations().iterator().next();
334                                StringAttribute serverAttr = (StringAttribute)_configuration.getAttribute("_server");
335                                if (serverAttr == null) {
336                                        doUpload = false;
337                                }
338                                if (doUpload) {
339                                        boolean success = uploadFiles(karFile);
340                                        if (success){
341                                                System.out.println("ReportingListener wrapup() - kar and karXML successfully uploaded.");
342                                        }
343                                        else{
344                                                System.out.println("ReportingListener wrapup() - problem with kar and/or karXML uplaod.");
345                                        }
346                                }
347                        }
348                }
349                
350                if (!keepKar && karFile != null){
351                        if (karFile.exists()){
352                                karFile.delete();
353                        }
354                }
355                
356        }

358        /**
359         * Upload karFile and karXML metadata (and two associated access docs)
360         * to repository. 
361         * @param karFile
362         * @return true only if both KAR and karXML uploaded successfully
363         */
364        public boolean uploadFiles(File karFile) {
365                try{
366                        
367                        // look up the save repository, or use config parameter
368                        String repoName = null;
369                        Repository saveRepository = RepositoryManager.getInstance().getSaveRepository();
370                        Configuration _configuration = (Configuration) Configuration.configurations().iterator().next();
371                        StringAttribute repositoryAttr = (StringAttribute)_configuration.getAttribute("_repository");
372                        if (repositoryAttr != null) {
373                                repoName = repositoryAttr.getExpression();
374                                saveRepository = RepositoryManager.getInstance().getRepository(repoName);
375                        }
376                        // authenticate using configured values
377                        String sessionId = null;
378                        String domainName = saveRepository.getAuthDomain();
379                        Domain domain = AuthenticationManager.getDomain(domainName);
380                        String userName = domain.getUsername();
381                        String password = domain.getPassword();
382                        AuthenticationService authService = AuthenticationManager.getAuthenticationService(domain);
383                        authService.setUserName(userName);
384                        authService.setPassword(password);
385                        ProxyEntity proxy = authService.authenticate(domain);
386                        sessionId = proxy.getCredential();
387                        
388                        // upload
389                        boolean isPublic = true;
390                        UploadToRepository uploader = new UploadToRepository(karFile);
391                        uploader.setPublicFile(isPublic);
392                        uploader.setSessionId(sessionId);
393                        uploader.setRepository(saveRepository);
394                        boolean success = uploader.uploadFile();
395                        if (success) {
396                                KarXmlGenerator kxg = new KarXmlGenerator(new KARFile(karFile));
397                                String metadataObj = kxg.getKarXml();
398                                boolean metadataUploadSuccess = uploader.uploadMetadata(metadataObj);
399                                if (metadataUploadSuccess){
400                                        return true;
401                                }
402                        }
403                        
404                } catch (Exception e) {
405                        e.printStackTrace();
406                }
407                return false;
408        }

410        /** A the Reporting Listener to a container if it is not present.
411         *
412         *  @param container The container in which to add the default reporting
413         *  listener.
414         *  @param effigy The effigy displaying the container. This parameter can
415     *  be null.
416         */
417        public static void addReportingListener(NamedObj container, Effigy effigy)      {
418        ReportingListener rl = null;
419        List<?> reportingListeners = container.attributeList(ReportingListener.class);
420        // there should only be one
421        if (reportingListeners != null && reportingListeners.size() > 0) {
422            rl = (ReportingListener) reportingListeners.get(0);
423        }
424        else {
425            // try getting by name - probably not necessary
426            rl = (ReportingListener) container.getAttribute(ReportingListener.TOP_REPORTING_LISTENER_NAME);
427        }
428        // if we don't have one, then add it
429        if (rl == null) {
430            try {
431                rl = new ReportingListener(container, ReportingListener.TOP_REPORTING_LISTENER_NAME);
432            } catch (Exception e) {
433                // TODO Auto-generated catch block
434                e.printStackTrace();
435            }
436        // the reporting listener was already present
437        // if it was stored in the moml, remove it.
438        } else if(rl.isPersistent() && effigy != null) {
439            System.out.println("removing old reporting listener");
440            effigy.setModified(true);
441        }
442        
443        // set persistence to false so it is not stored in moml.
444        rl.setPersistent(false);
445        }
