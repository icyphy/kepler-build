087        public static void reorderPNGReaders() {
088                Iterator<ImageReader> imageIter = ImageIO.getImageReadersByMIMEType("image/png");
089        
090                log.debug("--------- IMAGE READERS BEFORE ----------");
091        while (imageIter.hasNext()) {
092                ImageReader imageReader = imageIter.next();
093                log.debug(imageReader);
094        }
095        
096        Class category = javax.imageio.spi.ImageReaderSpi.class;
097
098        Class firstProviderClass = com.sun.imageio.plugins.png.PNGImageReaderSpi.class;
099                Object firstProvider = IIORegistry.getDefaultInstance().getServiceProviderByClass(firstProviderClass);
100                
101                Class secondProviderClass = com.sun.media.imageioimpl.plugins.png.CLibPNGImageReaderSpi.class;
102                Object secondProvider = IIORegistry.getDefaultInstance().getServiceProviderByClass(secondProviderClass);
103                
104                if (firstProvider !=null && secondProvider != null) {
105                        IIORegistry.getDefaultInstance().setOrdering(category, firstProvider, secondProvider);
106                }
107                
108                //AFTER
109                imageIter = ImageIO.getImageReadersByMIMEType("image/png");
110        log.debug("--------- IMAGE READERS AFTER ----------");
111        while (imageIter.hasNext()) {
112                ImageReader imageReader = imageIter.next();
113                log.debug(imageReader);
114        }
115                
116        }

118        /**
119         * Converts a Report object to a PDF file.
120         * 
121         * @param report
122         *            the Report object
123         * @param pdf
124         *            the target PDF file
125         * @param xsltStream
126         *            the stylesheet input stream
127         * @throws IOException
128         *             In case of an I/O problem
129         * @throws FOPException
130         *             In case of a FOP problem
131         * @throws TransformerException
132         *             In case of a XSL transformation problem
133         */
134        public static void convertReport2PDF(ReportInstance report, File pdf, InputStream xsltStream)
135                        throws IOException, FOPException, TransformerException {
136                
137                if (xsltStream == null) {
138                        // look up the stylesheet for generating the PDF
139                        xsltStream =  report.getClass().getResourceAsStream(DEFAULT_XSLT);
140                }
141                
142                FOUserAgent foUserAgent = fopFactory.newFOUserAgent();
143                // configure foUserAgent as desired
144
145                // Setup output
146                OutputStream out = new java.io.FileOutputStream(pdf);
147                out = new java.io.BufferedOutputStream(out);
148                try {
149                        // Construct fop with desired output format
150                        Fop fop = fopFactory.newFop(MimeConstants.MIME_PDF, foUserAgent,
151                                        out);
152
153                        // Setup XSLT
154                        TransformerFactory factory = TransformerFactory.newInstance();
155                        Transformer transformer = factory.newTransformer(new StreamSource(
156                                        xsltStream));
157
158                        // Set the value of a <param> in the stylesheet
159                        //TODO: this will be very useful, remember it!
160            //transformer.setParameter("myParamName", "myValue");
161            
162                        // Setup input for XSLT transformation
163                        Source src = ReportInputSource.getSourceForReport(report);
164
165                        // Resulting SAX events (the generated FO) must be piped through to
166                        // FOP
167                        Result res = new SAXResult(fop.getDefaultHandler());
168
169                        // Start XSLT transformation and FOP processing
170                        transformer.transform(src, res);
171                        
172                } finally {
173                        out.close();
174                }
175        }

192        /**
193         * Converts a Report object to XML.
194         * 
195         * @param report
196         *            the Report object
197         * @param xml
198         *            the target XML file
199         * @throws IOException
200         *             In case of an I/O problem
201         * @throws TransformerException
202         *             In case of a XSL transformation problem
203         * @throws ClassNotFoundException 
204         * @throws IllegalAccessException 
205         * @throws InstantiationException 
206         */
207        private static ReportLayout convertXML2Report(Source src) throws IOException,
208                        TransformerException {
209                        
210                // Setup XSLT
211                TransformerFactory factory = TransformerFactory.newInstance();
212                Transformer transformer = factory.newTransformer();
213                /*
214                 * Note: We use the identity transformer, no XSL transformation is done.
215                 * The transformer is basically just used to deserialize the XML to objects
216                 */
217        
218                // Setup output
219                DOMResult res = new DOMResult();
220        
221                // Start XSLT transformation
222                transformer.transform(src, res);
223                
224                ReportLayout report = null;
225                try {
226                        report = ReportParser.parse(res.getNode());
227                } catch (Exception e) {
228                        // TODO Auto-generated catch block
229                        e.printStackTrace();
230                } 
231                
232                return report;
233        }

275        /**
276         * Converts an XML file to an XSL-FO file using JAXP (XSLT).
277         * 
278         * @param xml
279         *            the XML file
280         * @param xslt
281         *            the stylesheet file
282         * @param fo
283         *            the target XSL-FO file
284         * @throws IOException
285         *             In case of an I/O problem
286         * @throws TransformerException
287         *             In case of a XSL transformation problem
288         */
289        public static void convertXML2FO(File xml, File xslt, File fo) throws IOException,
290                        TransformerException {
291
292                // Setup output
293                OutputStream out = new java.io.FileOutputStream(fo);
294                try {
295                        // Setup XSLT
296                        TransformerFactory factory = TransformerFactory.newInstance();
297                        Transformer transformer = factory.newTransformer(new StreamSource(
298                                        xslt));
299
300                        // Setup input for XSLT transformation
301                        Source src = new StreamSource(xml);
302
303                        // Resulting SAX events to XML
304                        Result res = new StreamResult(out);
305
306                        // Start XSLT transformation and FOP processing
307                        transformer.transform(src, res);
308                } finally {
309                        out.close();
310                }
311        }

371        /**
372         * Used for embedding a viewer pane in swing applications
373         * @param report Report to render
374         * @param xslt transform to use for rendering Report
375         * @return PreviewPane component that can be added to any GUI
376         * @throws IOException
377         * @throws FOPException
378         * @throws TransformerException
379         */
380        public static Component convertReport2AWT(ReportInstance report, InputStream xslt)
381                throws IOException, FOPException, TransformerException {
382                                
383                FOUserAgent foUserAgent = fopFactory.newFOUserAgent();
384                // configure foUserAgent as desired
385        
386                AWTRenderer renderer = new AWTRenderer();
387                foUserAgent.setRendererOverride(renderer);
388                renderer.setPreviewDialogDisplayed(false);
389                renderer.setUserAgent(foUserAgent);
390                renderer.clearViewportList();
391                 
392                // Construct fop with no output format since we supply the renderer
393                Fop fop = fopFactory.newFop(foUserAgent);
394                
395                // Setup XSLT
396                TransformerFactory factory = TransformerFactory.newInstance();
397                Transformer transformer = factory.newTransformer(new StreamSource(xslt));
398        
399                // Setup input for XSLT transformation
400                Source src = ReportInputSource.getSourceForReport(report);
401        
402                // Resulting SAX events (the generated FO) must be piped through to FOP
403                Result res = new SAXResult(fop.getDefaultHandler());
404        
405                // Start XSLT transformation and FOP processing
406                transformer.transform(src, res);
407                
408                // construct the panel to render the processed FO
409                PreviewPanel previewPanel = new PreviewPanel(foUserAgent, null, renderer);
410                //double scale = previewPanel.getScaleToFitWindow();
411                //previewPanel.setScaleFactor(scale);
412                previewPanel.setDisplayMode(PreviewPanel.CONTINUOUS);
413
414                return previewPanel;
415        }
