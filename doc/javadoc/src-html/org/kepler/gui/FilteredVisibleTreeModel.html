059public class FilteredVisibleTreeModel extends VisibleTreeModel {
060        
061        public FilteredVisibleTreeModel(TreeModel model, File filterFile) {
062                super(new CompositeEntity());
063                this.model = model;
064                this.filterFile = filterFile;
065                if (this.filterFile == null) {
066                        //System.out.println("Filter file is null");
067                        return;
068                }
069                if (filterFile.exists()) {
070                        _whitelistManager = new WhitelistManager(filterFile);
071                }
072        }
073
074        private Map<Object, TreeNode> nodesByObject = new HashMap<Object, TreeNode>();
075                
076        /**
077         * Returns a list of children of object (as defined by the TreeModel
078         * model) that are themselves or have as children concrete filtered
079         * actors.
080         * @param object
081         * @param model
082         * @return
083         */
084        private List<Object> getPrunedChildren(Object object, TreeModel model) {
085                
086                if (nodesByObject.containsKey(object)) {
087                        TreeNode node = nodesByObject.get(object);
088                        return _getFilteredChildren(node);
089                }
090
091                OpaqueTreeNode node = OpaqueTreeNode.getNode(object);
092                
093                List<Object> children = _getAllChildren(object, model);
094                
095                if (children.isEmpty()) {
096                        nodesByObject.put(object, node);
097                        return Collections.emptyList();
098                }
099                
100                // Iterate over all children
101                // Filter out all children that have no non-EntityLibrary children
102                // Remember to record that information into the cache
103                
104                List<Object> prunedItems = new ArrayList<Object>();
105                
106                for (Object child : children) {
107                        if (child instanceof EntityLibrary) {
108                                List<Object> grandchildren = getPrunedChildren(child, model);
109                                if (!grandchildren.isEmpty()) {
110                                        node.add(OpaqueTreeNode.getNode(child));
111                                        prunedItems.add(child);
112                                }
113                        }
114                        else if (passesFilter(child)) {
115                                node.add(OpaqueTreeNode.getNode(child));
116                                prunedItems.add(child);
117                        }
118                        nodesByObject.put(object, node);
119                }
120
121                return prunedItems;
122        }
123                
124        private boolean passesFilter(Object object) {
125                
126                NamedObj no = (NamedObj) object;
127                return checkAgainstWhitelist(no);
128        }
129
130        private boolean checkAgainstWhitelist(NamedObj no) {
131                
132                boolean success = _whitelistManager.checkLSID(((StringAttribute) no.getAttribute("entityId")).getExpression());
133                if (success) return true;
134                success = _whitelistManager.checkClassname(no.getClassName());
135                return success;
136        }
137        
138        private List<Object> _getAllChildren(Object node, TreeModel model) {
139                List<Object> list = new ArrayList<Object>();
140                for (int i = 0; i < model.getChildCount(node); i++) {
141                        list.add(model.getChild(node, i));
142                }
143                return list;
144        }
145        
146        private List<Object> _getFilteredChildren(TreeNode node) {
147                // Recover children from the cache.
148                List<Object> list = new ArrayList<Object>();
149                Enumeration children = node.children();
150                while (children.hasMoreElements()) {
151                        list.add(((DefaultMutableTreeNode) children.nextElement()).getUserObject());
152                }
153                return list;
154        }
155
156        public Object getChild(Object parent, int index) {
157
158                if (filterFile == null) {                       
159                        return model.getChild(parent, index);
160                }
161
162                List<Object> children = getPrunedChildren(parent, model);
163                return children.get(index);
164        }
165
166        public int getChildCount(Object parent) {
167                if (filterFile == null) {                       
168                        return model.getChildCount(parent);
169                }
170
171                List<Object> children = getPrunedChildren(parent, model);
172                return children.size();
173        }
174
175        public int getIndexOfChild(Object parent, Object child) {
176                if (filterFile == null) {                       
177                        return model.getIndexOfChild(parent, child);
178                }
179
180                List<Object> children = getPrunedChildren(parent, model);
181                return children.indexOf(child);
182        }

186                return model.getRoot();

191                return model.isLeaf(node);
192        }

196                model.valueForPathChanged(path, newValue);
197        }

201                model.addTreeModelListener(l);
202        }

206                model.removeTreeModelListener(l);
207        }

209        private WhitelistManager _whitelistManager = null;
210
211        private static class OpaqueTreeNode extends DefaultMutableTreeNode {
212                private OpaqueTreeNode(Object object, boolean allowsChildren) {
213                        super(object, allowsChildren);
214                }
215                
216                public static OpaqueTreeNode getNode(Object object) {
217                        return new OpaqueTreeNode(object, object instanceof EntityLibrary);
218                }
219        }
220        
221        private File filterFile;
222        private TreeModel model;
223
224        private class WhitelistManager {
225                public WhitelistManager(File filterFile) {
226                        BufferedReader br = null;
227                        try {
228                                br = new BufferedReader(new FileReader(filterFile));
229                                String line;
230                                while ((line = br.readLine()) != null) {
231                                        line = line.trim();
232                                        if (line.length() == 0) continue;
233                                        if (line.startsWith("urn:")) {
234                                                lsids.add(line);
235                                        }
236                                        else {
237                                                classes.add(line);
238                                        }
239                                }
240                        }
241                        catch(IOException ex) {
242                                ex.printStackTrace();
243                        }
244                        finally {
245                                if (br != null) {
246                                        try {
247                                                br.close();
248                                        }
249                                        catch(IOException ex) {
250                                                ex.printStackTrace();
251                                        }
252                                }
253                        }
254                }
255
256                public boolean checkLSID(String s) {
257                        return lsids.contains(s);
258                }
259
260                public boolean checkClassname(String s) {
261                        return classes.contains(s);
262                }
263
264                private List<String> lsids = new ArrayList<String>();
265                private List<String> classes = new ArrayList<String>();
266        }
