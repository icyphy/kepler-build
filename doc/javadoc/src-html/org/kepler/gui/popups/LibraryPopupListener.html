092        /**
093         * 
094         * @param selPath
095         * @param e
096         */
097        private void handlePopupInsideKar(TreePath selPath, MouseEvent e) {
098
099                Object ob = selPath.getLastPathComponent();
100                if (isDebugging)
101                        log.debug(ob.getClass().getName());
102
103                if (ob instanceof FolderEntityLibrary) {
104                        // User clicked on a Folder inside of a KAR
105                        KARFolderPopup kfp = new KARFolderPopup(selPath, _aptree
106                                        .getParentComponent());
107                        kfp.initialize();
108                        _aptree.add(kfp);
109                        _aptree.setSelectionPath(selPath);
110                        kfp.show(e.getComponent(), e.getX(), e.getY());
111
112                } else if (!(ob instanceof ptolemy.moml.EntityLibrary)) {
113                        // User clicked on a Component inside a KAR
114                        KARComponentPopup kcp = new KARComponentPopup(selPath, _aptree
115                                        .getParentComponent());
116                        kcp.initialize();
117                        _aptree.add(kcp);
118                        _aptree.setSelectionPath(selPath);
119                        kcp.show(e.getComponent(), e.getX(), e.getY());
120
121                }
122        }

124        /**
125         * 
126         * @param selPath
127         * @param e
128         */
129        private void handlePopupOutsideKar(TreePath selPath, MouseEvent e) {
130
131                Object ob = selPath.getLastPathComponent();
132                if (isDebugging)
133                        log.debug(ob.getClass().getName());
134
135                if (ob instanceof KAREntityLibrary) {
136                        // User clicked on a KAR
137                        if (isDebugging)
138                                log.debug("KAREntityLibrary selected");
139                        KARPopup kp = new KARPopup(selPath, _aptree.getParentComponent());
140                        kp.initialize();
141                        _aptree.add(kp);
142                        _aptree.setSelectionPath(selPath);
143                        kp.show(e.getComponent(), e.getX(), e.getY());
144                } else if (ob instanceof DownloadableKAREntityLibrary) {
145                        DownloadableKARPopup kp = new DownloadableKARPopup(selPath, _aptree.getParentComponent());
146                        kp.initialize();
147                        _aptree.add(kp);
148                        _aptree.setSelectionPath(selPath);
149                        kp.show(e.getComponent(), e.getX(), e.getY());
150                } else if (ob instanceof RemoteRepositoryEntityLibrary) {
151//                      System.out.println("Context click on RREL");
152
153                } else if (ob instanceof FolderEntityLibrary) {
154                        // User clicked on a Folder
155                        FolderPopup fp = new FolderPopup(selPath, _aptree
156                                        .getParentComponent());
157                        fp.initialize();
158                        _aptree.add(fp);
159                        _aptree.setSelectionPath(selPath);
160                        fp.show(e.getComponent(), e.getX(), e.getY());
161
162                } else if (ob instanceof OntologyEntityLibrary) {
163                        // User clicked on an ontology
164                        OntologyPopup op = new OntologyPopup(selPath, _aptree
165                                        .getParentComponent());
166                        op.initialize();
167                        _aptree.add(op);
168                        _aptree.setSelectionPath(selPath);
169                        op.show(e.getComponent(), e.getX(), e.getY());
170
171                } else if (ob instanceof KARErrorEntityLibrary) {
172                        KARErrorPopup kep = new KARErrorPopup(selPath, _aptree
173                                        .getParentComponent());
174                        kep.initialize();
175                        _aptree.add(kep);
176                        _aptree.setSelectionPath(selPath);
177                        kep.show(e.getComponent(), e.getX(), e.getY());
178
179                } else if (ob instanceof RemoteKARErrorEntityLibrary) {
180                        RemoteKARErrorPopup kep = new RemoteKARErrorPopup(selPath, _aptree
181                                        .getParentComponent());
182                        kep.initialize();
183                        _aptree.add(kep);
184                        _aptree.setSelectionPath(selPath);
185                        kep.show(e.getComponent(), e.getX(), e.getY());
186
187                } else if (!(ob instanceof ptolemy.moml.EntityLibrary)) {
188                        // User clicked on a Component
189                        if (!(ob instanceof NamedObj)) {
190                                // Shouldn't happen
191                                return;
192                        }
193                        NamedObj no = (NamedObj) ob;
194                        String alternateLibraryPopupClassName = _getAlternateLibraryPopupClassName(no);
195                        if (alternateLibraryPopupClassName == null) {
196                                //System.out.println("Using default context menu stuff for class: " + ob.getClass().getName());
197                                OntologyComponentPopup ocp = new OntologyComponentPopup(
198                                                selPath, _aptree.getParentComponent());
199                                ocp.initialize();
200                                _aptree.add(ocp);
201                                _aptree.setSelectionPath(selPath);
202                                ocp.show(e.getComponent(), e.getX(), e.getY());
203                        } else {
204                                try {
205                                        Class<?> libraryPopupClass = Class
206                                                        .forName(alternateLibraryPopupClassName);
207                                        Object object = libraryPopupClass.newInstance();
208                                        Method getPopupMethod = libraryPopupClass.getMethod(
209                                                        "getPopup", JTree.class, MouseEvent.class,
210                                                        TreePath.class, Component.class);
211                                        getPopupMethod.invoke(object, _aptree, e, selPath, _aptree
212                                                        .getParentComponent());
213                                } catch (Exception w) {
214                                        log.error("Error creating alternateGetLibraryPopup!", w);
215                                }
216                        }
217                }
218        }

220        /** Handle a double-click action. */
221        private void handleDoubleClickOutsideKar(TreePath selPath, MouseEvent event) {
222               Object ob = selPath.getLastPathComponent();
223
224               if (ob instanceof EntityLibrary) {
225                    return;
226               } else {
227
228                    int liid = LibraryManager.getLiidFor((ComponentEntity) ob);
229                    LibItem li = null;
230                try {
231                    li = LibraryManager.getInstance().getPopulatedLibItem(liid);
232                } catch (SQLException e) {
233                    MessageHandler.error("Error accessinc library item.", e);
234                    return;
235                }
236                
237                // open it if it's a MoML
238                    String filePath = li.getAttributeValue(LibIndex.ATT_XMLFILE);
239                    if(filePath != null) {
240                        Component component = _aptree.getParentComponent();
241                        while (component != null && !(component instanceof TableauFrame)) {
242                            component = component.getParent();
243                        }
244                        if(component == null) {
245                            MessageHandler.error("Could not find TableauFrame.");
246                            return;
247                        }
248                        Configuration configuration = ((TableauFrame) component).getConfiguration();
249                        try {
250                            URL url = new File(filePath).toURI().toURL();
251                            configuration.openModel(url, url, url.toExternalForm());
252                        } catch(Exception e) {
253                            MessageHandler.error("Error opening " + filePath, e);
254                        }
255                        
256                        // if we successfully opened a file, update the history menu and
257                        // set the last directory.
258
259                        // update the history menu
260                    if(component instanceof KeplerGraphFrame) {
261                        try {
262                            ((KeplerGraphFrame)component).updateHistory(filePath);
263                        } catch (IOException exception) {
264                            MessageHandler.error("Unable to update history menu.", exception);
265                        }
266                    }
267                    if(component instanceof BasicGraphFrame) {
268                        ((BasicGraphFrame)component).setLastDirectory(new File(filePath).getParentFile());
269                    }
270                    
271                    } else if(li.getLsid() != null) {
272                        // if it has an lsid, show the documentation
273                        Component component = _aptree.getParentComponent();
274                    while (component != null && !(component instanceof PtolemyFrame)) {
275                        component = component.getParent();
276                    }
277                    if(component == null) {
278                        MessageHandler.error("Could not find TableauFrame.");
279                        return;
280                    }
281                    ShowDocumentationAction action = new ShowDocumentationAction(selPath, _aptree.getParentComponent());
282                        action.setLsidToView(li.getLsid());
283                        action.setPtolemyFrame((PtolemyFrame) component);
284
285                        action.actionPerformed(new ActionEvent(this,
286                                ActionEvent.ACTION_FIRST, "open"));
287                    }
288                }
289        }

291        /**
292         * Description of the Method
293         * 
294         *@param e
295         *            Description of the Parameter
296         */
297        private void maybeShowPopup(MouseEvent e) {
298                if (isDebugging)
299                        log.debug("maybeShowPopup(" + e.toString() + ")");
300                if (e.isPopupTrigger() || _trigger) {
301                        _trigger = false;
302                        TreePath selPath = _aptree.getPathForLocation(e.getX(), e.getY());
303                        if (isDebugging)
304                                log.debug(selPath.toString());
305                        if ((selPath != null)) {
306
307                                if (isDebugging)
308                                        log.debug(selPath.getLastPathComponent().getClass().getName());
309
310                                if (_isPathInsideKAR(selPath)) {
311                                        handlePopupInsideKar(selPath, e);
312                                } else {
313                                        handlePopupOutsideKar(selPath, e);
314
315                                }
316                        }
317                // handle double clicks
318                } else if(e.getClickCount() == 2) {
319                    TreePath selPath = _aptree.getPathForLocation(e.getX(), e.getY());
320            if (isDebugging)
321                log.debug(selPath.toString());
322            if ((selPath != null)) {
323
324                if (isDebugging)
325                    log.debug(selPath.getLastPathComponent().getClass().getName());
326
327                if (_isPathInsideKAR(selPath)) {
328                    handlePopupInsideKar(selPath, e);
329                } else {
330                    handleDoubleClickOutsideKar(selPath, e);
331                }
332            }
333                }
334        }

337    /** Determine if this object is contained within a KAR
338     *  by checking all of the parent objects to see if they
339     *  are a KAREntityLibrary. do not check the object itself
340         */
341        private boolean _isPathInsideKAR(TreePath selPath) {
342        for (int i = (selPath.getPathCount() - 2); i >= 0; i--) {
343            if (selPath.getPathComponent(i) instanceof KAREntityLibrary) {
344                return true;
345            }
346        }
347        return false;
348        }
349
350        private String _getAlternateLibraryPopupClassName(NamedObj namedObj) {
351                Attribute attribute = namedObj
352                                .getAttribute(AnnotatedPTree.ALTERNATE_LIBRARY_POPUP_ATTRIBUTE_NAME);
353                if (attribute == null) {
354                        return null;
355                }
356                try {
357                        StringAttribute sa = (StringAttribute) attribute;
358                        return sa.getExpression();
359                } catch (ClassCastException ex) {
360                        log.warn(AnnotatedPTree.ALTERNATE_LIBRARY_POPUP_ATTRIBUTE_NAME
361                                        + " should be a StringAttribute"
362                                        + " specifying a class that should"
363                                        + " handle context popups from this object", ex);
364                }
365                return null;
366        }
