030/**
031 The exception of modifying graph elements in wrong ways. The
032 modification is due to the change of topology or improper weight
033 association. Function evaluations incurred by graph reading do not
034 cause this exception.
035
036 @author Mingyung Ko, Shuvra S. Bhattacharyya
037 @version $Id$
038 @since Ptolemy II 2.1
039 @Pt.ProposedRating Red (myko)
040 @Pt.AcceptedRating Red (ssb)
041 */
042@SuppressWarnings("serial")
043public class GraphConstructionException extends GraphException {
044    /** Constructor with an argument of text description.
045     *  @param message The exception message.
046     */
047    public GraphConstructionException(String message) {
048        super(message);
049    }
050}
