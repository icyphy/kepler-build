028package ptolemy.data.type;
029
030import ptolemy.data.ActorToken;
031import ptolemy.data.Token;
032import ptolemy.kernel.util.IllegalActionException;
033
034///////////////////////////////////////////////////////////////////
035//// ActorType
036
037/**
038 The base type of matrix token classes. This type functions as a union
039 of the various matrix types. It allows for the creation of arrays
040 that consist of diverse matrix types, because the array type will
041 be {matrix}.
042
043 @author Steve Neuendorffer and Edward A. Lee
044 @version $Id$
045 @since Ptolemy II 8.0
046 @Pt.ProposedRating Red (yuhong)
047 @Pt.AcceptedRating Red
048 */
049public class ActorType implements Type, Cloneable {
050
051    public ActorType() {
052        super();
053        BaseType._addType(this, "actor", ActorToken.class);
054    }
055
056    ///////////////////////////////////////////////////////////////////
057    ////                         public methods                    ////
058
059    /** Return a new type which represents the type that results from
060     *  adding a token of this type and a token of the given argument
061     *  type.
062     *  @param rightArgumentType The type to add to this type.
063     *  @return A new type, or BaseType.GENERAL, if the operation does
064     *  not make sense for the given types.
065     */
066    @Override
067    public Type add(Type rightArgumentType) {
068        return this;
069    }
070
071    /** Return this, that is, return the reference to this object.
072     *  @return A BaseType.
073     */
074    @Override
075    public Object clone() {
076        return this;
077    }
078
079    /** Convert the specified token to a token having the type
080     *  represented by this object.
081     *  @param token A token.
082     *  @return A token.
083     *  @exception IllegalActionException If lossless conversion cannot
084     *   be done.
085     */
086    @Override
087    public Token convert(Token token) throws IllegalActionException {
088        if (token instanceof ActorToken) {
089            return token;
090        } else {
091            throw new IllegalActionException("Attempt to convert token " + token
092                    + " into a test token, which is not possible.");
093        }
094    }
095
096    /** Return a new type which represents the type that results from
097     *  dividing a token of this type and a token of the given
098     *  argument type.
099     *  @param rightArgumentType The type to add to this type.
100     *  @return A new type, or BaseType.GENERAL, if the operation does
101     *  not make sense for the given types.
102     */
103    @Override
104    public Type divide(Type rightArgumentType) {
105        return this;
106    }
107
108    /** Determine if the argument represents the same BaseType as this
109     *  object.
110     *  @param t A Type.
111     *  @return True if the argument represents the same BaseType as
112     *   this object; false otherwise.
113     */
114    public boolean equals(Type t) {
115        return this == t;
116    }
117
118    /** Return the class for tokens that this basetype represents.
119     *  @return the class for tokens that this basetype represents.
120     */
121    @Override
122    public Class getTokenClass() {
123        return ActorToken.class;
124    }
125
126    /** Return this type's node index in the (constant) type lattice.
127     * @return this type's node index in the (constant) type lattice.
128     */
129    @Override
130    public int getTypeHash() {
131        return Type.HASH_INVALID;
132    }
133
134    /** Return true if this type does not correspond to a single token
135     *  class.  This occurs if the type is not instantiable, or it
136     *  represents either an abstract base class or an interface.
137     *  @return Always return false, this token is instantiable.
138     */
139    @Override
140    public boolean isAbstract() {
141        return false;
142    }
143
144    /** Model if the argument type is compatible with this type.
145     *  The method returns true if this type is UNKNOWN, since any type
146     *  is a substitution instance of it. If this type is not UNKNOWN,
147     *  this method returns true if the argument type is less than or
148     *  equal to this type in the type lattice, and false otherwise.
149     *  @param type An instance of Type.
150     *  @return True if the argument type is compatible with this type.
151     */
152    @Override
153    public boolean isCompatible(Type type) {
154        return type == this;
155    }
156
157    /** Model if this Type is UNKNOWN.
158     *  @return True if this Type is not UNKNOWN; false otherwise.
159     */
160    @Override
161    public boolean isConstant() {
162        return true;
163    }
164
165    /** Determine if this type corresponds to an instantiable token
166     *  classes. A BaseType is instantiable if it does not correspond
167     *  to an abstract token class, or an interface, or UNKNOWN.
168     *  @return True if this type is instantiable.
169     */
170    @Override
171    public boolean isInstantiable() {
172        return true;
173    }
174
175    /** Return true if the argument is a
176     *  substitution instance of this type.
177     *  @param type A Type.
178     *  @return True if this type is UNKNOWN; false otherwise.
179     */
180    @Override
181    public boolean isSubstitutionInstance(Type type) {
182        return this == type;
183    }
184
185    /** Return a new type which represents the type that results from
186     *  moduloing a token of this type and a token of the given
187     *  argument type.
188     *  @param rightArgumentType The type to add to this type.
189     *  @return A new type, or BaseType.GENERAL, if the operation does
190     *  not make sense for the given types.
191     */
192    @Override
193    public Type modulo(Type rightArgumentType) {
194        return this;
195    }
196
197    /** Return a new type which represents the type that results from
198     *  multiplying a token of this type and a token of the given
199     *  argument type.
200     *  @param rightArgumentType The type to add to this type.
201     *  @return A new type, or BaseType.GENERAL, if the operation does
202     *  not make sense for the given types.
203     */
204    @Override
205    public Type multiply(Type rightArgumentType) {
206        return this;
207    }
208
209    /** Return the type of the multiplicative identity for elements of
210     *  this type.
211     *  @return A new type, or BaseType.GENERAL, if the operation does
212     *  not make sense for the given types.
213     */
214    @Override
215    public Type one() {
216        return this;
217    }
218
219    /** Return a new type which represents the type that results from
220     *  subtracting a token of this type and a token of the given
221     *  argument type.
222     *  @param rightArgumentType The type to add to this type.
223     *  @return A new type, or BaseType.GENERAL, if the operation does
224     *  not make sense for the given types.
225     */
226    @Override
227    public Type subtract(Type rightArgumentType) {
228        return this;
229    }
230
231    /** Return the string representation of this type.
232     *  @return A String.
233     */
234    @Override
235    public String toString() {
236        return "actor";
237    }
238
239    /** Return the type of the additive identity for elements of
240     *  this type.
241     *  @return A new type, or BaseType.GENERAL, if the operation does
242     *  not make sense for the given types.
243     */
244    @Override
245    public Type zero() {
246        return this;
247    }
248}
