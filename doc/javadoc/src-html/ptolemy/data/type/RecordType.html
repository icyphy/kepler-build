047/**
048 A class representing the type of a RecordToken.
049 To set the type of a typeable object (such as a port or parameter)
050 to a record with particular fields, create an instance of this
051 class and call setTypeEquals() with that instance as an argument.
052 <p>
053 Note that a record type with more fields is a subtype of a record
054 type with a subset of the fields.  For example, {x = double, y = int}
055 is a subtype of {x = double}. When a record of type
056 {x = double, y = int} is converted to one of type {x = double},
057 the extra field is preserved. The converted record, therefore,
058 may have more fields than the type indicates, but not fewer.
059 <p>
060 A consequence of this is that all record types are subtypes
061 of the empty record type. Hence, to require that a typeable
062 object be a record type without specifying what the fields
063 are, use
064 <pre>
065 typeable.setTypeAtMost(BaseType.RECORD);
066 </pre>
067 Note, however, that by itself this type constraint will
068 not be useful because it does not, by itself, prevent the
069 type from resolving to unknown (the unknown type is at the
070 bottom of the type lattice, and hence satisfies this type
071 constraint). To force a port to resolve to the empty record
072 type, you can state
073 <pre>
074 typeable.setTypeEquals(BaseType.RECORD);
075 </pre>
076 Equivalently, in a port configurer dialog, you can set the type
077 of a port to record, which references an empty record.
078
079 @author Yuhong Xiong, Elaine Cheong and Steve Neuendorffer; contributors: J. S. Senecal, Marten Lohstroh
080 @version $Id$
081 @since Ptolemy II 3.0
082 @Pt.ProposedRating Red (neuendor)
083 @Pt.AcceptedRating Red (cxh)
084 */
085public class RecordType extends AssociativeType implements Cloneable {
086    /** Construct a new RecordType with the specified labels and types.
087     *  To leave the types of some fields undeclared, use BaseType.UNKNOWN.
088     *  The labels and the types are specified in two arrays. These two
089     *  arrays must have the same length, and their elements have one to
090     *  one correspondence. That is, the i'th entry in the types array is
091     *  the type for the i'th label in the labels array. Record labels may be
092     *  arbitrary strings. To construct the empty record type, set the length
093     *  of the argument arrays to 0.
094     *     *
095     *  @param labels An array of String.
096     *  @param types An array of Type.
097     *  @exception IllegalArgumentException If the two arrays do not have
098     *   the same size.
099     *  @exception NullPointerException If one of the arguments is null.
100     */
101    public RecordType(String[] labels, Type[] types) {
102        if (labels.length != types.length) {
103            throw new IllegalArgumentException("RecordType: the labels "
104                    + "and types arrays do not have the same size.");
105        }
106
107        for (int i = 0; i < labels.length; i++) {
108            if (labels[i] == null) {
109                throw new IllegalArgumentException("RecordType: the " + i
110                        + "'th element of the labels array is null.");
111            }
112            //labels[i] = StringUtilities.sanitizeName(labels[i]);
113            labels[i] = labels[i];
114            if (!_fields.containsKey(labels[i])) {
115                FieldType fieldType = new FieldType(types[i]);
116                _fields.put(labels[i], fieldType);
117            } else {
118                throw new IllegalArgumentException("RecordType: The "
119                        + "labels array contain duplicate element: "
120                        + labels[i]);
121            }
122        }
123    }
124
125    /** Construct a RecordType with the labels and values specified by
126     *  a given Map object. The object cannot contain any null keys or
127     *  values.
128     *  @param fieldMap A Map that has keys of type String and
129     *  values of type Token.
130     *  @exception IllegalActionException If the map contains null
131     *  keys or values, or if it contains non-String keys or non-Type
132     *  values
133     */
134    public RecordType(Map<String, FieldType> fieldMap)
135            throws IllegalActionException {
136        Iterator<Map.Entry<String, FieldType>> fields = fieldMap.entrySet()
137                .iterator();
138        while (fields.hasNext()) {
139            Map.Entry<String, FieldType> entry = fields.next();
140            if (entry.getKey() == null || entry.getValue() == null) {
141                throw new IllegalActionException("RecordType: given map"
142                        + " contains either null keys or null values.");
143            }
144            if (!(entry.getKey() instanceof String)
145                    || !(entry.getValue() instanceof Type)) {
146                throw new IllegalActionException("RecordType: given map"
147                        + " contains either non-String keys or"
148                        + " non-Type values.");
149            }
150            _fields.put(entry.getKey(), new FieldType((Type) entry.getValue()));
151        }
152    }

157    /** Return a deep copy of this RecordType if it is a variable, or
158     *  itself if it is a constant.
159     *  @return A RecordType.
160     */
161    @Override
162    public Object clone() {
163        if (isConstant()) {
164            return this;
165        } else {
166            // empty record is a constant, so this record type is not empty.
167            // construct the labels and declared types array
168            Object[] labelsObj = _fields.keySet().toArray();
169            String[] labels = new String[labelsObj.length];
170            Type[] types = new Type[labelsObj.length];
171
172            for (int i = 0; i < labels.length; i++) {
173                labels[i] = (String) labelsObj[i];
174
175                FieldType fieldType = _fields.get(labels[i]);
176                types[i] = fieldType._declaredType;
177            }
178
179            RecordType newObj = new RecordType(labels, types);
180
181            try {
182                newObj.updateType(this);
183            } catch (IllegalActionException ex) {
184                throw new InternalErrorException("RecordType.clone: Cannot "
185                        + "update new instance. " + ex.getMessage());
186            }
187
188            return newObj;
189        }
190    }

192    /** Convert the argument token into a RecordToken having this
193     *  type, if compatible.  The argument must be a RecordToken,
194     *  and its type must be a subtype of this record type.  The
195     *  argument token must have at least the fields of this type.
196     *  Extra fields in the argument token that are not in this
197     *  type are preserved.
198     *  @param token A token.
199     *  @return An RecordToken.
200     *  @exception IllegalActionException If token is incompatible
201     *  with this type.
202     */
203    @Override
204    public Token convert(Token token) throws IllegalActionException {
205        if (!isCompatible(token.getType())) {
206            throw new IllegalArgumentException(Token
207                    .notSupportedConversionMessage(token, this.toString()));
208        }
209
210        RecordToken recordToken = (RecordToken) token;
211
212        // Undeclared fields are preserved.
213        // If we don't do this, it would not be possible to express
214        // type constraints to accommodate actors that operate
215        // on the entirety of any received record, which may be just
216        // any record, or a record that must least contains a certain
217        // set of labels.
218        Object[] labelArray = recordToken.labelSet().toArray();
219
220        // Arrays that will be used to create the new token.
221        String[] labelStringArray = new String[labelArray.length];
222        Token[] values = new Token[labelArray.length];
223
224        for (int i = 0; i < labelArray.length; i++) {
225            String label = (String) labelArray[i];
226
227            // Convert each field of the record.
228            Token fieldToken = recordToken.get(label);
229            Type newFieldType = get(label);
230
231            // If the type of the field is specified, then convert it.
232            // Otherwise, leave it alone.
233            if (newFieldType != null) {
234                values[i] = newFieldType.convert(fieldToken);
235            } else {
236                values[i] = fieldToken;
237            }
238
239            // Store the label for each field.
240            labelStringArray[i] = label;
241        }
242
243        // If the original token preserves ordering of the
244        // fields, then the new one will too.
245        // FIXME: This seems questionable. But if we don't
246        // do this, then the ordering will be mysteriously lost
247        // when, for example, an OrderedRecordToken is put
248        // into an ArrayToken. A better solution would be
249        // to include OrderedRecordToken in the type system.
250        if (recordToken instanceof OrderedRecordToken) {
251            return new OrderedRecordToken(labelStringArray, values);
252        } else {
253            return new RecordToken(labelStringArray, values);
254        }
255    }

257    /** Return the depth of a record type. The depth of a
258     *  record type is the number of times it
259     *  contains other structured types. For example, a record
260     *  of arrays has depth 2.
261     *  @return the depth of a record type.
262     */
263    @Override
264    public int depth() {
265        Object[] labelsObj = _fields.keySet().toArray();
266        String[] labels = new String[labelsObj.length];
267        int[] depth = new int[labelsObj.length];
268        int maxDepth = 1;
269        for (int i = 0; i < labels.length; i++) {
270            labels[i] = (String) labelsObj[i];
271            Type fieldType = get(labels[i]);
272            depth[i] = 1;
273            if (fieldType instanceof StructuredType) {
274                depth[i] += ((StructuredType) fieldType).depth();
275            }
276            if (depth[i] > maxDepth) {
277                maxDepth = depth[i];
278            }
279        }
280        return maxDepth;
281    }

283    /** Determine if the argument represents the same RecordType as this
284     *  object.  Two record types are equal if they have the same field names
285     *  and the type of each field is the same.
286     *  @param object Another object.
287     *  @return True if the argument represents the same RecordType as
288     *  this object.
289     */
290    @Override
291    public boolean equals(Object object) {
292        if (!(object instanceof RecordType)) {
293            return false;
294        }
295
296        RecordType recordType = (RecordType) object;
297
298        // Check that the label sets are equal
299        Set<String> myLabelSet = _fields.keySet();
300        Set<String> argLabelSet = recordType._fields.keySet();
301
302        if (!myLabelSet.equals(argLabelSet)) {
303            return false;
304        }
305
306        Iterator<String> fieldNames = myLabelSet.iterator();
307
308        while (fieldNames.hasNext()) {
309            String label = fieldNames.next();
310            Type myType = this.get(label);
311            Type argType = recordType.get(label);
312
313            if (!myType.equals(argType)) {
314                return false;
315            }
316        }
317
318        return true;
319    }

369        // Loop through all of the fields.
370        Iterator<String> fieldNames = _fields.keySet().iterator();

372        while (fieldNames.hasNext()) {
373            String label = fieldNames.next();
374            Type type = this.get(label);
375
376            // Return false if the field is not instantiable.

410    /** Test whether the argument type is compatible with this type
411     *  (is less than or equal to this type in the type lattice).  The
412     *  given type will be compatible with this type if it is
413     *  BaseType.UNKNOWN, or it is a RecordType that contains at
414     *  least the fields of this record type.
415     *  @param type An instance of Type.
416     *  @return True if the argument is compatible with this type.
417     */
418    @Override
419    public boolean isCompatible(Type type) {
420        if (type.equals(BaseType.UNKNOWN)) {
421            return true;
422        }
423
424        if (!(type instanceof RecordType)) {
425            return false;
426        }
427
428        RecordType argumentRecordType = (RecordType) type;
429
430        // Loop through all of the fields of this type...
431        Iterator<String> iterator = _fields.keySet().iterator();
432
433        while (iterator.hasNext()) {
434            String label = iterator.next();
435
436            // The given type cannot be losslessly converted to this type
437            // if it does not contain one of the fields of this type.
438            Type argumentFieldType = argumentRecordType.get(label);
439
440            if (argumentFieldType == null) {
441                // argument token does not contain this label
442                return false;
443            }
444
445            // The given type cannot be losslessly converted to this type
446            // if the individual fields are not compatible.
447            Type thisFieldType = this.get(label);
448
449            if (!thisFieldType.isCompatible(argumentFieldType)) {
450                return false;
451            }
452        }
453
454        return true;
455    }

458     *  the declared type of all of its fields are constant.
459     *  @return True if this type is a constant.

461    @Override
462    public boolean isConstant() {
463        // Loop through all of the fields.
464        Iterator<FieldType> fieldTypes = _fields.values().iterator();
465
466        while (fieldTypes.hasNext()) {
467            FieldType fieldType = fieldTypes.next();
468            Type type = fieldType._declaredType;
469
470            // Return false if the field is not constant.
471            if (!type.isConstant()) {
472                return false;
473            }
474        }
475
476        return true;
477    }

555    /** Return the string representation of this type. The format is
556     *  {<i>label</i> = <i>type</i>, <i>label</i> = <i>type</i>, ...}.
557     *  The record fields are listed in the lexicographical order of the
558     *  labels determined by the java.lang.String.compareTo() method.
559     *  @return A String.
560     */
561    @Override
562    public String toString() {
563        Object[] labelArray = _fields.keySet().toArray();
564
565        // Order the labels
566        int size = labelArray.length;
567
568        for (int i = 0; i < size - 1; i++) {
569            for (int j = i + 1; j < size; j++) {
570                String labeli = (String) labelArray[i];
571                String labelj = (String) labelArray[j];
572
573                if (labeli.compareTo(labelj) >= 0) {
574                    Object temp = labelArray[i];
575                    labelArray[i] = labelArray[j];
576                    labelArray[j] = temp;
577                }
578            }
579        }
580
581        // construct the string representation of this token.
582        StringBuffer results = new StringBuffer("{");
583
584        for (int i = 0; i < size; i++) {
585            String label = (String) labelArray[i];
586            String type = this.get(label).toString();
587
588            if (i != 0) {
589                results.append(", ");
590            }
591
592            // quote and escape labels that are not valid Java identifiers
593            if (!StringUtilities.isValidIdentifier(label)) {
594                label = "\"" + StringUtilities.escapeString(label) + "\"";
595            }
596            results.append(label + " = " + type);
597        }
598
599        return results.toString() + "}";
600    }

602    /** Update this Type to the specified RecordType.
603     *  The specified type must be a RecordType and have the same structure
604     *  as this one, and have depth less than the MAXDEPTHDOUND.
605     *  This method will only update the component whose declared type is
606     *  BaseType.UNKNOWN, and leave the constant part of this type intact.
607     *  @param newType A StructuredType.
608     *  @exception IllegalActionException If the specified type is not a
609     *   RecordType or it does not have the same structure as this one.
610     */
611    @Override
612    public void updateType(StructuredType newType)
613            throws IllegalActionException {
614        super.updateType(newType);
615        if (this.isConstant()) {
616            if (this.equals(newType)) {
617                return;
618            }
619
620            throw new IllegalActionException("RecordType.updateType: "
621                    + "This type is a constant and the argument is not the"
622                    + " same as this type. This type: " + this.toString()
623                    + " argument: " + newType.toString());
624        }
625
626        // This type is a variable.
627        if (!this.isSubstitutionInstance(newType)) {
628            throw new IllegalActionException("RecordType.updateType: "
629                    + "Cannot update this type to the new type.");
630        }
631
632        // FindBugs: ptolemy.data.type.RecordType.updateType(StructuredType) makes inefficient use of keySet iterator instead of entrySet iterator
633        // Iterator<String> fieldNames = _fields.keySet().iterator();
634        // while (fieldNames.hasNext()) {
635        //    String label = fieldNames.next();
636        //    FieldType fieldType = _fields.get(label);
637        for (Map.Entry<String, FieldType> fields : _fields.entrySet()) {
638            FieldType fieldType = fields.getValue();
639            if (fieldType.isSettable()) {
640                //Type newFieldType = ((RecordType) newType).get(label);
641                Type newFieldType = ((RecordType) newType).get(fields.getKey());
642                fieldType.setValue(newFieldType);
643            }
644        }
645    }

657    /** Compare this type with the specified type. The specified type
658     *  must be a RecordType, otherwise an exception will be thrown.
659     *
660     *  This method returns one of ptolemy.graph.CPO.LOWER,
661     *  ptolemy.graph.CPO.SAME, ptolemy.graph.CPO.HIGHER,
662     *  ptolemy.graph.CPO.INCOMPARABLE, indicating this type is lower
663     *  than, equal to, higher than, or incomparable with the
664     *  specified type in the type hierarchy, respectively.
665     *  @param type a RecordType.
666     *  @return An integer.
667     *  @exception IllegalArgumentException If the specified type is
668     *   not a RecordType.
669     */
670    @Override
671    protected int _compare(StructuredType type) {
672        if (!(type instanceof RecordType)) {
673            throw new IllegalArgumentException("RecordType._compare: "
674                    + "The argument is not a RecordType.");
675        }
676
677        if (this.equals(type)) {
678            return CPO.SAME;
679        }
680
681        if (_isLessThanOrEqualTo(this, (RecordType) type)) {
682            return CPO.LOWER;
683        }
684
685        if (_isLessThanOrEqualTo((RecordType) type, this)) {
686            return CPO.HIGHER;
687        }
688
689        return CPO.INCOMPARABLE;
690    }

707     */
708    @Override
709    protected StructuredType _greatestLowerBound(StructuredType type) {
710        if (!(type instanceof RecordType)) {
711            throw new IllegalArgumentException(
712                    "RecordType.greatestLowerBound: The argument is not a "
713                            + "RecordType.");
714        }
715
716        RecordType recordType = (RecordType) type;
717
718        // the label set of the GLB is the union of the two label sets.
719        Set<String> unionSet = new HashSet<String>();
720        Set<String> myLabelSet = _fields.keySet();
721        Set<String> argLabelSet = recordType._fields.keySet();
722
723        unionSet.addAll(myLabelSet);
724        unionSet.addAll(argLabelSet);
725
726        // construct the GLB RecordToken
727        Object[] labelArray = unionSet.toArray();
728        int size = labelArray.length;
729        String[] labels = new String[size];
730        Type[] types = new Type[size];
731
732        for (int i = 0; i < size; i++) {
733            labels[i] = (String) labelArray[i];
734
735            Type type1 = this.get(labels[i]);
736            Type type2 = recordType.get(labels[i]);
737
738            if (type1 == null) {
739                types[i] = type2;
740            } else if (type2 == null) {
741                types[i] = type1;
742            } else {
743                types[i] = (Type) TypeLattice.lattice()
744                        .greatestLowerBound(type1, type2);
745            }
746        }
747
748        return new RecordType(labels, types);
749    }

751    /** Return the least Upper bound of this type with the specified
752     *  type. The specified type must be a RecordType, otherwise an
753     *  exception will be thrown.
754     *  @param type a RecordType.
755     *  @return a RecordType.
756     *  @exception IllegalArgumentException If the specified type is
757     *   not a RecordType.
758     */
759    @Override
760    protected StructuredType _leastUpperBound(StructuredType type) {
761        if (!(type instanceof RecordType)) {
762            throw new IllegalArgumentException("RecordType.leastUpperBound: "
763                    + "The argument is not a RecordType.");
764        }
765
766        RecordType recordType = (RecordType) type;
767
768        // the label set of the LUB is the intersection of the two label sets.
769        Set<String> intersectionSet = new HashSet<String>();
770        Set<String> myLabelSet = _fields.keySet();
771        Set<String> argLabelSet = recordType._fields.keySet();
772
773        intersectionSet.addAll(myLabelSet);
774        intersectionSet.retainAll(argLabelSet);
775
776        // construct the GLB RecordToken
777        Object[] labelArray = intersectionSet.toArray();
778        int size = labelArray.length;
779        String[] labels = new String[size];
780        Type[] types = new Type[size];
781
782        for (int i = 0; i < size; i++) {
783            labels[i] = (String) labelArray[i];
784
785            Type type1 = this.get(labels[i]);
786            Type type2 = recordType.get(labels[i]);
787            types[i] = (Type) TypeLattice.lattice().leastUpperBound(type1,
788                    type2);
789        }
790
791        return new RecordType(labels, types);
792    }

796    // Test if the first RecordType is less than or equal to the second
797    private boolean _isLessThanOrEqualTo(RecordType t1, RecordType t2) {
798        Set<String> labelSet1 = t1._fields.keySet();
799        Set<String> labelSet2 = t2._fields.keySet();
800
801        if (!labelSet1.containsAll(labelSet2)) {
802            return false;
803        }
804
805        // iterate over the labels of the second type
806        Iterator<String> iter = labelSet2.iterator();
807
808        while (iter.hasNext()) {
809            String label = iter.next();
810            Type type1 = t1.get(label);
811            Type type2 = t2.get(label);
812            int result = TypeLattice.compare(type1, type2);
813
814            if (result == CPO.HIGHER || result == CPO.INCOMPARABLE) {
815                return false;
816            }
817        }
818
819        return true;
820    }
