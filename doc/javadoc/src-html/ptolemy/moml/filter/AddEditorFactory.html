030import ptolemy.kernel.util.IllegalActionException;
031import ptolemy.kernel.util.NamedObj;
032import ptolemy.moml.MoMLParser;

034///////////////////////////////////////////////////////////////////
035//// AddEditorFactory
036
037/** Add a VisibleParameterEditorFactory named _editorFactory to certain
038 Parameters.
039
040 @author Christopher Hylands, Edward A. Lee
041 @version $Id$
042 @since Ptolemy II 2.0
043 @deprecated Use {@link ptolemy.moml.filter.AddMissingParameter} instead because it handles multiple missing parameter at once.
044 @Pt.ProposedRating Red (cxh)
045 @Pt.AcceptedRating Red (cxh)
046 */
047@Deprecated
048public class AddEditorFactory extends MoMLFilterSimple {
049    /**  Identify Parameters that need a VisibleParameterEditorFactory
050     *   named _editorFactory added.
051     *
052     *  @param container  The container for this attribute.
053     *  in this method.
054     *  @param element The XML element name.
055     *  @param attributeName The name of the attribute.
056     *  @param attributeValue The value of the attribute.
057     *  @param xmlFile The file currently being parsed.
058     *  @return the value of the attributeValue argument.
059     */
060    @Override
061    public String filterAttributeValue(NamedObj container, String element,
062            String attributeName, String attributeValue, String xmlFile) {
063        // This method gets called many times by the MoMLParser,
064        // so we try to be smart about the number of comparisons
065        // and we try to group comparisons together so that we
066        // are not making the same comparison more than once.
067        if (attributeValue == null) {
068            // attributeValue == null is fairly common, so we check for
069            // that first
070            return null;
071        }
072
073        if (attributeName.equals("name")) {
074            // Save the name of the for later use if we see a "class"
075            _lastNameSeen = attributeValue;
076
077            if (_currentlyProcessingActorThatMayNeedAnEditorFactory) {
078                if (attributeValue.equals("_editorFactory")) {
079                    // We are processing a Parameter that already has a
080                    // _editorFactory
081                    _currentlyProcessingActorThatMayNeedAnEditorFactory = false;
082                    _currentAttributeHasLocation = false;
083                } else if (attributeValue.equals("_location")) {
084                    // We only add _editorFactory to parameters that
085                    // have locations
086                    _currentAttributeHasLocation = true;
087                }
088            }
089        }
090
091        // If you change this class, you should run before and after
092        // timing tests on large moml files, a good command to run
093        // is:
094        // $PTII/bin/ptolemy -test $PTII/ptolemy/domains/ct/demo/CarTracking/CarTracking.xml
095        // which will open up a large xml file and then close after 2 seconds.
096        if (attributeName.equals("class")) {
097            if (attributeValue.equals("ptolemy.data.expr.Parameter")) {
098                _currentlyProcessingActorThatMayNeedAnEditorFactory = true;
099
100                if (container != null) {
101                    _currentActorFullName = container.getFullName() + "."
102                            + _lastNameSeen;
103                } else {
104                    _currentActorFullName = "." + _lastNameSeen;
105                }
106            } else if (_currentlyProcessingActorThatMayNeedAnEditorFactory
107                    && container != null
108                    && !container.getFullName().equals(_currentActorFullName)
109                    && !container.getFullName()
110                            .startsWith(_currentActorFullName)) {
111                // We found another class in a different container
112                // while handling a class with port name changes, so
113                _currentlyProcessingActorThatMayNeedAnEditorFactory = false;
114                _currentAttributeHasLocation = false;
115            }
116        }
117
118        return attributeValue;
119    }
120
121    /** Make modifications to the specified container, which is
122     *  defined in a MoML element with the specified name.
123     *  @param container The object created by this element.
124     *  @param elementName The element name.
125     *  @param currentCharData The character data, which appears
126     *   only in the doc and configure elements
127     *  @param xmlFile The file currently being parsed.
128     *  @exception Exception if there is a problem substituting
129     *  in the new value.
130     *  @deprecated Use {@link #filterEndElement(NamedObj, String, StringBuffer, String, MoMLParser)}
131     * instead and pass a MoMLParser.
132     */
133    @Deprecated
134    @Override
135    public void filterEndElement(NamedObj container, String elementName,
136            StringBuffer currentCharData, String xmlFile) throws Exception {
137        filterEndElement(container, elementName, currentCharData, xmlFile,
138                new MoMLParser());
139    }
140
141    /** Make modifications to the specified container, which is
142     *  defined in a MoML element with the specified name.
143     *  @param container The object created by this element.
144     *  @param elementName The element name.
145     *  @param currentCharData The character data, which appears
146     *   only in the doc and configure elements
147     *  @param xmlFile The file currently being parsed.
148     *  @param parser The parser in which MoML is optionally evaluated.
149     *  @exception Exception if there is a problem substituting
150     *  in the new value.
151     */
152    @Override
153    public void filterEndElement(NamedObj container, String elementName,
154            StringBuffer currentCharData, String xmlFile, MoMLParser parser)
155            throws Exception {
156        if (!_currentlyProcessingActorThatMayNeedAnEditorFactory) {
157            return;
158        } else if (_currentAttributeHasLocation && elementName != null
159                && elementName.equals("property") && container != null
160                && container.getFullName().equals(_currentActorFullName)) {
161            _currentlyProcessingActorThatMayNeedAnEditorFactory = false;
162            _currentAttributeHasLocation = false;
163
164            // In theory, we could do something like the lines below
165            // but that would mean that the moml package would depend
166            // on the vergil.toolbox package.
167            //
168            // VisibleParameterEditorFactor _editorFactory =
169            //        new VisibleParameterEditorFactory(container, "_editorFactory");
170
171            // setContext calls parser.reset()
172            parser.setContext(container);
173
174            String moml = "<property name=\"_editorFactory\""
175                    + "class=\"ptolemy.vergil.toolbox."
176                    + "VisibleParameterEditorFactory\">" + "</property>";
177
178            try {
179                // Do not call parse(moml) here, since that method
180                // will fail if we are in an applet because it tries
181                // to read user.dir
182                parser.parse(null, moml);
183                MoMLParser.setModified(true);
184            } catch (Exception ex) {
185                throw new IllegalActionException(null, ex,
186                        "Failed to parse\n" + moml);
187            }
188        }
189    }
190
191    /** Return a string that describes what the filter does.
192     *  @return the description of the filter that ends with a newline.
193     */
194    @Override
195    public String toString() {
196        return getClass().getName() + ": If a parameter has a _location, then\n"
197                + "add a VisibleParameterEditorFactory named _editorFactory.\n";
198    }
199
200    ///////////////////////////////////////////////////////////////////
201    ////                         private variables                 ////
202    // The the full name of the actor we are currently processing
203    private String _currentActorFullName;
204
205    // Set to true if the current attribute has a _location attribute.
206    // This variable is used to determine whether we need to add  a
207    // _editorFactory.
208    private boolean _currentAttributeHasLocation = false;
209
210    // Set to true if we are currently processing an actor that may
211    // need _editorFactory added, set to false when we are done.
212    private boolean _currentlyProcessingActorThatMayNeedAnEditorFactory = false;
213
214    // Last "name" value seen, for use if we see a "class".
215    private String _lastNameSeen;
216}
