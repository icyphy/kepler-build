054/** Generate actor/demo index files.
055 For each actor that is listed in a file, generate a html file that
056 lists the models in which the actor appears.
057
058 <p>For details, see $PTII/vergil/actor/docViewerHelp.htm
059
060 @author Christopher Brooks
061 @version $Id$
062 @since Ptolemy II 5.2
063 @Pt.ProposedRating Red (cxh)
064 @Pt.AcceptedRating Red (cxh)
065 */
066public class ActorIndex {
067    /** Generate the index files.
068     *  @param classesFileName The name of the file that contains the
069     *  dot separated class names - one class per line.
070     *  This file is usually called allActors.txt, which is generated by
071     *  running the $PTII/doc/doclets/PtDoc doclet.  For example, see
072     *  $PTII/doc/codeDoc/allActors.txt
073     *  @param modelsFileName The name of the file that contains the
074     *  urls that point to the models to be parsed, one model
075     *  per line.  This file is usually called models.txt and is generated by
076     *  running ptolemy.actor.gui.HTMLAbout.
077     *  @param outputDirectory Directory in which to write index files.
078     *  The files generated have the name <i>actorName</i>idx.htm, where
079     *  <i>actorName</i> is the name of the actor.  Typically, the value
080     *  for this parameter is doc/codeDoc.
081     *  @exception Exception If there is a problem reading or writing
082     *  a file.
083     */
084    public static void generateActorIndex(String classesFileName,
085            String modelsFileName, String outputDirectory) throws Exception {
086        // The class name is the key, a set of models is the value.
087        HashMap classesToBeIndexed = new HashMap();
088        BufferedReader classesReader = null;
089        BufferedReader modelReader = null;
090        try {
091            // The HandSimDroid work in $PTII/ptserver uses dependency
092            // injection to determine which implementation actors such as
093            // Const and Display to use.  This method reads the
094            // ptolemy/actor/ActorModule.properties file.</p>
095            ActorModuleInitializer.initializeInjector();
096
097            // Read classesFileName and populate the classes Set
098            classesReader = new BufferedReader(new java.io.InputStreamReader(
099                    new java.io.FileInputStream(classesFileName),
100                    java.nio.charset.Charset.defaultCharset()));
101            String className;
102            while ((className = classesReader.readLine()) != null) {
103                if (_verbose) {
104                    System.out.println("Going to index " + className);
105                }
106                classesToBeIndexed.put(className, new HashSet());
107            }
108
109            // Read modelsFileName and parse each model, looking
110            // for classes in which we are interested.
111            modelReader = new BufferedReader(new java.io.InputStreamReader(
112                    new java.io.FileInputStream(modelsFileName),
113                    java.nio.charset.Charset.defaultCharset()));
114            String modelName;
115            MoMLParser parser = new MoMLParser();
116
117            // Handle backward compatibility issues
118            MoMLParser.setMoMLFilters(BackwardCompatibility.allFilters());
119
120            // Add a MoML filter that updates the values of classesToBeIndexed
121            // with models that contain classes named by the key of
122            // classesToBeIndexed
123            NamedObjClassesSeen namedObjClassesSeen = new NamedObjClassesSeen(
124                    classesToBeIndexed);
125            MoMLParser.addMoMLFilter(namedObjClassesSeen);
126
127            // Add a filter to remove the graphical classes
128            // This is safe to do because our NamedObjClassesSeen filter
129            // runs before we then remove the graphical classes.
130            RemoveGraphicalClasses removeGraphicalClasses = new RemoveGraphicalClasses();
131            removeGraphicalClasses.setRemoveGR(true);
132            MoMLParser.addMoMLFilter(removeGraphicalClasses);
133
134            while ((modelName = modelReader.readLine()) != null) {
135                // Reset the list of classes seen, read the model
136                // The filter updates the classesToBeIndexed hashMap
137                namedObjClassesSeen.reset(modelName);
138                //URL modelURL = new File(modelName).toURL();
139                try {
140                    URL modelURL = FileUtilities.nameToURL(modelName, null,
141                            null);
142                    if (_verbose) {
143                        System.out.println("Parsing: " + modelURL);
144                    }
145                    parser.reset();
146                    parser.parse(null, modelURL);
147                } catch (Exception ex) {
148                    System.err.println("Warning, failed to parse " + modelName);
149                    ex.printStackTrace();
150                }
151            }
152
153        } finally {
154            if (classesReader != null) {
155                try {
156                    classesReader.close();
157                } catch (IOException ex) {
158                    System.out.println(
159                            "Failed to close " + classesReader + " " + ex);
160                }
161            }
162            if (modelReader != null) {
163                try {
164                    modelReader.close();
165                } catch (IOException ex) {
166                    System.out.println(
167                            "Failed to close " + modelReader + " " + ex);
168                }
169            }
170        }
171
172        // Write the output files.
173        Iterator classes = classesToBeIndexed.entrySet().iterator();
174        while (classes.hasNext()) {
175            Map.Entry entry = (Map.Entry) classes.next();
176            String actorClassName = (String) entry.getKey();
177            if (((Set) entry.getValue()).size() == 0) {
178                // Skip classes that are not used in a demo
179                // ptolemy.vergil.actor.DocManager checks to see if
180                // the Idx.htm file exists before creating a link to it.
181                continue;
182            }
183            BufferedWriter writer = null;
184            try {
185                String outputFileName = outputDirectory + File.separator
186                        + actorClassName.replace('.', File.separatorChar)
187                        + "Idx.htm";
188
189                // Determine the relative path to $PTII from this
190                // file.  We need this so that we can link to the models.
191                String canonicalOutputFileName = new File(outputFileName)
192                        .getCanonicalPath().replace('\\', '/');
193
194                // Get PTII as C:/cxh/ptII
195                String ptII = null;
196                try {
197                    ptII = new URI(StringUtilities
198                            .getProperty("ptolemy.ptII.dirAsURL")).normalize()
199                                    .getPath();
200                    // Under Windows, convert /C:/foo/bar to C:/foo/bar
201                    ptII = new File(ptII).getCanonicalPath().replace('\\', '/');
202                } catch (URISyntaxException ex) {
203                    throw new InternalErrorException(null, ex,
204                            "Failed to process PTII " + ptII);
205                }
206                if (ptII == null || ptII.length() == 0) {
207                    throw new InternalErrorException("Failed to process "
208                            + "ptolemy.ptII.dirAsURL property, ptII = null?");
209                }
210
211                String relativePath = "";
212                if (canonicalOutputFileName.startsWith(ptII)) {
213                    // If the canonical output file name starts with ptII
214                    // we then generate a relative path
215                    String relativeOutputFileName = StringUtilities
216                            .substitute(canonicalOutputFileName, ptII, "");
217                    StringBuffer relativePathBuffer = new StringBuffer();
218                    int index = 0;
219                    while (relativeOutputFileName.indexOf('/', index) != -1) {
220                        index = relativeOutputFileName.indexOf('/', index) + 1;
221                        relativePathBuffer.append("../");
222                    }
223                    relativePath = relativePathBuffer.toString();
224                    // Strip off the last ../
225                    relativePath = relativePath.substring(0,
226                            relativePath.length() - 3);
227                }
228
229                // Make directories if necessary
230                File outputDirectoryFile = new File(
231                        new File(outputFileName).getParent());
232                if (!outputDirectoryFile.exists()) {
233                    if (_verbose) {
234                        System.out.println("Creating " + outputDirectoryFile);
235                    }
236                    if (!outputDirectoryFile.mkdirs()) {
237                        throw new IOException("Directory \""
238                                + outputDirectoryFile
239                                + "\" does not exist and cannot be created.");
240                    }
241                }
242
243                if (_verbose) {
244                    System.out.println("Writing " + outputFileName);
245                }
246                writer = new BufferedWriter(new FileWriter(outputFileName));
247
248                writer.write("<html>\n<head>\n<title>Index for "
249                        + actorClassName + "</title>\n" + "<link href=\""
250                        + relativePath + "doc/default.css\""
251                        + "rel=\"stylesheet\" type=\"text/css\">\n"
252                        + "</head>\n<body>\n" + "<h2>" + actorClassName
253                        + "</h2>\n" + "Below are demonstration models that use "
254                        + actorClassName + "\n<ul>\n");
255
256                // Loop through all the models that use this actor
257                Iterator models = ((Set) entry.getValue()).iterator();
258                while (models.hasNext()) {
259                    String model = (String) models.next();
260                    if (model.startsWith("$CLASSPATH")) {
261                        model = model.substring(11);
262                    }
263                    writer.write("<li><a href=\"" + relativePath + model + "\">"
264                            + model + "</a>\n");
265                }
266                writer.write("</ul>\n</body>\n</html>\n");
267            } finally {
268                if (writer != null) {
269                    writer.close();
270                }
271            }
272        }
273    }
274
275    /** Generate index documentation.  The three arguments are passed to
276     *  {@link #generateActorIndex(String, String, String)}.
277     *  <pre>
278     *  java -classpath "$PTII;$PTII/lib/diva.jar" ptolemy.moml.filter.ActorIndex allActors.txt models.txt doc/codeDoc
279     *  </pre>
280     *  @param args An array of three Strings
281     *  <br> The name of the file that lists all the actors in which we are
282     *  interested.
283     *  <br> The name of the file that lists all the models to be indexed.
284     *  <br> The directory in which to write the index files.
285     *  @exception Exception If there is a problem reading or writing
286     *  a file.
287     */
288    public static void main(String[] args) throws Exception {
289        try {
290            ActorIndex.generateActorIndex(args[0], args[1], args[2]);
291        } catch (Throwable throwable) {
292            System.err.print(KernelException.stackTraceToString(throwable));
293            StringUtilities.exit(1);
294        }
295        // Need to call exit here because Jython and Jetty cause hangs during exit.
296        StringUtilities.exit(0);
297    }
