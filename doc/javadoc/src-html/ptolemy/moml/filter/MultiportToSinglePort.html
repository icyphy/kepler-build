036//// MultiportToSinglePort

038/** A filter to convert specific multiports of specific
039 actors to a single ports.
040
041 <p>When this class is registered with the MoMLParser.setMoMLFilter()
042 method, it will cause will cause MoMLParser to filter so that models from
043 earlier releases will run in the current release.
044
045 <p>The Autocorrelation actor changed between PtolemyII 2.x and 3.x
046 such that the output port is no longer a multiport.
047
048 <pre>
049 // Autocorrelation
050 _actorsWithMultiPortToSinglePortChanges
051 .put("ptolemy.actor.lib.Autocorrelation, "output")
052
053 </pre>
054
055 <p>The _actorsWithMultiPortToSinglePortChanges HashMap contains Strings
056 that name classes such as Autocorrelation that have multiports that should
057 be single ports.  The HashMap maps classnames to port names.
058
059
060 <p> Conceptually, how the code works is that when we see a class while
061 parsing, we check to see if the class is in
062 _actorsWithMultiPortToSinglePortChanges.
063
064 If the class was present in the HashMap, then as we go through the
065 code, we look for the named port and remove the multiport declaration
066
067 @author Christopher Hylands Brooks, Contributor: Edward A. Lee
068 @version $Id$
069 @since Ptolemy II 4.0
070 @Pt.ProposedRating Red (cxh)
071 @Pt.AcceptedRating Red (cxh)
072 */
073public class MultiportToSinglePort extends MoMLFilterSimple {
074    /** If the attributeName is "class" and attributeValue names a
075     *  class that has had its port names changed between releases,
076     *  then substitute in the new port names.
077     *
078     *  @param container  The container for this attribute.
079     *  in this method.
080     *  @param element The XML element name.
081     *  @param attributeName The name of the attribute.
082     *  @param attributeValue The value of the attribute.
083     *  @param xmlFile The file currently being parsed.
084     *  @return the value of the attributeValue argument.
085     */
086    @Override
087    public String filterAttributeValue(NamedObj container, String element,
088            String attributeName, String attributeValue, String xmlFile) {
089        //System.out.println("filterAttributeValue: " + container + "\t"
090        //  +  attributeName + "\t" + attributeValue);
091        // This method gets called many times by the MoMLParser,
092        // so we try to be smart about the number of comparisons
093        // and we try to group comparisons together so that we
094        // are not making the same comparison more than once.
095        if (attributeValue == null) {
096            // attributeValue == null is fairly common, so we check for
097            // that first
098            return null;
099        }
100
101        if (attributeName.equals("name")) {
102            // Save the name of the for later use if we see a "class"
103            _lastNameSeen = attributeValue;
104
105            if (_currentlyProcessingActorWithPropertyClassChanges) {
106                if (_portName != null && _portName.equals(attributeValue)) {
107                    // We will do the above checks only if we found a
108                    // class that had property class changes.
109                    //_foundChange = true;
110                } else {
111                    if (attributeValue.equals("multiport") && container != null
112                            && container.getName().equals(_portName)) {
113                        // What if the multiport is false?
114                        //_foundChange = false;
115                        MoMLParser.setModified(true);
116                        return null;
117                    }
118
119                    // Saw a name that did not match.
120                    // However, we might have other names that
121                    // did match, so keep looking
122                    //_foundChange = false;
123                }
124            }
125        }
126
127        // If you change this class, you should run before and after
128        // timing tests on large moml files, a good command to run
129        // is:
130        // $PTII/bin/ptolemy -test $PTII/ptolemy/domains/ct/demo/CarTracking/CarTracking.xml
131        // which will open up a large xml file and then close after 2 seconds.
132        if (attributeName.equals("class")) {
133            if (_actorsWithMultiPortToSinglePortChanges
134                    .containsKey(attributeValue)) {
135                // We found a class with a property class change.
136                _currentlyProcessingActorWithPropertyClassChanges = true;
137                if (container == null) {
138                    // Under certain circumstances, we can parse something
139                    // that has no container
140                    _currentActorFullName = "." + _lastNameSeen;
141                } else {
142                    _currentActorFullName = container.getFullName() + "."
143                            + _lastNameSeen;
144                }
145                _portName = _actorsWithMultiPortToSinglePortChanges
146                        .get(attributeValue);
147
148                //             } else if (_currentlyProcessingActorWithPropertyClassChanges
149                //                     && _foundChange) {
150                //                 // We found a property class to change, and now we
151                //                 // found the class itself that needs changing.
152                //                 // Only return the new class once, but we might
153                //                 // have other properties that need changing
154                //                 //_currentlyProcessingActorWithPropertyClassChanges = false;
155                //                 if (!attributeValue.equals(_newClass)) {
156                //                     MoMLParser.setModified(true);
157                //                 }
158                //                 _foundChange = false;
159                //                 return attributeValue;
160            } else if (_currentlyProcessingActorWithPropertyClassChanges
161                    && container != null
162                    && !container.getFullName().equals(_currentActorFullName)
163                    && !container.getFullName()
164                            .startsWith(_currentActorFullName)) {
165                // We found another class in a different container
166                // while handling a class with multiport change
167                //_foundChange = false;
168                _currentlyProcessingActorWithPropertyClassChanges = false;
169            }
170        }
171
172        return attributeValue;
173    }
174
175    /** In this class, do nothing.
176     *  @param container The object created by this element.
177     *  @param elementName The element name.
178     *  @param currentCharData The character data, which appears
179     *   only in the doc and configure elements
180     *  @param xmlFile The file currently being parsed.
181     *  @exception Exception Not thrown in this base class.
182     */
183    @Override
184    public void filterEndElement(NamedObj container, String elementName,
185            StringBuffer currentCharData, String xmlFile) throws Exception {
186    }
187
188    /** Return a string that describes what the filter does.
189     *  @return the description of the filter that ends with a newline.
190     */
191    @Override
192    public String toString() {
193        StringBuffer results = new StringBuffer(getClass().getName()
194                + ": Update any actor multiports that are now\n"
195                + "single ports.\n"
196                + "Below are the actors that are affected, along "
197                + "with the port name:\n");
198        //Iterator actors = _actorsWithMultiPortToSinglePortChanges.keySet()
199        //.iterator();
200
201        //while (actors.hasNext()) {
202        for (Map.Entry<String, String> actor : _actorsWithMultiPortToSinglePortChanges
203                .entrySet()) {
204            //String actor = (String) actors.next();
205            results.append("\t" + actor.getKey() + "\n"
206            //+ (String) _actorsWithMultiPortToSinglePortChanges.get(actor));
207                    + actor.getValue());
208        }
209
210        return results.toString();
211    }
212
213    ///////////////////////////////////////////////////////////////////
214    ////                         private variables                 ////
215    // Map of actor names a HashMap of class names to multiport names
216    // that should be single ports.
217    private static HashMap<String, String> _actorsWithMultiPortToSinglePortChanges;
218
219    // The the full name of the actor we are currently processing
220    private String _currentActorFullName;
221
222    // Set to true if we are currently processing an actor with parameter
223    // class changes, set to false when we are done.
224    private boolean _currentlyProcessingActorWithPropertyClassChanges = false;
225
226    // Last "name" value seen, for use if we see a "class".
227    private String _lastNameSeen;
228
229    // Keep track of whether a change was found.
230    //private static boolean _foundChange;
231
232    // The name of the port we are looking for.
233    private String _portName;
234
235    static {
236        ///////////////////////////////////////////////////////////
237        // Actors that have ports that have changed from multi to single
238        _actorsWithMultiPortToSinglePortChanges = new HashMap<String, String>();
239
240        // Autocorrelation
241        _actorsWithMultiPortToSinglePortChanges
242                .put("ptolemy.domains.sdf.lib.Autocorrelation", "output");
243
244        _actorsWithMultiPortToSinglePortChanges
245                .put("ptolemy.actor.lib.NonStrictTest", "input");
246
247        // In Ptolemy II 3.0.2, SOC_FSM_SR_HDE.xml has a Const with multiport.
248        _actorsWithMultiPortToSinglePortChanges.put("ptolemy.actor.lib.Const",
249                "output");
250
251        _actorsWithMultiPortToSinglePortChanges
252                .put("ptolemy.actor.lib.gui.MatrixViewer", "input");
253    }
