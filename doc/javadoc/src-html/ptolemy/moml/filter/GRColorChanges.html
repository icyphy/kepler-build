036///////////////////////////////////////////////////////////////////
037//// GRColorChanges
038
039/**
040 This class filters MoML files for backward compatibility between
041 GR models constructed in version 4.0 or earlier. In particular, it
042 handles the switch to using ColorAttribute for all colors, plus
043 some parameter renaming. It also handles some conversions from
044 matrix parameters to arrays.
045
046 @author Edward A. Lee
047 @version $Id$
048 @since Ptolemy II 4.1
049 @Pt.ProposedRating Yellow (eal)
050 @Pt.AcceptedRating Red (cxh)
051 */
052public class GRColorChanges extends MoMLFilterSimple {
053    ///////////////////////////////////////////////////////////////////
054    ////                         public methods                    ////
055
056    /** Handle parameter name changes.
057     *  @param container  The container for XML element.
058     *  @param element The XML element name.
059     *  @param attributeName The name of the attribute.
060     *  @param attributeValue The value of the attribute.
061     *  @param xmlFile The file currently being parsed.
062     *  @return A new value for the attribute, or the same value
063     *   to leave it unchanged, or null to cause the current element
064     *   to be ignored (unless the attributeValue argument is null).
065     */
066    @Override
067    public String filterAttributeValue(NamedObj container, String element,
068            String attributeName, String attributeValue, String xmlFile) {
069        if (attributeValue != null && attributeName != null) {
070            if (attributeValue.equals("RGB color")
071                    && attributeName.equals("name") && container != null
072                    && _actorsWithRGBColor
073                            .contains(container.getClass().getName())) {
074                // NOTE: This relies on their being no nested
075                // instance of this attribute, which there shouldn't
076                // be.
077                _foundOne = true;
078                MoMLParser.setModified(true);
079                return "diffuseColor";
080            }
081        }
082
083        return attributeValue;
084    }
085
086    /** If the container is a property named "diffuseColor" contained
087     *  by one of the GR actors, then check the format of its parameter
088     *  to change it, if necessary, from matrix format to array format.
089     *  Also fix the background color of ViewScreen3D.
090     *  If the property name is "polygon" contained by an instance
091     *  of PolyCylinder3D, perform a similar change from matrix
092     *  to array.
093     *  @param container The object defined by the element that this
094     *   is the end of.
095     *  @param elementName The element name.
096     *  @param currentCharData The character data, which appears
097     *   only in the doc and configure elements
098     *  @param xmlFile The file currently being parsed.
099     *  @exception Exception if there is a problem substituting
100     *  in the new value.
101     */
102    @Override
103    public void filterEndElement(NamedObj container, String elementName,
104            StringBuffer currentCharData, String xmlFile) throws Exception {
105        if (_foundOne) {
106            _foundOne = false;
107
108            if (elementName != null && elementName.equals("property")
109                    && container.getName().equals("diffuseColor")
110                    && ((Settable) container).getExpression().trim()
111                            .startsWith("[")) {
112                // Found one in matrix format.
113                String value = ((Settable) container).getExpression().trim();
114                value = value.replace('[', '{');
115                value = value.replace(']', '}');
116                ((Settable) container).setExpression(value);
117                MoMLParser.setModified(true);
118            }
119        }
120
121        // Fix the background color of the ViewScreen actor.
122        // Note that the ViewScreen actor also has a name change.
123        if (container != null
124                && container.getName().equals("backgroundColor")) {
125            NamedObj actor = container.getContainer();
126
127            if (actor != null && actor.getClass().getName()
128                    .startsWith("ptolemy.domains.gr.lib.ViewScreen")) {
129                String value = ((Settable) container).getExpression().trim();
130
131                if (value.startsWith("[")) {
132                    value = value.replace('[', '{');
133                    value = value.replace(']', '}');
134                    ((Settable) container).setExpression(value);
135                    MoMLParser.setModified(true);
136                }
137            }
138        }
139
140        // Fix the polygon attribute of the PolyCylinder3D actor.
141        if (container != null && container.getName().equals("polygon")) {
142            NamedObj actor = container.getContainer();
143
144            if (actor != null && actor.getClass().getName()
145                    .equals("ptolemy.domains.gr.lib.PolyCylinder3D")) {
146                String value = ((Settable) container).getExpression().trim();
147
148                if (value.startsWith("[")) {
149                    value = value.replace('[', '{');
150                    value = value.replace(']', '}');
151                    ((Settable) container).setExpression(value);
152                    MoMLParser.setModified(true);
153                }
154            }
155        }
156
157        // Fix the polyline attribute of the CircularSweep3D actor.
158        if (container != null && container.getName().equals("polyline")) {
159            NamedObj actor = container.getContainer();
160
161            if (actor != null && actor.getClass().getName()
162                    .equals("ptolemy.domains.gr.lib.CircularSweep3D")) {
163                String value = ((Settable) container).getExpression().trim();
164
165                if (value.startsWith("[")) {
166                    value = value.replace('[', '{');
167                    value = value.replace(']', '}');
168                    ((Settable) container).setExpression(value);
169                    MoMLParser.setModified(true);
170                }
171            }
172        }
173    }
174
175    /** Return a string that describes what the filter does.
176     *  @return A description of the filter (ending with a newline).
177     */
178    @Override
179    public String toString() {
180        StringBuffer results = new StringBuffer(getClass().getName()
181                + ": Update GR actor with the following changes:\n");
182        results.append("\tParameter name \"RGB color\" --> \"diffuseColor\"");
183        return results.toString();
184    }

186    ///////////////////////////////////////////////////////////////////

188
189    /** Set of actors with parameter named "RGB color" */
190    private static HashSet _actorsWithRGBColor = new HashSet();
191
192    static {
193        GRColorChanges._actorsWithRGBColor.add("ptolemy.domains.gr.lib.Box3D");
194        GRColorChanges._actorsWithRGBColor
195                .add("ptolemy.domains.gr.lib.CircularSweep3D");
196        GRColorChanges._actorsWithRGBColor.add("ptolemy.domains.gr.lib.Cone3D");
197        GRColorChanges._actorsWithRGBColor
198                .add("ptolemy.domains.gr.lib.Cylinder3D");
199        GRColorChanges._actorsWithRGBColor
200                .add("ptolemy.domains.gr.lib.Loader3D");
201        GRColorChanges._actorsWithRGBColor
202                .add("ptolemy.domains.gr.lib.PolyCylinder3D");
203        GRColorChanges._actorsWithRGBColor
204                .add("ptolemy.domains.gr.lib.Sphere3D");
205        GRColorChanges._actorsWithRGBColor
206                .add("ptolemy.domains.gr.lib.TextString3D");
207        GRColorChanges._actorsWithRGBColor
208                .add("ptolemy.domains.gr.lib.Torus3D");
209    }
210
211    /** Flag indicating that we found one whose name needed changing. */
212    private boolean _foundOne = false;
