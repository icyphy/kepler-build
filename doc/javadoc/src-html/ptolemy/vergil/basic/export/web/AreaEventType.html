029package ptolemy.vergil.basic.export.web;
030
031import ptolemy.data.expr.StringParameter;
032import ptolemy.kernel.util.IllegalActionException;
033import ptolemy.kernel.util.NameDuplicationException;
034import ptolemy.kernel.util.NamedObj;
035
036///////////////////////////////////////////////////////////////////
037//// AreaEventType
038/**
039 * A parameter used to specify the event type actions associated
040 * with an area in an HTML image map. This parameter offers the following events:
041 *  <ul>
042 *  <li><b>onblur</b>: Command to be run when an element loses focus.
043 *  <li><b>onclick</b>: Command to be run on a mouse click.
044 *  <li><b>ondblclick</b>: Command to be run on a mouse double-click.
045 *  <li><b>onfocus</b>: Command to be run when an element gets focus.
046 *  <li><b>onmousedown</b>: Command to be run when mouse button is pressed.
047 *  <li><b>onmousemove</b>: Command to be run when mouse pointer moves.
048 *  <li><b>onmouseout</b>: Command to be run when mouse pointer moves out of an element.
049 *  <li><b>onmouseover</b>: Command to be run when mouse pointer moves over an element.
050 *  <li><b>onmouseup</b>: Command to be run when mouse button is released.
051 *  <li><b>onkeydown</b>: Command to be run when a key is pressed.
052 *  <li><b>onkeypress</b>: Command to be run when a key is pressed and released.
053 *  <li><b>onkeyup</b>: Command to be run when a key is released.
054 *  </ul>
055 *  These are the events supported by the HTML area tag.
056 *  The default value is "onmouseover". It is not clear how
057 *  these areas get the focus, so the focus and key commands do not
058 *  appear to be useful.
059 *
060 * @author Edward A. Lee
061 * @version $Id$
062 * @since Ptolemy II 10.0
063 * @Pt.ProposedRating Red (cxh)
064 * @Pt.AcceptedRating Red (cxh)
065 */
066public class AreaEventType extends StringParameter {
067
068    /** Create an instance of this parameter.
069     *  @param container The container.
070     *  @param name The name.
071     *  @exception IllegalActionException If the superclass throws it.
072     *  @exception NameDuplicationException If the superclass throws it.
073     */
074    public AreaEventType(NamedObj container, String name)
075            throws IllegalActionException, NameDuplicationException {
076        super(container, name);
077
078        addChoice("onblur");
079        addChoice("onclick");
080        addChoice("ondblclick");
081        addChoice("onfocus");
082        addChoice("onmousedown");
083        addChoice("onmousemove");
084        addChoice("onmouseout");
085        addChoice("onmouseover");
086        addChoice("onmouseup");
087        addChoice("onkeydown");
088        addChoice("onkeypress");
089        addChoice("onkeyup");
090        setExpression("onmouseover");
091    }
092}
