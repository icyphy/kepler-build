063/**
064 * A drop target listener that provides drag and drop for the Ptolemy editor.
065 *
066 * <p>Derived classes could extend this class to provide a different
067 * drag and drop interface.</p>
068 *
069 * @author Sven Koehler, Contributor: Christopher Brooks.
070 * @version $Id$
071 * @since Ptolemy II 10.0
072 * @Pt.ProposedRating Red (eal)
073 * @Pt.AcceptedRating Red (johnr)
074 */
075public class EditorDropTargetListener implements DropTargetListener {
076
077    ///////////////////////////////////////////////////////////////////
078    ////                         public methods                    ////
079
080    /** Accept the event if the data is a known key.
081     *  This is called while a drag operation is ongoing,
082     *  when the mouse pointer enters the operable part of
083     *  the drop site for the DropTarget registered with
084     *  this listener.
085     *  @param dropEvent The drop event.
086     */
087    @Override
088    public void dragEnter(DropTargetDragEvent dropEvent) {
089        // Notify additionalListeners.
090        for (int i = 0; i < _dropTarget.getAdditionalListeners().size(); i++) {
091            DropTargetListener listener = _dropTarget.getAdditionalListeners()
092                    .elementAt(i);
093            listener.dragEnter(dropEvent);
094        }
095
096        if (dropEvent
097                .isDataFlavorSupported(PtolemyTransferable.namedObjFlavor)) {
098            dropEvent.acceptDrag(DnDConstants.ACTION_COPY_OR_MOVE);
099        } else {
100            dropEvent.rejectDrag();
101        }
102    }
103
104    /** Remove any highlighting that might be active.
105     *  This is called while a drag operation is ongoing, when the mouse
106     *  pointer has exited the operable part of the drop site for the
107     *  DropTarget registered with this listener.
108     *  @param dropEvent The drop event.
109     */
110    @Override
111    public void dragExit(DropTargetEvent dropEvent) {
112        // Notify additionalListeners.
113        for (int i = 0; i < _dropTarget.getAdditionalListeners().size(); i++) {
114            DropTargetListener listener = _dropTarget.getAdditionalListeners()
115                    .elementAt(i);
116            listener.dragExit(dropEvent);
117        }
118        if (_highlighted != null) {
119            _highlighter.renderDeselected(_highlightedFigure);
120            _highlighted = null;
121            _highlightedFigure = null;
122        }
123    }
124
125    /** If the location of the event is over an icon for
126     *  an instance of NamedObj, then highlight that icon.
127     *  This is called when a drag operation is ongoing,
128     *  while the mouse pointer is still over the operable
129     *  part of the drop site for the DropTarget registered
130     *  with this listener.
131     *  @param dropEvent The drop event.
132     */
133    @Override
134    public void dragOver(DropTargetDragEvent dropEvent) {
135        // Notify additionalListeners.
136        for (int i = 0; i < _dropTarget.getAdditionalListeners().size(); i++) {
137            DropTargetListener listener = _dropTarget.getAdditionalListeners()
138                    .elementAt(i);
139            listener.dragOver(dropEvent);
140        }
141        // See whether there is a container under the point.
142        Point2D originalPoint = SnapConstraint
143                .constrainPoint(dropEvent.getLocation());
144        NamedObj over = _getObjectUnder(originalPoint);
145
146        if (over != _highlighted) {
147            if (_highlighted != null) {
148                _highlighter.renderDeselected(_highlightedFigure);
149                _highlighted = null;
150                _highlightedFigure = null;
151            }
152
153            if (over != null && _dropTarget.isDropIntoEnabled()) {
154                if (_highlighter == null) {
155                    _highlighter = new AnimationRenderer(Color.white);
156                }
157
158                _highlighted = over;
159                _highlightedFigure = _getFigureUnder(originalPoint);
160                _highlighter.renderSelected(_highlightedFigure);
161            }
162        }
163
164        // Used to do this... Any reason for it?
165        // dragEnter(dropEvent);
166    }
167
168    /** If the transferable object is recognized as a Ptolemy II object,
169     *  then use the MoML description of the object to create a new
170     *  instance of the object at the drop location. If the drop
171     *  location is on top of an icon representing an instance of
172     *  NamedObj, then make that instance the container of the new
173     *  object. Otherwise, make the model associated with the graph
174     *  the container.
175     *  This is called when the drag operation has terminated with a
176     *  drop on the operable part of the drop site for the DropTarget
177     *  registered with this listener.
178     *  @param dropEvent The drop event.
179     */
180    @Override
181    public void drop(DropTargetDropEvent dropEvent) {
182        // Notify additionalListeners.
183        for (int i = 0; i < _dropTarget.getAdditionalListeners().size(); i++) {
184            DropTargetListener listener = _dropTarget.getAdditionalListeners()
185                    .elementAt(i);
186            listener.drop(dropEvent);
187        }
188        // Unhighlight the target. Do this first in case
189        // errors occur... Don't want to leave highlighting.
190        if (_highlighted != null) {
191            _highlighter.renderDeselected(_highlightedFigure);
192            _highlighted = null;
193            _highlightedFigure = null;
194        }
195
196        // See whether there is a container under the point.
197        Point2D originalPoint = SnapConstraint
198                .constrainPoint(dropEvent.getLocation());
199        NamedObj targetContainer = _getObjectUnder(originalPoint);
200
201        // Find the root container (the composite entity for the window).
202        GraphPane pane = ((JGraph) _dropTarget.getComponent()).getGraphPane();
203        GraphController controller = pane.getGraphController();
204        GraphModel model = controller.getGraphModel();
205        NamedObj rootContainer = (NamedObj) model.getRoot();
206
207        if (targetContainer == null || !_dropTarget.isDropIntoEnabled()) {
208            // Find the default container for the dropped object
209            targetContainer = rootContainer;
210        }
211
212        // Find the location for the dropped objects.
213        // Account for the scaling in the pane.
214        Point2D transformedPoint = new Point2D.Double();
215        pane.getTransformContext().getInverseTransform()
216                .transform(originalPoint, transformedPoint);
217
218        // Get an iterator over objects to drop.
219        Iterator iterator = null;
220
221        java.util.List dropObjects = null;
222        if (dropEvent
223                .isDataFlavorSupported(PtolemyTransferable.namedObjFlavor)) {
224            try {
225                dropEvent.acceptDrop(DnDConstants.ACTION_COPY_OR_MOVE);
226                dropObjects = (java.util.List) dropEvent.getTransferable()
227                        .getTransferData(PtolemyTransferable.namedObjFlavor);
228                iterator = dropObjects.iterator();
229            } catch (Exception e) {
230                MessageHandler
231                        .error("Can't find a supported data flavor for drop in "
232                                + dropEvent, e);
233                return;
234            }
235        } else {
236            dropEvent.rejectDrop();
237        }
238
239        if (iterator == null) {
240            // Nothing to drop!
241            return;
242        }
243
244        while (iterator.hasNext()) {
245            // Create the MoML change request to instantiate the new objects.
246            StringBuffer moml = new StringBuffer();
247
248            final NamedObj dropObj = (NamedObj) iterator.next();
249
250            // Figure out the destination container.
251            NamedObj destinationContainer = targetContainer;
252            boolean relativeLocation = false;
253            if (dropObj instanceof RelativeLocatable
254                    && targetContainer != rootContainer) {
255                relativeLocation = true;
256                destinationContainer = rootContainer;
257            }
258            final String name;
259
260            if (dropObj instanceof Singleton) {
261                name = dropObj.getName();
262            } else {
263                name = destinationContainer.uniqueName(dropObj.getName());
264            }
265
266            // Constrain point to snap to grid.
267            Point2D newPoint = SnapConstraint.constrainPoint(transformedPoint);
268            /* At this point we wish to provide for an alternative way to get the MOML
269             * code for the drop object. This is to allow for a drop object that may just
270             * contain a reference (e.g. an LSID) to the complete MOML rather than including
271             * all the information in the tree node object being dragged.
272             * DFH July 2007
273             */
274
275            // first try to get the LSID; if one cannot be found use
276            // the Ptolemy method for dropping
277            boolean lsidFlag = true;
278            try {
279                String lsidString = ((StringAttribute) dropObj
280                        .getAttribute("entityId")).getExpression();
281                if (lsidString == null || lsidString.equals("")) {
282                    lsidFlag = false;
283                }
284            } catch (Exception eee) {
285                lsidFlag = false;
286            }
287
288            String result = "";
289            String dropObjElementType = dropObj.getElementName();
290            Object object = null;
291
292            StringAttribute alternateGetMomlActionAttribute = null;
293            alternateGetMomlActionAttribute = (StringAttribute) dropObj
294                    .getAttribute("_alternateGetMomlAction");
295            if (alternateGetMomlActionAttribute == null && lsidFlag) {
296                Configuration config = null;
297                java.util.List configsList = Configuration.configurations();
298                for (Iterator it = configsList.iterator(); it.hasNext();) {
299                    config = (Configuration) it.next();
300                    if (config != null) {
301                        break;
302                    }
303                }
304                if (config == null) {
305                    throw new KernelRuntimeException(dropObj, "Could not find "
306                            + "configuration, list of configurations was "
307                            + configsList.size() + " elements, all were null.");
308                }
309                alternateGetMomlActionAttribute = (StringAttribute) config
310                        .getAttribute("_alternateGetMomlAction");
311            }
312
313            boolean appendGroupAuto = true;
314            if (alternateGetMomlActionAttribute != null) {
315                String alternateGetMomlClassName = alternateGetMomlActionAttribute
316                        .getExpression();
317                try {
318                    Class getMomlClass = Class
319                            .forName(alternateGetMomlClassName);
320                    object = getMomlClass.newInstance();
321                    try {
322                        Method getMomlMethod = getMomlClass.getMethod("getMoml",
323                                new Class[] { NamedObj.class, String.class });
324                        result = (String) getMomlMethod.invoke(object,
325                                new Object[] { dropObj, name });
326                        appendGroupAuto = false;
327                    } catch (NoSuchMethodException e) {
328                        Method getMomlMethod = getMomlClass.getMethod("getMoml",
329                                new Class[] { NamedObj.class });
330                        result = (String) getMomlMethod.invoke(object,
331                                new Object[] { dropObj });
332                        int int1 = 1;
333                        int int2 = result.indexOf(" ");
334                        dropObjElementType = result.substring(int1, int2);
335                        // following string substitution is needed to
336                        // replace possible name changes when multiple
337                        // copies of an actor are added to a workspace
338                        // canvas (name then has integer appended to it)
339                        // -- DFH
340                        int1 = result.indexOf("\"", 1);
341                        int2 = result.indexOf("\"", int1 + 1);
342
343                        result = result.substring(0, int1 + 1) + name
344                                + result.substring(int2, result.length());
345                    }
346                    moml.append(result);
347                } catch (Throwable throwable) {
348                    System.out.println("Error creating alternateGetMoml!");
349                }
350            } else { // default method for PtolemyII use
351                result = dropObj.exportMoML(name);
352                moml.append(result);
353            }
354
355            if (appendGroupAuto) {
356                moml.insert(0, "<group name=\"auto\">\n");
357                moml.append(
358                        "<" + dropObjElementType + " name=\"" + name + "\">\n");
359                if (relativeLocation) {
360                    moml.append("<property name=\"_location\" "
361                            + "class=\"ptolemy.kernel.util.RelativeLocation\" value=\"{"
362                            + RelativeLocation.INITIAL_OFFSET + ", "
363                            + RelativeLocation.INITIAL_OFFSET + "}\">"
364                            + "<property name=\"relativeTo\" value=\""
365                            + targetContainer.getName() + "\"/>\n"
366                            // Need to identify the targetContainer as an
367                            // Entity, Port, Relation, or Attribute.
368                            + "<property name=\"relativeToElementName\" value=\""
369                            + targetContainer.getElementName()
370                            + "\"/></property>\n</" + dropObjElementType
371                            + ">\n");
372
373                } else {
374                    moml.append("<property name=\"_location\" "
375                            + "class=\"ptolemy.kernel.util.Location\" value=\"{");
376                    moml.append((int) newPoint.getX());
377                    moml.append(", ");
378                    moml.append((int) newPoint.getY());
379                    moml.append("}\"/>\n</" + dropObjElementType + ">\n");
380                }
381                moml.append("</group>\n");
382            }
383            if (destinationContainer instanceof DropTargetHandler) {
384                try {
385                    ((DropTargetHandler) destinationContainer).dropObject(
386                            destinationContainer, dropObjects, moml.toString());
387                } catch (IllegalActionException e) {
388                    MessageHandler.error("Unable to drop the object to "
389                            + destinationContainer.getName() + ".", e);
390                }
391            } else {
392                MoMLChangeRequest request = new MoMLChangeRequest(this,
393                        destinationContainer, moml.toString());
394                request.setUndoable(true);
395                destinationContainer.requestChange(request);
396            }
397        }
398
399        dropEvent.dropComplete(true); //success!
400
401        //Added by MB 6Apr06 - without this, tooltips don't work
402        //after first actor is dragged to canvas from library, until
403        //pane loses & regains focus
404        JComponent comp = (JComponent) _dropTarget.getComponent();
405        if (comp != null) {
406            ToolTipManager.sharedInstance().registerComponent(comp);
407        }
408    }

410    /** Accept the event if the data is a known key.
411     *  This is called if the user has modified the current drop gesture.
412     *  @param dropEvent The drop event.
413     */
414    @Override
415    public void dropActionChanged(DropTargetDragEvent dropEvent) {
416        // Notify additionalListeners.
417        for (int i = 0; i < _dropTarget.getAdditionalListeners().size(); i++) {
418            DropTargetListener listener = _dropTarget.getAdditionalListeners()
419                    .elementAt(i);
420            listener.dropActionChanged(dropEvent);
421        }
422        // Used to do this... Not needed?
423        // dragEnter(dropEvent);
424    }

426    /** Links this Listener back to the EditorDropTarget it belongs to.
427     *  @param dropTarget The drop target.
428     */
429    public void setDropTarget(EditorDropTarget dropTarget) {
430        _dropTarget = dropTarget;

436    /** Return the figure that is an icon of a NamedObj and is
437     *  under the specified point, or null if there is none.
438     *  @param point The point in the graph pane.
439     *  @return The object under the specified point, or null if there
440     *   is none or it is not a NamedObj.
441     */
442    protected Figure _getFigureUnder(Point2D point) {
443        GraphPane pane = ((JGraph) _dropTarget.getComponent()).getGraphPane();
444
445        // Account for the scaling in the pane.
446        Point2D transformedPoint = new Point2D.Double();
447        pane.getTransformContext().getInverseTransform().transform(point,
448                transformedPoint);
449
450        return BasicGraphFrame.getFigureUnder(pane, transformedPoint,
451                new Object[] {});
452    }
453
454    /** Return the object under the specified point, or null if there
455     *  is none.
456     *  @param point The point in the graph pane.
457     *  @return The object under the specified point, or null if there
458     *   is none or it is not a NamedObj.
459     */
460    protected NamedObj _getObjectUnder(Point2D point) {
461        Figure figureUnderMouse = _getFigureUnder(point);
462
463        if (figureUnderMouse == null) {
464            return null;
465        }
466
467        Object objectUnderMouse = ((UserObjectContainer) figureUnderMouse)
468                .getUserObject();
469
470        // Object might be a Location, in which case we want its container.
471        if (objectUnderMouse instanceof Location) {
472            return ((NamedObj) objectUnderMouse).getContainer();
473        } else if (objectUnderMouse instanceof NamedObj) {
474            return (NamedObj) objectUnderMouse;
475        }
476
477        return null;
478    }
