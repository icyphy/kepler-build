064/**
065 This class provides interaction with nodes that represent hierarchical
066 states in an FSM graph. Hierarchical states are those with refinements,
067 and what this adds to the base class is the ability to add a new
068 refinement or remove a refinement via a context menu command.
069 The base class provides a double click binding to edit the parameters
070 of the state, and context menu commands to edit parameters ("Configure"),
071 rename, look inside, and get documentation.
072
073 @author Edward A. Lee
074 @version $Id$
075 @since Ptolemy II 8.0
076 @Pt.ProposedRating Red (eal)
077 @Pt.AcceptedRating Red (johnr)
078 */
079public class HierarchicalStateController extends StateController {
080
081    /** Create a state controller associated with the specified graph
082     *  controller.
083     *  @param controller The associated graph controller.
084     */
085    public HierarchicalStateController(GraphController controller) {
086        this(controller, FULL);
087    }
088
089    /** Create a state controller associated with the specified graph
090     *  controller.
091     *  @param controller The associated graph controller.
092     *  @param access The access level.
093     */
094    public HierarchicalStateController(GraphController controller,
095            Access access) {
096        super(controller, access);
097
098        _menuFactory.addMenuItemFactory(
099                new MenuActionFactory(_addRefinementAction));
100        _menuFactory.addMenuItemFactory(
101                new MenuActionFactory(_removeRefinementAction));
102        _menuFactory
103                .addMenuItemFactory(new MenuActionFactory(_lookInsideAction));
104    }

109    /** An action to add a new refinement.
110     */
111    @SuppressWarnings("serial")
112    public class AddRefinementAction extends FigureAction {
113
114        public AddRefinementAction() {
115            super("Add Refinement");
116        }
117
118        @Override
119        public void actionPerformed(ActionEvent e) {
120            super.actionPerformed(e);
121
122            NamedObj target = getTarget();
123
124            if (!(target instanceof State)) {
125                MessageHandler.error("Can only add refinements to states.");
126                return;
127            }
128
129            State state = (State) target;
130
131            // Check that all these containers exist.
132            Nameable immediateContainer = state.getContainer();
133
134            if (immediateContainer == null) {
135                MessageHandler.error("State has no container!");
136                return;
137            }
138
139            CompositeEntity container = (CompositeEntity) immediateContainer
140                    .getContainer();
141
142            if (container == null) {
143                MessageHandler.error("State container has no container!");
144                return;
145            }
146
147            // Open a dialog to get the refinement name and class.
148            Query query = new Query();
149            String defaultName = container.uniqueName(state.getName());
150            query.addLine("Name", "Name", defaultName);
151
152            Map refinementClasses;
153            // See whether the configuration offers state refinements.
154            Configuration configuration = ((FSMGraphController) getController())
155                    .getConfiguration();
156            Entity refinements = configuration.getEntity("_stateRefinements");
157            // Check the configuration to see whether the default is overridden.
158            if (refinements instanceof CompositeEntity) {
159                // There is a specification.
160                refinementClasses = new HashMap();
161                List refinementList = ((CompositeEntity) refinements)
162                        .entityList();
163                Iterator iterator = refinementList.iterator();
164                while (iterator.hasNext()) {
165                    Entity entity = (Entity) iterator.next();
166                    refinementClasses.put(entity.getName(),
167                            entity.getClass().getName());
168                }
169            } else {
170                refinementClasses = getRefinementClasses();
171            }
172
173            // If refinement extenders are defined, add the classes specified in
174            // them to the list of choices.
175            List<RefinementExtender> extenders = state
176                    .attributeList(RefinementExtender.class);
177            String firstExtenderDescription = null;
178            for (RefinementExtender extender : extenders) {
179                try {
180                    String description = extender.description.stringValue();
181                    if (firstExtenderDescription == null) {
182                        firstExtenderDescription = description;
183                    }
184                    refinementClasses.put(description,
185                            extender.className.stringValue());
186                } catch (IllegalActionException e1) {
187                    // Ignore.
188                }
189            }
190
191            String[] choiceNames = (String[]) refinementClasses.keySet()
192                    .toArray(new String[refinementClasses.size()]);
193            query.addChoice("Class", "Class", choiceNames, choiceNames[0],
194                    true);
195            if (firstExtenderDescription != null) {
196                query.set("Class", firstExtenderDescription);
197            }
198
199            // Need a frame owner for first arg. for the dialog constructor.
200            Frame owner = null;
201            GraphController controller = getController();
202            if (controller instanceof BasicGraphController) {
203                owner = ((BasicGraphController) controller).getFrame();
204            }
205            ComponentDialog dialog = new ComponentDialog(owner,
206                    "Specify Refinement", query);
207
208            if (!dialog.buttonPressed().equals("OK")) {
209                return;
210            }
211
212            String newName = query.getStringValue("Name");
213
214            if (container.getEntity(newName) != null) {
215                MessageHandler.error("There is already a refinement with name "
216                        + newName + ".");
217                return;
218            }
219
220            String choiceName = query.getStringValue("Class");
221            String newClass = (String) refinementClasses.get(choiceName);
222
223            /*
224            String currentRefinements = state.refinementName.getExpression();
225
226            if ((currentRefinements == null) || currentRefinements.equals("")) {
227                currentRefinements = newName;
228            } else {
229                currentRefinements = currentRefinements.trim() + ", " + newName;
230            }
231             */
232
233            // The MoML we create depends on whether the configuration
234            // specified a set of prototype refinements.
235            Entity template = null;
236            if (refinements instanceof CompositeEntity) {
237                template = ((CompositeEntity) refinements)
238                        .getEntity(choiceName);
239            }
240
241            try {
242                ((RefinementActor) state.getContainer()).addRefinement(state,
243                        newName, template, newClass, configuration);
244            } catch (IllegalActionException ex) {
245                MessageHandler.error(ex.getMessage());
246            }
247        }
248    }

250    /** Action to remove refinements. */
251    @SuppressWarnings("serial")
252    public static class RemoveRefinementAction extends FigureAction {
253        // FindBugs suggests making this class static so as to decrease
254        // the size of instances and avoid dangling references.
255
256        public RemoveRefinementAction() {
257            super("Remove Refinement");
258        }
259
260        @Override
261        public void actionPerformed(ActionEvent e) {
262            super.actionPerformed(e);
263
264            NamedObj target = getTarget();
265
266            if (!(target instanceof State)) {
267                MessageHandler
268                        .error("Can only remove refinements from states.");
269                return;
270            }
271
272            State state = (State) target;
273
274            // Check that all these containers exist.
275            CompositeEntity immediateContainer = (CompositeEntity) state
276                    .getContainer();
277
278            if (immediateContainer == null) {
279                MessageHandler.error("State has no container!");
280                return;
281            }
282
283            final CompositeEntity container = (CompositeEntity) immediateContainer
284                    .getContainer();
285
286            if (container == null) {
287                MessageHandler.error("State container has no container!");
288                return;
289            }
290
291            TypedActor[] refinements;
292
293            try {
294                refinements = state.getRefinement();
295            } catch (Exception ex) {
296                MessageHandler.error("Invalid refinements.", ex);
297                return;
298            }
299
300            if (refinements == null || refinements.length < 1) {
301                MessageHandler.error("No refinements to remove.");
302                return;
303            }
304
305            String[] choices = new String[refinements.length];
306
307            for (int i = 0; i < refinements.length; i++) {
308                choices[i] = ((Nameable) refinements[i]).getName();
309            }
310
311            // Open a dialog to get the refinement name and class.
312            Query query = new Query();
313            query.addChoice("Refinement", "Refinement", choices, choices[0],
314                    false);
315
316            // FIXME: Need a frame owner for first arg.
317            // Perhaps calling getController(), which returns a GraphController
318            // will be a good start.
319            ComponentDialog dialog = new ComponentDialog(null,
320                    "Specify Refinement", query);
321
322            if (!dialog.buttonPressed().equals("OK")) {
323                return;
324            }
325
326            String refinementName = query.getStringValue("Refinement");
327            StringBuffer newRefinements = new StringBuffer();
328            String currentRefinements = state.refinementName.getExpression();
329            StringTokenizer tokenizer = new StringTokenizer(currentRefinements,
330                    ",");
331
332            while (tokenizer.hasMoreTokens()) {
333                String token = tokenizer.nextToken();
334
335                if (!token.trim().equals(refinementName)) {
336                    if (newRefinements.length() > 0) {
337                        newRefinements.append(", ");
338                    }
339
340                    newRefinements.append(token.trim());
341                }
342            }
343
344            // Check to see whether any other state or transition has
345            // this refinment, and if not, remove it from its container.
346            Iterator states = immediateContainer.entityList().iterator();
347            boolean foundOne = false;
348
349            while (states.hasNext()) {
350                NamedObj other = (NamedObj) states.next();
351
352                if (other != state && other instanceof State) {
353                    String refinementList = ((State) other).refinementName
354                            .getExpression();
355
356                    if (refinementList == null) {
357                        continue;
358                    }
359
360                    tokenizer = new StringTokenizer(refinementList, ",");
361
362                    while (tokenizer.hasMoreTokens()) {
363                        String token = tokenizer.nextToken();
364
365                        if (token.equals(refinementName)) {
366                            foundOne = true;
367                            break;
368                        }
369                    }
370
371                    if (foundOne) {
372                        break;
373                    }
374                }
375            }
376
377            if (!foundOne) {
378                Iterator transitions = immediateContainer.relationList()
379                        .iterator();
380
381                while (transitions.hasNext()) {
382                    NamedObj other = (NamedObj) transitions.next();
383
384                    if (other instanceof Transition) {
385                        String refinementList = ((Transition) other).refinementName
386                                .getExpression();
387
388                        if (refinementList == null) {
389                            continue;
390                        }
391
392                        tokenizer = new StringTokenizer(refinementList, ",");
393
394                        while (tokenizer.hasMoreTokens()) {
395                            String token = tokenizer.nextToken();
396
397                            if (token.equals(refinementName)) {
398                                foundOne = true;
399                                break;
400                            }
401                        }
402
403                        if (foundOne) {
404                            break;
405                        }
406                    }
407                }
408            }
409
410            String removal = "";
411
412            if (!foundOne) {
413                removal = "<deleteEntity name=\"" + refinementName + "\"/>";
414            }
415
416            String moml = "<group><entity name=\"" + state.getName(container)
417                    + "\"><property name=\"refinementName\" value=\""
418                    + newRefinements.toString() + "\"/></entity>" + removal
419                    + "</group>";
420            MoMLChangeRequest change = new MoMLChangeRequest(this, container,
421                    moml);
422            container.requestChange(change);
423        }
424    }
425
426    /** Return a map with the keys as the names of the refinement types, and the
427     *  values as the names of the classes that implement those refinement
428     *  types.
429     *  @return The map of supported refinement types.
430     */
431    protected Map getRefinementClasses() {
432        Map map = new TreeMap();
433        map.put("Default Refinement", "ptolemy.domains.modal.modal.Refinement");
434        map.put("State Machine Refinement",
435                "ptolemy.domains.modal.modal.ModalRefinement");
436        return map;
437    }
