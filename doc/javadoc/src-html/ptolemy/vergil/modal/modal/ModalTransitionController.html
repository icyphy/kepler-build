065/**
066 This class provides interaction with relations that contain general purpose
067 refinements that are fired on state transitions in an FSM graph. It's
068 implementation is largely copied from ModalController.
069
070 @author David Hermann, Research In Motion Limited
071 @version $Id$
072 @since Ptolemy II 8.0
073 @Pt.ProposedRating Red (eal)
074 @Pt.AcceptedRating Red (johnr)
075 */
076public class ModalTransitionController extends TransitionController {
077    /** Create a transition controller associated with the specified graph
078     *  controller.
079     *  @param controller The associated graph controller.
080     */
081    public ModalTransitionController(final GraphController controller) {
082        super(controller);
083
084        _menuFactory.addMenuItemFactory(
085                new MenuActionFactory(new AddRefinementAction()));
086        _menuFactory.addMenuItemFactory(
087                new MenuActionFactory(new RemoveRefinementAction()));
088    }

093    /** An action to add a new refinement.
094     */
095    @SuppressWarnings("serial")
096    private class AddRefinementAction extends FigureAction {
097        public AddRefinementAction() {
098            super("Add Refinement");
099        }
100
101        @Override
102        public void actionPerformed(ActionEvent e) {
103            super.actionPerformed(e);
104
105            NamedObj target = getTarget();
106
107            if (!(target instanceof Transition)) {
108                MessageHandler
109                        .error("Can only add refinements to transitions.");
110                return;
111            }
112
113            Transition transition = (Transition) target;
114
115            // Check that all these containers exist.
116            Nameable immediateContainer = transition.getContainer();
117
118            if (immediateContainer == null) {
119                MessageHandler.error("Transition has no container!");
120                return;
121            }
122
123            final CompositeEntity container = (CompositeEntity) immediateContainer
124                    .getContainer();
125
126            if (container == null) {
127                MessageHandler.error("Transition container has no container!");
128                return;
129            }
130
131            // Open a dialog to get the refinement name and class.
132            Query query = new Query();
133            String defaultName = container.uniqueName(transition.getName());
134            query.addLine("Name", "Name", defaultName);
135
136            // See whether the configuration offers transition refinements.
137            Configuration configuration = ((FSMGraphController) getController())
138                    .getConfiguration();
139            Entity refinements = configuration
140                    .getEntity("_transitionRefinements");
141
142            // Default choices.
143            String[] choiceClasses = {
144                    "ptolemy.domains.modal.modal.TransitionRefinement" };
145            String[] choiceNames = { "Default Refinement" };
146
147            // Check the configuration to see whether the default is overridden.
148            if (refinements instanceof CompositeEntity) {
149                // There is a specification.
150                List refinementList = ((CompositeEntity) refinements)
151                        .entityList();
152                choiceNames = new String[refinementList.size()];
153                choiceClasses = new String[refinementList.size()];
154
155                Iterator iterator = refinementList.iterator();
156                int count = 0;
157
158                while (iterator.hasNext()) {
159                    Entity entity = (Entity) iterator.next();
160                    choiceNames[count] = entity.getName();
161                    choiceClasses[count++] = entity.getClass().getName();
162                }
163            }
164
165            query.addChoice("Class", "Class", choiceNames, choiceNames[0],
166                    true);
167
168            // Need a frame owner for first arg. for the dialog constructor.
169            Frame owner = null;
170            GraphController controller = getController();
171            if (controller instanceof BasicGraphController) {
172                owner = ((BasicGraphController) controller).getFrame();
173            }
174            ComponentDialog dialog = new ComponentDialog(owner,
175                    "Specify Refinement", query);
176
177            if (!dialog.buttonPressed().equals("OK")) {
178                return;
179            }
180
181            final String newName = query.getStringValue("Name");
182
183            if (container.getEntity(newName) != null) {
184                MessageHandler.error("There is already a refinement with name "
185                        + newName + ".");
186                return;
187            }
188
189            int choiceIndex = query.getIntValue("Class");
190            String newClass = choiceClasses[choiceIndex];
191
192            String currentRefinements = transition.refinementName
193                    .getExpression();
194
195            if (currentRefinements == null || currentRefinements.equals("")) {
196                currentRefinements = newName;
197            } else {
198                currentRefinements = currentRefinements.trim() + ", " + newName;
199            }
200
201            String moml;
202
203            // The MoML we create depends on whether the configuration
204            // specified a set of prototype refinements.
205            if (refinements instanceof CompositeEntity) {
206                String choiceName = choiceNames[choiceIndex];
207                Entity template = ((CompositeEntity) refinements)
208                        .getEntity(choiceName);
209                String templateDescription = template.exportMoML(newName);
210                moml = "<group>" + templateDescription + "<relation name=\""
211                        + transition.getName(container)
212                        + "\"><property name=\"refinementName\" value=\""
213                        + currentRefinements + "\"/></relation></group>";
214            } else {
215                moml = "<group><entity name=\"" + newName + "\" class=\""
216                        + newClass + "\"/>" + "<relation name=\""
217                        + transition.getName(container)
218                        + "\"><property name=\"refinementName\" value=\""
219                        + currentRefinements + "\"/></relation></group>";
220            }
221
222            MoMLChangeRequest change = new MoMLChangeRequest(this, container,
223                    moml) {
224                @Override
225                protected void _execute() throws Exception {
226                    super._execute();
227
228                    // Mirror the ports of the container in the refinement.
229                    // Note that this is done here rather than as part of
230                    // the MoML because we have set protected variables
231                    // in the refinement to prevent it from trying to again
232                    // mirror the changes in the container.
233                    // The following gets the newly created entity.
234                    Entity entity = container.getEntity(newName);
235
236                    // Get the initial port configuration from the container.
237                    Iterator ports = container.portList().iterator();
238
239                    while (ports.hasNext()) {
240                        Port port = (Port) ports.next();
241
242                        try {
243                            if (entity instanceof RefinementActor) {
244                                // Prevent the newly created entity from creating
245                                // ports in its container.
246                                ((RefinementActor) entity).setMirrorDisable(1);
247                            }
248
249                            Port newPort = entity.newPort(port.getName());
250
251                            if (newPort instanceof RefinementPort
252                                    && port instanceof IOPort) {
253                                try {
254                                    ((RefinementPort) newPort)
255                                            .setMirrorDisable(true);
256
257                                    if (((IOPort) port).isInput()) {
258                                        ((RefinementPort) newPort)
259                                                .setInput(true);
260                                    }
261
262                                    if (((IOPort) port).isOutput()) {
263                                        ((RefinementPort) newPort)
264                                                .setOutput(true);
265                                    }
266
267                                    if (((IOPort) port).isMultiport()) {
268                                        ((RefinementPort) newPort)
269                                                .setMultiport(true);
270                                    }
271
272                                    /* No longer needed since Yuhong modified
273                                     * the type system to allow UNKNOWN. EAL
274                                     if (port instanceof TypedIOPort
275                                     && newPort instanceof TypedIOPort) {
276                                     ((TypedIOPort)newPort).setTypeSameAs(
277                                     (TypedIOPort)port);
278                                     }
279                                     */
280                                } finally {
281                                    ((RefinementPort) newPort)
282                                            .setMirrorDisable(false);
283                                }
284                            }
285                        } finally {
286                            if (entity instanceof RefinementActor) {
287                                ((RefinementActor) entity).setMirrorDisable(0);
288                            }
289                        }
290                    }
291
292                    if (_configuration != null) {
293                        // Look inside.
294                        _configuration.openModel(entity);
295                    }
296                }
297            };
298
299            container.requestChange(change);
300        }
301    }

303    /** Action to remove refinements. */
304    @SuppressWarnings("serial")
305    private static class RemoveRefinementAction extends FigureAction {
306        // FindBugs suggests making this class static so as to decrease
307        // the size of instances and avoid dangling references.
308
309        public RemoveRefinementAction() {
310            super("Remove Refinement");
311        }
312
313        @Override
314        public void actionPerformed(ActionEvent e) {
315            super.actionPerformed(e);
316
317            NamedObj target = getTarget();
318
319            if (!(target instanceof Transition)) {
320                MessageHandler
321                        .error("Can only remove refinements from transitions.");
322                return;
323            }
324
325            Transition transition = (Transition) target;
326
327            // Check that all these containers exist.
328            CompositeEntity immediateContainer = (CompositeEntity) transition
329                    .getContainer();
330
331            if (immediateContainer == null) {
332                MessageHandler.error("Transition has no container!");
333                return;
334            }
335
336            final CompositeEntity container = (CompositeEntity) immediateContainer
337                    .getContainer();
338
339            if (container == null) {
340                MessageHandler.error("Transition container has no container!");
341                return;
342            }
343
344            TypedActor[] refinements;
345
346            try {
347                refinements = transition.getRefinement();
348            } catch (Exception ex) {
349                MessageHandler.error("Invalid refinements.", ex);
350                return;
351            }
352
353            if (refinements == null || refinements.length < 1) {
354                MessageHandler.error("No refinements to remove.");
355                return;
356            }
357
358            String[] choices = new String[refinements.length];
359
360            for (int i = 0; i < refinements.length; i++) {
361                choices[i] = ((Nameable) refinements[i]).getName();
362            }
363
364            // Open a dialog to get the refinement name and class.
365            Query query = new Query();
366            query.addChoice("Refinement", "Refinement", choices, choices[0],
367                    false);
368
369            // FIXME: Need a frame owner for first arg.
370            // Perhaps calling getController(), which returns a GraphController
371            // will be a good start.
372            ComponentDialog dialog = new ComponentDialog(null,
373                    "Specify Refinement", query);
374
375            if (!dialog.buttonPressed().equals("OK")) {
376                return;
377            }
378
379            String refinementName = query.getStringValue("Refinement");
380            StringBuffer newRefinements = new StringBuffer();
381            String currentRefinements = transition.refinementName
382                    .getExpression();
383            StringTokenizer tokenizer = new StringTokenizer(currentRefinements,
384                    ",");
385
386            while (tokenizer.hasMoreTokens()) {
387                String token = tokenizer.nextToken();
388
389                if (!token.trim().equals(refinementName)) {
390                    if (newRefinements.length() > 0) {
391                        newRefinements.append(", ");
392                    }
393
394                    newRefinements.append(token.trim());
395                }
396            }
397
398            // Check to see whether any other transition has
399            // this refinment, and if not, remove it from its container.
400            Iterator transitions = immediateContainer.relationList().iterator();
401            boolean foundOne = false;
402
403            while (transitions.hasNext()) {
404                NamedObj other = (NamedObj) transitions.next();
405
406                if (other != transition && other instanceof Transition) {
407                    String refinementList = ((Transition) other).refinementName
408                            .getExpression();
409
410                    if (refinementList == null) {
411                        continue;
412                    }
413
414                    tokenizer = new StringTokenizer(refinementList, ",");
415
416                    while (tokenizer.hasMoreTokens()) {
417                        String token = tokenizer.nextToken();
418
419                        if (token.equals(refinementName)) {
420                            foundOne = true;
421                            break;
422                        }
423                    }
424
425                    if (foundOne) {
426                        break;
427                    }
428                }
429            }
430
431            if (!foundOne) {
432                Iterator states = immediateContainer.entityList().iterator();
433
434                while (states.hasNext()) {
435                    NamedObj other = (NamedObj) states.next();
436
437                    if (other instanceof State) {
438                        String refinementList = ((State) other).refinementName
439                                .getExpression();
440
441                        if (refinementList == null) {
442                            continue;
443                        }
444
445                        tokenizer = new StringTokenizer(refinementList, ",");
446
447                        while (tokenizer.hasMoreTokens()) {
448                            String token = tokenizer.nextToken();
449
450                            if (token.equals(refinementName)) {
451                                foundOne = true;
452                                break;
453                            }
454                        }
455
456                        if (foundOne) {
457                            break;
458                        }
459                    }
460                }
461            }
462
463            String removal = "";
464
465            if (!foundOne) {
466                removal = "<deleteEntity name=\"" + refinementName + "\"/>";
467            }
468
469            String moml = "<group><relation name=\""
470                    + transition.getName(container)
471                    + "\"><property name=\"refinementName\" value=\""
472                    + newRefinements.toString() + "\"/></relation>" + removal
473                    + "</group>";
474            MoMLChangeRequest change = new MoMLChangeRequest(this, container,
475                    moml);
476            container.requestChange(change);
477        }
478    }
