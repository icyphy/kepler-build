066/**
067 On each firing send an expression for evaluation to a matlab {@link
068 Engine}. The expression is any valid matlab expression, e.g.:
069
070 <pre>
071 [out1, out2, ... ] = SomeMatlabFunctionOrExpression( in1, in2, ... );...
072 </pre>
073
074 <p>The expression may include references to the input port names, current
075 time (<i>time</i>), and a count of the firing (<i>iteration</i>). This
076 is similar to <a href="../../ptolemy/actor/lib/Expression.html">Expression</a>.
077 To refer to parameters in scope, use $name or ${name} within
078 the expression.</p>
079
080 <p>The matlab engine is opened (started) during prefire() by the first
081 matlab Expression actor. Subsequent open()s simply increment a use
082 count.</p>
083
084 <p>At the start of fire(), <i>clear variables;clear globals</i> commands are
085 sent to matlab to clear its workspace. This helps detect errors where the
086 matlab expression refers to a matlab variable not initialized from the
087 input ports of this actor instance.</p>
088
089 <p>After the evaluation of the matlab expression is complete, the fire()
090 method iterates through names of output ports and converts matlab
091 variables with corresponding names to Tokens that are sent to the
092 corresponding output ports. Incorrect expressions are usually first
093 detected at this point by not finding the expected variables. If an
094 output port variable is not found in the matlab {@link Engine}, an
095 exception is thrown. The exception description string contains the last
096 stdout of the matlab engine that usually describes the error.</p>
097
098 <p>The {@link #get1x1asScalars} and {@link #getIntegerMatrices} control
099 data conversion (see {@link Engine} and
100 {@link Engine.ConversionParameters}).</p>
101
102 <p>A Parameter named <i>packageDirectories</i> may be added to this actor
103 to augment the search path of the matlab engine during the firing of this
104 actor. The value of this parameter should evaluate to a StringToken,
105 e.g.:
106
107 <pre>
108 "path1, path2, ..."
109 </pre>
110
111 <p> containing a comma-separated list of paths to be prepended to the matlab
112 engine search path before <i>expression</i> is evaluated. The list may
113 contain paths relative to the directory in which ptolemy was started,
114 or any directory listed in the current classpath (in that order, first
115 match wins). See {@link ptolemy.data.expr.UtilityFunctions#findFile(String)}.
116 After evaluation, the previous search path is restored.</p>
117
118 <p>A Parameter named <i>_debugging</i> may be used to turn on debug print
119 statements to stdout from {@link Engine} and the ptmatlab JNI. An IntToken
120 with a value of 1 turns on Engine debug statements, a value of 2 adds
121 ptmatlab debug statements as well.  A value of 0 or the absence of the
122 <i>_debugging</i> parameter yields normal operation.</p>
123
124 <p>For this actor to work, Matlab must be installed on your local
125 machine.  In addition, your environment must be set properly. The
126 <code>$PTII/bin/vergil</code> script does this for you, below are
127 instructions for users, such as Eclipse users, who are not using
128 <code>$PTII/bin/vergil</code>.</p>
129
130 <p>Add the Matlab shared libraries to your the library path. In the examples
131 below <code>$MATLAB</code>
132 should be the location of your Matlab installation. For example, if
133 /usr/bin/matlab is a link:
134 <pre>
135 bash-3.2$ which matlab
136 /usr/bin/matlab
137 bash-3.2$ ls -l /usr/bin/matlab
138 lrwxr-xr-x  1 root  wheel  42 Jan 15 20:57 /usr/bin/matlab -&gt; /Applications/MATLAB_R2009b.app/bin/matlab
139 bash-3.2$
140 </pre>
141
142 Then $MATLAB would be /Applications/MATLAB_R2009b.app
143
144 <dl>
145 <dt>32 Bit Mac (10.5?, Edit ~/.MacOSX/environment.plist)</dt>
146 <dd><code>export DYLD_LIBRARY_PATH=$MATLAB/bin/maci</code></dd>
147 <dt>64 Bit Mac (10.6?, Edit ~/.MacOSX/environment.plist)</dt>
148 <dd><code>export DYLD_LIBRARY_PATH=$MATLAB/bin/maci64</code></dd>
149 <dt>32 Bit Linux (Edit</dt>
150 <dd><code>export LD_LIBRARY_PATH=$MATLAB/bin/gnlx86</code></dd>
151 <dt>64 Bit Linux</dt>
152 <dd><code>export LD_LIBRARY_PATH=$MATLAB/bin/glnxa64</code></dd>
153 <dt>Windows (Start|My Computer|Properties|Advanced
154       Environment Variables| Add the directory that contains
155       matlab.exe to your path)</dt>
156 <dd><code>Be sure that the matlab binary is in your path</code></dd>

257     *  @param container The container.

264    public Expression(CompositeEntity container, String name)
265            throws NameDuplicationException, IllegalActionException {
266        super(container, name);
267
268        output = new TypedIOPort(this, "output", false, true);
269        expression = new StringParameter(this, "expression");
270        new TextStyle(expression, "Matlab expression");
271
272        _dataParameters = new Engine.ConversionParameters();
273
274        get1x1asScalars = new Parameter(this, "get1x1asScalars",
275                new BooleanToken(_dataParameters.getScalarMatrices));
276        new CheckBoxStyle(get1x1asScalars, "style");
277
278        getIntegerMatrices = new Parameter(this, "getIntegerMatrices",
279                new BooleanToken(_dataParameters.getIntMatrices));
280        new CheckBoxStyle(getIntegerMatrices, "style");
281
282        clearEnvironment = new Parameter(this, "clearEnvironment",
283                new BooleanToken(true));
284        clearEnvironment.setTypeEquals(BaseType.BOOLEAN);
285        clearEnvironment.setToken(BooleanToken.TRUE);
286
287        // _time is not needed, fire() sets a matlab variable directly
288        _iteration = new Variable(this, "_iteration", new IntToken(1));
289    }

349    /** Initialize the iteration count to 1.
350     *  @exception IllegalActionException If the parent class throws it.
351     */
352    @Override
353    public void initialize() throws IllegalActionException {
354        super.initialize();
355
356        // Check to make sure that the engine has been initialized.
357        synchronized (Engine.semaphore) {
358            if (engine == null) {
359                _initializeEngine();
360            }
361        }
362
363        _iterationCount = 1;
364        _iteration.setToken(new IntToken(_iterationCount));
365
366        // Process any additional directories to be added to matlab's
367        // path. The list may contain paths relative to the directory in
368        // which ptolemy was started or any directory listed in the current
369        // classpath (in this order, first match wins). See
370        // UtilityFunctions.findFile()
371        _addPathCommand = null; // Assume none
372        _previousPath = null;
373
374        Parameter packageDirectories = (Parameter) getAttribute(
375                "packageDirectories");
376
377        if (packageDirectories != null) {
378            StringTokenizer dirs = new StringTokenizer(
379                    ((StringToken) packageDirectories.getToken()).stringValue(),
380                    ",");
381            StringBuffer cellFormat = new StringBuffer(512);
382            cellFormat.append("{");
383
384            if (dirs.hasMoreTokens()) {
385                cellFormat.append("'"
386                        + UtilityFunctions.findFile(dirs.nextToken()) + "'");
387            }
388
389            while (dirs.hasMoreTokens()) {
390                cellFormat.append(",'"
391                        + UtilityFunctions.findFile(dirs.nextToken()) + "'");
392            }
393
394            cellFormat.append("}");
395
396            if (cellFormat.length() > 2) {
397                _addPathCommand = "addedPath_ = " + cellFormat.toString()
398                        + ";addpath(addedPath_{:});";
399
400                synchronized (Engine.semaphore) {
401                    matlabEngine.evalString(engine, "previousPath_=path");
402                    _previousPath = matlabEngine.get(engine, "previousPath_");
403                }
404            }
405        }
406
407        _dataParameters.getScalarMatrices = ((BooleanToken) get1x1asScalars
408                .getToken()).booleanValue();
409        _dataParameters.getIntMatrices = ((BooleanToken) getIntegerMatrices
410                .getToken()).booleanValue();
411    }

413    /** Return true if all input ports have at least one token.
414     *  @return True if this actor is ready for firing, false otherwise.
415     *  @exception IllegalActionException Not thrown in this base class.
416     */
417    @Override
418    public boolean prefire() throws IllegalActionException {
419        Iterator inputPorts = inputPortList().iterator();
420
421        while (inputPorts.hasNext()) {
422            IOPort port = (IOPort) inputPorts.next();
423
424            if (!port.hasToken(0)) {
425                return false;
426            }
427        }
428
429        return super.prefire();
430    }

432    /** Evaluate the expression and send its result to the output.
433     *  @exception IllegalActionException If the evaluation of the expression
434     *   triggers it, or the evaluation yields a null result, or the evaluation
435     *   yields an incompatible type, or if there is no director.
436     */
437    @Override
438    public void fire() throws IllegalActionException {
439        super.fire();
440        Director director = getDirector();
441
442        if (director == null) {
443            throw new IllegalActionException(this, "No director!");
444        }
445
446        boolean clearEnvironmentValue = ((BooleanToken) clearEnvironment
447                .getToken()).booleanValue();
448
449        try {
450
451            // Read the input ports before acquiring the engine lock since
452            // get() may block depending on the director, e.g., PN.
453            for (TypedIOPort port : inputPortList()) {
454                _inputTokens.put(port.getName(), port.get(0));
455            }
456
457            synchronized (Engine.semaphore) {
458
459                if (clearEnvironmentValue) {
460                    // The following clears variables, but preserves any
461                    // persistent storage created by a function (this usually
462                    // for speed-up purposes to avoid recalculation on every
463                    // function call)
464                    matlabEngine.evalString(engine,
465                            "clear variables;clear globals");
466                }
467
468                if (_addPathCommand != null) {
469                    matlabEngine.evalString(engine, _addPathCommand);
470                }
471
472                try {
473                    matlabEngine.put(engine, "time", new DoubleToken(
474                            director.getModelTime().getDoubleValue()));
475                } catch (IllegalActionException ex) {
476                    throw new IllegalActionException(this, ex,
477                            "Failed to set the \"time\" variable in the Matlab "
478                                    + "engine to " + new DoubleToken(director
479                                            .getModelTime().getDoubleValue()));
480                }
481                try {
482                    matlabEngine.put(engine, "iteration",
483                            _iteration.getToken());
484                } catch (IllegalActionException ex) {
485                    throw new IllegalActionException(this, ex,
486                            "Failed to set the \"iteration\" variable in the Matlab "
487                                    + "engine to " + _iteration.getToken());
488                }
489
490                for (Map.Entry<String, Token> entry : _inputTokens.entrySet()) {
491                    matlabEngine.put(engine, entry.getKey(), entry.getValue());
492                }
493
494                matlabEngine.evalString(engine, expression.stringValue());
495
496                Iterator outputPorts = outputPortList().iterator();
497
498                while (outputPorts.hasNext()) {
499                    IOPort port = (IOPort) outputPorts.next();
500
501                    // FIXME: Handle multiports
502                    if (port.isOutsideConnected()) {
503                        port.send(0, matlabEngine.get(engine, port.getName(),
504                                _dataParameters));
505                    }
506                }
507
508                // Restore previous path if path was modified above
509                if (_previousPath != null) {
510                    matlabEngine.put(engine, "previousPath_", _previousPath);
511                    matlabEngine.evalString(engine, "path(previousPath_);");
512                }
513            }
514        } finally {
515            // Remove references to any tokens that were read from input ports.
516            _inputTokens.clear();
517        }
518    }
519
520    /** Increment the iteration count.
521     *  @exception IllegalActionException If the superclass throws it.
522     */
523    @Override
524    public boolean postfire() throws IllegalActionException {
525        _iterationCount++;
526        _iteration.setToken(new IntToken(_iterationCount));
527
528        // This actor never requests termination.
529        return super.postfire();
530    }

532    /** Close matlab engine if it was open.
533     *  @exception IllegalActionException Not thrown in this base class.
534     */
535    @Override
536    public void wrapup() throws IllegalActionException {
537        super.wrapup();
538
539        synchronized (Engine.semaphore) {
540            if (matlabEngine != null) {
541                matlabEngine.close(engine);
542            }
543        }
544
545        engine = null;
546    }

564    /** Initialize the Matlab Engine.
565     *  @exception IllegalActionException If the connection to the
566     *  Matlab engine cannot be intialized.
567     */
568    private void _initializeEngine() throws IllegalActionException {
569        synchronized (Engine.semaphore) {
570            // The intialization used to occur in preinitialize(), but this
571            // meant that the Matlab Expression actor could not be used
572            // in a RunComposite because RunComposite invokes _executeInsideModel()
573            // which does not invoke preinitialize().  If _executeInsideModel()
574            // invokes preinitialize(), then the PortParameters of the RunComposite
575            // are constantly reset to the value of the _persistentExpression.
576            try {
577                matlabEngine = new Engine();
578            } catch (Throwable throwable) {
579                // LinkageError is and Error, not an exceptoin
580                throw new IllegalActionException(this, throwable,
581                        "There was a problem invoking the Ptolemy II Matlab "
582                                + "interface.\nThe interface has been tested under "
583                                + "Linux, Mac OS X, and Windows.\n"
584                                + "The interface requires that Matlab be installed "
585                                + "on the local machine and that the ptmatlab "
586                                + "shared library available.\n"
587                                + "* Under Linux, you must have the LD_LIBRARY_PATH "
588                                + "environment variable set to include the directories "
589                                + "that contain libmx.so and libptmatlab.so.\n"
590                                + "* Under Mac OS X, you must have the DYLD_LIBRARY_PATH "
591                                + "environment variable set to include the directories "
592                                + "that contain libmx.dylib and libptmatlab.jnilib.\n"
593                                + "* Under Windows, you must have your PATH set to include "
594                                + "the Matlab bin/win32 or equivalent directory so that "
595                                + "libmex.dll is found and the directory that contains "
596                                + "libptmatlab.dll. "
597                                + "In addition, if you are running under Windows from "
598                                + "the Cygwin bash prompt, then you must start Vergil with "
599                                + "the -jni argument: $PTII/bin/vergil -jni. For details, "
600                                + "see $PTII/jni/package.html.\n"
601                                + "Refer to $PTII/ptolemy/matlab/makefile for more "
602                                + "information.");
603            }
604
605            // First set default debugging level, then check for more
606            matlabEngine.setDebugging((byte) 0);
607
608            Parameter debugging = (Parameter) getAttribute("_debugging");
609
610            if (debugging != null) {
611                Token t = debugging.getToken();
612
613                if (t instanceof IntToken) {
614                    matlabEngine.setDebugging((byte) ((IntToken) t).intValue());
615                }
616            }
617
618            engine = matlabEngine.open();
619        }
620    }
