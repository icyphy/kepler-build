064//// Expression

066/**
067 On each firing send an expression for evaluation to a matlab {@link
068 Engine}. The expression is any valid matlab expression, e.g.:
069
070 <pre>
071 [out1, out2, ... ] = SomeMatlabFunctionOrExpression( in1, in2, ... );...
072 </pre>
073
074 <p>The expression may include references to the input port names, current
075 time (<i>time</i>), and a count of the firing (<i>iteration</i>). This
076 is similar to <a href="../../ptolemy/actor/lib/Expression.html">Expression</a>.
077 To refer to parameters in scope, use $name or ${name} within
078 the expression.</p>
079
080 <p>The matlab engine is opened (started) during prefire() by the first
081 matlab Expression actor. Subsequent open()s simply increment a use
082 count.</p>
083
084 <p>At the start of fire(), <i>clear variables;clear globals</i> commands are
085 sent to matlab to clear its workspace. This helps detect errors where the
086 matlab expression refers to a matlab variable not initialized from the
087 input ports of this actor instance.</p>
088
089 <p>After the evaluation of the matlab expression is complete, the fire()
090 method iterates through names of output ports and converts matlab
091 variables with corresponding names to Tokens that are sent to the
092 corresponding output ports. Incorrect expressions are usually first
093 detected at this point by not finding the expected variables. If an
094 output port variable is not found in the matlab {@link Engine}, an
095 exception is thrown. The exception description string contains the last
096 stdout of the matlab engine that usually describes the error.</p>
097
098 <p>The {@link #get1x1asScalars} and {@link #getIntegerMatrices} control
099 data conversion (see {@link Engine} and
100 {@link Engine.ConversionParameters}).</p>
101
102 <p>A Parameter named <i>packageDirectories</i> may be added to this actor
103 to augment the search path of the matlab engine during the firing of this
104 actor. The value of this parameter should evaluate to a StringToken,
105 e.g.:
106
107 <pre>
108 "path1, path2, ..."
109 </pre>
110
111 <p> containing a comma-separated list of paths to be prepended to the matlab
112 engine search path before <i>expression</i> is evaluated. The list may
113 contain paths relative to the directory in which ptolemy was started,
114 or any directory listed in the current classpath (in that order, first
115 match wins). See {@link ptolemy.data.expr.UtilityFunctions#findFile(String)}.
116 After evaluation, the previous search path is restored.</p>
117
118 <p>A Parameter named <i>_debugging</i> may be used to turn on debug print
119 statements to stdout from {@link Engine} and the ptmatlab JNI. An IntToken
120 with a value of 1 turns on Engine debug statements, a value of 2 adds
121 ptmatlab debug statements as well.  A value of 0 or the absence of the
122 <i>_debugging</i> parameter yields normal operation.</p>
123
124 <p>For this actor to work, Matlab must be installed on your local
125 machine.  In addition, your environment must be set properly. The
126 <code>$PTII/bin/vergil</code> script does this for you, below are
127 instructions for users, such as Eclipse users, who are not using
128 <code>$PTII/bin/vergil</code>.</p>
129
130 <p>Add the Matlab shared libraries to your the library path. In the examples
131 below <code>$MATLAB</code>
132 should be the location of your Matlab installation. For example, if
133 /usr/bin/matlab is a link:
134 <pre>
135 bash-3.2$ which matlab
136 /usr/bin/matlab
137 bash-3.2$ ls -l /usr/bin/matlab
138 lrwxr-xr-x  1 root  wheel  42 Jan 15 20:57 /usr/bin/matlab -&gt; /Applications/MATLAB_R2009b.app/bin/matlab
139 bash-3.2$
140 </pre>
141
142 Then $MATLAB would be /Applications/MATLAB_R2009b.app
143
144 <dl>
145 <dt>32 Bit Mac (10.5?, Edit ~/.MacOSX/environment.plist)</dt>
146 <dd><code>export DYLD_LIBRARY_PATH=$MATLAB/bin/maci</code></dd>
147 <dt>64 Bit Mac (10.6?, Edit ~/.MacOSX/environment.plist)</dt>
148 <dd><code>export DYLD_LIBRARY_PATH=$MATLAB/bin/maci64</code></dd>
149 <dt>32 Bit Linux (Edit</dt>
150 <dd><code>export LD_LIBRARY_PATH=$MATLAB/bin/gnlx86</code></dd>
151 <dt>64 Bit Linux</dt>
152 <dd><code>export LD_LIBRARY_PATH=$MATLAB/bin/glnxa64</code></dd>
153 <dt>Windows (Start|My Computer|Properties|Advanced
154       Environment Variables| Add the directory that contains
155       matlab.exe to your path)</dt>
156 <dd><code>Be sure that the matlab binary is in your path</code></dd>
157 </dl>

166    /** Construct an actor with the given container and name.
167     *  @param container The container.
168     *  @param name The name of this actor.
169     *  @exception IllegalActionException If the actor cannot be contained
170     *   by the proposed container.
171     *  @exception NameDuplicationException If the container already has an
172     *   actor with this name.
173     */
174    public Expression(CompositeEntity container, String name)
175            throws NameDuplicationException, IllegalActionException {
176        super(container, name);
177
178        output = new TypedIOPort(this, "output", false, true);
179        expression = new StringParameter(this, "expression");
180        new TextStyle(expression, "Matlab expression");
181
182        _dataParameters = new Engine.ConversionParameters();
183
184        get1x1asScalars = new Parameter(this, "get1x1asScalars",
185                new BooleanToken(_dataParameters.getScalarMatrices));
186        new CheckBoxStyle(get1x1asScalars, "style");
187
188        getIntegerMatrices = new Parameter(this, "getIntegerMatrices",
189                new BooleanToken(_dataParameters.getIntMatrices));
190        new CheckBoxStyle(getIntegerMatrices, "style");
191
192        clearEnvironment = new Parameter(this, "clearEnvironment",
193                new BooleanToken(true));
194        clearEnvironment.setTypeEquals(BaseType.BOOLEAN);
195        clearEnvironment.setToken(BooleanToken.TRUE);
196
197        // _time is not needed, fire() sets a matlab variable directly
198        _iteration = new Variable(this, "_iteration", new IntToken(1));
199    }

230    /** Clone the actor into the specified workspace. This calls the
231     *  base class and then sets the <code>iteration</code>
232     *  public member to the parameters of the new actor.
233     *  @param workspace The workspace for the new object.
234     *  @return A new actor.
235     *  @exception CloneNotSupportedException If a derived class contains
236     *   an attribute that cannot be cloned.
237     */
238    @Override
239    public Object clone(Workspace workspace) throws CloneNotSupportedException {
240        Expression newObject = (Expression) super.clone(workspace);
241
242        newObject._addPathCommand = null;
243        newObject._iteration = (Variable) newObject.getAttribute("_iteration");
244        newObject._iterationCount = 1;
245        newObject._previousPath = null;
246        newObject._inputTokens = new HashMap<String, Token>();
247        return newObject;
248    }

250    /** Open a matlab engine.
251     *  @exception IllegalActionException If matlab engine not found.
252     */
253    @Override
254    public void preinitialize() throws IllegalActionException {
255        super.preinitialize();
256        _initializeEngine();
257    }
258
259    /** Initialize the iteration count to 1.
260     *  @exception IllegalActionException If the parent class throws it.
261     */
262    @Override
263    public void initialize() throws IllegalActionException {
264        super.initialize();
265
266        // Check to make sure that the engine has been initialized.
267        synchronized (Engine.semaphore) {
268            if (engine == null) {
269                _initializeEngine();
270            }
271        }
272
273        _iterationCount = 1;
274        _iteration.setToken(new IntToken(_iterationCount));
275
276        // Process any additional directories to be added to matlab's
277        // path. The list may contain paths relative to the directory in
278        // which ptolemy was started or any directory listed in the current
279        // classpath (in this order, first match wins). See
280        // UtilityFunctions.findFile()
281        _addPathCommand = null; // Assume none
282        _previousPath = null;
283
284        Parameter packageDirectories = (Parameter) getAttribute(
285                "packageDirectories");
286
287        if (packageDirectories != null) {
288            StringTokenizer dirs = new StringTokenizer(
289                    ((StringToken) packageDirectories.getToken()).stringValue(),
290                    ",");
291            StringBuffer cellFormat = new StringBuffer(512);
292            cellFormat.append("{");
293
294            if (dirs.hasMoreTokens()) {
295                cellFormat.append("'"
296                        + UtilityFunctions.findFile(dirs.nextToken()) + "'");
297            }
298
299            while (dirs.hasMoreTokens()) {
300                cellFormat.append(",'"
301                        + UtilityFunctions.findFile(dirs.nextToken()) + "'");
302            }
303
304            cellFormat.append("}");
305
306            if (cellFormat.length() > 2) {
307                _addPathCommand = "addedPath_ = " + cellFormat.toString()
308                        + ";addpath(addedPath_{:});";
309
310                synchronized (Engine.semaphore) {
311                    matlabEngine.evalString(engine, "previousPath_=path");
312                    _previousPath = matlabEngine.get(engine, "previousPath_");
313                }
314            }
315        }
316
317        _dataParameters.getScalarMatrices = ((BooleanToken) get1x1asScalars
318                .getToken()).booleanValue();
319        _dataParameters.getIntMatrices = ((BooleanToken) getIntegerMatrices
320                .getToken()).booleanValue();
321    }
322
323    /** Return true if all input ports have at least one token.
324     *  @return True if this actor is ready for firing, false otherwise.
325     *  @exception IllegalActionException Not thrown in this base class.
326     */
327    @Override
328    public boolean prefire() throws IllegalActionException {
329        Iterator inputPorts = inputPortList().iterator();
330
331        while (inputPorts.hasNext()) {
332            IOPort port = (IOPort) inputPorts.next();
333
334            if (!port.hasToken(0)) {
335                return false;
336            }
337        }
338
339        return super.prefire();
340    }
341
342    /** Evaluate the expression and send its result to the output.
343     *  @exception IllegalActionException If the evaluation of the expression
344     *   triggers it, or the evaluation yields a null result, or the evaluation
345     *   yields an incompatible type, or if there is no director.
346     */
347    @Override
348    public void fire() throws IllegalActionException {
349        super.fire();
350        Director director = getDirector();
351
352        if (director == null) {
353            throw new IllegalActionException(this, "No director!");
354        }
355
356        boolean clearEnvironmentValue = ((BooleanToken) clearEnvironment
357                .getToken()).booleanValue();
358
359        try {
360
361            // Read the input ports before acquiring the engine lock since
362            // get() may block depending on the director, e.g., PN.
363            for (TypedIOPort port : inputPortList()) {
364                _inputTokens.put(port.getName(), port.get(0));
365            }
366
367            synchronized (Engine.semaphore) {
368
369                if (clearEnvironmentValue) {
370                    // The following clears variables, but preserves any
371                    // persistent storage created by a function (this usually
372                    // for speed-up purposes to avoid recalculation on every
373                    // function call)
374                    matlabEngine.evalString(engine,
375                            "clear variables;clear globals");
376                }
377
378                if (_addPathCommand != null) {
379                    matlabEngine.evalString(engine, _addPathCommand);
380                }
381
382                try {
383                    matlabEngine.put(engine, "time", new DoubleToken(
384                            director.getModelTime().getDoubleValue()));
385                } catch (IllegalActionException ex) {
386                    throw new IllegalActionException(this, ex,
387                            "Failed to set the \"time\" variable in the Matlab "
388                                    + "engine to " + new DoubleToken(director
389                                            .getModelTime().getDoubleValue()));
390                }
391                try {
392                    matlabEngine.put(engine, "iteration",
393                            _iteration.getToken());
394                } catch (IllegalActionException ex) {
395                    throw new IllegalActionException(this, ex,
396                            "Failed to set the \"iteration\" variable in the Matlab "
397                                    + "engine to " + _iteration.getToken());
398                }
399
400                for (Map.Entry<String, Token> entry : _inputTokens.entrySet()) {
401                    matlabEngine.put(engine, entry.getKey(), entry.getValue());
402                }
403
404                matlabEngine.evalString(engine, expression.stringValue());
405
406                Iterator outputPorts = outputPortList().iterator();
407
408                while (outputPorts.hasNext()) {
409                    IOPort port = (IOPort) outputPorts.next();
410
411                    // FIXME: Handle multiports
412                    if (port.isOutsideConnected()) {
413                        port.send(0, matlabEngine.get(engine, port.getName(),
414                                _dataParameters));
415                    }
416                }
417
418                // Restore previous path if path was modified above
419                if (_previousPath != null) {
420                    matlabEngine.put(engine, "previousPath_", _previousPath);
421                    matlabEngine.evalString(engine, "path(previousPath_);");
422                }
423            }
424        } finally {
425            // Remove references to any tokens that were read from input ports.
426            _inputTokens.clear();
427        }
428    }
429
430    /** Increment the iteration count.
431     *  @exception IllegalActionException If the superclass throws it.
432     */
433    @Override
434    public boolean postfire() throws IllegalActionException {
435        _iterationCount++;
436        _iteration.setToken(new IntToken(_iterationCount));
437
438        // This actor never requests termination.
439        return super.postfire();
440    }
441
442    /** Close matlab engine if it was open.
443     *  @exception IllegalActionException Not thrown in this base class.
444     */
445    @Override
446    public void wrapup() throws IllegalActionException {
447        super.wrapup();
448
449        synchronized (Engine.semaphore) {
450            if (matlabEngine != null) {
451                matlabEngine.close(engine);
452            }
453        }
454
455        engine = null;
456    }

474    /** Initialize the Matlab Engine.
475     *  @exception IllegalActionException If the connection to the
476     *  Matlab engine cannot be intialized.
477     */
478    private void _initializeEngine() throws IllegalActionException {
479        synchronized (Engine.semaphore) {
480            // The intialization used to occur in preinitialize(), but this
481            // meant that the Matlab Expression actor could not be used
482            // in a RunComposite because RunComposite invokes _executeInsideModel()
483            // which does not invoke preinitialize().  If _executeInsideModel()
484            // invokes preinitialize(), then the PortParameters of the RunComposite
485            // are constantly reset to the value of the _persistentExpression.
486            try {
487                matlabEngine = new Engine();
488            } catch (Throwable throwable) {
489                // LinkageError is and Error, not an exceptoin
490                throw new IllegalActionException(this, throwable,
491                        "There was a problem invoking the Ptolemy II Matlab "
492                                + "interface.\nThe interface has been tested under "
493                                + "Linux, Mac OS X, and Windows.\n"
494                                + "The interface requires that Matlab be installed "
495                                + "on the local machine and that the ptmatlab "
496                                + "shared library available.\n"
497                                + "* Under Linux, you must have the LD_LIBRARY_PATH "
498                                + "environment variable set to include the directories "
499                                + "that contain libmx.so and libptmatlab.so.\n"
500                                + "* Under Mac OS X, you must have the DYLD_LIBRARY_PATH "
501                                + "environment variable set to include the directories "
502                                + "that contain libmx.dylib and libptmatlab.jnilib.\n"
503                                + "* Under Windows, you must have your PATH set to include "
504                                + "the Matlab bin/win32 or equivalent directory so that "
505                                + "libmex.dll is found and the directory that contains "
506                                + "libptmatlab.dll. "
507                                + "In addition, if you are running under Windows from "
508                                + "the Cygwin bash prompt, then you must start Vergil with "
509                                + "the -jni argument: $PTII/bin/vergil -jni. For details, "
510                                + "see $PTII/jni/package.html.\n"
511                                + "Refer to $PTII/ptolemy/matlab/makefile for more "
512                                + "information.");
513            }
514
515            // First set default debugging level, then check for more
516            matlabEngine.setDebugging((byte) 0);
517
518            Parameter debugging = (Parameter) getAttribute("_debugging");
519
520            if (debugging != null) {
521                Token t = debugging.getToken();
522
523                if (t instanceof IntToken) {
524                    matlabEngine.setDebugging((byte) ((IntToken) t).intValue());
525                }
526            }
527
528            engine = matlabEngine.open();
529        }
530    }
