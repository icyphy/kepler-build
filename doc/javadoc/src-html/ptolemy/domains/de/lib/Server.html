047//// Server

049/**
050 This actor models a presemptive server with a fixed or variable service time.
051 A server is either busy (serving a customer) or not busy at any given time.
052 If an input arrives when the server is not busy, then the input token is
053 produced on the output with a delay given by the <i>serviceTime</i>
054 parameter.
055 If an input arrives while the server is busy, then that input is
056 queued until the server becomes free, at which point it is produced
057 on the output with a delay given by the <i>serviceTime</i> parameter
058 value at the time that the input arrived.
059 If several inputs arrive while the server is busy, then they are
060 served on a first-come, first-served basis.
061 On every firing, produce an output indicating the final queue size.
062 <p>
063 The service time and priority used for a job are the most recently arrived
064 values prior to or simultaneous with the <i>arrival</i> of the job
065 (not with the time at which service begins). Thus, if you want
066 each job to have an independent service time or priority, you should
067 provide each parameter as an input synchronized with each new job arrival.
068
069 @see ptolemy.actor.lib.TimeDelay
070
071 @author Lukito Muliadi, Edward A. Lee, Haiyang Zheng
072 @version $Id$
073 @since Ptolemy II 0.3
074 @Pt.ProposedRating Yellow (hyzheng)
075 @Pt.AcceptedRating Yellow (hyzheng)
076 */
077public class Server extends DETransformer {
078    /** Construct an actor with the specified container and name.
079     *  @param container The composite entity to contain this one.
080     *  @param name The name of this actor.
081     *  @exception IllegalActionException If the entity cannot be contained
082     *   by the proposed container.
083     *  @exception NameDuplicationException If the container already has an
084     *   actor with this name.
085     */
086    public Server(CompositeEntity container, String name)
087            throws NameDuplicationException, IllegalActionException {
088        super(container, name);
089
090        output.setTypeSameAs(input);
091
092        // FIXME: Put in a name change from newServiceTime to serviceTime in MoML filters.
093        serviceTime = new PortParameter(this, "serviceTime");
094        serviceTime.setExpression("1.0");
095        serviceTime.setTypeEquals(BaseType.DOUBLE);
096        // Put the delay port at the bottom of the icon by default.
097        StringAttribute cardinality = new StringAttribute(serviceTime.getPort(),
098                "_cardinal");
099        cardinality.setExpression("SOUTH");
100
101        priority = new PortParameter(this, "priority");
102        priority.setExpression("0.0");
103        priority.setTypeEquals(BaseType.DOUBLE);
104        // Put the priority port at the bottom of the icon by default.
105        cardinality = new StringAttribute(priority.getPort(),
106                "_cardinal");
107        cardinality.setExpression("SOUTH");
108
109        _queue = new PriorityQueue();
110
111        _queueCounter = 0;
112
113        size = new TypedIOPort(this, "size", false, true);
114        size.setTypeEquals(BaseType.INT);
115        // Put it at the bottom of the icon by default.
116        cardinality = new StringAttribute(size, "_cardinal");
117        cardinality.setExpression("SOUTH");
118
119        capacity = new Parameter(this, "capacity");
120        capacity.setTypeEquals(BaseType.INT);
121        capacity.setExpression("0");
122    }

127    /** The capacity of the queue. If the value is positive, then
128     *  it specifies the capacity of the queue. If it is negative
129     *  or 0, then it specifies that the capacity is infinite.
130     *  This is an integer with default 0.

132    public Parameter capacity;
133
134    /** The current size of the queue. This port produces an output
135     *  whenever the size changes. It has type int.
136     */
137    public TypedIOPort size;
138
139    /** The service time. This is a double with default 1.0.
140     *  It is required to be non-negative.
141     */
142    public PortParameter serviceTime;
143
144    /** The priority. This is a double with default 0.0.
145     *  A higher priority implies the task has precedence over tasks
146     *  with lower priority values.
147     */
148    public PortParameter priority;

150    ///////////////////////////////////////////////////////////////////

152
153    /** If the attribute is <i>serviceTime</i>, then ensure that the value
154     *  is non-negative, and if the attribute is
155     *  <i>capacity</i>, then change the capacity of the queue.
156     *  If the size of the queue currently exceeds the specified
157     *  capacity, then throw an exception.
158     *  @param attribute The attribute that changed.
159     *  @exception IllegalActionException If the service time is negative.
160     */
161    @Override
162    public void attributeChanged(Attribute attribute)
163            throws IllegalActionException {
164        if (attribute == serviceTime) {
165            double value = ((DoubleToken) serviceTime.getToken()).doubleValue();
166
167            if (value < 0.0 || Double.isNaN(value)) {
168                throw new IllegalActionException(this,
169                        "Cannot have negative or NaN serviceTime: " + value);
170            }
171        } else if (attribute == priority) {
172            double value = ((DoubleToken) priority.getToken()).doubleValue();
173
174            if (Double.isNaN(value)) {
175                throw new IllegalActionException(this,
176                        "Cannot have NaN priority: " + value);
177            }
178        } else if (attribute == capacity) {
179            int newCapacity = ((IntToken) capacity.getToken()).intValue();
180            if (newCapacity > 0 && _queue.size() > newCapacity) {
181                throw new IllegalActionException(this,
182                        "Queue size (" + _queue.size()
183                                + ") exceed requested capacity "
184                                + newCapacity + ").");
185            }
186        } else {
187            super.attributeChanged(attribute);
188        }
189    }
190
191    /** Clone the actor into the specified workspace. Set a type
192     *  constraint that the output type is the same as the that of input.
193     *  @param workspace The workspace for the new object.
194     *  @return A new actor.
195     *  @exception CloneNotSupportedException If a derived class has
196     *   has an attribute that cannot be cloned.
197     */
198    @Override
199    public Object clone(Workspace workspace) throws CloneNotSupportedException {
200        Server newObject = (Server) super.clone(workspace);
201        newObject.output.setTypeSameAs(newObject.input);
202        newObject._queue = new PriorityQueue<Job>();
203        return newObject;
204    }
205
206    /** Declare that the <i>output</i> does not depend on
207     *  the <i>input</i>, <i>serviceTime</i>, and <i>priority</i>, in a firing.
208     *  @exception IllegalActionException If the causality interface
209     *  cannot be computed.
210     *  @see #getCausalityInterface()
211     */
212    @Override
213    public void declareDelayDependency() throws IllegalActionException {
214        // Declare that output does not immediately depend on the input,
215        // though there is no lower bound on the time delay.
216        _declareDelayDependency(input, output, 0.0);
217        _declareDelayDependency(serviceTime.getPort(), output, 0.0);
218        _declareDelayDependency(priority.getPort(), output, 0.0);
219    }
220
221    /** If there is input, read it and put it in the queue.
222     *  If the service time has expired for a token currently
223     *  in the queue, then send that token on the output.
224     *  Produce an output indicating the current queue size.
225     *  @exception IllegalActionException If the serviceTime
226     *  or priority is invalid, or if an error occurs sending
227     *  the output token.
228     */
229    @Override
230    public void fire() throws IllegalActionException {
231        super.fire();
232        Time currentTime = getDirector().getModelTime();
233
234        serviceTime.update();
235        priority.update();
236
237        Job currentJob = this.peekQueue();
238        if (currentJob != null) {
239            currentJob.serviceTimeRemaining =
240                    _nextTimeFree.subtractToDouble(currentTime);
241        }
242
243        long nextQueueCounter = _queueCounter;
244        // Consume the input.
245        if (input.hasToken(0)) {
246            double serviceTimeValue = ((DoubleToken) serviceTime.getToken())
247                    .doubleValue();
248            double priorityValue = ((DoubleToken) priority.getToken())
249                    .doubleValue();
250            Token token = input.get(0);
251            this.enqueue(new Job(token, serviceTimeValue, priorityValue, currentTime));
252            if (_debugging) {
253                _debug("Read input with value " + token
254                        + ", and put into queue, which now has size"
255                        + _queue.size() + " at time " + currentTime
256                        + ". Event will be processes with service time "
257                        + serviceTimeValue);
258            }
259        }
260
261        // If appropriate, produce output.
262        Job job = this.peekQueue();
263        if (job != null && currentTime.compareTo(_nextTimeFree) == 0 &&
264                /* We check the queueCounter to ensure at least a microstep delay */
265                job.queueCounter < nextQueueCounter) {
266            job = this.dequeue();
267            Token outputToken = job.payload;
268            output.send(0, outputToken);
269            // Indicate that the server is free.
270            if (_debugging) {
271                _debug("Produced output " + outputToken
272                        + ", so queue now has size " + _queue.size()
273                        + " at time " + currentTime);
274            }
275        }
276        size.send(0, new IntToken(_queue.size()));
277    }
278
279    private Job peekQueue() {
280        return _queue.peek();
281    }
282
283    private void enqueue(Job newJob) throws IllegalActionException {
284        int currentCapacity = ((IntToken) capacity.getToken()).intValue();
285        if (currentCapacity > 0 && _queue.size() >= currentCapacity) {
286            throw new IllegalActionException(this,
287                    "Queue size (" + _queue.size()
288                            + ") is already at maximum capacity "
289                            + currentCapacity + ").");
290        }
291        newJob.queueCounter = _queueCounter;
292        _queue.add(newJob);
293        ++_queueCounter;
294        updateNextTimeFree();
295    }
296
297    private void updateNextTimeFree() {
298        Job job = this.peekQueue();
299        if (job != null) {
300            Time now = getDirector().getModelTime();
301            _nextTimeFree = now.add(job.serviceTimeRemaining);
302        }
303    }
304
305    private Job dequeue() throws IllegalActionException {
306        Job job = _queue.poll();
307        if (job == null) {
308            throw new IllegalActionException(this, "Queue is empty");
309        }
310        updateNextTimeFree();
311        return job;
312    }
313
314    /** Reset the states of the server to indicate that the server is ready
315     *  to serve.
316     *  @exception IllegalActionException If the base class throws it.
317     */
318    @Override
319    public void initialize() throws IllegalActionException {
320        super.initialize();
321        _nextTimeFree = Time.NEGATIVE_INFINITY;
322        _queue.clear();
323        _queueCounter = 0;
324    }
325
326    /** If the server is free and there is at least one token in the queue,
327     *  request a firing at the current time plus the service time.
328     *  @exception IllegalActionException If there is no director.
329     *  @return Whatever the superclass returns.
330     */
331    @Override
332    public boolean postfire() throws IllegalActionException {
333        updateNextTimeFree();
334        if (!_nextTimeFree.equals(Time.NEGATIVE_INFINITY) && _queue.size() > 0) {
335            if (_debugging) {
336                _debug("In postfire, requesting a refiring at time "
337                        + _nextTimeFree);
338            }
339            _fireAt(_nextTimeFree);
340        }
341        return super.postfire();
342    }
343    
344    /** Clear the queue so that the capacity can be changed.
345     *  @exception IllegalActionException If the superclass throws it.
346     */
347    @Override
348    public void wrapup() throws IllegalActionException {
349        super.wrapup();
350        _queue.clear();
351        _queueCounter = 0;
352    }

358    protected Time _nextTimeFree;

361    protected PriorityQueue<Job> _queue;

363    /** The counter for tie-breaking the queue by insertion order. */
364    protected long _queueCounter;

366    ///////////////////////////////////////////////////////////////////
367    ////                         inner classes                     ////
368
369    /** A data structure containing a token and a service time. */
370    private static class Job implements Comparable<Job> {
371        public Job(Token payload, double serviceTime,
372                double priority, Time creationTime) {
373            this.payload = payload;
374            this.serviceTimeRemaining = serviceTime;
375            this.priority = priority;
376            this.creationTime = creationTime;
377        }
378
379        public Token payload;
380        public double serviceTimeRemaining;
381        public double priority;    //     priority=1 >     priority=0
382        public Time creationTime;  // creationTime=1 < creationTime=0
383        public long queueCounter;  // queueCounter=1 < queueCounter=0
384
385        @Override
386        public int compareTo(Job other) {
387            int result = 0;
388            if (result == 0) {
389                result = Double.compare(this.priority, other.priority);
390                result = -result;  // higher priorities are smaller for min heap
391            }
392            if (result == 0) {
393                result = this.creationTime.compareTo(other.creationTime);
394            }
395            if (result == 0) {
396                result = Long.compare(this.queueCounter, other.queueCounter);
397            }
398            return result;
399        }
400    }
