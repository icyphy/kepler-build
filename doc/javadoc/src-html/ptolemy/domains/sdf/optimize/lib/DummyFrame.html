034/**
035<h1>Class comments</h1>
036A DummyFrame mimics a shared data object.
037It is used for testing the OptimizingSDFDirector. The shared data object holds an
038integer value.
039<p>
040See {@link ptolemy.domains.sdf.optimize.OptimizingSDFDirector} for more information.
041</p>
042@see ptolemy.domains.sdf.optimize.OptimizingSDFDirector
043
044@author Marc Geilen
045@version $Id$
046@since Ptolemy II 10.0
047@Pt.ProposedRating Red (mgeilen)
048@Pt.AcceptedRating Red ()
049 */
050public class DummyFrame implements Cloneable {
051
052    /**
053     * Represents the value encapsulated by the shared data frame.
054     */
055    public int value;
056
057    /**
058     * Provides a string representation of the object.
059     * @return string representation of the object
060     */
061    @Override
062    public String toString() {
063        return "Frame(" + Integer.toString(value) + ")";
064    }
065
066    /**
067     * Clone the DummyFrame.
068     * @return the cloned frame
069     */
070    @Override
071    public DummyFrame clone() {
072        DummyFrame f;
073        // TODO: below seems a silly solution, but I don't want expose callers
074        // of this method to the exception...
075        try {
076            f = (DummyFrame) super.clone();
077        } catch (CloneNotSupportedException e) {
078            f = new DummyFrame();
079        }
080        f.value = value;
081        return f;
082    }
083
084}
