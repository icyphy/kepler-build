046/**
047  A base class that contains common functionality for ModelPort and
048  ModalRefinementPort.
049
050  <p>ModalPort and RefinementPort both extend this class.
051  ModalRefinementPort in turn extends RefinementPort.  RefinementPort
052  has different functionality for methods like setInput(),
053  setMultiport(), setName() and setOutput().  Therefore, this base
054  class defines protected versions of the common implementations of
055  these methods that are called by the public versions of these
056  methods in ModalPort and ModelRefinementPort, thus avoiding
057  code duplication.</p>
058
059 @see ModalPort
060 @see ModalRefinementPort
061 @author Christopher Brooks
062 @version $Id$
063 @since Ptolemy II 10.0
064 @Pt.ProposedRating Red (cxh)
065 @Pt.AcceptedRating Red (cxh)
066 */
067public class ModalBasePort extends TypedIOPort {
068    /** Construct a port in the given workspace.
069     *  @param workspace The workspace.
070     *  @exception IllegalActionException If the port is not of an acceptable
071     *   class for the container, or if the container does not implement the
072     *   TypedActor interface.
073     */
074    public ModalBasePort(Workspace workspace) throws IllegalActionException {
075        super(workspace);
076    }
077
078    /** Construct a port with a containing actor and a name
079     *  that is neither an input nor an output.  The specified container
080     *  must implement the TypedActor interface, or an exception will be
081     *  thrown.
082     *  @param container The container actor.
083     *  @param name The name of the port.
084     *  @exception IllegalActionException If the port is not of an acceptable
085     *   class for the container, or if the container does not implement the
086     *   TypedActor interface.
087     *  @exception NameDuplicationException If the name coincides with
088     *   a port already in the container.
089     */
090    public ModalBasePort(ComponentEntity container, String name)
091            throws IllegalActionException, NameDuplicationException {
092        super(container, name);
093    }

098    /** Move this object down by one in the list of attributes of
099     *  its container. If this object is already last, do nothing.
100     *  This method overrides the base class to mirror the change
101     *  in any mirror ports.
102     *  Increment the version of the workspace.
103     *  @return The index of the specified object prior to moving it,
104     *   or -1 if it is not moved.
105     *  @exception IllegalActionException If this object has
106     *   no container.
107     */
108    protected int _moveDown() throws IllegalActionException {
109        try {
110            _workspace.getWriteAccess();
111
112            int result = super.moveDown();
113
114            if (result != -1) {
115                // Mirror the change in mirror ports.
116                CompositeEntity container = (CompositeEntity) getContainer();
117                Iterator entities = container.entityList().iterator();
118
119                while (entities.hasNext()) {
120                    Entity entity = (Entity) entities.next();
121                    Port mirrorPort = entity.getPort(getName());
122
123                    if (mirrorPort instanceof RefinementPort) {
124                        RefinementPort castPort = (RefinementPort) mirrorPort;
125                        boolean disableStatus = castPort._mirrorDisable;
126
127                        try {
128                            castPort._mirrorDisable = true;
129                            castPort.moveDown();
130                        } finally {
131                            castPort._mirrorDisable = disableStatus;
132                        }
133                    }
134                }
135            }
136
137            return result;
138        } finally {
139            _workspace.doneWriting();
140        }
141    }

326    /** Remove a port from the current container.
327     *  @param oldContainer The container of the port
328     *  @exception IllegalActionException If the container or the
329     *  container of the relation cannot be set to null.
330     *  @exception NameDuplicationException If the thrown will setting
331     *  the container or the container of the relation to null.
332     */
333    protected void _removePort(CompositeEntity oldContainer)
334            throws IllegalActionException, NameDuplicationException {
335        Iterator entities = oldContainer.entityList().iterator();
336
337        while (entities.hasNext()) {
338            Entity entity = (Entity) entities.next();
339            Port mirrorPort = entity.getPort(getName());
340
341            if (mirrorPort instanceof RefinementPort) {
342                RefinementPort castPort = (RefinementPort) mirrorPort;
343                boolean disableStatus = castPort._mirrorDisable;
344
345                try {
346                    castPort._mirrorDisable = true;
347                    castPort.setContainer(null);
348                } finally {
349                    castPort._mirrorDisable = disableStatus;
350                }
351            }
352        }
353        // Remove the relation as well.
354        ComponentRelation relation = oldContainer
355                .getRelation(getName() + "Relation");
356
357        if (relation != null) {
358            relation.setContainer(null);
359        }
360    }

449    /** Set the name of the port, and mirror the change in all the
450     *  mirror ports.
451     *  This method is write-synchronized on the workspace, and increments
452     *  the version of the workspace.
453     *  @param name The name.
454     *  @exception IllegalActionException If the name has a period.
455     *  @exception NameDuplicationException If there is already a port
456     *   with the same name in the container.
457     */
458    protected void _setName(String name)
459            throws IllegalActionException, NameDuplicationException {
460        try {
461            _workspace.getWriteAccess();
462
463            String oldName = getName();
464            super.setName(name);
465
466            // Mirror the change in mirror ports.
467            CompositeEntity container = (CompositeEntity) getContainer();
468
469            // NOTE: This is called before there is even a container
470            // to originally set the name.
471            if (container != null) {
472                Iterator entities = container.entityList().iterator();
473
474                while (entities.hasNext()) {
475                    Entity entity = (Entity) entities.next();
476                    Port mirrorPort = entity.getPort(oldName);
477
478                    if (mirrorPort instanceof RefinementPort) {
479                        RefinementPort castPort = (RefinementPort) mirrorPort;
480                        boolean disableStatus = castPort._mirrorDisable;
481
482                        try {
483                            castPort._mirrorDisable = true;
484                            castPort.setName(name);
485                        } finally {
486                            castPort._mirrorDisable = disableStatus;
487                        }
488                    }
489                }
490
491                // Rename the corresponding relation.
492                Relation relation = container.getRelation(oldName + "Relation");
493
494                if (relation != null) {
495                    relation.setName(name + "Relation");
496                }
497            }
498        } finally {
499            _workspace.doneWriting();
500        }
501    }

503    /** If the argument is true, make the port an output port.
504     *  If the argument is false, make the port not an output port.
505     *  This method overrides the base class to make the same
506     *  change on the mirror ports in the controller and state refinments.
507     *  This method invalidates the schedule and resolved types of the
508     *  director of the container, if there is one.
509     *  It is write-synchronized on the workspace, and increments
510     *  the version of the workspace.
511     *  @param isOutput True to make the port an output.
512     *  @exception IllegalActionException If changing the port status is
513     *   not permitted.
514     */
515    protected void _setOutput(boolean isOutput) throws IllegalActionException {
516        try {
517            _workspace.getWriteAccess();
518
519            super.setOutput(isOutput);
520
521            // Mirror the change in mirror ports.
522            CompositeEntity container = (CompositeEntity) getContainer();
523            Iterator entities = container.entityList().iterator();
524
525            while (entities.hasNext()) {
526                Entity entity = (Entity) entities.next();
527                Port mirrorPort = entity.getPort(getName());
528
529                if (mirrorPort instanceof RefinementPort) {
530                    RefinementPort castPort = (RefinementPort) mirrorPort;
531                    boolean disableStatus = castPort._mirrorDisable;
532
533                    try {
534                        castPort._mirrorDisable = true;
535                        castPort.setOutput(isOutput);
536                    } finally {
537                        castPort._mirrorDisable = disableStatus;
538                    }
539
540                    // If the entity is a controller, then set the
541                    // port to also be an input.
542                    if (entity.getName().equals("_Controller")) {
543                        boolean controlPortStatus = castPort._mirrorDisable;
544
545                        try {
546                            castPort._mirrorDisable = true;
547                            castPort.setInput(true);
548
549                            // Mark that the input property is
550                            // automatically set, so that if it
551                            // is changed, that change is not
552                            // mirrored.
553                            if (!isInput()) {
554                                castPort._automaticallyInput = true;
555                            }
556                        } finally {
557                            castPort._mirrorDisable = controlPortStatus;
558                        }
559                    }
560                }
561            }
562        } finally {
563            _workspace.doneWriting();
564        }
565    }

567    /** Override the base class to ensure that the proposed container
568     *  is a CompositeEntity or null.
569     *  @param container The proposed container.
570     *  @exception IllegalActionException If the proposed container is not a
571     *   TypedActor, or if the base class throws it.
572     */
573    @Override
574    protected void _checkContainer(Entity container)
575            throws IllegalActionException {
576        if (!(container instanceof CompositeEntity) && container != null) {
577            throw new IllegalActionException(container, this,
578                    "ModalPort can only be contained by CompositeEntity objects. "
579                            + "The container was: " + container);
580        }
