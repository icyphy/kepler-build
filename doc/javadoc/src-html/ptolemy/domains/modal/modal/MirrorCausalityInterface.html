047///////////////////////////////////////////////////////////////////
048//// MirrorCausalityInterface
049
050/**
051This class infers a causality interface from causality interfaces provided
052in the constructor and in the {@link #composeWith} method. For each of these
053interfaces, this interface finds ports in its own actor that match the names
054of those for the specified interfaces, and constructs dependencies that
055are oPlus compositions of the dependencies in the specified interfaces for
056ports with the same names. For equivalence classes, it merges the equivalence
057classes so that if two ports are equivalent in any of the provided causality
058interfaces, then the corresponding (same named) ports in the actor are also
059equivalent.
060
061@author Edward A. Lee
062@version $Id$
063@since Ptolemy II 8.0
064@Pt.ProposedRating Yellow (eal)
065@Pt.AcceptedRating Red (eal)
066 */
067public class MirrorCausalityInterface extends CausalityInterfaceForComposites {
068
069    /** Construct a causality interface that mirrors the specified causality
070     *  interface.
071     *  @param actor The actor for which this is an interface.
072     *  @param causality The interface to mirror.
073     * @exception IllegalArgumentException If the actor parameter is not
074     * an instance of CompositeEntity.
075     */
076    public MirrorCausalityInterface(Actor actor, CausalityInterface causality)
077            throws IllegalArgumentException {
078        super(actor, causality.getDefaultDependency());
079        _composedInterfaces.add(causality);
080    }
081
082    ///////////////////////////////////////////////////////////////////
083    ////                         public methods                    ////
084
085    /** Add the specified causality interface for the specified actor.
086     *  @param causality The interface to compose with the one specified
087     *   in the constructor.
088     */
089    public void composeWith(CausalityInterface causality) {
090        _composedInterfaces.add(causality);
091    }
092
093    /** Return the dependency between the specified input port
094     *  and the specified output port.  This is done by checking
095     *  the guards and actions of all the transitions.
096     *  When called for the first time since a change in the model
097     *  structure, this method performs the complete analysis of
098     *  the FSM and caches the result. Subsequent calls just
099     *  look up the result.
100     *  @param input The input port.
101     *  @param output The output port, or null to update the
102     *   dependencies (and record equivalence classes) without
103     *   requiring there to be an output port.
104     *  @return The dependency between the specified input port
105     *   and the specified output port, or null if a null output
106     *   is port specified.
107     *  @exception IllegalActionException If a guard expression cannot be parsed.
108     */
109    @Override
110    public Dependency getDependency(IOPort input, IOPort output)
111            throws IllegalActionException {
112        // If the dependency is not up-to-date, then update it.
113        long workspaceVersion = ((NamedObj) _actor).workspace().getVersion();
114        if (_dependencyVersion != workspaceVersion) {
115            // Need to update dependencies. The cached version
116            // is obsolete.
117            try {
118                ((NamedObj) _actor).workspace().getReadAccess();
119                _reverseDependencies = new HashMap<IOPort, Map<IOPort, Dependency>>();
120                _forwardDependencies = new HashMap<IOPort, Map<IOPort, Dependency>>();
121
122                // Iterate over all the associated interfaces.
123                for (CausalityInterface causality : _composedInterfaces) {
124                    List<IOPort> mirrorInputs = causality.getActor()
125                            .inputPortList();
126                    for (IOPort mirrorInput : mirrorInputs) {
127                        Port localInput = ((Entity) _actor)
128                                .getPort(mirrorInput.getName());
129                        if (!(localInput instanceof IOPort)) {
130                            throw new IllegalActionException(_actor,
131                                    mirrorInput.getContainer(),
132                                    "No matching port with name "
133                                            + mirrorInput.getName());
134                        }
135                        // The localInput may not be an input port...
136                        // It may have been an output port that the FSMActor controller
137                        // also has as an input port. But we don't want to set a forward
138                        // depedency in this case.
139                        if (!((IOPort) localInput).isInput()) {
140                            continue;
141                        }
142                        Map<IOPort, Dependency> forwardMap = _forwardDependencies
143                                .get(localInput);
144                        if (forwardMap == null) {
145                            forwardMap = new HashMap<IOPort, Dependency>();
146                            _forwardDependencies.put((IOPort) localInput,
147                                    forwardMap);
148                        }
149                        for (IOPort dependentOutput : causality
150                                .dependentPorts(mirrorInput)) {
151                            Port localOutput = ((Entity) _actor)
152                                    .getPort(dependentOutput.getName());
153                            if (!(localOutput instanceof IOPort)) {
154                                throw new IllegalActionException(_actor,
155                                        mirrorInput.getContainer(),
156                                        "No matching port with name "
157                                                + mirrorInput.getName());
158                            }
159                            Dependency dependency = causality.getDependency(
160                                    mirrorInput, dependentOutput);
161                            forwardMap.put((IOPort) localOutput, dependency);
162                            // Now handle the reverse dependencies.
163                            Map<IOPort, Dependency> backwardMap = _reverseDependencies
164                                    .get(localOutput);
165                            if (backwardMap == null) {
166                                backwardMap = new HashMap<IOPort, Dependency>();
167                                _reverseDependencies.put((IOPort) localOutput,
168                                        backwardMap);
169                            }
170                            backwardMap.put((IOPort) localInput, dependency);
171                        }
172                    }
173                }
174                // Next do equivalence classes.
175                // We iterate over the input ports, and for each one,
176                // find the ports for which it has equivalents in any
177                // associated causality.
178                _equivalenceClasses = new HashMap<IOPort, Collection<IOPort>>();
179                Collection<IOPort> localInputs = _actor.inputPortList();
180                for (IOPort localInput : localInputs) {
181                    Collection<IOPort> equivalences = _equivalenceClasses
182                            .get(localInput);
183                    if (equivalences != null) {
184                        // This input port is done.
185                        continue;
186                    }
187                    equivalences = new HashSet<IOPort>();
188                    // Iterate over all the associated interfaces.
189                    for (CausalityInterface causality : _composedInterfaces) {
190                        IOPort mirrorInput = (IOPort) ((Entity) causality
191                                .getActor()).getPort(localInput.getName());
192                        if (mirrorInput == null) {
193                            throw new IllegalActionException(_actor, localInput,
194                                    "Expected matching port in " + causality
195                                            .getActor().getFullName());
196                        }
197                        equivalences.addAll(_localMirrors(
198                                causality.equivalentPorts(mirrorInput)));
199                    }
200                    // Set the equivalence class for all ports in the set.
201                    for (IOPort equivalentPort : equivalences) {
202                        _equivalenceClasses.put(equivalentPort, equivalences);
203                    }
204                }
205            } finally {
206                ((NamedObj) _actor).workspace().doneReading();
207            }
208            _dependencyVersion = workspaceVersion;
209        }
210        if (output == null) {
211            return null;
212        }
213        Map<IOPort, Dependency> inputMap = _forwardDependencies.get(input);
214        if (inputMap != null) {
215            Dependency result = inputMap.get(output);
216            if (result != null) {
217                return result;
218            }
219        }
220        // If there is no recorded dependency, then reply
221        // with the additive identity (which indicates no
222        // dependency).
223        return _defaultDependency.oPlusIdentity();
224    }
225
226    ///////////////////////////////////////////////////////////////////
227    ////                         private methods                   ////
228
229    /** Return the local ports whose names match the ports in the
230     *  specified collection.
231     *  @param ports A collection of ports.
232     *  @exception IllegalActionException If no matching port is found.
233     */
234    private Collection<IOPort> _localMirrors(Collection<IOPort> ports)
235            throws IllegalActionException {
236        Set<IOPort> result = new HashSet<IOPort>();
237        for (IOPort port : ports) {
238            IOPort localPort = (IOPort) ((Entity) _actor)
239                    .getPort(port.getName());
240            if (localPort == null) {
241                throw new IllegalActionException(port.getContainer(), port,
242                        "Expected matching port in " + _actor.getFullName());
243            }
244            result.add(localPort);
245        }
246        return result;
247    }
248
249    ///////////////////////////////////////////////////////////////////
250    ////                         private fields                    ////
251
252    /** The set of causality interfaces that this one composes. */
253    private Set<CausalityInterface> _composedInterfaces = new HashSet<CausalityInterface>();
254}
