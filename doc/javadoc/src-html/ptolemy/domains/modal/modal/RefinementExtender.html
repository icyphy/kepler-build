028package ptolemy.domains.modal.modal;
029
030import ptolemy.data.expr.StringParameter;
031import ptolemy.kernel.util.Attribute;
032import ptolemy.kernel.util.IllegalActionException;
033import ptolemy.kernel.util.NameDuplicationException;
034import ptolemy.kernel.util.NamedObj;
035
036//////////////////////////////////////////////////////////////////////////
037//// RefinementExtender
038
039/**
040 An attribute to add a class to the list of refinement classes for a state. The
041 name of this attribute should be an easy-to-remember name of the refinement
042 class to be shown to the user in the dialog displayed when the user chooses to
043 add a refinement. The <code>className</code> property of this attribute refers
044 to the actual class, which is either a Java class with which the refinement is
045 to be created, or a Ptolemy class specified in MoML at the path pointed to by
046 the class name.
047
048 @author Thomas Huining Feng
049 @version $Id$
050 @since Ptolemy II 8.0
051 @Pt.ProposedRating Red (tfeng)
052 @Pt.AcceptedRating Red (tfeng)
053 */
054public class RefinementExtender extends Attribute {
055
056    /** Construct a refinement extender attribute with the given name contained
057     *  by the specified entity. The container argument must not be null, or a
058     *  NullPointerException will be thrown.  This attribute will use the
059     *  workspace of the container for synchronization and version counts.
060     *  If the name argument is null, then the name is set to the empty string.
061     *  Increment the version of the workspace.
062     *  @param container The container.
063     *  @param name The name of this attribute.
064     *  @exception IllegalActionException If the attribute is not of an
065     *   acceptable class for the container, or if the name contains a period.
066     *  @exception NameDuplicationException If the name coincides with
067     *   an attribute already in the container.
068     */
069    public RefinementExtender(NamedObj container, String name)
070            throws IllegalActionException, NameDuplicationException {
071        super(container, name);
072
073        className = new StringParameter(this, "className");
074        description = new StringParameter(this, "description");
075    }
076
077    /** The full name of the refinement class.
078     */
079    public StringParameter className;
080
081    /** A string description of the refinement class, which is shown in the
082    dialog for the user to choose a refinement to add. */
083    public StringParameter description;
084
085}
