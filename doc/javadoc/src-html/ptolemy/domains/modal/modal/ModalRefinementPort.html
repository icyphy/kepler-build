037///////////////////////////////////////////////////////////////////
038//// ModalRefinementPort
039
040/**
041 A port for refinements in modal models that are themselves modal models.  This port
042 mirrors certain changes to it in the ports of the container of the container and
043 in the ports of the controller and refinements of the modal model that contains it.
044 It is designed to work closely with RefinementPort and ModalPort, since changes to the
045 ports can be initiated in either class.
046
047 @see ModalPort
048 @author Edward A. Lee
049 @version $Id$
050 @since Ptolemy II 8.0
051 @Pt.ProposedRating Red (eal)
052 @Pt.AcceptedRating Red (liuxj)
053 */
054public class ModalRefinementPort extends RefinementPort {
055    /** Construct a port with a containing actor and a name
056     *  that is neither an input nor an output.  The specified container
057     *  must implement the TypedActor interface, or an exception will be
058     *  thrown.
059     *  @param container The container actor.
060     *  @param name The name of the port.
061     *  @exception IllegalActionException If the port is not of an acceptable
062     *   class for the container, or if the container does not implement the
063     *   TypedActor interface.
064     *  @exception NameDuplicationException If the name coincides with
065     *   a port already in the container.
066     */
067    public ModalRefinementPort(ComponentEntity container, String name)
068            throws IllegalActionException, NameDuplicationException {
069        super(container, name);
070    }
071
072    ///////////////////////////////////////////////////////////////////
073    ////                         public methods                    ////
074
075    /** Move this object down by one in the list of attributes of
076     *  its container. If this object is already last, do nothing.
077     *  This method overrides the base class to mirror the change
078     *  in any mirror ports.
079     *  Increment the version of the workspace.
080     *  @return The index of the specified object prior to moving it,
081     *   or -1 if it is not moved.
082     *  @exception IllegalActionException If this object has
083     *   no container.
084     */
085    @Override
086    public int moveDown() throws IllegalActionException {
087        return _moveDown();
088    }
089
090    /** Move this object to the first position in the list
091     *  of attributes of the container. If  this object is already first,
092     *  do nothing. Increment the version of the workspace.
093     *  This method overrides the base class to mirror the change
094     *  in any mirror ports.
095     *  @return The index of the specified object prior to moving it,
096     *   or -1 if it is not moved.
097     *  @exception IllegalActionException If this object has
098     *   no container.
099     */
100    @Override
101    public int moveToFirst() throws IllegalActionException {
102        return _moveToFirst();
103    }
104
105    /** Move this object to the specified position in the list
106     *  of attributes of the container. If this object is already at the
107     *  specified position, do nothing.
108     *  This method overrides the base class to mirror the change
109     *  in any mirror ports.
110     *  Increment the version of the workspace.
111     *  @param index The position to move this object to.
112     *  @return The index of the specified object prior to moving it,
113     *   or -1 if it is not moved.
114     *  @exception IllegalActionException If this object has
115     *   no container or if the index is out of bounds.
116     */
117    @Override
118    public int moveToIndex(int index) throws IllegalActionException {
119        return _moveToIndex(index);
120    }
121
122    /** Move this object to the last position in the list
123     *  of attributes of the container.  If this object is already last,
124     *  do nothing. This method overrides the base class to mirror the change
125     *  in any mirror ports.
126     *  Increment the version of the workspace.
127     *  @return The index of the specified object prior to moving it,
128     *   or -1 if it is not moved.
129     *  @exception IllegalActionException If this object has
130     *   no container.
131     */
132    @Override
133    public int moveToLast() throws IllegalActionException {
134        return _moveToLast();
135    }
136
137    /** Move this object up by one in the list of
138     *  attributes of the container. If  this object is already first, do
139     *  nothing.
140     *  This method overrides the base class to mirror the change
141     *  in any mirror ports.
142     *  Increment the version of the workspace.
143     *  @return The index of the specified object prior to moving it,
144     *   or -1 if it is not moved.
145     *  @exception IllegalActionException If this object has
146     *   no container.
147     */
148    @Override
149    public int moveUp() throws IllegalActionException {
150        return _moveUp();
151    }
152
153    /** Override the base class so that if the port is being removed
154     *  from the current container, then it is also removed from the
155     *  controller and from each of the refinements.
156     *  @param container The proposed container.
157     *  @exception IllegalActionException If the proposed container is not a
158     *   ComponentEntity, doesn't implement Actor, or has no name,
159     *   or the port and container are not in the same workspace. Or
160     *   it's not null
161     *  @exception NameDuplicationException If the container already has
162     *   a port with the name of this port.
163     */
164    @Override
165    public void setContainer(Entity container)
166            throws IllegalActionException, NameDuplicationException {
167        NamedObj oldContainer = getContainer();
168
169        // FIXME: Why is this method is slightly different from ModalPort.setContainer()
170        // In particular, this method does not call getWriteAccess()
171        if (_mirrorDisable || oldContainer == null) {
172            // Mirroring changes from above, or there
173            // is no pre-existing above.
174
175            // Use the superclass to delegate up the hierarchy and mirror the change
176            // to ports in the enclosing modal model and its refinements.
177            super.setContainer(container);
178
179            if (oldContainer instanceof CompositeEntity
180                    && container != oldContainer) {
181                // The port is being removed from the current container.
182                // Remove it from the mirrored ports.
183                _removePort((CompositeEntity) oldContainer);
184            }
185        } else {
186            // Not mirroring changes from above.
187            // Use the superclass to delegate up the hierarchy and mirror the change
188            // to ports in the enclosing modal model and its refinements.
189            // But don't mirror below in the hierarchy because the code
190            // above will be called as a result of delegating upwards.
191            super.setContainer(container);
192        }
193    }
194
195    /** If the argument is true, make the port an input port.
196     *  If the argument is false, make the port not an input port.
197     *  This method overrides the base class to make the same
198     *  change on the mirror ports in the controller and state refinments.
199     *  This method invalidates the schedule and resolved types of the
200     *  director of the container, if there is one.
201     *  It is write-synchronized on the workspace, and increments
202     *  the version of the workspace.
203     *  @param isInput True to make the port an input.
204     *  @exception IllegalActionException If changing the port status is
205     *   not permitted.
206     */
207    @Override
208    public void setInput(boolean isInput) throws IllegalActionException {
209        _setInput(isInput);
210    }
211
212    /** If the argument is true, make the port a multiport.
213     *  If the argument is false, make the port not a multiport.
214     *  This method overrides the base class to make the same
215     *  change on the mirror ports in the controller and state refinments.
216     *  This method invalidates the schedule and resolved types of the
217     *  director of the container, if there is one.
218     *  It is write-synchronized on the workspace, and increments
219     *  the version of the workspace.
220     *  @param isMultiport True to make the port a multiport.
221     *  @exception IllegalActionException If changing the port status is
222     *   not permitted.
223     */
224    @Override
225    public void setMultiport(boolean isMultiport)
226            throws IllegalActionException {
227        _setMultiport(isMultiport);
228    }
229
230    /** Set the name of the port, and mirror the change in all the
231     *  mirror ports.
232     *  This method is write-synchronized on the workspace, and increments
233     *  the version of the workspace.
234     *  @exception IllegalActionException If the name has a period.
235     *  @exception NameDuplicationException If there is already a port
236     *   with the same name in the container.
237     */
238    @Override
239    public void setName(String name)
240            throws IllegalActionException, NameDuplicationException {
241        _setName(name);
242    }
243
244    /** If the argument is true, make the port an output port.
245     *  If the argument is false, make the port not an output port.
246     *  In addition, if the container is an instance of Refinement,
247     *  and the argument is true, find the corresponding port of the
248     *  controller and make it an input and not an output.  This makes
249     *  it possible for the controller to see the outputs of the refinements.
250     *  This method overrides the base class to make the same
251     *  change on the mirror ports in the controller and state refinments.
252     *  This method invalidates the schedule and resolved types of the
253     *  director of the container, if there is one.
254     *  It is write-synchronized on the workspace, and increments
255     *  the version of the workspace.
256     *  @param isOutput True to make the port an output.
257     *  @exception IllegalActionException If changing the port status is
258     *   not permitted.
259     */
260    @Override
261    public void setOutput(boolean isOutput) throws IllegalActionException {
262        _setOutput(isOutput);
263    }
