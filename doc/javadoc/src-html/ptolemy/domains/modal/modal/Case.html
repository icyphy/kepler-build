042///////////////////////////////////////////////////////////////////
043//// Case
044
045/**
046 An actor that executes one of several refinements depending on the
047 value provided by the <i>control</i> port-parameter. To use this,
048 look inside, add refinement cases, and populate them with computations.
049 Each refinement is a composite that is required to have its own director.
050 The name of the refinement is value that the control must have to
051 execute this refinement.
052 This actor always provides one case called "default". This is
053 the refinement that is executed if no other refinement matches
054 the control input.  All refinements have the same ports,
055 and adding ports to any one refinement or to the case actor
056 itself results in identical ports being added to all refinements.
057
058 @author Edward A. Lee
059 @version $Id$
060 @since Ptolemy II 8.0
061 @Pt.ProposedRating Yellow (eal)
062 @Pt.AcceptedRating Red (reviewmoderator)
063 @deprecated Use ptolemy.actor.lib.hoc.Case instead.
064 */
065@Deprecated
066public class Case extends ModalModel {
067
068    /** Construct
069     * a modal model with a name and a container.
070     *  The container argument must not be null, or a
071     *  NullPointerException will be thrown.
072     *  @param container The container.
073     *  @param name The name of this actor.
074     *  @exception IllegalActionException If the container is incompatible
075     *   with this actor.
076     *  @exception NameDuplicationException If the name coincides with
077     *   an actor already in the container.
078     */
079    public Case(CompositeEntity container, String name)
080            throws IllegalActionException, NameDuplicationException {
081        super(container, name);
082        _init();
083    }
084
085    ///////////////////////////////////////////////////////////////////
086    ////                         public variables                  ////
087
088    /** The input port-parameter on which the control token is provided.
089     *  This can have any type, and is initialized with a default value
090     *  of true.
091     */
092    public PortParameter control;
093
094    ///////////////////////////////////////////////////////////////////
095    ////                         public methods                    ////
096
097    /** Override the base class to ensure that the _default member
098     *  points to the default refinement.
099     *  @param workspace The workspace for the new object.
100     *  @return A new Case.
101     *  @exception CloneNotSupportedException If any of the attributes
102     *   cannot be cloned.
103     */
104    @Override
105    public Object clone(Workspace workspace) throws CloneNotSupportedException {
106        Case newObject = (Case) super.clone(workspace);
107        newObject._default = (Refinement) newObject.getEntity("default");
108        return newObject;
109    }
110
111    ///////////////////////////////////////////////////////////////////
112    ////                         protected methods                 ////
113
114    /** Override the base class to create the transition associated
115     *  with this refinement if the argument is an instance of Refinement.
116     *  @param entity Entity to contain.
117     *  @exception IllegalActionException If the entity has no name, or the
118     *   action would result in a recursive containment structure.
119     *  @exception NameDuplicationException If the name collides with a name
120     *  already in the entity.
121     */
122    @Override
123    protected void _addEntity(ComponentEntity entity)
124            throws IllegalActionException, NameDuplicationException {
125        super._addEntity(entity);
126        if (entity instanceof Refinement) {
127            String controlValue = entity.getName();
128            // Create a self-loop transition if one does not exist.
129            // FIXME: This assumes it is right!  Can it be wrong?
130            // Need to ensure that these relations are deleted when a
131            // refinement is deleted.
132            if (_controller.getRelation(controlValue) == null) {
133                Transition transition = (Transition) _controller
134                        .newRelation(controlValue);
135                transition.guardExpression
136                        .setExpression("control == " + controlValue);
137                transition.refinementName.setExpression(controlValue);
138                transition.preemptive.setToken(BooleanToken.TRUE);
139                if (controlValue.equals("default")) {
140                    transition.defaultTransition.setToken(BooleanToken.TRUE);
141                }
142                // Create a self loop connection.
143                _state.incomingPort.link(transition);
144                _state.outgoingPort.link(transition);
145            }
146            // Ensure that the default refinement remains the last one.
147            // Note however that this is called on the default itself,
148            // at which time the local member has not been set.
149            if (_default != null) {
150                _default.moveToLast();
151            }
152        }
153    }
154
155    ///////////////////////////////////////////////////////////////////
156    ////                         protected variables               ////
157
158    /** The one and only state. */
159    protected State _state;
160
161    /** The default refinement. */
162    protected Refinement _default;
163
164    ///////////////////////////////////////////////////////////////////
165    ////                         private methods                   ////
166
167    /** Initialize the model with a single state.
168     */
169    private void _init()
170            throws IllegalActionException, NameDuplicationException {
171        if (!_printedDeprecatedMessage) {
172            _printedDeprecatedMessage = true;
173            System.out.println("Warning: " + getFullName()
174                    + ": modal.modal.Case has been deprecated"
175                    + " since March, 2006.  Use actor.lib.hoc.Case instead.");
176        }
177        // Create the one and only state in the controller.
178        _state = new State(_controller, "State");
179        _controller.initialStateName.setExpression("State");
180        // FIXME: The following doesn't give much info.
181        // _controller.addDebugListener(this);
182        // Make the controller transient (it is reconstructed
183        // at construction time).
184
185        // Hide the directorClass parameter.
186        directorClass.setVisibility(Settable.EXPERT);
187
188        // Create the control port.
189        control = new PortParameter(this, "control");
190        // FIXME: This is awkward... If I provide some
191        // non-boolean control input, I get obscure type
192        // conflict error messages and have to change this
193        // to match.
194        control.setExpression("true");
195        ParameterPort port = control.getPort();
196        // Put the control input on the bottom of the actor.
197        StringAttribute controlCardinal = new StringAttribute(port,
198                "_cardinal");
199        controlCardinal.setExpression("SOUTH");
200
201        // Create the default refinement.
202        // NOTE: We do not use a TransitionRefinement because we don't
203        // want the sibling input ports that come with output ports.
204        _default = new Refinement(this, "default");
205    }
206
207    ///////////////////////////////////////////////////////////////////
208    ////                         private variables                 ////
209
210    /** True if we have printed the deprecated message. */
211    private static boolean _printedDeprecatedMessage = false;
212}
