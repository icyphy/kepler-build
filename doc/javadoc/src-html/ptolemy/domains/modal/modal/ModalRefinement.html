064/**
065 This modal model actor supports mirroring of its ports in its container
066 (which is required to be a refinement of a state in a ModalModel).
067 This in turn assures
068 mirroring of ports in each of the refinements.
069 <p>
070 Note that this actor has no attributes of its own.
071 Requests for attributes are delegated to the container.
072
073 @author Edward A. Lee
074 @version $Id$
075 @since Ptolemy II 8.0
076 @Pt.ProposedRating Red (eal)
077 @Pt.AcceptedRating Red (reviewmoderator)
078 */
079public class ModalRefinement extends ModalModel
080        implements DropTargetHandler, RefinementActor {
081
082    /** Construct a modal controller with a name and a container.
083     *  The container argument must not be null, or a
084     *  NullPointerException will be thrown.
085     *  @param container The container.
086     *  @param name The name of this actor.
087     *  @exception IllegalActionException If the container is incompatible
088     *   with this actor.
089     *  @exception NameDuplicationException If the name coincides with
090     *   an actor already in the container.
091     */
092    public ModalRefinement(CompositeEntity container, String name)
093            throws IllegalActionException, NameDuplicationException {
094        super(container, name);
095        ContainmentExtender containmentExtender = new ContainmentExtender(this,
096                "_containmentExtender");
097        containmentExtender.setPersistent(false);
098    }

103    /** React to a list of objects being dropped onto a target.
104     *
105     *  @param target The target on which the objects are dropped.
106     *  @param dropObjects The list of objects dropped onto the target.
107     *  @param moml The moml string generated for the dropped objects.
108     *  @exception IllegalActionException If the handling is unsuccessful.
109     */
110    @Override
111    public void dropObject(NamedObj target, List dropObjects, String moml)
112            throws IllegalActionException {
113        boolean merge = false;
114        if (target instanceof State) {
115            State state = (State) target;
116
117            TreeMap<Class<? extends Entity>, String> map = _getRefinementClasses();
118            String refinementClass = null;
119            boolean conflict = false;
120            RefinementSuggestion suggestion = (RefinementSuggestion) target
121                    .getAttribute("refinementSuggestion");
122            for (Object dropObject : dropObjects) {
123                if (suggestion != null) {
124                    String className = suggestion
125                            .getRefinementClass((NamedObj) dropObject);
126                    if (refinementClass == null) {
127                        refinementClass = className;
128                    } else if (!refinementClass.equals(className)) {
129                        conflict = true;
130                    }
131                } else {
132                    for (Class<? extends Entity> keyClass : map.keySet()) {
133                        if (keyClass.isInstance(dropObject)) {
134                            String value = map.get(keyClass);
135                            if (refinementClass == null) {
136                                refinementClass = value;
137                                break;
138                            } else if (!refinementClass.equals(value)) {
139                                conflict = true;
140                                break;
141                            }
142                        }
143                    }
144                }
145                if (conflict) {
146                    break;
147                }
148            }
149
150            if (conflict || refinementClass == null) {
151                throw new IllegalActionException(this, "Unable to determine "
152                        + "the type of all the dropped objects.");
153            }
154
155            TypedActor[] refinements = state.getRefinement();
156            TypedActor refinement = null;
157            if (refinements != null) {
158                for (TypedActor actor : refinements) {
159                    if (((NamedObj) actor).getClassName()
160                            .equals(refinementClass)) {
161                        refinement = actor;
162                        break;
163                    }
164                }
165            }
166            if (refinement == null) {
167                CompositeEntity containerContainer = (CompositeEntity) state
168                        .getContainer().getContainer();
169                String name = containerContainer.uniqueName(state.getName());
170                addRefinement(state, name, null, refinementClass, null);
171                merge = true;
172                refinement = (TypedActor) containerContainer.getEntity(name);
173            }
174            target = (NamedObj) refinement;
175        }
176        MoMLChangeRequest request = new MoMLChangeRequest(this, target, moml);
177        request.setUndoable(true);
178        if (merge) {
179            request.setMergeWithPreviousUndo(true);
180        }
181        target.requestChange(request);
182    }

184    /** Get the state in any ModalRefinement within this ModalModel that has
185     *  this ModalRefinement as its refinement, if any. Return null if no such
186     *  state is found.
187     *
188     *  @return The state with this ModalRefinement as its refinement, or null.
189     *  @exception IllegalActionException If the specified refinement cannot be
190     *   found in a state, or if a comma-separated list is malformed.
191     */
192    @Override
193    public State getRefinedState() throws IllegalActionException {
194        NamedObj container = getContainer();
195        if (container instanceof ModalModel) {
196            List<?> controllers = ((ModalModel) container)
197                    .entityList(ModalRefinement.class);
198            for (Object controllerObject : controllers) {
199                ModalRefinement controller = (ModalRefinement) controllerObject;
200                List<?> states = controller.entityList(State.class);
201                for (Object stateObject : states) {
202                    State state = (State) stateObject;
203                    TypedActor[] refinements = state.getRefinement();
204                    if (refinements != null) {
205                        for (TypedActor refinement : refinements) {
206                            if (refinement == this) {
207                                return state;
208                            }
209                        }
210                    }
211                }
212            }
213        }
214        return null;
215    }

217    /** Create a new port with the specified name in the container of
218     *  this controller, which in turn creates a port in this controller
219     *  and all the refinements.
220     *  This method is write-synchronized on the workspace.
221     *  @param name The name to assign to the newly created port.
222     *  @return The new port.
223     *  @exception NameDuplicationException If the entity already has a port
224     *   with the specified name.
225     */
226    @Override
227    public Port newPort(String name) throws NameDuplicationException {
228        try {
229            _workspace.getWriteAccess();
230
231            ModalModel container = (ModalModel) getContainer();
232
233            if (_mirrorDisable == 1 || container == null) {
234                // We are mirroring a change above in the hierarchy
235                // (or there is no above in the hierarchy),
236                // so we should not mirror this change upwards.
237                // But we should mirror it downwards.
238                ModalRefinementPort port = new ModalRefinementPort(this, name);
239
240                // Create the links on the outside of the new port.
241                if (container != null) {
242                    String relationName = name + "Relation";
243                    Relation relation = container.getRelation(relationName);
244
245                    if (relation == null) {
246                        relation = container.newRelation(relationName);
247
248                        Port containerPort = container.getPort(name);
249                        containerPort.link(relation);
250                    }
251
252                    port.link(relation);
253                }
254                // Mirror the change downwards in the hierarchy.
255                Iterator entities = entityList().iterator();
256                while (entities.hasNext()) {
257                    Entity entity = (Entity) entities.next();
258
259                    if (entity instanceof RefinementActor) {
260                        if (entity.getPort(name) == null) {
261                            try {
262                                ((RefinementActor) entity).setMirrorDisable(1);
263                                entity.newPort(name);
264                            } finally {
265                                ((RefinementActor) entity).setMirrorDisable(0);
266                            }
267                        }
268                    }
269                }
270                return port;
271            } else {
272                // We are originating the change, or it originated from
273                // below us in the hierarchy, and hence should delegate
274                // it upwards. This will cause this newPort() method
275                // to be called again by the container after it has
276                // created its own port. That will result in the code
277                // above executing because it will set _mirrorPort to 1
278                // on this port before doing the call.
279                ModalRefinementPort containerPort = container == null ? null
280                        : (ModalRefinementPort) container.getPort(name);
281                if (containerPort == null) {
282                    // The container does not have a mirror port.
283                    // Delegate upwards.
284                    // Note that this will result in the container
285                    // creating the port here by calling this
286                    // newPort() method. It will set my
287                    // _mirrorDisable to 1 before doing that,
288                    // and set it back to 0 after.
289                    container.newPort(name);
290                    return getPort(name);
291                } else {
292                    // The container already has a mirror port, so
293                    // we cannot use it to create our port.
294                    // This can happen if the container was created first
295                    // and populated with its ports before this refinement
296                    // was created.
297                    ModalRefinementPort port = new ModalRefinementPort(this,
298                            name);
299                    // Create the link on the outside of the port.
300                    String relationName = name + "Relation";
301                    Relation relation = container.getRelation(relationName);
302                    // The container should already have this relation,
303                    // but in case not, we create it.
304                    if (relation == null) {
305                        relation = container.newRelation(relationName);
306                        containerPort.link(relation);
307                    }
308                    port.link(relation);
309
310                    // Mirror the change downwards in the hierarchy.
311                    Iterator entities = entityList().iterator();
312                    while (entities.hasNext()) {
313                        Entity entity = (Entity) entities.next();
314
315                        if (entity instanceof RefinementActor) {
316                            if (entity.getPort(name) == null) {
317                                try {
318                                    ((RefinementActor) entity)
319                                            .setMirrorDisable(1);
320                                    entity.newPort(name);
321                                } finally {
322                                    ((RefinementActor) entity)
323                                            .setMirrorDisable(0);
324                                }
325                            }
326                        }
327                    }
328                    return port;
329                }
330            }
331        } catch (IllegalActionException ex) {
332            // This exception should not occur, so we throw a runtime
333            // exception.
334            throw new InternalErrorException(
335                    "ModalRefinement.newPort: Internal error: "
336                            + ex.getMessage());
337        } finally {
338            _mirrorDisable = 0;
339            _workspace.doneWriting();
340        }
341    }

343    /** Control whether adding a port should be mirrored in the modal
344     *  model and refinements.
345     *  This is added to allow control by the UI.
346     *  @param disable 0 if mirroring should occur, -1
347     *   if mirroring should not occur downwards in the hierarchy,
348     *   1 if mirroring should not occur upwards in the hierarchy.
349     */
350    @Override
351    public void setMirrorDisable(int disable) {
352        _mirrorDisable = disable;
353    }

355    /** Override the base class to ensure that the proposed container
356     *  is a ModalModel or null.
357     *  @param container The proposed container.
358     *  @exception IllegalActionException If the proposed container is not a
359     *   TypedActor, or if the base class throws it.
360     */
361    protected void _checkContainer(Entity container)
362            throws IllegalActionException {
363        if (!(container instanceof ModalModel) && container != null) {
364            throw new IllegalActionException(container, this,
365                    "ModalRefinement can only be contained by "
366                            + "ModalModel objects.");
367        }
368    }

373    /** Return a map from the classes of the entities to be dropped into a state
374     *  and the class names of the refinements that can be used to contain those
375     *  entities.
376     *
377     *  @return The map.
378     */
379    protected TreeMap<Class<? extends Entity>, String> _getRefinementClasses() {
380        TreeMap map = new TreeMap<Class<? extends Entity>, String>(
381                new ClassComparator());
382        map.put(State.class, ModalRefinement.class.getName());
383        map.put(ComponentEntity.class, TypedCompositeActor.class.getName());
384        return map;
385    }

418    /** Create a refinement for the given state.
419     *
420     *  @param state The state that will contain the new refinement.
421     *  @param name The name of the composite entity that stores the refinement.
422     *  @param template The template used to create the refinement, or null if
423     *   template is not used.
424     *  @param className The class name for the refinement, which is used when
425     *   template is null.
426     *  @param instanceOpener The instanceOpener, typically a
427     *   Configuration, that is used to open the refinement (as a
428     *   look-inside action) after it is created, or null if it is not
429     *   needed to open the refinement.
430     *  @param container The container.
431     *  @exception IllegalActionException If error occurs while creating the
432     *   refinement.
433     */
434    public static void addRefinement(State state, final String name,
435            Entity template, String className,
436            final InstanceOpener instanceOpener,
437            final CompositeEntity container) throws IllegalActionException {
438
439        // This method is static so that ModalController can call it
440        // and avoid code duplication.
441
442        if (container == null) {
443            throw new IllegalActionException(state,
444                    "State container has no " + "container!");
445        }
446
447        if (container.getEntity(name) != null) {
448            throw new IllegalActionException(state, "There is already a "
449                    + "refinement with name " + name + ".");
450        }
451
452        Attribute allowRefinement = state.getAttribute("_allowRefinement");
453        if (allowRefinement instanceof Parameter
454                && !((BooleanToken) ((Parameter) allowRefinement).getToken())
455                        .booleanValue()) {
456            throw new IllegalActionException(state,
457                    "State does not support " + "refinement.");
458        }
459
460        String currentRefinements = state.refinementName.getExpression();
461
462        if (currentRefinements == null || currentRefinements.equals("")) {
463            currentRefinements = name;
464        } else {
465            currentRefinements = currentRefinements.trim() + ", " + name;
466        }
467
468        String moml;
469
470        // The MoML we create depends on whether the configuration
471        // specified a set of prototype refinements.
472        if (template != null) {
473            String templateDescription = template.exportMoML(name);
474            moml = "<group>" + templateDescription + "<entity name=\""
475                    + state.getName(container)
476                    + "\"><property name=\"refinementName\" value=\""
477                    + currentRefinements + "\"/></entity></group>";
478        } else {
479            moml = "<group><entity name=\"" + name + "\" class=\"" + className
480                    + "\"/>" + "<entity name=\"" + state.getName(container)
481                    + "\"><property name=\"refinementName\" value=\""
482                    + currentRefinements + "\"/></entity></group>";
483        }
484
485        MoMLChangeRequest change = new MoMLChangeRequest(state, container,
486                moml) {
487            @Override
488            protected void _execute() throws Exception {
489                super._execute();
490
491                // Mirror the ports of the container in the refinement.
492                // Note that this is done here rather than as part of
493                // the MoML because we have set protected variables
494                // in the refinement to prevent it from trying to again
495                // mirror the changes in the container.
496                // The following entity is newly created refinement.
497                Entity entity = container.getEntity(name);
498
499                // Get the initial port configuration from the container.
500                Iterator ports = container.portList().iterator();
501
502                while (ports.hasNext()) {
503                    Port port = (Port) ports.next();
504
505                    try {
506                        if (entity instanceof RefinementActor) {
507                            // Prevent the newly created refinement from
508                            // creating a port in its container.
509                            ((RefinementActor) entity).setMirrorDisable(1);
510                        }
511
512                        String name = port.getName();
513                        Port newPort = entity.getPort(name);
514                        if (newPort == null) {
515                            newPort = entity.newPort(port.getName());
516                        }
517
518                        if (newPort instanceof RefinementPort
519                                && port instanceof IOPort) {
520                            try {
521                                ((RefinementPort) newPort)
522                                        .setMirrorDisable(true);
523
524                                if (((IOPort) port).isInput()) {
525                                    ((RefinementPort) newPort).setInput(true);
526                                }
527
528                                if (((IOPort) port).isOutput()) {
529                                    ((RefinementPort) newPort).setOutput(true);
530                                }
531
532                                if (((IOPort) port).isMultiport()) {
533                                    ((RefinementPort) newPort)
534                                            .setMultiport(true);
535                                }
536
537                                // Copy the location to the new port if any.
538                                // (tfeng 08/29/08)
539                                if (container instanceof ModalModel) {
540                                    FSMActor controller = ((ModalModel) container)
541                                            .getController();
542                                    if (controller != null
543                                            && controller != container) {
544                                        Port controllerPort = controller
545                                                .getPort(port.getName());
546                                        if (controllerPort != null) {
547                                            Location location = (Location) controllerPort
548                                                    .getAttribute("_location",
549                                                            Location.class);
550                                            if (location != null) {
551                                                location = (Location) location
552                                                        .clone(newPort
553                                                                .workspace());
554                                                location.setContainer(newPort);
555                                            }
556                                        }
557                                    }
558                                }
559                            } finally {
560                                ((RefinementPort) newPort)
561                                        .setMirrorDisable(false);
562                            }
563                        }
564                    } finally {
565                        if (entity instanceof RefinementActor) {
566                            ((RefinementActor) entity).setMirrorDisable(0);
567                        }
568                    }
569                }
570
571                if (instanceOpener != null) {
572                    // Look inside.
573                    instanceOpener.openAnInstance(entity);
574                }
575            }
576        };
577
578        container.requestChange(change);
579    }
