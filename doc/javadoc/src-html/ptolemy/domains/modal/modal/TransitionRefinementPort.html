045///////////////////////////////////////////////////////////////////
046//// TransitionRefinementPort
047
048/**
049 A port for transition refinements in modal models.  This port
050 mirrors certain changes to it in the ports of the container of the container.
051 That container in turn mirrors those changes in other refinements and/or
052 controllers.
053 <p>
054 For output ports, this class creates a sibling "input" port. The sibling
055 port is normally treated like any other port but mirrors changes via it's
056 output sibling. This is so that TransitionRefinement instances can get the
057 outputs from the State refinements without having the port be an input/output.
058 This sibling port is labeled as "port_in" where port is the name of the
059 corresponding output port.
060
061 @see ModalPort
062 @see TransitionRefinement
063 @author David Hermann, Research In Motion Limited
064 @version $Id$
065 @since Ptolemy II 8.0
066 @Pt.ProposedRating Red (eal)
067 @Pt.AcceptedRating Red (liuxj)
068 */
069public class TransitionRefinementPort extends RefinementPort {
070    /** Construct a port in the given workspace.
071     *  @param workspace The workspace.
072     *  @exception IllegalActionException If the port is not of an acceptable
073     *   class for the container, or if the container does not implement the
074     *   TypedActor interface.
075     *  @exception NameDuplicationException If the name coincides with
076     *   a port already in the container.
077     */
078    public TransitionRefinementPort(Workspace workspace)
079            throws IllegalActionException, NameDuplicationException {
080        super(workspace);
081    }
082
083    /** Construct a port with a containing actor and a name
084     *  that is neither an input nor an output.  The specified container
085     *  must implement the TypedActor interface, or an exception will be
086     *  thrown.
087     *  @param container The container actor.
088     *  @param name The name of the port.
089     *  @exception IllegalActionException If the port is not of an acceptable
090     *   class for the container, or if the container does not implement the
091     *   TypedActor interface.
092     *  @exception NameDuplicationException If the name coincides with
093     *   a port already in the container.
094     */
095    public TransitionRefinementPort(ComponentEntity container, String name)
096            throws IllegalActionException, NameDuplicationException {
097        super(container, name);
098    }
099
100    ///////////////////////////////////////////////////////////////////
101    ////                         public methods                    ////
102
103    /** Override the base class to avoid linking a sibling input port to
104     *  the same relation as the sibling output port multiple times.
105     *  @param relation The relation to link to.
106     *  @exception IllegalActionException If the link crosses levels of
107     *   the hierarchy, or the port has no container, or the relation
108     *   is not an instance of IORelation.
109     */
110    @Override
111    public void link(Relation relation) throws IllegalActionException {
112        if (isInput() && _hasSibling && isLinked(relation)) {
113            return;
114        }
115
116        super.link(relation);
117    }
118
119    /** Override the base class so that if the port is being removed
120     *  from the current container, then it is also removed from the
121     *  controller and from each of the refinements.
122     *  @param container The proposed container.
123     *  @exception IllegalActionException If the proposed container is not a
124     *   ComponentEntity, doesn't implement Actor, or has no name,
125     *   or the port and container are not in the same workspace. Or
126     *   it's not null
127     *  @exception NameDuplicationException If the container already has
128     *   a port with the name of this port.
129     */
130    @Override
131    public void setContainer(Entity container)
132            throws IllegalActionException, NameDuplicationException {
133        NamedObj oldContainer = getContainer();
134
135        if (container == oldContainer) {
136            // Nothing to do.
137            return;
138        }
139
140        boolean disableStatus = _mirrorDisable;
141
142        try {
143            _workspace.getWriteAccess();
144
145            if (_mirrorDisable || getContainer() == null) {
146                // process request for the sibling
147                if (_hasSibling && isOutput() && getContainer() != null) {
148                    TransitionRefinement transContainer = (TransitionRefinement) oldContainer;
149                    TransitionRefinementPort sibling = (TransitionRefinementPort) transContainer
150                            .getPort(getName() + "_in");
151
152                    sibling._mirrorDisable = true;
153                    sibling.setContainer(container);
154                    sibling._mirrorDisable = false;
155                }
156
157                // Have already called the super class.
158                // This time, process the request.
159                super.setContainer(container);
160            } else {
161                // if this is the input port of a pair of siblings,
162                // then forward request to the output port of the pair
163                _mirrorDisable = true;
164
165                boolean success = false;
166                String portName = getName();
167
168                if (_hasSibling && isInput() && !isOutput()) {
169                    // we are the input sibling, extract "real" port name
170                    portName = getName().substring(0, getName().length() - 3);
171                }
172
173                if (oldContainer != null) {
174                    Nameable modal = oldContainer.getContainer();
175
176                    if (modal instanceof ModalModel) {
177                        Port port = ((ModalModel) modal).getPort(portName);
178
179                        if (port != null) {
180                            port.setContainer(null);
181                            success = true;
182                        }
183                    }
184                }
185
186                if (!success) {
187                    super.setContainer(container);
188                }
189            }
190        } finally {
191            _mirrorDisable = disableStatus;
192            _workspace.doneWriting();
193        }
194    }
195
196    /** If the argument is true, make the port an input port.
197     *  If the argument is false, make the port not an input port.
198     *  This method overrides the base class to make the same
199     *  change on the mirror ports in the controller and state refinments.
200     *  This method invalidates the schedule and resolved types of the
201     *  director of the container, if there is one.
202     *  It is write-synchronized on the workspace, and increments
203     *  the version of the workspace.
204     *  @param isInput True to make the port an input.
205     *  @exception IllegalActionException If changing the port status is
206     *   not permitted.
207     */
208    @Override
209    public void setInput(boolean isInput) throws IllegalActionException {
210        boolean disableStatus = _mirrorDisable;
211
212        try {
213            _workspace.getWriteAccess();
214
215            if (_mirrorDisable || getContainer() == null) {
216                // Have already called the super class.
217                // This time, process the request.
218                super.setInput(isInput);
219            } else {
220                _mirrorDisable = true;
221
222                boolean success = false;
223                Nameable container = getContainer();
224
225                if (container != null) {
226                    Nameable modal = container.getContainer();
227
228                    if (modal instanceof ModalModel) {
229                        Port port = ((ModalModel) modal).getPort(getName());
230
231                        if (port instanceof IOPort) {
232                            ((IOPort) port).setInput(isInput);
233                            success = true;
234                        }
235                    }
236                }
237
238                if (!success) {
239                    super.setInput(isInput);
240                }
241            }
242        } finally {
243            _mirrorDisable = disableStatus;
244            _workspace.doneWriting();
245        }
246    }
247
248    /** If the argument is true, make the port a multiport.
249     *  If the argument is false, make the port not a multiport.
250     *  This method overrides the base class to make the same
251     *  change on the mirror ports in the controller and state refinments.
252     *  This method invalidates the schedule and resolved types of the
253     *  director of the container, if there is one.
254     *  It is write-synchronized on the workspace, and increments
255     *  the version of the workspace.
256     *  @param isMultiport True to make the port a multiport.
257     *  @exception IllegalActionException If changing the port status is
258     *   not permitted.
259     */
260    @Override
261    public void setMultiport(boolean isMultiport)
262            throws IllegalActionException {
263        boolean disableStatus = _mirrorDisable;
264
265        try {
266            _workspace.getWriteAccess();
267
268            if (_mirrorDisable || getContainer() == null) {
269                // Have already called the super class.
270                // This time, process the request.
271                // process request for the sibling
272                if (_hasSibling && isOutput() && getContainer() != null) {
273                    TransitionRefinement container = (TransitionRefinement) getContainer();
274                    TransitionRefinementPort sibling = (TransitionRefinementPort) container
275                            .getPort(getName() + "_in");
276
277                    sibling._mirrorDisable = true;
278                    sibling.setMultiport(isMultiport);
279                    sibling._mirrorDisable = false;
280                }
281
282                super.setMultiport(isMultiport);
283            } else {
284                _mirrorDisable = true;
285
286                boolean success = false;
287                Nameable container = getContainer();
288
289                if (container != null) {
290                    Nameable modal = container.getContainer();
291
292                    if (modal instanceof ModalModel) {
293                        Port port = ((ModalModel) modal).getPort(getName());
294
295                        if (port instanceof IOPort) {
296                            ((IOPort) port).setMultiport(isMultiport);
297                            success = true;
298                        }
299                    }
300                }
301
302                if (!success) {
303                    super.setMultiport(isMultiport);
304                }
305            }
306        } finally {
307            _mirrorDisable = disableStatus;
308            _workspace.doneWriting();
309        }
310    }

312    /** Set the name of the port, and mirror the change in all the
313     *  mirror ports.
314     *  This method is write-synchronized on the workspace, and increments
315     *  the version of the workspace.
316     *  @exception IllegalActionException If the name has a period.
317     *  @exception NameDuplicationException If there is already a port
318     *   with the same name in the container.
319     */
320    @Override
321    public void setName(String name)
322            throws IllegalActionException, NameDuplicationException {
323        boolean disableStatus = _mirrorDisable;
324
325        try {
326            _workspace.getWriteAccess();
327
328            if (_mirrorDisable || getContainer() == null) {
329                //change sibling
330                if (_hasSibling && isOutput() && getContainer() != null) {
331                    TransitionRefinement container = (TransitionRefinement) getContainer();
332                    TransitionRefinementPort sibling = (TransitionRefinementPort) container
333                            .getPort(getName() + "_in");
334                    sibling._mirrorDisable = true;
335                    sibling.setName(name + "_in");
336                    sibling._mirrorDisable = false;
337                }
338
339                // Have already called the super class.
340                // This time, process the request.
341                super.setName(name);
342            } else {
343                _mirrorDisable = true;
344
345                boolean success = false;
346                Nameable container = getContainer();
347
348                if (container != null) {
349                    Nameable modal = container.getContainer();
350
351                    if (modal instanceof ModalModel) {
352                        Port port = ((ModalModel) modal).getPort(getName());
353
354                        if (port != null) {
355                            port.setName(name);
356                            success = true;
357                        }
358                    }
359                }
360
361                if (!success) {
362                    super.setName(name);
363                }
364            }
365        } finally {
366            _mirrorDisable = disableStatus;
367            _workspace.doneWriting();
368        }

371    /** If the argument is true, make the port an output port.
372     *  If the argument is false, make the port not an output port.
373     *  In addition, if the container is an instance of Refinement,
374     *  and the argument is true, find the corresponding port of the
375     *  controller and make it an input and not an output.  This makes
376     *  it possible for the controller to see the outputs of the refinements.
377     *  This method overrides the base class to make the same
378     *  change on the mirror ports in the controller and state refinments.
379     *  This method invalidates the schedule and resolved types of the
380     *  director of the container, if there is one.
381     *  It is write-synchronized on the workspace, and increments
382     *  the version of the workspace.
383     *  @param isOutput True to make the port an output.
384     *  @exception IllegalActionException If changing the port status is
385     *   not permitted.
386     */
387    @Override
388    public void setOutput(boolean isOutput) throws IllegalActionException {
389        boolean disableStatus = _mirrorDisable;
390
391        // check first that this isn't an input sibling port,
392        // if it is then it *cannot* be set as an output too
393        if (_hasSibling && isInput() && !isOutput()) {
394            if (isOutput) {
395                throw new InternalErrorException(
396                        "TransitionRefinementPort.setOutput:"
397                                + " cannot set input sibling port to be an output");
398            } else {
399                return;
400            }
401        }
402
403        try {
404            _workspace.getWriteAccess();
405
406            if (_mirrorDisable || getContainer() == null) {
407                // Have already called the super class.
408                // This time, process the request.
409                super.setOutput(isOutput);
410
411                // now create a sibling if we
412                // don't otherwise have one
413                if (!_hasSibling && isOutput) {
414                    try {
415                        TransitionRefinement container = (TransitionRefinement) getContainer();
416                        // Temporarily turn off mirroring while we create a port.
417                        // Needed by $PTII/ptolemy/domains/modal/demo/ModalBSC/ModalBSC.xml
418                        // and $PTII/ptolemy/domains/modal/demo/StateTracker/StateTracker.xml
419                        TransitionRefinementPort sibling = null;
420                        try {
421                            container.setMirrorDisable(-1);
422                            sibling = new TransitionRefinementPort(container,
423                                    getName() + "_in");
424                        } finally {
425                            container.setMirrorDisable(0);
426                        }
427
428                        sibling._hasSibling = true;
429                        sibling._mirrorDisable = true;
430
431                        // set attributes of sibling
432                        sibling.setInput(true);
433                        sibling.setMultiport(isMultiport());
434
435                        sibling._mirrorDisable = false;
436
437                        // link the port relation should already exist
438                        // from this port's creation in newPort()
439                        String relationName = getName() + "Relation";
440                        ModalModel model = (ModalModel) container
441                                .getContainer();
442                        Relation relation = model.getRelation(relationName);
443
444                        if (relation != null) {
445                            sibling.link(relation);
446                        }
447
448                        _hasSibling = true;
449                    } catch (NameDuplicationException ex) {
450                        throw new InternalErrorException(this, ex,
451                                "TransitionRefinementPort.setOutput: Internal error.");
452                    }
453                }
454            } else {
455                _mirrorDisable = true;
456
457                boolean success = false;
458                Nameable container = getContainer();
459
460                if (container != null) {
461                    Nameable modal = container.getContainer();
462
463                    if (modal instanceof ModalModel) {
464                        Port port = ((ModalModel) modal).getPort(getName());
465
466                        if (port instanceof IOPort) {
467                            ((IOPort) port).setOutput(isOutput);
468                            success = true;
469                        }
470                    }
471                }
472
473                if (!success) {
474                    super.setOutput(isOutput);
475                }
476            }
477        } finally {
478            _mirrorDisable = disableStatus;
479            _workspace.doneWriting();
480        }
481    }
