064/**
065 A TypedCompositeActor is an aggregation of typed actors.
066 <p>
067 When exporting MoML, instances of this class identify their class name
068 as TypedCompositeActor. If a derived class does not change this, then it
069 too will be identified as a TypedCompositeActor. To change this in a
070 derived class, put the following line in the constructor:</p>
071 <pre>
072 setClassName(<i>full class name</i>");
073 </pre>
074 <p>If you do this, you will probably also want to override _exportMoMLContents()
075 to not generate a description of the contents of the composite, since
076 they will be already defined in the Java class.
077 </p>
078
079 <p>
080 The ports of a TypedCompositeActor are constrained to be TypedIOPorts,
081 the relations to be TypedIORelations, and the actors to be instances of
082 ComponentEntity that implement the TypedActor interface.  Derived classes
083 may impose further constraints by overriding newPort(), _addPort(),
084 newRelation(), _addRelation(), and _addEntity(). Also, derived classes may
085 constrain the container by overriding _checkContainer().
086 </p>
087
088 @author Yuhong Xiong, Marten Lohstroh
089 @version $Id$
090 @since Ptolemy II 0.2
091 @Pt.ProposedRating Green (yuhong)
092 @Pt.AcceptedRating Green (lmuliadi)
093 @see ptolemy.actor.TypedIOPort
094 @see ptolemy.actor.TypedIORelation
095 @see ptolemy.actor.TypedActor
096 @see ptolemy.kernel.ComponentEntity
097 @see ptolemy.actor.CompositeActor
098 */

178     *  is "enableBackwardTypeInference" then invalidate resolved types.
179     *  @param attribute The attribute that changed.
180     *  @exception IllegalActionException Not thrown in this base class.
181     *   Derived classes can throw this exception if type change is not allowed.
182     */
183    @Override
184    public void attributeChanged(Attribute attribute)
185            throws IllegalActionException {
186        if (attribute.getName().equals("enableBackwardTypeInference")) {
187            Director director = getDirector();

189            if (director != null) {
190                director.invalidateResolvedTypes();
191            }
192        } else {
193            super.attributeChanged(attribute);
194        }

219    /** Return true if backward type inference is enabled.
220     *  If this composite actor is opaque, then it looks for an attribute
221     *  named "enableBackwardTypeInference" inside of it, and if it exists,
222     *  returns its boolean value. If it does not exist, it looks up the
223     *  hierarchy until the top-level for such an attribute, and if it exists,
224     *  again given that the container is opaque, returns its boolean value.
225     *  If it does not exist, then return false.
226     *  For backward compatibility, if it does not exist, then look for a
227     *  parameter named "disableBackwardTypeInference" at the top level and
228     *  return its value, if it exists.
229     *  @return True if backward type inference is enabled.
230     */
231    @Override
232    public boolean isBackwardTypeInferenceEnabled() {
233        try {
234            Parameter backwardTypeInf = (Parameter) getAttribute(
235                    "enableBackwardTypeInference", Parameter.class);
236            if (!isOpaque() || backwardTypeInf == null) {
237                // Look up the hierarchy.
238                NamedObj container = getContainer();
239                if (container instanceof CompositeEntity
240                        && container instanceof TypedActor
241                        && ((CompositeEntity) container).isOpaque()) {
242                    return ((TypedActor) container)
243                            .isBackwardTypeInferenceEnabled();
244                }
245            } else if (isOpaque()) {
246                // Parameter exists.
247                return ((BooleanToken) backwardTypeInf.getToken())
248                        .booleanValue();
249            }
250
251            return false;
252
253        } catch (IllegalActionException e) {
254            // This should not happen
255            throw new InternalErrorException(e);
256        }
257    }

311    /** Do type checking and type resolution on the specified composite actor.
312     *  The specified actor must be the top level container of the model.
313     *  @param topLevel The top level TypedCompositeActor.
314     *  @exception IllegalArgumentException If the specified actor is not the
315     *   top level container. That is, its container is not null.
316     *  @exception TypeConflictException If a type conflict is detected.
317     */
318    public static void resolveTypes(TypedCompositeActor topLevel)
319            throws TypeConflictException {
320        if (topLevel.getContainer() != null) {
321            throw new IllegalArgumentException(
322                    "TypedCompositeActor.resolveTypes: The specified actor is "
323                            + "not the top level container.");
324        }
325
326        try {
327            List<Inequality> conflicts = new LinkedList<Inequality>();
328            List<InequalityTerm> unacceptable = new LinkedList<InequalityTerm>();
329
330            // Check declared types across all connections.
331            List<Inequality> typeConflicts = topLevel._checkDeclaredTypes();
332            conflicts.addAll(typeConflicts);
333
334            // Collect and solve type constraints.
335            Set<Inequality> constraintList = topLevel.typeConstraints();
336
337            // NOTE: To view all type constraints, uncomment these.
338
339            /*
340            Iterator constraintsIterator = constraintList.iterator();
341            while (constraintsIterator.hasNext()) {
342                System.out.println(constraintsIterator.next().toString());
343            }
344             */
345
346            if (constraintList.size() > 0) {
347                CPO cpo = TypeLattice.lattice();
348                InequalitySolver solver = new InequalitySolver(cpo);
349                Iterator<Inequality> constraints = constraintList.iterator();
350
351                solver.addInequalities(constraints);
352
353                try {
354                    // Find the least solution (most specific types)
355                    solver.solveLeast();
356                } catch (InvalidStateException ex) {
357                    throw new InvalidStateException(topLevel, ex,
358                            "Invalid state in type system. The basic type lattice was: "
359                                    + TypeLattice.basicLattice());
360                }
361
362                // If some inequalities are not satisfied, or type variables
363                // are resolved to unacceptable types, such as
364                // BaseType.UNKNOWN, add the inequalities to the list of type
365                // conflicts.
366                Iterator<Inequality> inequalities = constraintList.iterator();
367
368                while (inequalities.hasNext()) {
369                    Inequality inequality = inequalities.next();
370
371                    if (!inequality.isSatisfied(TypeLattice.lattice())) {
372                        conflicts.add(inequality);
373                    } else {
374                        // Check if type variables are resolved to unacceptable
375                        //types
376                        InequalityTerm[] lesserVariables = inequality
377                                .getLesserTerm().getVariables();
378                        InequalityTerm[] greaterVariables = inequality
379                                .getGreaterTerm().getVariables();
380                        boolean added = false;
381
382                        for (InequalityTerm lesserVariable : lesserVariables) {
383                            InequalityTerm variable = lesserVariable;
384
385                            if (!variable.isValueAcceptable()) {
386                                unacceptable.add(variable);
387                                added = true;
388                                break;
389                            }
390                        }
391
392                        if (added == false) {
393                            for (InequalityTerm greaterVariable : greaterVariables) {
394                                InequalityTerm variable = greaterVariable;
395
396                                if (!variable.isValueAcceptable()) {
397                                    unacceptable.add(variable);
398                                    break;
399                                }
400                            }
401                        }
402                    }
403                }
404            }
405
406            if (conflicts.size() > 0) {
407                throw new TypeConflictException(conflicts,
408                        "Type conflicts occurred in " + topLevel.getFullName()
409                                + " on the following inequalities:");
410            }
411            if (unacceptable.size() > 0) {
412                throw new TypeConflictException(unacceptable,
413                        "Types resolved to unacceptable types in "
414                                + topLevel.getFullName()
415                                + " due to the following objects:");
416            }
417        } catch (IllegalActionException ex) {
418            // This should not happen. The exception means that
419            // _checkDeclaredType or typeConstraints is called on a
420            // transparent actor.
421            throw new InternalErrorException(topLevel, ex,
422                    "Type resolution failed because of an error "
423                            + "during type inference");
424        }
425    }

427    /** Return the type constraints of this typed composite actor.
428     *  The constraints have the form of a set of inequalities.  The
429     *  constraints come from three sources, the contained actors, the
430     *  contained Typeables, and (for opaque actors) the topology of
431     *  connections between actors. To generate the constraints based
432     *  on the topology, this method scans all the connections within
433     *  this composite between opaque TypedIOPorts.  If the type of
434     *  the ports on one or both ends of a connection is not declared,
435     *  a type constraint is formed that requires the type of the port
436     *  at the source end of the connection to be less than or equal
437     *  to the type at the destination port.  To collect the type
438     *  constraints from the contained actors, This method recursively
439     *  calls the typeConstraints() method of the contained actors
440     *  and combine all the constraints together.  The type
441     *  constraints from contained Typeables (ports and parameters)
442     *  are collected by calling the typeConstraints() method of
443     *  all the contained Typeables.  <p> This method is
444     *  read-synchronized on the workspace.
445     *  @return a list of instances of Inequality.
446     *  @exception IllegalActionException If the typeConstraints
447     *  of one of the deeply contained objects throws it.
448     *  @see ptolemy.graph.Inequality
449     */
450    @Override
451    public Set<Inequality> typeConstraints() throws IllegalActionException {
452        try {
453            workspace().getReadAccess();
454
455            Set<Inequality> result = new HashSet<Inequality>();
456
457            if (isOpaque()) {
458                Iterator<ComponentEntity> entities = deepEntityList()
459                        .iterator();
460
461                while (entities.hasNext()) {
462                    // Collect type constraints from contained actors.
463                    TypedActor actor = (TypedActor) entities.next();
464
465                    // Collect constraints on all the ports in the contained
466                    // actor to the ports that the actor can send data to.
467                    Iterator<IOPort> ports = actor.outputPortList().iterator();
468
469                    while (ports.hasNext()) {
470                        result.addAll(_destinationTypeConstraints(
471                                (TypedIOPort) ports.next()));
472                    }
473                }
474
475                // Also need to check connection from the input ports on
476                // this composite actor to input ports of contained actors.
477                Iterator<IOPort> boundaryPorts = inputPortList().iterator();
478
479                while (boundaryPorts.hasNext()) {
480                    TypedIOPort sourcePort = (TypedIOPort) boundaryPorts.next();
481                    result.addAll(_destinationTypeConstraints(sourcePort));
482                }
483            }
484
485            // Collect type constraints from contained actors.
486            Iterator entities = entityList().iterator();
487
488            while (entities.hasNext()) {
489                TypedActor actor = (TypedActor) entities.next();
490                result.addAll(actor.typeConstraints());
491            }
492
493            // Collect constraints from contained ports.
494            Iterator ports = portList().iterator();
495
496            while (ports.hasNext()) {
497                Typeable port = (Typeable) ports.next();
498                result.addAll(port.typeConstraints());
499            }
500
501            // Collect constraints from contained attributes.
502            Iterator typeables = attributeList(Typeable.class).iterator();
503
504            while (typeables.hasNext()) {
505                Typeable typeable = (Typeable) typeables.next();
506                result.addAll(typeable.typeConstraints());
507            }
508
509            // Collect constraints from instances of ScopeExtender,
510            // such as ScopeExtendingAttribute.
511            Iterator extenders = attributeList(ScopeExtender.class).iterator();
512
513            while (extenders.hasNext()) {
514                ScopeExtender extender = (ScopeExtender) extenders.next();
515                Iterator extenderAttributes = extender.attributeList()
516                        .iterator();
517                while (extenderAttributes.hasNext()) {
518                    Attribute extenderAttribute = (Attribute) extenderAttributes
519                            .next();
520                    if (extenderAttribute instanceof Variable) {
521                        result.addAll(((Variable) extenderAttribute)
522                                .typeConstraints());
523                    }
524                }
525            }
526
527            return result;
528        } finally {
529            workspace().doneReading();
530        }
531    }

634    /** Check types from a source port to a group of destination ports,
635     *  assuming the source port is connected to all the ports in the
636     *  group of destination ports.  Return a list of instances of
637     *  Inequality that have type conflicts.
638     *  @param sourcePort The source port.
639     *  @param destinationPortList A list of destination ports.
640     *  @return A list of instances of Inequality indicating the
641     *   type constraints that are not satisfied.
642     */
643    protected List<Inequality> _checkTypesFromTo(TypedIOPort sourcePort,
644            List<TypedIOPort> destinationPortList) {
645        List<Inequality> result = new LinkedList<Inequality>();
646
647        boolean isUndeclared = sourcePort.getTypeTerm().isSettable();
648
649        if (!isUndeclared) {
650            // sourcePort has a declared type.
651            Type srcDeclared = sourcePort.getType();
652            Iterator<TypedIOPort> destinationPorts = destinationPortList
653                    .iterator();
654
655            while (destinationPorts.hasNext()) {
656                TypedIOPort destinationPort = destinationPorts.next();
657                isUndeclared = destinationPort.getTypeTerm().isSettable();
658
659                if (!isUndeclared) {
660                    // both source/destination ports are declared,
661                    // check type
662                    Type destDeclared = destinationPort.getType();
663                    int compare = TypeLattice.compare(srcDeclared,
664                            destDeclared);
665
666                    if (compare == CPO.HIGHER || compare == CPO.INCOMPARABLE) {
667                        Inequality inequality = new Inequality(
668                                sourcePort.getTypeTerm(),
669                                destinationPort.getTypeTerm());
670                        result.add(inequality);
671                    }
672                }
673            }
674        }
675
676        return result;
677    }

680    // so we do not override it.

682    /** Return the type constraints on all connections starting from the
683     *  specified source port to all the ports in a group of destination
684     *  ports.
685     *  @param source The source port.
686     *  @return A list of instances of Inequality.
687     */
688    protected List<Inequality> _destinationTypeConstraints(TypedIOPort source) {
689        Iterator<IOPort> destinationPorts;
690        List<Inequality> result = new LinkedList<Inequality>();
691
692        boolean srcUndeclared = source.getTypeTerm().isSettable();
693
694        // NOTE: Do not only check whether the port is an input,
695        // because it can be an input and an output.
696        if (source.isInput() && source.isOutput()) {
697            List<IOPort> sinks = source.sinkPortList();
698            sinks.addAll(source.insideSinkPortList());
699            destinationPorts = sinks.iterator();
700        } else if (source.isInput()) {
701            destinationPorts = source.insideSinkPortList().iterator();
702        } else {
703            destinationPorts = source.sinkPortList().iterator();
704        }
705
706        while (destinationPorts.hasNext()) {
707            TypedIOPort destinationPort = (TypedIOPort) destinationPorts.next();
708            boolean destUndeclared = destinationPort.getTypeTerm().isSettable();
709
710            if (srcUndeclared || destUndeclared) {
711                // At least one of the source/destination ports does
712                // not have declared type, form type constraint.
713                Inequality ineq = new Inequality(source.getTypeTerm(),
714                        destinationPort.getTypeTerm());
715                result.add(ineq);
716            }
717
718        }
719
720        // Set the constraint for backwards type inference where the
721        // source port type is greater than or equal to the GLB of all
722        // its destination ports.
723        // 1) only setup type constraint if source has no type declared
724        if (srcUndeclared) {
725            // 2) only setup type constraint if bidirectional type
726            // inference is enabled.
727            if (isBackwardTypeInferenceEnabled()) {
728                result.add(new Inequality(new GLBFunction(source),
729                        source.getTypeTerm()));
730            }
731        }
732
733        return result;
734    }

739    // If this composite actor is opaque, perform static type checking.
740    // Specifically, this method scans all the connections within this
741    // composite between opaque TypedIOPorts, if the ports on both ends
742    // of the connection have declared types, the types of both ports
743    // are examined to see if the type of the port at the source end
744    // of the connection is less than or equal to the type at the
745    // destination port. If not, the two ports have a type conflict.
746    // If the type of the ports on one or both ends of a connection is
747    // not declared, the connection is skipped by this method and left
748    // to the type resolution mechanism.
749    // This method returns a List of instances of Inequality that have
750    // type conflicts. If no type conflict is detected, an empty
751    // list is returned.
752    // If this TypedCompositeActor contains other opaque
753    // TypedCompositeActors, the _checkDeclaredType() methods of the contained
754    // TypedCompositeActors are called to check types further down the
755    // hierarchy.
756    private List<Inequality> _checkDeclaredTypes()
757            throws IllegalActionException {
758        if (!isOpaque()) {
759            throw new IllegalActionException(this,
760                    "Cannot check types on a non-opaque actor.");
761        }
762
763        List<Inequality> result = new LinkedList<Inequality>();
764
765        Iterator entities = deepEntityList().iterator();
766
767        while (entities.hasNext()) {
768            // Check types on contained actors.
769            TypedActor actor = (TypedActor) entities.next();
770
771            if (actor instanceof TypedCompositeActor) {
772                result.addAll(
773                        ((TypedCompositeActor) actor)._checkDeclaredTypes());
774            }
775
776            // Type check from all the ports on the contained actor.
777            // to the ports that the actor can send data to.
778            Iterator<TypedIOPort> ports = ((Entity<TypedIOPort>) actor)
779                    .portList().iterator();
780
781            while (ports.hasNext()) {
782                TypedIOPort sourcePort = ports.next();
783                Receiver[][] receivers = sourcePort.getRemoteReceivers();
784
785                List destinationPorts = _receiverToPort(receivers);
786                result.addAll(_checkTypesFromTo(sourcePort, destinationPorts));
787            }
788        }
789
790        // Also need to check connection from the input ports on
791        // this composite actor to input ports of contained actors.
792        Iterator boundaryPorts = portList().iterator();
793
794        while (boundaryPorts.hasNext()) {
795            TypedIOPort sourcePort = (TypedIOPort) boundaryPorts.next();
796            Receiver[][] receivers = sourcePort.deepGetReceivers();
797            List destinationPorts = _receiverToPort(receivers);
798            result.addAll(_checkTypesFromTo(sourcePort, destinationPorts));
799        }
800
801        return result;
802    }
803
804    // Return all the ports containing the specified receivers.
805    private List _receiverToPort(Receiver[][] receivers) {
806        List result = new LinkedList();
807
808        if (receivers != null) {
809            for (Receiver[] receiver : receivers) {
810                if (receiver != null) {
811                    for (int j = 0; j < receiver.length; j++) {
812                        result.add(receiver[j].getContainer());
813                    }
814                }
815            }
816        }
817
818        return result;
819    }
