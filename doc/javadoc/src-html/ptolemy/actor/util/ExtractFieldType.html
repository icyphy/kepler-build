028package ptolemy.actor.util;
029
030import ptolemy.actor.TypedIOPort;
031import ptolemy.data.type.AssociativeType;
032import ptolemy.data.type.BaseType;
033import ptolemy.data.type.MonotonicFunction;
034import ptolemy.data.type.Type;
035import ptolemy.graph.InequalityTerm;
036import ptolemy.kernel.util.IllegalActionException;
037
038///////////////////////////////////////////////////////////////////
039////ExtractFieldType
040
041/**
042 A partial monotonic function of the given port that returns a type
043 associated with the given field name, to be found in the type definition
044 of the port.
045 <p>
046 The <code>getValue()</code> method is used to get the current value of the
047 function. If the port type is an <code>AssociateType</code> with a field for
048 the specified field name, then the function returns the type of that field.
049 If the port type is <code>BaseType.GENERAL</code>, then return <code>
050 BaseType.GENERAL</code>, or if the type is <code>BaseType.UNKNOWN</code>,
051 then return <code>BaseType.UNKNOWN</code>.
052 If the port type is <code>AssociateType</code> but it has no
053 corresponding field, then return <code>BaseType.GENERAL</code>.
054 Otherwise, the getValue() method throws an exception, which makes the
055 function partial.
056 </p>
057 @author Edward A. Lee, Marten Lohstroh
058 @version $Id$
059 @since Ptolemy II 10.0
060 @Pt.ProposedRating Red (marten)
061 @Pt.AcceptedRating Red
062 */
063public class ExtractFieldType extends MonotonicFunction {
064
065    /** Construct a new monotonic function.
066     *  @param port The port of which the type is extracted the field from
067     *  @param name The name of the field of interest
068     */
069    public ExtractFieldType(TypedIOPort port, String name) {
070        _name = name;
071        _port = port;
072    }
073
074    ///////////////////////////////////////////////////////////////////
075    ////                   public inner methods                ////
076
077    /** Return the current value of this monotonic function.
078     *  Specifically, this is a function of one variable, the type variable
079     *  of the given port. If the port type is <code>BaseType.GENERAL</code>,
080     *  then return <code>BaseType.GENERAL</code>, or if the type is
081     *  <code>BaseType.UNKNOWN</code>, then return
082     *  <code>BaseType.UNKNOWN</code>. If the port type is
083     *  <code>AssociateType</code> but it has no corresponding field,
084     *  then return <code>BaseType.GENERAL</code>.
085     *  Otherwise, throw an exception.
086     *  @return A Type.
087     *  @exception IllegalActionException If the port type is
088     *   not <code>BaseType.UNKNOWN</code> or <code>BaseType.GENERAL</code>
089     *   and is not an instance of <code>AssociateType</code>.
090     */
091    @Override
092    public Object getValue() throws IllegalActionException {
093        Type portType;
094
095        if (_port.getTypeTerm().isSettable()) {
096            portType = (Type) _port.getTypeTerm().getValue();
097        } else {
098            portType = _port.getType();
099        }
100
101        // If the port type was declared as or did resolve to GENERAL,
102        // return GENERAL
103        if (portType.equals(BaseType.GENERAL)) {
104            return BaseType.GENERAL;
105        }
106
107        // If the port type was declared as or did resolve to UNKNOWN,
108        // return UNKNOWN
109        if (portType.equals(BaseType.UNKNOWN)) {
110            return BaseType.UNKNOWN;
111        }
112
113        // Extract the field type if the port type is a associative type,
114        // or else throw an exception
115        if (portType instanceof AssociativeType) {
116            Type fieldType = ((AssociativeType) portType).get(_name);
117
118            // The field is missing from the associative type
119            if (fieldType == null) {
120                return BaseType.GENERAL; // NOTE: to ensure monotonicity
121            } else {
122                return fieldType;
123            }
124        } else {
125            throw new IllegalActionException(_port,
126                    "Invalid type for given port: " + portType);
127        }
128    }
129
130    /** Return an additional string describing the current value
131     *  of this function.
132     */
133    @Override
134    public String getVerboseString() {
135        if (_port.getType() instanceof AssociativeType) {
136            AssociativeType type = (AssociativeType) _port.getType();
137            Type fieldType = type.get(_name);
138
139            if (fieldType == null) {
140                return "AssociativeType doesn't have field named " + _name;
141            }
142        }
143
144        return null;
145    }
146
147    /** Return the type variables in this inequality term. If the
148     *  type of the input port is not declared, return a one
149     *  element array containing the inequality term representing
150     *  the type of the port; otherwise, return an empty array.
151     *  @return An array of InequalityTerm.
152     */
153    @Override
154    public InequalityTerm[] getVariables() {
155        InequalityTerm portTerm = _port.getTypeTerm();
156
157        if (portTerm.isSettable()) {
158            InequalityTerm[] variable = new InequalityTerm[1];
159            variable[0] = portTerm;
160            return variable;
161        }
162
163        return new InequalityTerm[0];
164    }
165
166    ///////////////////////////////////////////////////////////////////
167    ////                  private inner variable               ////
168
169    /** The name of the field of interest. */
170    private String _name;
171
172    /** The port of which the type is extracted the field type from. */
173    private TypedIOPort _port;
174
175}
