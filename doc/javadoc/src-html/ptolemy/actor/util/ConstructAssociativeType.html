029package ptolemy.actor.util;
030
031import java.util.ArrayList;
032import java.util.Collection;
033import java.util.LinkedList;
034import java.util.List;
035
036import ptolemy.actor.TypedIOPort;
037import ptolemy.data.type.AssociativeType;
038import ptolemy.data.type.MonotonicFunction;
039import ptolemy.data.type.Type;
040import ptolemy.graph.InequalityTerm;
041import ptolemy.kernel.util.IllegalActionException;
042
043/** A function that, given a list of ports, returns a <code>StructuredType</code>
044 *  of which the fields names and field types correspond with the given ports.
045 *  The arguments to this function (the array returned by getVariables())
046 *  represent the types of the ports, in the same order as the ports
047 *  are defined. The concrete subclass of <code>StructuredType</code> that
048 *  is to be instantiated by this function depends on the type parameter
049 *  passed to the constructor.
050 *
051 * @author Edward A. Lee, Marten Lohstroh
052 * @version $Id$
053 * @since Ptolemy II 10.0
054 * @Pt.ProposedRating Red (marten)
055 * @Pt.AcceptedRating Red
056 */
057public class ConstructAssociativeType extends MonotonicFunction {
058
059    /** Construct a new monotonic function.
060     *  @param ports A list of ports used to construct the AssociativeType
061     *  @param type A specific subclass to instantiate
062     */
063    public ConstructAssociativeType(Collection<TypedIOPort> ports,
064            Class<? extends AssociativeType> type) {
065
066        _ports = new LinkedList<TypedIOPort>();
067
068        for (TypedIOPort port : ports) {
069            // only consider ports that are connected
070            if (port.isOutput() && port.numberOfSinks() > 0) {
071                _ports.add(port);
072            } else if (port.isInput() && port.numberOfSources() > 0) {
073                _ports.add(port);
074            }
075        }
076
077        _type = type;
078    }
079
080    ///////////////////////////////////////////////////////////////////
081    ////                       public inner methods            ////
082
083    /** Return a RecordType with field names equal to the given port names,
084     *  and field types equal to the current inferred or declared type of the
085     *  corresponding ports.
086     *  @return A RecordType.
087     *  @exception IllegalActionException If thrown while getting the
088     *  value of the typeTerm of the port or while instantiating a type.
089     *
090     */
091    @Override
092    public Object getValue() throws IllegalActionException {
093        Type[] types = new Type[_ports.size()];
094        String[] labels = new String[_ports.size()];
095
096        int i = 0;
097        for (TypedIOPort port : _ports) {
098            InequalityTerm portTypeTerm = port.getTypeTerm();
099            if (port.getDisplayName() == null
100                    || port.getDisplayName().equals("")) {
101                labels[i] = port.getName();
102            } else {
103                labels[i] = port.getDisplayName();
104            }
105
106            if (portTypeTerm.isSettable()) {
107                types[i] = (Type) portTypeTerm.getValue();
108            } else {
109                types[i] = port.getType();
110            }
111            i++;
112        }
113
114        // construct a new AssociativeType (RecordType, UnionType, ..)
115        Object[] arglist = { labels, types };
116        try {
117            return _type.getConstructor(String[].class, Type[].class)
118                    .newInstance(arglist);
119        } catch (Exception e) {
120            throw new IllegalActionException(e.getCause().getMessage());
121        }
122    }
123
124    /** Return the type variables for this function, which are
125     *  the type variables for all the ports that do not have declared types.
126     *  @return An array of InequalityTerm.
127     */
128    @Override
129    public InequalityTerm[] getVariables() {
130        ArrayList<InequalityTerm> portTypeTermList = new ArrayList<InequalityTerm>();
131        for (TypedIOPort port : _ports) {
132            InequalityTerm portTypeTerm = port.getTypeTerm();
133            if (portTypeTerm.isSettable()) {
134                portTypeTermList.add(portTypeTerm);
135            }
136        }
137        return portTypeTermList
138                .toArray(new InequalityTerm[portTypeTermList.size()]);
139    }
140
141    /** The list of ports used to construct the AssociativeType. */
142    private List<TypedIOPort> _ports;
143
144    /** The specific subclass to instantiate. */
145    private Class<? extends AssociativeType> _type;
146}
