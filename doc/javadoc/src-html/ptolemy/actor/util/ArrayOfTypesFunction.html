029package ptolemy.actor.util;
030
031import ptolemy.data.type.ArrayType;
032import ptolemy.data.type.MonotonicFunction;
033import ptolemy.data.type.Type;
034import ptolemy.data.type.Typeable;
035import ptolemy.graph.InequalityTerm;
036import ptolemy.kernel.util.IllegalActionException;
037
038///////////////////////////////////////////////////////////////////
039//// ArrayOfTypesFunction
040
041/** This class implements a monotonic function that returns an array type
042 *  with the element type equal to its argument.
043 * @author Edward A. Lee, Marten Lohstroh
044 * @version $Id: ArrayOfTypesFunction.java$
045 * @since Ptolemy II 10.0
046 * @Pt.ProposedRating Red (eal)
047 * @Pt.AcceptedRating Red (cxh)
048 */
049public class ArrayOfTypesFunction extends MonotonicFunction {
050
051    /** Construct a ArrayElementTypeFunction whose argument is the
052     *  type of the specified object.
053     *  @param typeable A Typeable object.
054     */
055    public ArrayOfTypesFunction(Typeable typeable) {
056        _typeable = typeable;
057        _arrayLength = -1;
058    }
059
060    /** Construct a ArrayElementTypeFunction whose argument is the
061     *  type of the specified object.
062     *  @param typeable A Typeable object.
063     *  @param arrayLength The length of the array.
064     */
065    public ArrayOfTypesFunction(Typeable typeable, int arrayLength) {
066        _typeable = typeable;
067        _arrayLength = arrayLength;
068    }
069
070    ///////////////////////////////////////////////////////////////////
071    ////                         public methods                    ////
072
073    /** Return the current value of this monotonic function.
074     *  @return A Type.
075     *  @exception IllegalActionException If the type of the argument
076     *   cannot be determined.
077     */
078    @Override
079    public Object getValue() throws IllegalActionException {
080        Type type = _typeable.getType();
081        if (_arrayLength > 0) {
082            return new ArrayType(type, _arrayLength);
083        } else {
084            return new ArrayType(type);
085        }
086    }
087
088    /** Return the type variables for this function, which is
089     *  the type term of the specified typeable, unless it has a constant type,
090     *  in which case return an empty array.
091     *  @return An array of InequalityTerms.
092     */
093    @Override
094    public InequalityTerm[] getVariables() {
095        InequalityTerm term = _typeable.getTypeTerm();
096        if (term.isSettable()) {
097            InequalityTerm[] result = new InequalityTerm[1];
098            result[0] = term;
099            return result;
100        } else {
101            return EMPTY_TERM_ARRAY;
102        }
103    }
104
105    ///////////////////////////////////////////////////////////////////
106    ////                         private variables                 ////
107
108    /** The argument. */
109    private Typeable _typeable;
110
111    /** The length of the array */
112    private int _arrayLength;
113
114    private static InequalityTerm[] EMPTY_TERM_ARRAY = new InequalityTerm[0];
115}
