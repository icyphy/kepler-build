028package ptolemy.actor;
029
030import java.util.HashSet;
031import java.util.Set;
032
033import ptolemy.graph.Inequality;
034import ptolemy.kernel.CompositeEntity;
035import ptolemy.kernel.Port;
036import ptolemy.kernel.util.IllegalActionException;
037import ptolemy.kernel.util.InternalErrorException;
038import ptolemy.kernel.util.NameDuplicationException;
039import ptolemy.kernel.util.Workspace;
040
041///////////////////////////////////////////////////////////////////
042//// TypeOpaqueCompositeActor
043
044/**
045 * A composite actor whose ports have types, but the actors inside are
046 * not required to be typed. This actor does not impose any type
047 * constraints between its ports, so any output ports of this actor
048 * have to have explicitly declared types. There is no mechanism for
049 * inferring the types of the output ports. Note that the ports of
050 * this actor cannot be linked on the inside to untyped relations, and
051 * by default, any relation created on the inside will be untyped.
052 *
053 * <p> The intended use of this actor is for scenarios where the
054 * inside structure is not part of the Ptolemy type system. For
055 * example, the inside structure may define a component that will be
056 * translated into executable code by a code generator. A subclass of
057 * this actor would typically include the code generator an a
058 * mechanism for executing the generating code.
059 *
060 * @author Elaine Cheong and Edward A. Lee
061 * @version $Id$
062 * @since Ptolemy II 4.1
063 * @Pt.ProposedRating Red (cxh)
064 * @Pt.AcceptedRating Red (cxh)
065 * @see ptolemy.actor.TypedCompositeActor
066 * @see ptolemy.actor.TypedIOPort
067 */
068public class TypeOpaqueCompositeActor extends CompositeActor
069        implements TypedActor {
070    /** Construct an actor in the default workspace with an empty string
071     *  as its name.  The object is added to the workspace directory.
072     *  Increment the version number of the workspace.
073     */
074    public TypeOpaqueCompositeActor() {
075        super();
076    }
077
078    /** Construct an actor in the specified workspace with an empty
079     *  string as a name. You can then change the name with setName().
080     *  If the workspace argument is null, then use the default workspace.
081     *  The object is added to the workspace directory.
082     *  Increment the version number of the workspace.
083     *  @param workspace The workspace that will list the entity.
084     */
085    public TypeOpaqueCompositeActor(Workspace workspace) {
086        super(workspace);
087    }
088
089    /** Create a new actor in the specified container with the specified
090     *  name.  The name must be unique within the container or an exception
091     *  is thrown. The container argument must not be null, or a
092     *  NullPointerException will be thrown.
093     *  @param container The container.
094     *  @param name The name of this actor within the container.
095     *  @exception IllegalActionException If this actor cannot be contained
096     *   by the proposed container (see the setContainer() method).
097     *  @exception NameDuplicationException If the name coincides with
098     *   an entity already in the container.
099     */
100    public TypeOpaqueCompositeActor(CompositeEntity container, String name)
101            throws IllegalActionException, NameDuplicationException {
102        super(container, name);
103    }
104
105    ///////////////////////////////////////////////////////////////////
106    ////                         public methods                    ////
107
108    /** Return false because backward type inference is not implemented
109     *  for this actor.
110     *  @return false
111     */
112    @Override
113    public boolean isBackwardTypeInferenceEnabled() {
114        return false;
115    }
116
117    /** Create a new TypedIOPort with the specified name.
118     *  The container of the port is set to this actor.
119     *  This method is write-synchronized on the workspace.
120     *  @param name The name for the new port.
121     *  @return The new port.
122     *  @exception NameDuplicationException If the actor already has a port
123     *   with the specified name.
124     */
125    @Override
126    public Port newPort(String name) throws NameDuplicationException {
127        try {
128            _workspace.getWriteAccess();
129
130            TypedIOPort port = new TypedIOPort(this, name);
131            return port;
132        } catch (IllegalActionException ex) {
133            // This exception should not occur, so we throw a runtime
134            // exception.
135            throw new InternalErrorException(this, ex, null);
136        } finally {
137            _workspace.doneWriting();
138        }
139    }
140
141    /** Return the type constraints of this actor.
142     *  This is always an empty list because the types of ports
143     *  have to be explicitly declared.
144     *  @return An empty list.
145     *  @see ptolemy.graph.Inequality
146     */
147    @Override
148    public Set<Inequality> typeConstraints() {
149        return new HashSet<Inequality>();
150    }
151
152    ///////////////////////////////////////////////////////////////////
153    ////                         protected methods                 ////
154
155    /** Override the base class to throw an exception if the added port
156     *  is not an instance of TypedIOPort.  This method should not be used
157     *  directly.  Call the setContainer() method of the port instead.
158     *  This method does not set the container of the port to point to
159     *  this entity. It assumes that the port is in the same workspace
160     *  as this actor.
161     *  Derived classes may override this method to further constrain the
162     *  port to be a subclass of TypedIOPort. This method is <i>not</i>
163     *  synchronized on the workspace, so the caller should be.
164     *
165     *  @param port The port to add to this actor.
166     *  @exception IllegalActionException If the port is not an instance
167     *   of TypedIOPort, or the port has no name.
168     *  @exception NameDuplicationException If the port name coincides with
169     *   the name of another port already in the actor.
170     */
171    @Override
172    protected void _addPort(Port port)
173            throws IllegalActionException, NameDuplicationException {
174        // In the future, this method can be changed to allow IOPort to be
175        // added. In that case, the type system just ignores instances of
176        // IOPort during type checking. Since there is no intended application
177        // for that change yet, constrain the port to be TypedIOPort for now.
178        if (!(port instanceof TypedIOPort)) {
179            throw new IllegalActionException(this, port,
180                    "Incompatible port class for this actor.");
181        }
182
183        super._addPort(port);
184    }
185}
