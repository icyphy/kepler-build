029package ptolemy.actor.lib;
030
031//////////////////////////////////////////////////////////////////////////
032//// ExceptionSubscriber
033
034/**
035 * An ExceptionSubscriber is an entity that is informed of exceptions and the
036 * handling policy for exceptions caught by
037 * {@link ptolemy.actor.lib.CatchExceptionAttribute}.
038 * The ExceptionSubscriber can then take some action, if desired.
039 * For example, actors that communicate with outside clients may want to pass
040 * along information about the error that has occurred, instead of returning
041 * a generic message or a timeout error response.
042 *
043 * This follows the Command design pattern, where the invoker is
044 * {@link ptolemy.actor.lib.CatchExceptionAttribute}, the client is
045 * the Ptolemy developer (defines commands by dragging and dropping
046 * attributes into the model), and the receiver is a Ptolemy entity
047 * (such as an attribute that writes to a file, or an actor such as
048 * {@link org.ptolemy.ptango.lib.HttpActor} which sends a retry
049 * message upon exception with restart policy.
050 *
051 * @author Elizabeth Latronico
052 * @version $Id: ExceptionSubscriber.java 69467 2014-06-29 14:35:19Z beth@berkeley.edu$
053 * @since Ptolemy II 10.0
054 * @Pt.ProposedRating Red (beth)
055 * @Pt.AcceptedRating Red (beth)
056 * @see ptolemy.actor.lib.CatchExceptionAttribute
057 */
058
059public interface ExceptionSubscriber {
060    /** Invoked by an exception handler (e.g. {@link ptolemy.actor.lib.ExceptionManager}) when an
061     *  exception occurs.  Some subscribers may need to set up access to
062     *  resources (such as opening a file) prior to being notified of an
063     *  exception. These could extend AbstractInitalizableAttribute to do so.
064     *
065     * @param policy  The exception handling policy (for example, restart) of
066     * the exception handler; see {@link ptolemy.actor.lib.CatchExceptionAttribute}
067     */
068
069    /** Action to execute upon the occurrence of an exception.
070     *  Some ExceptionReactions may need to set up and close access to resources
071     *  (such as opening and closing a file).  These could extend
072     *  AbstractInitalizableAttribute to do so.
073     *
074     * @param policy The handling policy of the exception manager (for example,
075     *  restart); see {@link ptolemy.actor.lib.CatchExceptionAttribute}
076     * @param exception The exception
077     * @return True if the subscriber successfully processed the information;
078     *  false otherwise (for example, an ExceptionEmailer that fails to send
079     *  an email)
080     */
081
082    public boolean exceptionOccurred(String policy, Throwable exception);
083
084    /** Invoked by an exception handler (e.g. {@link ptolemy.actor.lib.ExceptionManager}) after
085     *  an exception has been handled.
086     *
087     * @param successful True if the exception was successfully handled; false
088     * otherwise
089     * @param message A status message from the exception handler
090     * @return True if the subscriber successfully processed the information;
091     *  false otherwise (for example, an ExceptionEmailer that fails to send
092     *  an email)
093     */
094    public boolean exceptionHandled(boolean successful, String message);
095}
