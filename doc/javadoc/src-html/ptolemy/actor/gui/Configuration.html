077/**
078 The configuration of an application that uses Ptolemy II classes.
079 An instance of this class is in charge of the user interface,
080 and coordinates multiple views of multiple models. One of its
081 functions, for example, is to manage the opening of new models,
082 ensuring that an appropriate view is used. It also makes sure that
083 if a model is opened that is already open, then existing views are
084 shown rather than creating new views.
085 <p>
086 The applications <i>vergil</i> and <i>moml</i> (at least) use
087 configurations defined in MoML files, typically located in
088 ptII/ptolemy/configs. The <i>moml</i> application takes as
089 command line arguments a list of MoML files, the first of which
090 is expected to define an instance of Configuration and its contents.
091 That configuration is then used to open subsequent MoML files on the
092 command line, and to manage the user interface.
093 <p>
094 Rather than performing all these functions itself, this class
095 is a container for a model directory, effigy factories, and tableau
096 factories that actually realize these functions. An application
097 is configured by populating an instance of this class with
098 a suitable set of these other classes. A minimal configuration
099 defined in MoML is shown below:
100 <pre>
101 &lt;?xml version="1.0" standalone="no"?&gt;
102 &lt;!DOCTYPE entity PUBLIC "-//UC Berkeley//DTD MoML 1//EN"
103 "http://ptolemy.eecs.berkeley.edu/xml/dtd/MoML_1.dtd"&gt;
104 &lt;entity name="configuration" class="ptolemy.actor.gui.Configuration"&gt;
105 &lt;doc&gt;Configuration to run but not edit Ptolemy II models&lt;/doc&gt;
106 &lt;entity name="directory" class="ptolemy.actor.gui.ModelDirectory"/&gt;
107 &lt;entity name="effigyFactory" class="ptolemy.actor.gui.PtolemyEffigy$Factory"/&gt;
108 &lt;property name="tableauFactory" class="ptolemy.actor.gui.RunTableau$Factory"/&gt;
109 &lt;/entity&gt;
110 </pre>
111 <p>
112 It must contain, at a minimum, an instance of ModelDirectory, named
113 "directory", and an instance of EffigyFactory, named "effigyFactory".
114 The openModel() method delegates to the effigy factory the opening of a model.
115 It may also contain an instance of TextEditorTableauFactory, named "tableauFactory".
116 A tableau is a visual representation of the model in a top-level window.
117 The above minimal configuration can be used to run Ptolemy II models
118 by opening a run panel only.
119 <p>
120 When the directory becomes empty (all models have been closed),
121 it removes itself from the configuration. When this happens, the
122 configuration calls System.exit() to exit the application.
123
124 <p>To access the configuration from a random place, if you have a
125 NamedObj <code>foo</code>, then you can call:
126 <pre>
127 Effigy effigy = Configuration.findEffigy(foo.toplevel());
128 Configuration configuration = effigy.toplevel();
129 </pre>
130
131
132 @author Steve Neuendorffer and Edward A. Lee
133 @version $Id$
134 @since Ptolemy II 1.0
135 @Pt.ProposedRating Green (eal)
136 @Pt.AcceptedRating Yellow (celaine)
137 @see EffigyFactory
138 @see ModelDirectory
139 @see Tableau
140 @see TextEditorTableau
141 */

202    /** React to a change in an attribute.
203     *  @param attribute The attribute that changed.
204     *  @exception IllegalActionException If the change is not acceptable
205     *   to this container (not thrown in this base class).
206     */
207    @Override
208    public void attributeChanged(Attribute attribute)
209            throws IllegalActionException {
210        if (attribute == classesToRemove) {
211            // FIXME: There is code duplication here, but
212            // combining the code caused problems because getting
213            // calling getToken() on the other attribute resulted in
214            // attributeChanged() being called for the other token
215            // which caused rentry problems.  Basically, we ended
216            // up with two RemoveGraphicalClasses filters in the list.
217
218            // Find the RemoveClasses element, if any.
219            RemoveClasses removeClassesFilter = null;
220            List momlFilters = MoMLParser.getMoMLFilters();
221            if (momlFilters == null) {
222                momlFilters = new LinkedList();
223            } else {
224                Iterator filters = momlFilters.iterator();
225                while (filters.hasNext()) {
226                    MoMLFilter filter = (MoMLFilter) filters.next();
227                    if (filter instanceof RemoveClasses) {
228                        removeClassesFilter = (RemoveClasses) filter;
229                        break;
230                    }
231                }
232            }
233
234            // Get the token
235            ArrayToken classesToRemoveToken = (ArrayToken) classesToRemove
236                    .getToken();
237            if (removeClassesFilter == null) {
238                // We did not find a RemoveGraphicalClasses, so create one.
239                removeClassesFilter = new RemoveClasses();
240                momlFilters.add(removeClassesFilter);
241            }
242
243            // We always clear
244            RemoveClasses.clear();
245
246            // _classesToRemove is an array of Strings where each element
247            // names a class to be added to the MoMLFilter for removal.
248            for (int i = 0; i < classesToRemoveToken.length(); i++) {
249                String classNameToRemove = ((StringToken) classesToRemoveToken
250                        .getElement(i)).stringValue();
251                removeClassesFilter.put(classNameToRemove, null);
252            }
253
254            MoMLParser.setMoMLFilters(momlFilters);
255        } else if (attribute == removeGraphicalClasses) {
256            // Find the RemoveGraphicalClasses element, if any
257            RemoveGraphicalClasses removeGraphicalClassesFilter = null;
258            List momlFilters = MoMLParser.getMoMLFilters();
259            if (momlFilters == null) {
260                momlFilters = new LinkedList();
261            } else {
262                Iterator filters = momlFilters.iterator();
263                while (filters.hasNext()) {
264                    MoMLFilter filter = (MoMLFilter) filters.next();
265                    if (filter instanceof RemoveGraphicalClasses) {
266                        removeGraphicalClassesFilter = (RemoveGraphicalClasses) filter;
267                        break;
268                    }
269                }
270            }
271            // Get the token
272            BooleanToken removeGraphicalClassesToken = (BooleanToken) removeGraphicalClasses
273                    .getToken();
274            if (removeGraphicalClassesToken.booleanValue()) {
275                if (removeGraphicalClassesFilter == null) {
276                    // We did not find a RemoveGraphicalClasses, so create one.
277                    removeGraphicalClassesFilter = new RemoveGraphicalClasses();
278                    momlFilters.add(removeGraphicalClassesFilter);
279                }
280            } else {
281                if (removeGraphicalClassesFilter != null) {
282                    momlFilters.remove(removeGraphicalClassesFilter);
283                }
284            }
285            MoMLParser.setMoMLFilters(momlFilters);
286        }
287        super.attributeChanged(attribute);
288    }

290    /** Check the configuration for common style problems.
291     *  @return HTML describing the problems
292     *  @exception Exception If there is a problem cloning the configuration.
293     */
294    public String check() throws Exception {
295        StringBuffer results = new StringBuffer();
296        Configuration cloneConfiguration = (Configuration) clone(
297                new Workspace("clonedCheckWorkspace"));
298
299        // Check TypedAtomicActors and Attributes
300        Iterator containedObjects = deepNamedObjList().iterator();
301        while (containedObjects.hasNext()) {
302            NamedObj containedObject = (NamedObj) containedObjects.next();
303            // Check the clone fields on AtomicActors and Attributes.
304            // Note that Director extends Attribute, so we get the
305            // Directors as well
306            if (containedObject instanceof TypedAtomicActor
307                    || containedObject instanceof Attribute) {
308                try {
309                    results.append(checkCloneFields(containedObject));
310                } catch (Throwable throwable) {
311                    throw new InternalErrorException(containedObject, null,
312                            throwable,
313                            "The check for " + "clone methods properly setting "
314                                    + "the fields failed.");
315                }
316            }
317        }
318
319        for (CompositeEntity composite : deepCompositeEntityList()) {
320            CompositeEntity clonedComposite = (CompositeEntity) composite
321                    .clone(new Workspace("clonedCompositeWorkspace"));
322            Iterator attributes = composite.attributeList().iterator();
323            while (attributes.hasNext()) {
324                Attribute attribute = (Attribute) attributes.next();
325                if (!attribute.getClass().isMemberClass()) {
326                    // If an attribute is an inner class, it makes
327                    // no sense to clone to a different workspace because
328                    // the outer class will remain in the old workspace.
329                    // For example, IterateOverArray has an inner class
330                    // (IterateDirector) that is an attribute.
331                    try {
332                        results.append(checkCloneFields(attribute));
333                    } catch (Throwable throwable) {
334                        throw new InternalErrorException(attribute, null,
335                                throwable,
336                                "The check for "
337                                        + "clone methods properly setting "
338                                        + "the fields failed.");
339                    }
340                } else {
341                    // Clone the composite and check that any inner
342                    // classes are properly cloned.
343                    String name = attribute.getName();
344                    Attribute clonedAttribute = clonedComposite
345                            .getAttribute(name);
346                    if (clonedAttribute == null) {
347                        results.append("Could not find \"" + name + "\" in "
348                                + composite.getFullName() + "\n"
349                                + clonedComposite.description());
350                    } else {
351                        if (attribute.workspace()
352                                .equals(clonedAttribute.workspace())) {
353                            results.append("\nIn attribute "
354                                    + attribute.getFullName()
355                                    + ", the workspace is the same in the master and "
356                                    + " the clone?");
357                        }
358                        // Check that the Workspace of the outer class is different.
359                        Class attributeClass = attribute.getClass();
360
361                        if (!Modifier.isStatic(attributeClass.getModifiers())) {
362                            // Get the this$0 field, which refers to the outer class.
363                            Field thisZeroField = null;
364                            try {
365                                thisZeroField = attributeClass
366                                        .getDeclaredField("this$0");
367                            } catch (Throwable throwable) {
368                                results.append("Class "
369                                        + attributeClass.getName()
370                                        + " does not have a this$0 field?\n");
371                            }
372                            if (thisZeroField != null) {
373                                thisZeroField.setAccessible(true);
374
375                                Object outer = thisZeroField.get(attribute);
376                                if (Class
377                                        .forName("ptolemy.kernel.util.NamedObj")
378                                        .isAssignableFrom(outer.getClass())) {
379                                    NamedObj outerNamedObj = (NamedObj) outer;
380                                    NamedObj clonedOuterNamedObj = (NamedObj) thisZeroField
381                                            .get(clonedAttribute);
382                                    if (outerNamedObj.workspace().equals(
383                                            clonedOuterNamedObj.workspace())) {
384                                        results.append(
385                                                "\nAn inner class instance has the same workspace in the outer "
386                                                        + "class in both the original and the cloned attribute."
387                                                        + "\n Attribute:        "
388                                                        + attribute
389                                                                .getFullName()
390                                                        + "\n Cloned attribute: "
391                                                        + clonedAttribute
392                                                                .getFullName()
393                                                        + "\n Outer Workspace:       "
394                                                        + outerNamedObj
395                                                                .workspace()
396                                                        + "\n ClonedOuter Workspace: "
397                                                        + clonedOuterNamedObj
398                                                                .workspace()
399                                                        + "\n Outer Object:        "
400                                                        + outerNamedObj
401                                                                .getFullName()
402                                                        + "\n Cloned Outer Object: "
403                                                        + clonedOuterNamedObj
404                                                                .getFullName());
405                                    }
406                                }
407                            }
408                        }
409                    }
410                }
411            }
412        }
413
414        // Check atomic actors for clone problems related to types
415        List entityList = allAtomicEntityList();
416        Iterator entities = entityList.iterator();
417        while (entities.hasNext()) {
418            Object entity = entities.next();
419            if (entity instanceof TypedAtomicActor) {
420                // Check atomic actors for clone problems
421                try {
422                    results.append(checkCloneFields((TypedAtomicActor) entity));
423                } catch (Throwable throwable) {
424                    throw new InternalErrorException((TypedAtomicActor) entity,
425                            null, throwable,
426                            "The check for " + "clone methods properly setting "
427                                    + "the fields failed.");
428                }
429                TypedAtomicActor actor = (TypedAtomicActor) entity;
430                String fullName = actor.getName(this);
431                TypedAtomicActor clone = (TypedAtomicActor) cloneConfiguration
432                        .getEntity(fullName);
433                if (clone == null) {
434                    results.append("Actor " + fullName + " was not cloned!");
435                } else {
436
437                    // First, check type constraints
438                    Set<Inequality> constraints = actor.typeConstraints();
439                    Set<Inequality> cloneConstraints = clone.typeConstraints();
440
441                    // Make sure the clone has the same number of constraints.
442                    if (constraints.size() != cloneConstraints.size()) {
443                        results.append(actor.getFullName() + " has "
444                                + constraints.size() + " constraints that "
445                                + "differ from " + cloneConstraints.size()
446                                + " constraints its clone has.\n"
447                                + " Constraints: \nMaster Constraints:\n");
448                        Iterator constraintIterator = constraints.iterator();
449                        while (constraintIterator.hasNext()) {
450                            Inequality constraint = (Inequality) constraintIterator
451                                    .next();
452                            results.append(constraint.toString() + "\n");
453                        }
454                        results.append("Clone constraints:\n");
455                        Iterator cloneConstraintIterator = cloneConstraints
456                                .iterator();
457                        while (cloneConstraintIterator.hasNext()) {
458                            Inequality constraint = (Inequality) cloneConstraintIterator
459                                    .next();
460                            results.append(constraint.toString() + "\n");
461                        }
462
463                    }
464
465                    // Make sure the constraints are the same.
466                    // First, iterate through the constraints of the master
467                    // and create a HashSet of string descriptions.
468                    // This is likely to work since the problem is usually
469                    // the the clone is missing constraints.
470                    HashSet<String> constraintsDescription = new HashSet<String>();
471                    try {
472                        Iterator constraintIterator = constraints.iterator();
473                        while (constraintIterator.hasNext()) {
474                            Inequality constraint = (Inequality) constraintIterator
475                                    .next();
476                            constraintsDescription.add(constraint.toString());
477                        }
478                    } catch (Throwable throwable) {
479                        throw new IllegalActionException(actor, throwable,
480                                "Failed to iterate through constraints.");
481                    }
482                    // Make sure that each constraint in the clone is present
483                    // in the master
484                    Iterator cloneConstraintIterator = cloneConstraints
485                            .iterator();
486                    while (cloneConstraintIterator.hasNext()) {
487                        Inequality constraint = (Inequality) cloneConstraintIterator
488                                .next();
489                        if (!constraintsDescription
490                                .contains(constraint.toString())) {
491                            results.append(
492                                    "Master object of " + actor.getFullName()
493                                            + " is missing constraint:\n"
494                                            + constraint.toString() + ".\n");
495                        }
496                    }
497
498                    // Now do the same for the clone.
499                    HashSet<String> cloneConstraintsDescription = new HashSet<String>();
500                    try {
501                        Iterator constraintIterator = cloneConstraints
502                                .iterator();
503                        while (constraintIterator.hasNext()) {
504                            Inequality constraint = (Inequality) constraintIterator
505                                    .next();
506                            cloneConstraintsDescription
507                                    .add(constraint.toString());
508                        }
509                    } catch (Throwable throwable) {
510                        throw new IllegalActionException(actor, throwable,
511                                "Failed to iterate through constraints.");
512                    }
513                    // Make sure that each constraint in the clone is present
514                    // in the master
515                    Iterator constraintIterator = constraints.iterator();
516                    while (constraintIterator.hasNext()) {
517                        Inequality constraint = (Inequality) constraintIterator
518                                .next();
519                        if (!cloneConstraintsDescription
520                                .contains(constraint.toString())) {
521                            results.append("Clone of " + actor.getFullName()
522                                    + " is missing constraint:\n"
523                                    + constraint.toString() + ".\n");
524                        }
525                    }
526
527                    // Check that the type constraint is between ports
528                    // and Parameters of the same object.
529
530                    cloneConstraintIterator = cloneConstraints.iterator();
531                    while (cloneConstraintIterator.hasNext()) {
532                        Inequality constraint = (Inequality) cloneConstraintIterator
533                                .next();
534                        InequalityTerm greaterTerm = constraint
535                                .getGreaterTerm();
536                        InequalityTerm lesserTerm = constraint.getLesserTerm();
537
538                        Object greaterAssociatedObject = greaterTerm
539                                .getAssociatedObject();
540                        Object lesserAssociatedObject = lesserTerm
541                                .getAssociatedObject();
542                        if (greaterAssociatedObject instanceof NamedObj
543                                && lesserAssociatedObject instanceof NamedObj) {
544                            // FIXME: what about non-NamedObjs?
545                            NamedObj greaterNamedObj = (NamedObj) greaterAssociatedObject;
546                            NamedObj lesserNamedObj = (NamedObj) lesserAssociatedObject;
547                            greaterNamedObj.getContainer().getClass().getName();
548                            if (lesserNamedObj != null && greaterNamedObj
549                                    .getContainer() != lesserNamedObj
550                                            .getContainer()
551                            // actor.lib.qm.CompositeQM was causing false
552                            // positives because the contstraints had different
553                            // containers, but were contained within the Composite.
554                                    && greaterNamedObj.getContainer()
555                                            .getContainer() != lesserNamedObj
556                                                    .getContainer()
557                                    // PubSubPort that contains an
558                                    // initialTokens that is used to
559                                    // set the type.
560                                    && !(lesserNamedObj
561                                            .getContainer() instanceof PubSubPort)) {
562                                results.append(clone.getFullName()
563                                        + " has type constraints with "
564                                        + "associated objects that don't have "
565                                        + "the same container:\n"
566                                        + greaterNamedObj.getFullName()
567                                        + " has a container:\n"
568                                        + greaterNamedObj.getContainer() + "\n"
569                                        + lesserNamedObj.getFullName()
570                                        + " has a container:\n"
571                                        + lesserNamedObj.getContainer()
572                                        + "\nThe constraint was: " + constraint
573                                        + "\n"
574                                        + "This can occur if the clone(Workspace) "
575                                        + "method is not present or does not set "
576                                        + "the constraints like the constructor "
577                                        + "does or if a Parameter or Port is not "
578                                        + "declared public.\n");
579                            }
580                        }
581                    }
582                }
583            }
584        }
585        return results.toString();
586    }

588    /** Check that clone(Workspace) method properly sets the fields.
589     *  In a cloned Director, Attribute or Actor, all
590     *  private fields should either point to null or to
591     *  distinct objects.
592     *  @param namedObj The NamedObj, usually a Director, Attribute
593     *  or actor to be checked.
594     *  @return A string containing an error message if there is a problem,
595     *  otherwise return the empty string.
596     *  @exception CloneNotSupportedException If namedObj does not support
597     *  clone(Workspace).
598     *  @exception IllegalAccessException If there is a problem getting
599     *  a field.
600     *  @exception ClassNotFoundException If a class cannot be found.
601     */
602    public static String checkCloneFields(NamedObj namedObj)
603            throws CloneNotSupportedException, IllegalAccessException,
604            ClassNotFoundException {
605        NamedObj namedObjClone = (NamedObj) namedObj.clone(new Workspace());
606
607        StringBuffer results = new StringBuffer();
608        Class namedObjClass = namedObj.getClass();
609
610        // We check only the public, protected and private fields
611        // declared in this class, but not inherited fields.
612        Field[] namedObjFields = namedObjClass.getDeclaredFields();
613        for (Field field : namedObjFields) {
614            results.append(_checkCloneField(namedObj, namedObjClone, field));
615        }
616
617        // Get the fields of the parent classes
618        Class clazz = namedObjClass;
619        while (clazz != NamedObj.class && clazz != null) {
620            clazz = clazz.getSuperclass();
621            namedObjFields = clazz.getDeclaredFields();
622            for (Field field : namedObjFields) {
623                field.setAccessible(true);
624                results.append(
625                        _checkCloneField(namedObj, namedObjClone, field));
626            }
627        }
628        return results.toString();
629    }

631    /** Close all the tableaux.
632     *  @exception IllegalActionException If thrown while accessing the Configuration
633     *  or while closing the tableaux.
634     */
635    public static void closeAllTableaux() throws IllegalActionException {
636        // Based on code by Chad Berkley.
637        Configuration configuration = Configuration.configurations().iterator()
638                .next();
639        // Get the directory from the configuration.
640        ModelDirectory directory = configuration.getDirectory();
641        if (directory == null) {
642            return;
643        }
644
645        Iterator effigies = directory.entityList(Effigy.class).iterator();

665    /** Create the first tableau for the given effigy, using the
666     *  tableau factory.  This is called after an effigy is first opened,
667     *  or when a new effigy is created.  If the method fails
668     *  to create a tableau, then it removes the effigy from the directory.
669     *  This prevents us from having lingering effigies that have no
670     *  user interface.
671     *  @return A tableau for the specified effigy, or null if none
672     *   can be opened.
673     *  @param effigy The effigy for which to create a tableau.
674     */
675    public Tableau createPrimaryTableau(final Effigy effigy) {
676        // NOTE: It used to be that the body of this method was
677        // actually executed later, in the event thread, so that it can
678        // safely interact with the user interface.
679        // However, this does not appear to be necessary, and it
680        // makes it impossible to return the tableau.
681        // So we no longer do this.
682        // If the object referenced by the effigy contains
683        // an attribute that is an instance of TextEditorTableauFactory,
684        // then use that factory to create the tableau.
685        // Otherwise, use the first factory encountered in the
686        // configuration that agrees to represent this effigy.
687        TableauFactory factory = null;
688        if (effigy instanceof PtolemyEffigy) {
689            NamedObj model = ((PtolemyEffigy) effigy).getModel();
690
691            if (model != null) {
692                Iterator factories = model.attributeList(TableauFactory.class)
693                        .iterator();
694
695                // If there are more than one of these, use the first
696                // one that agrees to open the model.
697                while (factories.hasNext() && factory == null) {
698                    factory = (TableauFactory) factories.next();
699
700                    try {
701                        Tableau tableau = factory.createTableau(effigy);
702
703                        if (tableau != null) {
704                            // The first tableau is a master if the container
705                            // of the containing effigy is the model directory.
706                            // Used to do this:
707                            // if (effigy.getContainer() instanceof ModelDirectory) {
708                            if (effigy.masterEffigy() == effigy) {
709                                tableau.setMaster(true);
710                            }
711
712                            tableau.setEditable(effigy.isModifiable());
713                            tableau.show();
714                            return tableau;
715                        }
716                    } catch (Exception ex) {
717                        // Ignore so we keep trying.
718                        // NOTE: Uncomment this line to detect bugs when
719                        // you try to open a model and you get a text editor.
720                        // ex.printStackTrace();
721                        factory = null;
722                    }
723                }
724            }
725        }
726
727        // Defer to the configuration.
728        // Create a tableau if there is a tableau factory.
729        factory = (TableauFactory) getAttribute("tableauFactory");
730
731        if (factory != null) {
732            // If this fails, we do not want the effigy to linger
733            try {
734                Tableau tableau = factory.createTableau(effigy);
735
736                if (tableau == null) {
737                    throw new Exception("Tableau factory returns null.");
738                }
739
740                // The first tableau is a master if the container
741                // of the containing effigy is the model directory.
742                if (effigy.getContainer() instanceof ModelDirectory) {
743                    tableau.setMaster(true);
744                }
745
746                tableau.setEditable(effigy.isModifiable());
747                tableau.show();
748                return tableau;
749            } catch (Exception ex) {
750                // NOTE: Uncomment this line to detect bugs when
751                // you try to open a model and you get a text editor.
752                // ex.printStackTrace();
753                // Remove the effigy.  We were unable to open a tableau for it.
754
755                // If we have a link to a missing .htm file, we want to
756                // avoid popping up two MessageHandlers.
757                boolean calledMessageHandler = false;
758                try {
759                    if (effigy == null) {
760                        // Coverity Scan 1352685 warned that effigy can be null.
761                        throw new InternalErrorException(
762                                "createPrimaryTableau() "
763                                        + "called with a null Effigy?");
764                    } else if (effigy
765                            .getContainer() instanceof ModelDirectory) {
766                        // This is the master.
767                        // Calling setContainer() = null will exit, so
768                        // we display the error message here.
769                        //
770                        // We will get to here if
771                        // diva.graph.AbstractGraphController.rerender()
772                        // throws an NullPointerException when starting
773                        // vergil.
774                        if (effigy instanceof PtolemyEffigy) {
775                            if (((PtolemyEffigy) effigy).getModel() != null) {
776                                MessageHandler.error("Failed to open "
777                                        + ((PtolemyEffigy) effigy).getModel()
778                                                .getFullName(),
779                                        ex);
780                            } else {
781                                MessageHandler.error("Failed to open " + effigy,
782                                        ex);
783                            }
784                            calledMessageHandler = true;
785                        } else {
786                            // Opening a link to a non-existent .htm file
787                            // might get us to here because the effigy is
788                            // not a PtolemyEffigy.
789                            //
790                            // Note that because we call MessageHandler here,
791                            // this means that putting a try/catch around
792                            // configuration.openModel() does not do much good
793                            // if a .htm file is not found because the
794                            // MessageHandler pops up before we return
795                            // from the exception.
796                            // In addition, we cannot catch HeadlessExceptions.
797
798                            MessageHandler.error(
799                                    "Failed to open "
800                                            + effigy.identifier.getExpression(),
801                                    ex);
802                            calledMessageHandler = true;
803                        }
804                    }
805
806                    // This eventually calls Configuration._removeEntity()
807                    // which calls System.exit();
808                    effigy.setContainer(null);
809                } catch (Throwable throwable) {
810                    calledMessageHandler = false;
811                    throw new InternalErrorException(this, throwable, null);
812                }
813
814                // As a last resort, attempt to open source code
815                // associated with the object.
816                if (effigy instanceof PtolemyEffigy) {
817                    NamedObj object = ((PtolemyEffigy) effigy).getModel();
818
819                    // Source code is found by name.
820                    String filename = StringUtilities
821                            .objectToSourceFileName(object);
822
823                    try {
824                        URL toRead = getClass().getClassLoader()
825                                .getResource(filename);
826
827                        // If filename was not found in the classpath, then search
828                        // each element in the classpath for a directory named
829                        // src and then search for filename.  This is needed
830                        // by Eclipse for Kepler.  See also TextEffigy.newTextEffigy()
831                        if (toRead == null) {
832                            toRead = ClassUtilities.sourceResource(filename);
833                            System.out.println("Configuration: sourceResource "
834                                    + filename + " " + toRead);
835                        }
836
837                        if (toRead != null) {
838                            return openModel(null, toRead,
839                                    toRead.toExternalForm());
840                        } else {
841                            MessageHandler.error(
842                                    "Cannot find a tableau or the source code for "
843                                            + object.getFullName());
844                        }
845                    } catch (Exception exception) {
846                        MessageHandler
847                                .error("Failed to open the source code for "
848                                        + object.getFullName(), exception);
849                    }
850                }
851
852                // Note that we can't rethrow the exception here
853                // because removing the effigy may result in
854                // the application exiting.
855                if (!calledMessageHandler) {
856                    MessageHandler.error("Failed to open tableau for "
857                            + effigy.identifier.getExpression(), ex);
858                }
859            }
860        }
861        return null;
862    }

1037    /** Open the specified URL using the specified effigy factory.
1038     *  If a model with the specified identifier is present in the directory,
1039     *  then find all the tableaux of that model and make them
1040     *  visible; otherwise, read a model from the specified URL <i>in</i>
1041     *  and create a default tableau for the model and add the tableau
1042     *  to this directory.
1043     *  @param base The base for relative file references, or null if
1044     *   there are no relative file references.
1045     *  @param in The input URL.
1046     *  @param identifier The identifier that uniquely identifies the model.
1047     *  @param factory The effigy factory to use.
1048     *  @return The tableau that is created, or null if none.
1049     *  @exception Exception If the URL cannot be read.
1050     */
1051    public Tableau openModel(URL base, URL in, String identifier,
1052            EffigyFactory factory) throws Exception {
1053        ModelDirectory directory = (ModelDirectory) getEntity(_DIRECTORY_NAME);
1054
1055        if (directory == null) {
1056            throw new InternalErrorException("No model directory!");
1057        }
1058
1059        // Check to see whether the model is already open.
1060        Effigy effigy = directory.getEffigy(identifier);
1061
1062        if (effigy == null) {
1063            // No previous effigy exists that is identified by this URL.
1064            // Find an effigy factory to read it.
1065            if (factory == null) {
1066                // Check to see whether the URL includes the special
1067                // target "#in_browser", and if so, use a BrowserEffigy factory.
1068                // NOTE: This used to be handled only by HTMLViewer, but
1069                // this limited where the #in_browser notation could be used.
1070                // The following is adapted from HTMLViewer.
1071
1072                // NOTE: It would be nice to use target="_browser" or some
1073                // such, but this doesn't work. Targets aren't
1074                // seen unless the link is inside a frame,
1075                // regrettably.  An alternative might be to
1076                // use the "userInfo" part of the URL,
1077                // defined at http://www.ncsa.uiuc.edu/demoweb/url-primer.html
1078                boolean useBrowser = false;
1079                String ref = in.getRef();
1080                if (ref != null) {
1081                    useBrowser = ref.equals("in_browser");
1082                }
1083                String protocol = in.getProtocol();
1084                if (protocol != null) {
1085                    // Suggested mailto: extension from Paul Lieverse.
1086                    // Unfortunately, it doesn't work for me on a Mac.
1087                    // Leaving it here in case it works for someone else.
1088                    useBrowser |= protocol.equals("mailto");
1089                }
1090                if (useBrowser) {
1091                    if (BrowserEffigy.staticFactory == null) {
1092                        // The following will set BrowserEffigy.staticFactory.
1093                        new BrowserEffigy.Factory(this, "browserEffigyFactory");
1094                    }
1095                    factory = BrowserEffigy.staticFactory;
1096                } else {
1097                    // Not a browser or mailto URL.
1098                    // Get an effigy factory from this configuration.
1099                    factory = (EffigyFactory) getEntity("effigyFactory");
1100                }
1101            }
1102
1103            if (factory == null) {
1104                throw new InternalErrorException(
1105                        "No effigy factories in the configuration!");
1106            }
1107
1108            effigy = factory.createEffigy(directory, base, in);
1109
1110            if (effigy == null) {
1111                MessageHandler.error(
1112                        "Unsupported file type or connection not available: "
1113                                + in.toExternalForm());
1114                return null;
1115            }
1116
1117            if (effigy.identifier.getExpression().compareTo("Unnamed") == 0) {
1118                // If the value identifier field of the effigy we just
1119                // created is "Unnamed", then set it to the value of
1120                // the identifier parameter.
1121                //
1122                // HSIFEffigyFactory sets effiigy.identifier because it
1123                // converts the file we specified from HSIF to MoML and then
1124                // opens up a file other than the one we specified.
1125                effigy.identifier.setExpression(identifier);
1126            }
1127
1128            // Check the URL to see whether it is a file,
1129            // and if so, whether it is writable.
1130            if (in != null && in.getProtocol().equals("file")) {
1131                String filename = in.getFile();
1132                File file = new File(filename);
1133
1134                try {
1135                    if (!file.canWrite()) {
1136                        // FIXME: we need a better way to check if
1137                        // a URL is writable.
1138
1139                        // Sigh.  If the filename has spaces in it,
1140                        // then the URL will have %20s.  However,
1141                        // the file does not have %20s.
1142                        // See
1143                        // https://chess.eecs.berkeley.edu/bugzilla/show_bug.cgi?id=153
1144                        filename = StringUtilities.substitute(filename, "%20",
1145                                " ");
1146                        file = new File(filename);
1147                        if (!file.canWrite()) {
1148                            effigy.setModifiable(false);
1149                        }
1150                    }
1151                } catch (java.security.AccessControlException accessControl) {
1152                    // If we are running in a sandbox, then canWrite()
1153                    // may throw an AccessControlException.
1154                    effigy.setModifiable(false);
1155                }
1156            } else {
1157                effigy.setModifiable(false);
1158            }
1159
1160            return createPrimaryTableau(effigy);
1161        } else {
1162            // Model already exists.
1163            return effigy.showTableaux();
1164        }
1165    }

1305    /** Check that clone(Workspace) method properly sets the fields.
1306     *  In a cloned Director, Attribute or Actor, all
1307     *  private fields should either point to null or to
1308     *  distinct objects.
1309     *  @param namedObj The NamedObj, usually a Director, Attribute
1310     *  or actor to be checked.
1311     *  @param namedObjClone the clone of the namedObj, created with
1312     *  clone(new Workspace())
1313     *  @param field The field to be checked.
1314     *  @return A string containing an error message if there is a problem,
1315     *  otherwise return the empty string.
1316     *  @exception CloneNotSupportedException If namedObj does not support
1317     *  clone(Workspace).
1318     *  @exception IllegalAccessException If there is a problem getting
1319     *  a field.
1320     *  @exception ClassNotFoundException If a class cannot be found.
1321     */
1322    private static String _checkCloneField(NamedObj namedObj,
1323            NamedObj namedObjClone, Field field)
1324            throws CloneNotSupportedException, IllegalAccessException,
1325            ClassNotFoundException {
1326        Class namedObjClass = namedObj.getClass();
1327        StringBuffer results = new StringBuffer();
1328        // Tell the security manager we want to read private fields.
1329        // This will fail in an applet.
1330        field.setAccessible(true);
1331        Class fieldType = field.getType();
1332        if (!fieldType.isPrimitive() && field.get(namedObj) != null
1333                && !Modifier.isStatic(field.getModifiers())
1334                && !Modifier.isStatic(fieldType.getModifiers()) //matlab.Engine.ConversionParameters.
1335                /*&& !fieldType.isArray()*/
1336                // Skip fields introduced by javascope
1337                && !fieldType.toString().equals(
1338                        "COM.sun.suntest.javascope.database.CoverageUnit")
1339                && !field.getName().equals("js$p")
1340                // Skip fields introduced by backtracking
1341                && !(field.getName().indexOf("$RECORD$") != -1)
1342                && !(field.getName().indexOf("$RECORDS") != -1)
1343                && !(field.getName().indexOf("$CHECKPOINT") != -1)
1344                // Skip dependency injection fields
1345                && !(field.getName().indexOf("_implementation") != -1)
1346                // Skip immutables
1347                && !fieldType.equals(java.net.InetAddress.class)
1348                && !fieldType.equals(java.util.regex.Pattern.class)
1349                // SharedParameter has a _containerClass field
1350                && !fieldType.equals(Boolean.class)
1351                && !fieldType.equals(Class.class)
1352                && !fieldType.equals(String.class)
1353                && !fieldType.equals(Token.class)
1354                // Variable has various type fields
1355                && !fieldType.equals(ptolemy.data.type.Type.class)
1356                && !fieldType.equals(Settable.Visibility.class)) {
1357
1358            // If an object is equal and the default hashCode() from
1359            // Object is the same, then we have a problem.
1360            if (field.get(namedObj).equals(field.get(namedObjClone))
1361                    && System.identityHashCode(field.get(namedObj)) == System
1362                            .identityHashCode(field.get(namedObjClone))) {
1363
1364                String message = "";
1365                if (Class.forName("ptolemy.kernel.util.NamedObj")
1366                        .isAssignableFrom(fieldType)) {
1367                    NamedObj fieldNamedObj = (NamedObj) Class
1368                            .forName("ptolemy.kernel.util.NamedObj")
1369                            .cast(field.get(namedObj));
1370                    NamedObj cloneNamedObj = (NamedObj) Class
1371                            .forName("ptolemy.kernel.util.NamedObj")
1372                            .cast(field.get(namedObjClone));
1373                    message = "Field: " + fieldNamedObj.workspace().getName()
1374                            + " Clone: " + cloneNamedObj.workspace().getName();
1375                }
1376
1377                // Determine what code should go in clone(W)
1378                String assignment = field.getName();
1379                // FIXME: extend this to more types
1380                if (Class.forName("ptolemy.kernel.Port")
1381                        .isAssignableFrom(fieldType)) {
1382                    assignment = ".getPort(\"" + assignment + "\")";
1383                    //                       } else if (fieldType.isInstance( new Attribute())) {
1384                } else if (Class.forName("ptolemy.kernel.util.Attribute")
1385                        .isAssignableFrom(fieldType)) {
1386                    Attribute fieldAttribute = (Attribute) field
1387                            .get(namedObjClone);
1388
1389                    if (fieldAttribute.getContainer() != namedObjClone) {
1390                        // If the attribute is actually contained by a Port
1391                        // and not by the AtomicActor, then get its value.
1392                        // SDF actors that have ports that have
1393                        // tokenConsumptionRate and tokenProductionRate
1394                        // such as ConvolutionalCoder need this.
1395                        assignment = "."
1396                                + fieldAttribute.getContainer().getName()
1397                                + ".getAttribute(\"" + fieldAttribute.getName()
1398                                + "\")";
1399                    } else {
1400                        assignment = ".getAttribute(\"" + assignment + "\")";
1401                    }
1402                } else {
1403                    assignment = "\n\t/* Get the object method "
1404                            + "or null?  */ " + assignment;
1405                }
1406
1407                String shortClassName = field.getType().getName().substring(
1408                        field.getType().getName().lastIndexOf(".") + 1);
1409
1410                //new Exception("Configuration._checkCloneField()").printStackTrace();
1411
1412                results.append("The " + field.getName() + " "
1413                        + field.getType().getName() + " field"
1414                        + "\n\tin the clone of \"" + namedObjClass.getName()
1415                        + "\"\n\tdoes not point to an "
1416                        + "object distinct from the "
1417                        + "master.  \n\tThis may cause problems with "
1418                        + "actor oriented classes."
1419                        + "\n\tThe clone(Workspace) "
1420                        + "method should have a line " + "like:\n newObject."
1421                        + field.getName() + " = (" + shortClassName
1422                        + ")newObject" + assignment + ";\n" + message);
1423            }
1424
1425        }
1426        return results.toString();
1427
1428    }

1430    /** Return an identifier for the specified effigy based on its
1431     *  container (if any) and its name.
1432     *  @return An identifier for the effigy.
1433     */
1434    private String _effigyIdentifier(Effigy effigy, NamedObj entity) {
1435        // Set the identifier of the effigy to be that
1436        // of the parent with the model name appended.
1437        NamedObj parent = effigy.getContainer();
1438
1439        if (!(parent instanceof Effigy)) {
1440            return effigy.getFullName();
1441        }
1442
1443        Effigy parentEffigy = (Effigy) parent;
1444
1445        // Note that we add a # the first time, and
1446        // then add . after that.  So
1447        // file:/c:/foo.xml#bar.bif is ok, but
1448        // file:/c:/foo.xml#bar#bif is not
1449        // If the title does not contain a legitimate
1450        // way to reference the submodel, then the user
1451        // is likely to look at the title and use the wrong
1452        // value if they xml edit files by hand. (cxh-4/02)
1453        String entityName = parentEffigy.identifier.getExpression();
1454        String separator = "#";
1455
1456        if (entityName.indexOf("#") >= 0) {
1457            separator = ".";
1458        }
1459
1460        return entityName + separator + entity.getName();
1461    }

1463    // Recursively search the specified composite for an instance of
1464    // PtolemyEffigy that matches the specified model.
1465    private PtolemyEffigy _findEffigyForModel(CompositeEntity composite,
1466            NamedObj model) {
1467        if (composite != null) {
1468            Iterator effigies = composite.entityList(PtolemyEffigy.class)
1469                    .iterator();
1470
1471            while (effigies.hasNext()) {
1472                PtolemyEffigy effigy = (PtolemyEffigy) effigies.next();
1473
1474                // First see whether this effigy matches.
1475                if (effigy.getModel() == model) {
1476                    return effigy;
1477                }
1478
1479                // Then see whether any effigy inside this one matches.
1480                PtolemyEffigy inside = _findEffigyForModel(effigy, model);
1481
1482                if (inside != null) {
1483                    return inside;
1484                }
1485            }
1486        }
1487
1488        return null;
1489    }

1491    /** Open the specified model without deferring to its class definition.
1492     *  @param entity The model to open.
1493     *  @param container The container for any new effigy.
1494     *  @return The tableau that is created, or the first one found,
1495     *   or null if none is created or found.
1496     *  @exception IllegalActionException If constructing an effigy or tableau
1497     *   fails.
1498     *  @exception NameDuplicationException If a name conflict occurs (this
1499     *   should not be thrown).
1500     */
1501    private Tableau _openModel(NamedObj entity, CompositeEntity container)
1502            throws IllegalActionException, NameDuplicationException {
1503        if (entity == null) {
1504            throw new IllegalActionException("Nothing to open.");
1505        }
1506        // Search the model directory for an effigy that already
1507        // refers to this model.
1508        PtolemyEffigy effigy = getEffigy(entity);
1509
1510        if (effigy != null) {
1511            // Found one.  Display all open tableaux.
1512            return effigy.showTableaux();
1513        } else {
1514            // There is no pre-existing effigy.  Create one.
1515            effigy = new PtolemyEffigy(workspace());
1516            effigy.setModel(entity);
1517
1518            // Look to see whether the model has a URIAttribute.
1519            List attributes = entity.attributeList(URIAttribute.class);
1520
1521            if (attributes.size() > 0) {
1522                // The entity has a URI, which was probably
1523                // inserted by MoMLParser.
1524                URI uri = ((URIAttribute) attributes.get(0)).getURI();
1525
1526                // Set the URI and identifier of the effigy.
1527                effigy.uri.setURI(uri);
1528
1529                // NOTE: The uri might be null, which results in
1530                // a null pointer exception below. In particular,
1531                // the class Effigy always has a URI attribute, but
1532                // the value might not get set.
1533                if (uri == null) {
1534                    effigy.identifier
1535                            .setExpression(_effigyIdentifier(effigy, entity));
1536                } else {
1537                    effigy.identifier.setExpression(uri.toString());
1538                }
1539
1540                if (container == null) {
1541                    // Put the effigy into the directory
1542                    ModelDirectory directory = getDirectory();
1543                    effigy.setName(directory.uniqueName(entity.getName()));
1544                    effigy.setContainer(directory);
1545                } else {
1546                    effigy.setName(container.uniqueName(entity.getName()));
1547                    effigy.setContainer(container);
1548                }
1549
1550                // Create a default tableau.
1551                return createPrimaryTableau(effigy);
1552            } else {
1553                // If we get here, then we are looking inside a model
1554                // that is defined within the same file as the parent,
1555                // probably.  Create a new PtolemyEffigy
1556                // and open a tableau for it.
1557                // Put the effigy inside the effigy of the parent,
1558                // rather than directly into the directory.
1559                NamedObj parent = entity.getContainer();
1560                PtolemyEffigy parentEffigy = null;
1561
1562                // Find the first container above in the hierarchy that
1563                // has an effigy.
1564                while (parent != null && parentEffigy == null) {
1565                    parentEffigy = getEffigy(parent);
1566                    parent = parent.getContainer();
1567                }
1568
1569                boolean isContainerSet = false;
1570
1571                if (parentEffigy != null) {
1572                    // OK, we can put it into this other effigy.
1573                    effigy.setName(parentEffigy.uniqueName(entity.getName()));
1574                    effigy.setContainer(parentEffigy);
1575
1576                    // Set the uri of the effigy to that of
1577                    // the parent.
1578                    effigy.uri.setURI(parentEffigy.uri.getURI());
1579
1580                    // Indicate success.
1581                    isContainerSet = true;
1582                }
1583
1584                // If the above code did not find an effigy to put
1585                // the new effigy within, then put it into the
1586                // directory directly or the specified container.
1587                if (!isContainerSet) {
1588                    if (container == null) {
1589                        CompositeEntity directory = getDirectory();
1590                        effigy.setName(directory.uniqueName(entity.getName()));
1591                        effigy.setContainer(directory);
1592                    } else {
1593                        effigy.setName(container.uniqueName(entity.getName()));
1594                        effigy.setContainer(container);
1595                    }
1596                }
1597
1598                effigy.identifier
1599                        .setExpression(_effigyIdentifier(effigy, entity));
1600
1601                return createPrimaryTableau(effigy);
1602            }
1603        }
1604    }
