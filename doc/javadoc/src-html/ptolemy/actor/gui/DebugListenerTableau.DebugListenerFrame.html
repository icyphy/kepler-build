027package ptolemy.actor.gui;
028
029import java.awt.event.WindowAdapter;
030import java.awt.event.WindowEvent;
031
032import ptolemy.kernel.util.DebugEvent;
033import ptolemy.kernel.util.DebugListener;
034import ptolemy.kernel.util.Debuggable;
035import ptolemy.kernel.util.IllegalActionException;
036import ptolemy.kernel.util.NameDuplicationException;
037import ptolemy.kernel.util.NamedObj;
038
039///////////////////////////////////////////////////////////////////
040//// DebugListenerTableau
041
042/**
043 A tableau representing a debug listener window. The constructor of this
044 class creates the window.  You must then attached any object implementing
045 the Debuggable interface using the setDebuggable() method.
046 Once attached, the window will display any debug messages produced by
047 that object.  The listener window itself is an instance of the inner class
048 DebugListenerFrame, which extends TextEditor, and can be
049 accessed using the getFrame() method. As with other tableaux,
050 this is an entity that is contained by an effigy of the model.
051 There can be any number of instances of this class in an effigy.
052
053 @author  Steve Neuendorffer and Edward A. Lee, Elaine Cheong
054 @version $Id$
055 @since Ptolemy II 1.0
056 @Pt.ProposedRating Yellow (eal)
057 @Pt.AcceptedRating Red (cxh)
058 @see Effigy
059 */
060public class DebugListenerTableau extends Tableau {
061    /** Construct a new tableau for the model represented by the given effigy.
062     *  @param container The container.
063     *  @param name The name.
064     *  @exception IllegalActionException If the container does not accept
065     *   this entity (this should not occur).
066     *  @exception NameDuplicationException If the name coincides with an
067     *   attribute already in the container.
068     */
069    public DebugListenerTableau(Effigy container, String name)
070            throws IllegalActionException, NameDuplicationException {
071        super(container, name);
072
073        final DebugListenerFrame frame = new DebugListenerFrame();
074        setFrame(frame);
075        frame.setTableau(this);
076
077        // Listen for window closing events to unregister.
078        frame.addWindowListener(new WindowAdapter() {
079            @Override
080            public void windowClosing(WindowEvent e) {
081                setDebuggable(null);
082            }
083        });
084
085        if (container instanceof TextEffigy) {
086            ((TextEffigy) container).setDocument(frame.text.getDocument());
087        }
088
089        frame.setVisible(true);
090        frame.pack();
091    }
092
093    ///////////////////////////////////////////////////////////////////
094    ////                         public methods                    ////
095
096    /** Return the object that this tableau is listening to, or null
097     *  if none has been set.
098     *  @return The current debuggable.
099     *  @see #setDebuggable(Debuggable)
100     */
101    public Debuggable getDebuggable() {
102        return _debug;
103    }
104
105    /** Set the object for this tableau to listen to, or null to stop
106     *  listening to the current one.  If this tableau is already
107     *  listening to an object, then first stop listening to that
108     *  object.  Set the title of the window to show name of object.
109     *  @param debuggable The object to listen to.
110     *  @see #getDebuggable()
111     */
112    public void setDebuggable(Debuggable debuggable) {
113        if (_debug != null) {
114            _debug.removeDebugListener((DebugListenerFrame) getFrame());
115        }
116
117        _debug = debuggable;
118
119        if (_debug != null) {
120            DebugListenerFrame debugFrame = (DebugListenerFrame) getFrame();
121            _debug.addDebugListener(debugFrame);
122
123            debugFrame.setTitle(((NamedObj) _debug).getFullName());
124        }
125    }
126
127    ///////////////////////////////////////////////////////////////////
128    ////                         private members                   ////
129    // The object this is listening to.
130    private Debuggable _debug;
131
132    ///////////////////////////////////////////////////////////////////
133    ////                         inner classes                     ////
134
135    /** This class is a top-level window for displaying textual debug
136     *  output from an instance of Debuggable.
137     */
138    @SuppressWarnings("serial")
139    public static class DebugListenerFrame extends TextEditor
140            implements DebugListener {
141        /** Create a debug listener that displays messages in a top-level
142         *  window.
143         */
144        public DebugListenerFrame() {
145            super();
146            text.setEditable(false);
147            text.setColumns(80);
148            text.setRows(20);
149            pack();
150        }
151
152        ///////////////////////////////////////////////////////////////////
153        ////                         public methods                    ////
154
155        /** Display a string representation of the specified event.
156         */
157        @Override
158        public void event(DebugEvent event) {
159            text.append(event.toString() + "\n");
160            scrollToEnd();
161        }
162
163        /** Display the specified message.
164         */
165        @Override
166        public void message(String message) {
167            text.append(message + "\n");
168            scrollToEnd();
169        }
170    }
171}
