072    /** Abbreviate a string.
073     *  If the string is longer than 80 characters, truncate it by
074     *  displaying the first 37 chars, then ". . .", then the last 38
075     *  characters.
076     *  If the <i>longName</i> argument is null, then the string
077     *  "&gt;Unnamed&lt;" is returned.
078     *  @param longName The string to be abbreviated.
079     *  @return The possibly abbreviated name.
080     *  @see #split(String)
081     */
082    public static String abbreviate(String longName) {
083        // This method is used to abbreviate window titles so that long
084        // file names may appear in the window title bar.  It is not
085        // parameterized so that we can force a unified look and feel.
086        // FIXME: it would be nice to split on a nearby space.
087        if (longName == null) {
088            return "<Unnamed>";
089        }
090
091        if (longName.length() <= 80) {
092            return longName;
093        }
094
095        return longName.substring(0, 37) + ". . ."
096                + longName.substring(longName.length() - 38);
097    }

099    /** Add a directory to the java.library.path directory..
100     *  The java.library.path directory determines where the JVM
101     *  looks for native shared libraries.  It is typically read once
102     *  when the JVM is started and no longer read after that.
103     *  <p>This code may only work on certain JVMs</p>
104     *
105     *  <p>Based on code from http://forums.sun.com/thread.jspa?threadID=707176
106     *  and http://stackoverflow.com/questions/5419039/is-djava-library-path-equivalent-to-system-setpropertyjava-library-path</p>
107     *
108     *  @param directoryName The directory to be added.
109     *  @exception IOException If there are insufficient permissions to
110     *  get set the java.library.path environment variable or there
111     *  is no such field as usr_paths in the ClassLoader.
112     */
113    public static void addDirectoryToJavaLibraryPath(String directoryName)
114            throws IOException {
115        try {
116            // Java 1.9 throws a warning unless we use this:
117            // https://stackoverflow.com/a/46458447
118            Field theUnsafe = Unsafe.class.getDeclaredField("theUnsafe");
119            theUnsafe.setAccessible(true);
120            Unsafe unsafe = (Unsafe) theUnsafe.get(null);
121
122            Field usrPathsField = ClassLoader.class
123                    .getDeclaredField("usr_paths");
124            String[] libraryPathsArray = (String[]) unsafe.getObjectVolatile(
125                    ClassLoader.class, unsafe.staticFieldOffset(usrPathsField));
126            ArrayList<String> libraryPaths = new ArrayList<String>(
127                    Arrays.asList(libraryPathsArray));
128            if (libraryPaths.contains(directoryName)) {
129                return;
130            }
131            libraryPaths.add(directoryName);
132            unsafe.putObjectVolatile(ClassLoader.class,
133                    unsafe.staticFieldOffset(usrPathsField),
134                    libraryPaths.toArray(new String[libraryPaths.size()]));
135            System.setProperty("java.library.path",
136                    System.getProperty("java.library.path") + File.pathSeparator
137                            + directoryName);
138        } catch (IllegalAccessException ex) {
139            IOException ioException = new IOException(
140                    "Failed to get permissions to set library path");
141            ioException.initCause(ex);
142            throw ioException;
143        } catch (NoSuchFieldException ex2) {
144            IOException ioException = new IOException(
145                    "Failed to get field handle to set library path");
146            ioException.initCause(ex2);
147            throw ioException;
148        }
149    }

177    /** Return a string with a maximum line length of <i>length</i>
178     *  characters, limited to the given number of characters.
179     *  If there are more than 10 newlines, then the string is truncated
180     *  after 10 lines.
181     *  If the string is truncated, an ellipsis (three periods in a
182     *  row: "...") will be appended to the end of the string.
183     *  Lines that are longer than 160 characters are split into lines
184     *  that are shorter than 160 characters.
185     *  @param string The string to truncate.
186     *  @param length The number of characters to which to truncate the string.
187     *  @return The possibly truncated string with ellipsis possibly added.
188     */
189    public static String ellipsis(String string, int length) {
190        // If necessary, insert newlines into long strings.
191        // If we don't do split long lines and we throw an exception
192        // with a very long line, then the window close button and
193        // possible the dismiss button will be off the right side of
194        // the screen.
195        // The number 160 was generated by trying different sizes and
196        // seeing what fits on a 1024 wide screen.
197        string = StringUtilities.split(string, 160);
198
199        // Third argument being true means return the delimiters as tokens.
200        StringTokenizer tokenizer = new StringTokenizer(string, LINE_SEPARATOR,
201                true);
202
203        // If there are more than 42 lines and 42 newlines, return
204        // truncate after the first 42 lines and newlines.
205        // This is necessary so that we can deal with very long lines
206        // of text without spaces.
207        if (tokenizer.countTokens() > 42) {
208            StringBuffer results = new StringBuffer();
209
210            for (int i = 0; i < 42 && tokenizer.hasMoreTokens(); i++) {
211                results.append(tokenizer.nextToken());
212            }
213
214            results.append("...");
215            string = results.toString();
216        }
217
218        if (string.length() > length) {
219            return string.substring(0, length - 3) + "...";
220        }
221
222        return string;
223    }

239     *

242     */

244        return escapeForXML(string, true);

247    /** Given a string, replace all the instances of XML special characters
248     *  with their corresponding XML entities.  This is necessary to
249     *  allow arbitrary strings to be encoded within XML.
250     *
251     *  <p>In this method, we make the following translations:
252     *  <pre>
253     *  &amp; becomes &amp;amp;
254     *  " becomes &amp;quot;
255     *  &lt; becomes &amp;lt;
256     *  &gt; becomes &amp;gt;
257     *  newline becomes &amp;#10;, if requested.
258     *  carriage return becomes $amp;#13;, if requested.
259     *  </pre>
260     *  @see #unescapeForXML(String)
261     *
262     *  @param string The string to escape.
263     *  @param string Whether or not to escape line break characters.
264     *  @return A new string with special characters replaced.
265     */
266    public static String escapeForXML(String string, boolean escapeLinebreaks) {
267        if (string != null) {
268            StringBuffer buffer = new StringBuffer();
269            buffer.ensureCapacity(string.length());
270            for (int i = 0, n = string.length(); i < n; ++i) {
271                char c = string.charAt(i);
272                switch (c) {
273                case '\n':
274                    if (escapeLinebreaks) {
275                        buffer.append("&#10;");
276                    } else {
277                        buffer.append(c);
278                    }
279                    break;
280                case '\r':
281                    if (escapeLinebreaks) {
282                        buffer.append("&#13;");
283                    } else {
284                        buffer.append(c);
285                    }
286                    break;
287                case '"':
288                    buffer.append("&quot;");
289                    break;
290                case '&':
291                    buffer.append("&amp;");
292                    break;
293                case '<':
294                    buffer.append("&lt;");
295                    break;
296                case '>':
297                    buffer.append("&gt;");
298                    break;
299                default:
300                    buffer.append(c);
301                    break;
302                }
303            }
304            string = buffer.toString();
305        }
306        return string;
307    }

309    /** Given a string, return a string that when fed to the
310     *  Ptolemy expression parser, will turn into the argument
311     *  string. That is, replace all the instances of backslashes
312     *  with double backslashes, all quotation marks with \",
313     *  etc.
314     *  For example
315     *  <pre>
316     *  x"y becomes x\"y;
317     *  x\"y becomes x\\\"y;
318     *  x\y"z becomes x\\y\"z;
319     *  x\\y\"z becomes x\\\\y\\\"
320     *  </pre>
321     *  Similarly, this method replaces the following characters
322     *  exactly as defined in Java strings: \n, \t, \b, \r, and \f.
323     *  @param string The string to escape.
324     *  @return A new string with that can be put between quotation marks.
325     */
326    public static String escapeString(String string) {
327        // Since the first string is a regular expression, it needs extra escaping.
328        // I have no idea why the extra escaping is needed on the second argument.
329        string = string.replaceAll("\\\\", "\\\\\\\\");
330        string = string.replaceAll("\"", "\\\\\"");
331        string = string.replaceAll("\n", "\\\\n");
332        string = string.replaceAll("\t", "\\\\t");
333        string = string.replaceAll("\b", "\\\\b");
334        string = string.replaceAll("\r", "\\\\r");
335        // Not needed.
336        // string = string.replaceAll("\'", "\\\\'");
337        return string;
338    }

340    /** If the ptolemy.ptII.exitAfterWrapup or the
341     *  ptolemy.ptII.doNotExit properties are not set, then call
342     *  System.exit().
343     *  Ptolemy code should call this method instead of directly calling
344     *  System.exit() so that we can test code that would usually exit.
345     *  @param returnValue The return value of this process, where
346     *  non-zero values indicate an error.
347     */
348    public static void exit(int returnValue) {
349        try {
350            if (StringUtilities.getProperty("ptolemy.ptII.doNotExit")
351                    .length() > 0) {
352                return;
353            }
354        } catch (SecurityException ex) {
355            System.out.println("Warning: failed to get property \""
356                    + "ptolemy.ptII.doNotExit\". "
357                    + "(-sandbox always causes this)");
358        }
359
360        try {
361            if (StringUtilities.getProperty("ptolemy.ptII.exitAfterWrapup")
362                    .length() > 0) {
363                throw new RuntimeException("Normally, we would "
364                        + "exit here because Manager.exitAfterWrapup() "
365                        + "was called.  However, because the "
366                        + "ptolemy.ptII.exitAfterWrapup property "
367                        + "is set, we throw this exception instead.");
368            }
369        } catch (SecurityException ex) {
370            System.out.println("Warning: failed to get property \""
371                    + "ptolemy.ptII.exitAfterWrapup\". "
372                    + "(-sandbox always causes this)");
373
374        }
375
376        if (!inApplet()) {
377            // Only call System.exit if we are not in an applet.
378            // Non-zero indicates a problem.
379            System.exit(returnValue);
380        }
381    }

383    /** Return a number of spaces that is proportional to the argument.
384     *  If the argument is negative or zero, return an empty string.
385     *  @param level The level of indenting represented by the spaces.
386     *  @return A string with zero or more spaces.
387     */
388    public static String getIndentPrefix(int level) {
389        if (level <= 0) {
390            return "";
391        }
392
393        StringBuffer result = new StringBuffer(level * 4);
394
395        for (int i = 0; i < level; i++) {
396            result.append("    ");
397        }
398
399        return result.toString();
400    }

402    /** Get the specified property from the environment. An empty
403     *  string is returned if the property named by the "propertyName"
404     *  argument environment variable does not exist, though if
405     *  certain properties are not defined, then we make various
406     *  attempts to determine them and then set them.  See the javadoc
407     *  page for java.util.System.getProperties() for a list of system
408     *  properties.
409
410     *  <p>The following properties are handled specially
411     *  <dl>
412     *  <dt> "ptolemy.ptII.dir"
413     *  <dd> vergil usually sets the ptolemy.ptII.dir property to the
414     *  value of $PTII.  However, if we are running under Web Start,
415     *  then this property might not be set, in which case we look
416     *  for "ptolemy/util/StringUtilities.class" and set the
417     *  property accordingly.
418     *  <dt> "ptolemy.ptII.dirAsURL"
419     *  <dd> Return $PTII as a URL.  For example, if $PTII was c:\ptII,
420     *  then return file:/c:/ptII/.
421     *  <dt> "user.dir"
422     *  <dd> Return the canonical path name to the current working
423     *  directory.  This is necessary because under Windows with
424     *  JDK1.4.1, the System.getProperty() call returns
425     *  <code><b>c</b>:/<i>foo</i></code> whereas most of the other
426     *  methods that operate on path names return
427     *  <code><b>C</b>:/<i>foo</i></code>.
428     *  </dl>
429     *  @param propertyName The name of property.
430     *  @return A String containing the string value of the property.
431     *  If the property is not found, then we return the empty string.
432     */
433    public static String getProperty(String propertyName) {
434        // NOTE: getProperty() will probably fail in applets, which
435        // is why this is in a try block.
436        String property = null;
437
438        try {
439            property = System.getProperty(propertyName);
440            // if (propertyName.equals("ptolemy.ptII.dir")) {
441            //     System.out.println("StringUtilities.getProperty(" + propertyName + "): " + property);
442            // }
443        } catch (SecurityException ex) {
444            if (!propertyName.equals("ptolemy.ptII.dir")) {
445                // Constants.java depends on this when running with
446                // -sandbox.
447                SecurityException security = new SecurityException(
448                        "Could not find '" + propertyName
449                                + "' System property");
450                security.initCause(ex);
451                throw security;
452            }
453        }
454
455        if (propertyName.equals("user.dir")) {
456            try {
457                if (property == null) {
458                    return property;
459                }
460                File userDirFile = new File(property);
461                return userDirFile.getCanonicalPath();
462            } catch (IOException ex) {
463                return property;
464            }
465        }
466
467        // Check for cases where the ptII property starts with
468        // the string "/cygdrive".  This can happen if the property
469        // was set by doing "PTII=`pwd`" under Cygwin bash.
470        //
471        // If the property starts with $JAVAROOT, and the
472        // propertyName is ptolemy.ptII.dir, then don't return
473        // the property yet, instead, refine it.
474        if (property != null && (!propertyName.equals("ptolemy.ptII.dir")
475                && !property.startsWith("$JAVAROOT"))) {
476            if (propertyName.equals("ptolemy.ptII.dir")
477                    && property.startsWith("/cygdrive")
478                    && !_printedCygwinWarning) {
479                // This error only occurs when users build their own,
480                // so it is safe to print to stderr
481                _printedCygwinWarning = true;
482                System.err.println("ptolemy.ptII.dir property = \"" + property
483                        + "\", which contains \"cygdrive\". "
484                        + "This is almost always an error under Cygwin that "
485                        + "is occurs when one does PTII=`pwd`.  Instead, do "
486                        + "PTII=c:/foo/ptII");
487            }
488
489            return property;
490        } else {
491
492            if (propertyName.equals("ptolemy.ptII.dirAsURL")) {
493                // Return $PTII as a URL.  For example, if $PTII was c:\ptII,
494                // then return file:/c:/ptII/
495                File ptIIAsFile = new File(getProperty("ptolemy.ptII.dir"));
496
497                try {
498                    // Convert first to a URI, then to a URL so that we
499                    // properly handle cases where $PTII has spaces in it.
500                    URI ptIIAsURI = ptIIAsFile.toURI();
501                    URL ptIIAsURL = ptIIAsURI.toURL();
502                    return ptIIAsURL.toString();
503                } catch (java.net.MalformedURLException malformed) {
504                    throw new RuntimeException("While trying to find '"
505                            + propertyName + "', could not convert '"
506                            + ptIIAsFile + "' to a URL", malformed);
507                }
508            }
509
510            if (propertyName.equals("ptolemy.ptII.dir")) {
511                if (_ptolemyPtIIDir != null) {
512                    // Return the previously calculated value
513                    // System.out.println("StringUtilities.getProperty(" + propertyName + "): returning previous " + _ptolemyPtIIDir);
514                    return _ptolemyPtIIDir;
515                } else {
516                    String stringUtilitiesPath = "ptolemy/util/StringUtilities.class";
517
518                    // PTII variable was not set
519                    URL namedObjURL = ClassUtilities
520                            .getResource(stringUtilitiesPath);
521
522                    if (namedObjURL != null) {
523                        // Get the file portion of URL
524                        String namedObjFileName = namedObjURL.getFile();
525
526                        // System.out.println("StringUtilities.getProperty(" + propertyName + "): namedObjURL: " + namedObjURL);
527                        // FIXME: How do we get from a URL to a pathname?
528                        if (namedObjFileName.startsWith("file:")) {
529                            if (namedObjFileName.startsWith("file://")
530                                    || namedObjFileName
531                                            .startsWith("file:\\\\")) {
532                                // We get rid of either file:/ or file:\
533                                namedObjFileName = namedObjFileName
534                                        .substring(6);
535                            } else {
536                                // Get rid of file:
537                                namedObjFileName = namedObjFileName
538                                        .substring(5);
539                            }
540                        }
541
542                        String abnormalHome = namedObjFileName.substring(0,
543                                namedObjFileName.length()
544                                        - stringUtilitiesPath.length());
545
546                        // abnormalHome will have values like: "/C:/ptII/"
547                        // which cause no end of trouble, so we construct a File
548                        // and call toString().
549                        _ptolemyPtIIDir = new File(abnormalHome).toString();
550
551                        // If we are running under Web Start, then strip off
552                        // the trailing "!"
553                        if (_ptolemyPtIIDir.endsWith("/!")
554                                || _ptolemyPtIIDir.endsWith("\\!")) {
555                            _ptolemyPtIIDir = _ptolemyPtIIDir.substring(0,
556                                    _ptolemyPtIIDir.length() - 1);
557                        }
558
559                        // Web Start, we might have
560                        // RMptsupport.jar or
561                        // XMptsupport.jar1088483703686
562                        String ptsupportJarName = File.separator + "DMptolemy"
563                                + File.separator + "RMptsupport.jar";
564
565                        if (_ptolemyPtIIDir.endsWith(ptsupportJarName)) {
566                            _ptolemyPtIIDir = _ptolemyPtIIDir.substring(0,
567                                    _ptolemyPtIIDir.length()
568                                            - ptsupportJarName.length());
569                        } else {
570                            ptsupportJarName = "/DMptolemy/XMptsupport.jar";
571
572                            if (_ptolemyPtIIDir
573                                    .lastIndexOf(ptsupportJarName) != -1) {
574                                _ptolemyPtIIDir = _ptolemyPtIIDir.substring(0,
575                                        _ptolemyPtIIDir
576                                                .lastIndexOf(ptsupportJarName));
577                            } else {
578                                // Ptolemy II 6.0.1 under Windows: remove
579                                // "\ptolemy\ptsupport.jar!"
580                                // If we don't do this, then ptolemy.ptII.dir
581                                // is set incorrectly and then links to the javadoc
582                                // files will not be found if the javadoc only
583                                // exists in codeDoc.jar and lib/ptII.properties
584                                // is not present.
585                                ptsupportJarName = File.separator + "ptolemy"
586                                        + File.separator + "ptsupport.jar";
587
588                                if (_ptolemyPtIIDir
589                                        .lastIndexOf(ptsupportJarName) != -1) {
590                                    _ptolemyPtIIDir = _ptolemyPtIIDir.substring(
591                                            0, _ptolemyPtIIDir.lastIndexOf(
592                                                    ptsupportJarName));
593                                }
594                            }
595                        }
596                    }
597
598                    // Convert %20 to spaces because if a URL has %20 in it,
599                    // then we know we have a space, but file names do not
600                    // recognize %20 as being a single space, instead file names
601                    // see %20 as three characters: '%', '2', '0'.
602                    if (_ptolemyPtIIDir != null) {
603                        _ptolemyPtIIDir = StringUtilities
604                                .substitute(_ptolemyPtIIDir, "%20", " ");
605                    }
606                    //*.class files are compiled into classes.dex file; therefore, check for StringUtilities.class fails
607                    //it's OK to set _ptolemyPtIIDir to an empty string on Android
608                    if (_ptolemyPtIIDir == null && System
609                            .getProperty("java.vm.name").equals("Dalvik")) {
610                        _ptolemyPtIIDir = "";
611                    }
612                    if (_ptolemyPtIIDir == null) {
613                        throw new RuntimeException("Could not find "
614                                + "'ptolemy.ptII.dir'" + " property.  "
615                                + "Also tried loading '" + stringUtilitiesPath
616                                + "' as a resource and working from that. "
617                                + "Vergil should be "
618                                + "invoked with -Dptolemy.ptII.dir"
619                                + "=\"$PTII\", "
620                                + "otherwise the following features will not work: "
621                                + "PtinyOS, Ptalon, the Python actor, "
622                                + "actor document, cg code generation and possibly "
623                                + "other features will not work.");
624                    }
625
626                    try {

660     *  Section 3.8 of the Java language spec says:
661     *  <blockquote>
662     *  "An identifier is an unlimited-length sequence of Java letters
663     *  and Java digits, the first of which must be a Java letter. An
664     *  identifier cannot have the same spelling (Unicode character
665     *  sequence) as a keyword (3.9), boolean literal (3.10.3), or
666     *  the null literal (3.10.7)."
667     *  </blockquote>
668     *  Java characters are A-Z, a-z, $ and _.
669     *  <p> Characters that are not permitted in a Java identifier are changed

682        }

685                return false;
686            }
687        }
688        return true;

889    /** If the string is longer than <i>length</i> characters,
890     *  split the string up by adding newlines in all
891     *  newline delimited substrings that are longer than <i>length</i>
892     *  characters.
893     *  If the <i>longName</i> argument is null, then the string
894     *  "&gt;Unnamed&lt;" is returned.
895     *  @see #abbreviate(String)
896     *  @see #split(String)
897     *  @param longName The string to optionally split.
898     *  @param length The maximum length of the sequence of characters
899     *  before a newline is inserted.
900     *  @return Either the original string, or the string with newlines
901     *  inserted.
902     */
903    public static String split(String longName, int length) {
904        if (longName == null) {
905            return "<Unnamed>";
906        }
907
908        if (longName.length() <= length) {
909            return longName;
910        }
911
912        StringBuffer results = new StringBuffer();
913
914        // The third argument is true, which means return the delimiters
915        // as part of the tokens.
916        StringTokenizer tokenizer = new StringTokenizer(longName,
917                LINE_SEPARATOR, true);
918
919        while (tokenizer.hasMoreTokens()) {
920            String token = tokenizer.nextToken();
921            int mark = 0;
922
923            while (mark < token.length() - length) {
924                // We look for the space from the end of the first length
925                // characters.  If we find one, then we use that
926                // as the place to insert a newline.
927                int lastSpaceIndex = token.substring(mark, mark + length)
928                        .lastIndexOf(" ");
929
930                if (lastSpaceIndex < 0) {
931                    // No space found, just insert a new line after length
932                    results.append(token.substring(mark, mark + length)
933                            + LINE_SEPARATOR);
934                    mark += length;
935                } else {
936                    results.append(token.substring(mark, mark + lastSpaceIndex)
937                            + LINE_SEPARATOR);
938                    mark += lastSpaceIndex + 1;
939                }
940            }
941
942            results.append(token.substring(mark));
943        }
944
945        return results.toString();
946    }

974    /** Replace all occurrences of <i>pattern</i> in the specified
975     *  string with <i>replacement</i>.  Note that the pattern is NOT
976     *  a regular expression, and that relative to the
977     *  String.replaceAll() method in jdk1.4, this method is extremely
978     *  slow.  This method does not work well with back slashes.
979     *  @param string The string to edit.
980     *  @param pattern The string to replace.
981     *  @param replacement The string to replace it with.
982     *  @return A new string with the specified replacements.
983     */
984    public static String substitute(String string, String pattern,
985            String replacement) {
986        if (string == null) {
987            return null;
988        }
989        int start = string.indexOf(pattern);
990
991        while (start != -1) {
992            StringBuffer buffer = new StringBuffer(string);
993            buffer.delete(start, start + pattern.length());
994            buffer.insert(start, replacement);
995            string = new String(buffer);
996            start = string.indexOf(pattern, start + replacement.length());
997        }
998
999        return string;
1000    }

1051    /** Tokenize a String to an array of Strings for use with
1052     *  Runtime.exec(String []).
1053     *
1054     *  <p>Lines that begin with an octothorpe '#' are ignored.
1055     *  Substrings that start and end with a double quote are considered
1056     *  to be a single token and are returned as a single array element.
1057     *
1058     *  @param inputString  The String to tokenize
1059     *  @return An array of substrings.
1060     *  @exception IOException If StreamTokenizer.nextToken() throws it.
1061     */
1062    public static String[] tokenizeForExec(String inputString)
1063            throws IOException {
1064        // The java.lang.Runtime.exec(String command) call uses
1065        // java.util.StringTokenizer() to parse the command string.
1066        // Unfortunately, this means that double quotes are not handled
1067        // in the same way that the shell handles them in that 'ls "foo
1068        // 'bar"' will interpreted as three tokens 'ls', '"foo' and
1069        // 'bar"'.  In the shell, the string would be two tokens 'ls' and
1070        // '"foo bar"'.  What is worse is that the exec() behaviour is
1071        // slightly different under Windows and Unix.  To solve this
1072        // problem, we preprocess the command argument using
1073        // java.io.StreamTokenizer, which converts quoted substrings into
1074        // single tokens.  We then call java.lang.Runtime.exec(String []
1075        // commands);
1076        // Parse the command into tokens
1077        List<String> commandList = new LinkedList<String>();
1078
1079        StreamTokenizer streamTokenizer = new StreamTokenizer(
1080                new StringReader(inputString));
1081
1082        // We reset the syntax so that we don't convert to numbers,
1083        // otherwise, if PTII is "d:\\tmp\\ptII\ 2.0", then
1084        // we have no end of problems.
1085        streamTokenizer.resetSyntax();
1086        streamTokenizer.whitespaceChars(0, 32);
1087        streamTokenizer.wordChars(33, 127);
1088
1089        // We can't use quoteChar here because it does backslash
1090        // substitution, so "c:\ptII" ends up as "c:ptII"
1091        // Substituting forward slashes for backward slashes seems like
1092        // overkill.
1093        // streamTokenizer.quoteChar('"');
1094        streamTokenizer.ordinaryChar('"');
1095
1096        streamTokenizer.eolIsSignificant(true);
1097
1098        streamTokenizer.commentChar('#');
1099
1100        // Current token
1101        String token = "";
1102
1103        // Single character token, usually a -
1104        String singleToken = "";
1105
1106        // Set to true if we are inside a double quoted String.
1107        boolean inDoubleQuotedString = false;
1108
1109        while (streamTokenizer.nextToken() != StreamTokenizer.TT_EOF) {
1110            switch (streamTokenizer.ttype) {
1111            case StreamTokenizer.TT_WORD:
1112
1113                if (inDoubleQuotedString) {
1114                    if (token.length() > 0) {
1115                        // FIXME: multiple spaces will get compacted here
1116                        token += " ";
1117                    }
1118
1119                    token += singleToken + streamTokenizer.sval;
1120                } else {
1121                    token = singleToken + streamTokenizer.sval;
1122                    commandList.add(token);
1123                }
1124
1125                singleToken = "";
1126                break;
1127
1128            case StreamTokenizer.TT_NUMBER:
1129                throw new RuntimeException("Internal error: Found TT_NUMBER: '"
1130                        + streamTokenizer.nval + "'.  We should not be "
1131                        + "tokenizing numbers");
1132
1133                //break;
1134            case StreamTokenizer.TT_EOL:
1135            case StreamTokenizer.TT_EOF:
1136                break;
1137
1138            default:
1139                singleToken = Character.toString((char) streamTokenizer.ttype);
1140
1141                if (singleToken.equals("\"")) {
1142                    if (inDoubleQuotedString) {
1143                        commandList.add(token);
1144                    }
1145
1146                    inDoubleQuotedString = !inDoubleQuotedString;
1147                    singleToken = "";
1148                    token = "";
1149                }
1150
1151                break;
1152            }
1153        }
1154
1155        return commandList.toArray(new String[commandList.size()]);
1156    }

1159     *  and a maximum number of lines <i>numberOfLines</i>.
1160     *  Each line that exceeds the line length is replaced with a line that
1161     *  ends with "...". If the number of lines exceeds <i>numberOfLines</i>,
1162     *  then the returned string will have exactly <i>numberOfLines</i> lines
1163     *  where the last line is "...".
1164     *  @param string The string to truncate.
1165     *  @param lineLength The number of characters to which to truncate each line.
1166     *  @param numberOfLines The maximum number of lines.
1167     *  @return The possibly truncated string with ellipsis possibly added.
1168     */
1169    public static String truncateString(String string, int lineLength,
1170            int numberOfLines) {
1171
1172        // Third argument being true means the delimiters (LINE_SEPARATOR) are
1173        // included in as tokens in the parsed results.
1174        StringTokenizer tokenizer = new StringTokenizer(string, LINE_SEPARATOR,
1175                true);
1176
1177        StringBuffer results = new StringBuffer();
1178        // Count the lines + newlines.
1179        int lineCount = 0;
1180        while (tokenizer.hasMoreTokens()) {
1181            if (lineCount >= numberOfLines * 2) {

1191                line = line.substring(0, lineLength - 3) + "...";
1192            }
1193            results.append(line);
1194        }
1195        return results.toString();

1252    }

1254    /** Return a string that contains a description of how to use a
1255     *  class that calls this method.  For example, this method is
1256     *  called by "$PTII/bin/vergil -help".
1257     *  @param commandTemplate  A string naming the command and the
1258     *  format of the arguments, for example
1259     *  "moml [options] [file . . .]"
1260     *  @param commandOptions A 2xN array of Strings that list command-line
1261     *  options that take arguments where the first
1262     *  element is a String naming the command line option, and the
1263     *  second element is the argument, for example
1264     *  <code>{"-class", "&lt;classname&gt;")</code>
1265     *  @param commandFlagsWithDescriptions A 2xM array of Strings that list
1266     *  command-line options that are either present or not and a description
1267     *  of what the command line option does.
1268     *  @return A string that describes the command.
1269     */
1270    public static String usageString(String commandTemplate,
1271            String[][] commandOptions,
1272            String[][] commandFlagsWithDescriptions) {
1273        // This method is static so that we can reuse it in places
1274        // like copernicus/kernel/Copernicus and actor/gui/MoMLApplication
1275        StringBuffer result = new StringBuffer("Usage: " + commandTemplate
1276                + "\n\n" + "Options that take values:\n");
1277
1278        int i;
1279
1280        for (i = 0; i < commandOptions.length; i++) {
1281            result.append(" " + commandOptions[i][0]);
1282            if (commandOptions[i][1].length() > 0) {
1283                result.append(" " + commandOptions[i][1]);
1284            }
1285            result.append("\n");
1286        }
1287
1288        result.append("\nBoolean flags:\n");
1289
1290        for (i = 0; i < commandFlagsWithDescriptions.length; i++) {
1291            result.append(" " + commandFlagsWithDescriptions[i][0]);
1292            if (commandFlagsWithDescriptions[i][1].length() > 0) {
1293                result.append("\t" + commandFlagsWithDescriptions[i][1]);
1294            }
1295            result.append("\n");
1296        }
1297
1298        return result.toString();
1299    }
