017package ptolemy.util;
018
019// Remove the dependency on google classes.
020//import static com.google.common.base.Preconditions.checkArgument;
021import static java.lang.Double.MAX_EXPONENT;
022import static java.lang.Double.MIN_EXPONENT;
023import static java.lang.Double.POSITIVE_INFINITY;
024import static java.lang.Double.doubleToRawLongBits;
025import static java.lang.Double.isNaN;
026import static java.lang.Double.longBitsToDouble;
027import static java.lang.Math.getExponent;
028
029import java.math.BigInteger;
030
031/**
032 * Utilities for double primitives.
033 *
034 * This code is used by ptolemy.actor.util.Time to improve performance.
035 *
036 * @author Louis Wasserman, based on https://code.google.com/p/guava-libraries/source/browse/guava/src/com/google/common/math/DoubleUtils.java
037 */
038public final class DoubleUtilities {
039    private DoubleUtilities() {
040    }
041
042    static double nextDown(double d) {
043        return -Math.nextUp(-d);
044    }
045
046    static long getSignificand(double d) {
047        // Remove check argument and replace with conditionals
048        // to remove dependency on google classes.
049        //checkArgument(isFinite(d), "not a normal value");
050        if (!isFinite(d)) {
051            throw new IllegalArgumentException("not a normal value");
052        }
053        int exponent = getExponent(d);
054        long bits = doubleToRawLongBits(d);
055        bits &= SIGNIFICAND_MASK;
056        return (exponent == MIN_EXPONENT - 1) ? bits << 1 : bits | IMPLICIT_BIT;
057    }
058
059    static boolean isFinite(double d) {
060        return getExponent(d) <= MAX_EXPONENT;
061    }
062
063    static boolean isNormal(double d) {
064        return getExponent(d) >= MIN_EXPONENT;
065    }
066
067    /*
068     * Returns x scaled by a power of 2 such that it is in the range [1, 2). Assumes x is positive,
069     * normal, and finite.
070     */
071    static double scaleNormalize(double x) {
072        long significand = doubleToRawLongBits(x) & SIGNIFICAND_MASK;
073        return longBitsToDouble(significand | ONE_BITS);
074    }
075
076    /** Convert a BigInteger to a double using a fast implementation.
077     *  @param x The BigInteger to be converted.
078     *  @return the value of the BigInteger as a double.
079     */
080    public static double bigToDouble(BigInteger x) {
081        // This is an extremely fast implementation of BigInteger.doubleValue().  JDK patch pending.
082        BigInteger absX = x.abs();
083        int exponent = absX.bitLength() - 1;
084        // exponent == floor(log2(abs(x)))
085        if (exponent < Long.SIZE - 1) {
086            return x.longValue();
087        } else if (exponent > MAX_EXPONENT) {
088            return x.signum() * POSITIVE_INFINITY;
089        }
090
091        /*
092         * We need the top SIGNIFICAND_BITS + 1 bits, including the "implicit" one bit. To make
093         * rounding easier, we pick out the top SIGNIFICAND_BITS + 2 bits, so we have one to help us
094         * round up or down. twiceSignifFloor will contain the top SIGNIFICAND_BITS + 2 bits, and
095         * signifFloor the top SIGNIFICAND_BITS + 1.
096         *
097         * It helps to consider the real number signif = absX * 2^(SIGNIFICAND_BITS - exponent).
098         */
099        int shift = exponent - SIGNIFICAND_BITS - 1;
100        long twiceSignifFloor = absX.shiftRight(shift).longValue();
101        long signifFloor = twiceSignifFloor >> 1;
102        signifFloor &= SIGNIFICAND_MASK; // remove the implied bit
103
104        /*
105         * We round up if either the fractional part of signif is strictly greater than 0.5 (which is
106         * true if the 0.5 bit is set and any lower bit is set), or if the fractional part of signif is
107         * >= 0.5 and signifFloor is odd (which is true if both the 0.5 bit and the 1 bit are set).
108         */
109        boolean increment = (twiceSignifFloor & 1) != 0
110                && ((signifFloor & 1) != 0 || absX.getLowestSetBit() < shift);
111        long signifRounded = increment ? signifFloor + 1 : signifFloor;
112        long bits = (long) ((exponent + EXPONENT_BIAS)) << SIGNIFICAND_BITS;
113        bits += signifRounded;
114        /*
115         * If signifRounded == 2^53, we'd need to set all of the significand bits to zero and add 1 to
116         * the exponent. This is exactly the behavior we get from just adding signifRounded to bits
117         * directly.  If the exponent is MAX_DOUBLE_EXPONENT, we round up (correctly) to
118         * Double.POSITIVE_INFINITY.
119         */
120        bits |= x.signum() & SIGN_MASK;
121        return longBitsToDouble(bits);
122    }
123
124    /**
125     * Returns its argument if it is non-negative, zero if it is negative.
126     * @param value The value to be checked.
127     * @return If value is non-negative, return the value.  If the value
128     * is negative, then return zero.
129     */
130    static double ensureNonNegative(double value) {
131        // Remove check argument and replace with conditionals
132        // to remove dependency on google classes.
133        //checkArgument(!isNaN(value));
134        if (isNaN(value)) {
135            throw new IllegalArgumentException("not a number");
136        }
137        if (value > 0.0) {
138            return value;
139        } else {
140            return 0.0;
141        }
142    }
143
144    ///////////////////////////////////////////////////////////////////
145    ////                    static variables                      ////
146
147    // The mask for the significand, according to the {@link
148    // Double#doubleToRawLongBits(double)} spec.
149    static final long SIGNIFICAND_MASK = 0x000fffffffffffffL;
150
151    // The mask for the exponent, according to the {@link
152    // Double#doubleToRawLongBits(double)} spec.
153    static final long EXPONENT_MASK = 0x7ff0000000000000L;
154
155    // The mask for the sign, according to the {@link
156    // Double#doubleToRawLongBits(double)} spec.
157    static final long SIGN_MASK = 0x8000000000000000L;
158
159    // The significands bits
160    static final int SIGNIFICAND_BITS = 52;
161
162    // The exponent bias
163    static final int EXPONENT_BIAS = 1023;
164
165    // The implicit 1 bit that is omitted in significands of normal doubles.
166    static final long IMPLICIT_BIT = SIGNIFICAND_MASK + 1;
167
168    ///////////////////////////////////////////////////////////////////
169    ////                      static private variables            ////
170
171    private static final long ONE_BITS = doubleToRawLongBits(1.0);
172}
