073    /** If the VertxHelperBase class is present, then invoke the
074     *  closeVertx() method so that this process does not wait around
075     *  for the Vert.x threads.
076     */
077    @AfterClass
078    public static void afterClass() {
079        try {
080            Class clazz = Class
081                .forName("ptolemy.actor.lib.jjs.VertxHelperBase");
082            if (clazz != null) {
083                Method method = clazz.getMethod("closeVertx");
084                System.out.println("AutoTests.java: About to close Vertx.");
085                method.invoke(null);
086                System.out.println("AutoTests.java: Vertx closed.");
087            }
088        } catch (NoClassDefFoundError ex) {
089            // Ignore this, it means that MoMLSimpleApplication was invoked without the Vert.x jar files.
090        } catch (Throwable throwable) {
091            System.err.println(
092                               "AutoTests: Failed to invoke VertxHelperBase.closeVertx() during exit.  This can be ignored. Error was: "
093                               + throwable);
094        }
095
096    }

098    /** If the fullPath is a hlacerti or accessor demo, then
099     *  delay so that the system can stabilize.
100     *  @param fullPath The forward slash separated path of the demo.
101     */
102    public static void delayIfNecessary(String fullPath) {
103        int delay = 5000;
104        boolean match = fullPath.matches(".*(org/hlacerti|org/terraswarm/accessors|ptolemy/actor/lib/jjs).*");
105        // System.out.println("AutoTests.java: fullPath: " + fullPath
106        //                   + " match: " + match);
107        if (match) {
108            System.out.println("----------------- " + (new java.util.Date())
109                               + " About to sleep for " + delay / 1000.0
110                               + " seconds before running or rerunning.  Test is: " + fullPath);
111            try {
112                Thread.sleep(delay);
113            } catch (InterruptedException ex) {
114                System.err.println(
115                                   "Sleep before rerunning was interrupted: "
116                                   + ex);
117            }
118            System.out.println("Done sleeping");
119        }
120    }

122    /**
123     * Execute a model and time out after 900000 ms.
124     *
125     * <p>To check the time, go to the list of tests for AutoTest, for example
126     * <a href="http://sisyphus.eecs.berkeley.edu:8079/hudson/job/ptII/724/testReport/ptolemy.util.test.junit/AutoTests/"><code>http://sisyphus.eecs.berkeley.edu:8079/hudson/job/ptII/724/testReport/ptolemy.util.test.junit/AutoTests/</code></a>,
127     * where 724 is a recent run of the build.  Look for thhe longest
128     * passing test, typically Trilateration, and round up by a factor
129     * of 1.5.</p>
130     *
131     * @param fullPath
132     *            The full path to the model file to be executed. If the
133     *            fullPath ends with the value of the
134     *            {@link #THERE_ARE_NO_AUTO_TESTS}, then the method returns
135     *            immediately.
136     * @exception Throwable
137     *                If thrown while executing the model.
138     */
139    @Test(timeout = 900000)
140    @Parameters(method = "modelValues")
141    public void RunModel(String fullPath) throws Throwable {
142        if (fullPath.endsWith(THERE_ARE_NO_AUTO_TESTS)) {
143            System.out.println(
144                               "No auto/*.xml tests in " + System.getProperty("user.dir"));
145            return;
146        }
147        if (fullPath.endsWith(THERE_ARE_NO_AUTO_ARCH_TESTS)) {
148            System.out.println(
149                               "No " + AutoNameArchTests.autoNameArch()
150                               + "*.xml tests in " + System.getProperty("user.dir"));
151            return;
152        }
153
154        // Only check for the JSAccessor class if there are auto tests.
155        // MoMLParser.tcl had a test that was reporting different
156        // results between running using ptjacl and JUnit because
157        // JUnit was always instantiating a JSAccessor, even
158        // if there were no auto tests.  This was causing
159        // "StreamChangeRequest.changeExecuted(): AccessorIconLoader succeeded"
160        // to appear in the listener.
161        if (!AutoTests._haveCheckedForJSAccessor) {
162            AutoTests._haveCheckedForJSAccessor = true;
163            AutoTests._checkForJSAccessor();
164        }
165        // Use a whole row of equals to signify the start of a new test.
166        System.out.println("\n===========================================================================================");
167        if (modelFileIsOK(fullPath)) {
168
169            System.out.println("----------------- testing " + (new java.util.Date()) + " " + fullPath);
170            System.out.flush();
171            if (_applicationConstructor == null) {
172                // Delay instantiating MoMLSimpleApplication so that we
173                // can run the kernel tests without requiring moml
174                _applicationClass = Class
175                    .forName("ptolemy.moml.MoMLSimpleApplication");
176                _applicationConstructor = _applicationClass
177                    .getConstructor(String.class);
178            }
179
180            // _applicationConstructor might have been initialized in
181            // AutoKnownFailedTests, so we initialize
182            // _applicationToplevelMethod here.
183            if (_applicationToplevelMethod == null) {
184                _applicationToplevelMethod = _applicationClass
185                    .getMethod("toplevel", new Class[] {});
186            }
187
188            // If a model is in various directories, including
189            // org/terraswarm/accessors/test, the delay before
190            // reloading.  See
191            // https://chess.eecs.berkeley.edu/ptexternal/wiki/Main/WebSocketDeadlock#Starvation
192            AutoTests.delayIfNecessary(fullPath);
193
194            System.out.println("----------------- Instantiating " + fullPath);
195            Object instance = _applicationConstructor.newInstance(fullPath);
196            Method rerunMethod = _applicationClass.getMethod("rerun",
197                                                             (Class<?>[]) null);
198
199            // If JSAccessor is present and the model contains one, then
200            // reload all the JSAccessors and rerun the model
201
202            if (_jsAccessorClass == null) {
203                // JsAccessor is not present, just rerun.
204                System.out.println("----------------- testing again " + fullPath);
205                System.out.flush();
206                rerunMethod.invoke(instance, (Object[]) null);
207
208            } else {
209                // JSAccessor is present, reload and rerun.
210                
211                System.out.println("----------------- Invoking toplevel() on " + instance);
212                _applicationToplevelMethod = _applicationClass
213                    .getMethod("toplevel", new Class[] {});
214                Object toplevel = _applicationToplevelMethod.invoke(instance,
215                                                                    (Object[]) null);
216                System.out.println("----------------- Done invoking toplevel() on " + instance);
217
218                if (_jsAccessorReloadAllAccessorsMethod == null) {
219                    throw new InternalError(
220                                            "Found the JSAccessor class, but not the reloadAllAccessors() method?");
221                }
222                // Reload all the accessors and invoke rerun.
223                if (((Boolean) _jsAccessorReloadAllAccessorsMethod.invoke(null,
224                                                                          new Object[] { toplevel })).booleanValue()) {
225                    System.out.println("-------------- Reloaded accessors, but skipping rerun for now. "
226                                       + (new java.util.Date()) + " "
227                                       + fullPath);
228                    System.out.flush();
229                    // Autotests.delayIfNecessary(fullPath);
230
231                    // System.out.println(
232                    //         "----------------- Reloaded Accessors and testing again " + (new java.util.Date()) + " "
233                    //                 + fullPath);
234                    // System.out.flush();
235                    // rerunMethod.invoke(instance, (Object[]) null);
236                }
237            }
238        }
239    }

344    private static void _checkForJSAccessor() {
345        try {
346            _jsAccessorClass = Class
347                .forName("org.terraswarm.accessor.JSAccessor");
348            Class compositeEntityClass = Class
349                .forName("ptolemy.kernel.CompositeEntity");
350            _jsAccessorReloadAllAccessorsMethod = _jsAccessorClass.getMethod(
351                                                                             "reloadAllAccessors", new Class[] { compositeEntityClass });
352        } catch (Throwable throwable) {
353            // Ignore this, it could be that the JSAccessor class
354            // is not present.
355            _jsAccessorClass = null;
356            _jsAccessorReloadAllAccessorsMethod = null;
357        }
358    }
