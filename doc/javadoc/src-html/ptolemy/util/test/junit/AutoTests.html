040///////////////////////////////////////////////////////////////////
041//// AutoTests
042/**
043 * Run the Ptolemy model tests in the auto/ directory using JUnit.
044 *
045 * <p>
046 * This test must be run from the directory that contains the auto/ directory,
047 * for example:
048 * </p>
049 *
050 * <pre>
051 * (cd ~/ptII/ptolemy/actor/lib/io/test; java -classpath ${PTII}:${PTII}/lib/junit-4.8.2.jar:${PTII}/lib/JUnitParams-0.3.0.jar org.junit.runner.JUnitCore ptolemy.util.test.junit.AutoTests)
052 * </pre>
053 *
054 * <p>
055 * This test uses JUnitParams from <a
056 * href="http://code.google.com/p/junitparams/#in_browser"
057 * >http://code.google.com/p/junitparams/</a>, which is released under <a
058 * href="http://www.apache.org/licenses/LICENSE-2.0#in_browser">Apache License
059 * 2.0</a>.
060 * </p>
061 *
062 * @author Christopher Brooks
063 * @version $Id$
064 * @since Ptolemy II 10.0
065 * @Pt.ProposedRating Red (cxh)
066 * @Pt.AcceptedRating Red (cxh)
067 */
068@RunWith(JUnitParamsRunner.class)
069public class AutoTests extends ModelTests {
070
071    /**
072     * Execute a model and time out after 900000 ms.
073     *
074     * <p>To check the time, go to the list of tests for AutoTest, for example
075     * <a href="http://sisyphus.eecs.berkeley.edu:8079/hudson/job/ptII/724/testReport/ptolemy.util.test.junit/AutoTests/"><code>http://sisyphus.eecs.berkeley.edu:8079/hudson/job/ptII/724/testReport/ptolemy.util.test.junit/AutoTests/</code></a>,
076     * where 724 is a recent run of the build.  Look for thhe longest
077     * passing test, typically Trilateration, and round up by a factor
078     * of 1.5.</p>
079     *
080     * @param fullPath
081     *            The full path to the model file to be executed. If the
082     *            fullPath ends with the value of the
083     *            {@link #THERE_ARE_NO_AUTO_TESTS}, then the method returns
084     *            immediately.
085     * @exception Throwable
086     *                If thrown while executing the model.
087     */
088    @Test(timeout = 900000)
089    @Parameters(method = "modelValues")
090    public void RunModel(String fullPath) throws Throwable {
091        if (fullPath.endsWith(THERE_ARE_NO_AUTO_TESTS)) {
092            System.out.println(
093                    "No auto/*.xml tests in " + System.getProperty("user.dir"));
094            return;
095        }
096        // Only check for the JSAccessor class if there are auto tests.
097        // MoMLParser.tcl had a test that was reporting different
098        // results between running using ptjacl and JUnit because
099        // JUnit was always instantiating a JSAccessor, even
100        // if there were no auto tests.  This was causing
101        // "StreamChangeRequest.changeExecuted(): AccessorIconLoader succeeded"
102        // to appear in the listener.
103        if (!AutoTests._haveCheckedForJSAccessor) {
104            AutoTests._haveCheckedForJSAccessor = true;
105            AutoTests._checkForJSAccessor();
106        }
107        if (modelFileIsOK(fullPath)) {
108            System.out.println("----------------- testing " + fullPath);
109            System.out.flush();
110            if (_applicationConstructor == null) {
111                // Delay instantiating MoMLSimpleApplication so that we
112                // can run the kernel tests without requiring moml
113                _applicationClass = Class
114                        .forName("ptolemy.moml.MoMLSimpleApplication");
115                _applicationConstructor = _applicationClass
116                        .getConstructor(String.class);
117            }
118
119            // _applicationConstructor might have been initialized in
120            // AutoKnownFailedTests, so we initialize
121            // _applicationToplevelMethod here.
122            if (_applicationToplevelMethod == null) {
123                _applicationToplevelMethod = _applicationClass
124                        .getMethod("toplevel", new Class[] {});
125            }
126
127            // If a model is in org/terraswarm/accessors/test, the delay before reloading.
128            // See https://chess.eecs.berkeley.edu/ptexternal/wiki/Main/WebSocketDeadlock#Starvation
129            _delay(fullPath);
130
131            Object instance = _applicationConstructor.newInstance(fullPath);
132
133            System.out.println("----------------- testing again " + fullPath);
134            System.out.flush();
135            Method rerunMethod = _applicationClass.getMethod("rerun",
136                    (Class<?>[]) null);
137            rerunMethod.invoke(instance, (Object[]) null);
138
139            // If JSAccessor is present and the model contains one, then
140            // reload all the JSAccessors and rerun the model
141            if (_jsAccessorClass != null) {
142                _applicationToplevelMethod = _applicationClass
143                        .getMethod("toplevel", new Class[] {});
144
145                // If a model is in org/terraswarm/accessors/test, the delay before reloading.
146                // See https://chess.eecs.berkeley.edu/ptexternal/wiki/Main/WebSocketDeadlock#Starvation
147                _delay(fullPath);
148
149                if (fullPath.indexOf("org/terraswarm/accessors/test") != -1) {
150                    System.out.println(
151                            "----------------- About to sleep for 5 seconds before reloading accessors in org/terraswarm/accessors/test"
152                                    + fullPath);
153                    try {
154                        Thread.sleep(5000);
155                    } catch (InterruptedException ex) {
156                        System.err.println(
157                                "Sleep before reloading of accessors was interrupted: "
158                                        + ex);
159                    }
160                }
161                Object toplevel = _applicationToplevelMethod.invoke(instance,
162                        (Object[]) null);
163                if (_jsAccessorReloadAllAccessorsMethod == null) {
164                    throw new InternalError(
165                            "Found the JSAccessor class, but not the reloadAllAccessors() method?");
166                }
167                if (((Boolean) _jsAccessorReloadAllAccessorsMethod.invoke(null,
168                        new Object[] { toplevel })).booleanValue()) {
169                    System.out.println(
170                            "----------------- Reloaded Accessors and testing again "
171                                    + fullPath);
172                    System.out.flush();
173                    rerunMethod.invoke(instance, (Object[]) null);
174                }
175            }
176        } else {
177            System.err.println(
178                    "----------------- *** Skipping testing of " + fullPath);
179            System.err.flush();
180
181        }
182    }
183
184    /** Return true if the model should be run.
185     *  This is a hack to avoid a problem where certain models
186     *  interact badly with the Cobertura code coverage tool.
187     *  @param fullPath The full path of the model to be executed
188     *  @return true if the model should be run.
189     */
190    public boolean modelFileIsOK(String fullPath) {
191        if (fullPath.endsWith("de/test/auto/ThreadedComposite.xml")
192                && !StringUtilities
193                        .getProperty("net.sourceforge.cobertura.datafile")
194                        .equals("")) {
195            System.err.println(
196                    "ModelTests: Skipping de/test/auto/ThreadedComposite.xml because it interacts badly with Cobertura.");
197            return false;
198        }
199        return true;
200    }

202    ///////////////////////////////////////////////////////////////////
203    ////                         protected fields                  ////
204
205    /** The org.terraswarm.accessor.JSAccessor class, which is tested
206     * by reloading Accessors.
207     */
208    protected static Class<?> _jsAccessorClass = null;
209
210    /** The method that reloads all the accessors in a CompositeEntity. */
211    protected static Method _jsAccessorReloadAllAccessorsMethod = null;
212
213    ///////////////////////////////////////////////////////////////////
214    ////                         private methods                   ////
215
216    private static void _checkForJSAccessor() {
217        try {
218            _jsAccessorClass = Class
219                    .forName("org.terraswarm.accessor.JSAccessor");
220            Class compositeEntityClass = Class
221                    .forName("ptolemy.kernel.CompositeEntity");
222            _jsAccessorReloadAllAccessorsMethod = _jsAccessorClass.getMethod(
223                    "reloadAllAccessors", new Class[] { compositeEntityClass });
224        } catch (Throwable throwable) {
225            // Ignore this, it could be that the JSAccessor class
226            // is not present.
227            _jsAccessorClass = null;
228            _jsAccessorReloadAllAccessorsMethod = null;
229        }
230    }
231
232    private static void _delay(String fullPath) {
233        String accessorTests = "org/terraswarm/accessor";
234        int delay = 5000;
235        if (fullPath.indexOf(accessorTests) != -1) {
236            System.out.println("----------------- " + (new java.util.Date())
237                    + " About to sleep for " + delay / 1000
238                    + " seconds before rerunning in " + accessorTests
239                    + ".  Test is: " + fullPath);
240            try {
241                Thread.sleep(5000);
242            } catch (InterruptedException ex) {
243                System.err.println(
244                        "Sleep before reloading of accessors was interrupted: "
245                                + ex);
246            }
247        }
248    }
249
250    ///////////////////////////////////////////////////////////////////
251    ////                         private fields                    ////
252
253    /** True if we have checked for JSAccessor. */
254    private static boolean _haveCheckedForJSAccessor = false;
