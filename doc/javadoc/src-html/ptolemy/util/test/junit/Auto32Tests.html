031import java.io.IOException;
032import java.lang.reflect.Method;
033
034import org.junit.Test;

037import junitparams.JUnitParamsRunner;

041///////////////////////////////////////////////////////////////////
042//// Auto32Tests
043/**
044 * Run the Ptolemy model tests in the auto/ directory using JUnit.
045 *
046 * <p>
047 * This test must be run from the directory that contains the auto/ directory,
048 * for example:
049 * </p>
050 *
051 * <pre>
052 * (cd ~/ptII/ptolemy/actor/lib/io/test; java -classpath ${PTII}:${PTII}/lib/junit-4.8.2.jar:${PTII}/lib/JUnitParams-0.3.0.jar org.junit.runner.JUnitCore ptolemy.util.test.junit.AutoTests)
053 * </pre>
054 *
055 * <p>
056 * This test uses JUnitParams from <a
057 * href="http://code.google.com/p/junitparams/#in_browser"
058 * >http://code.google.com/p/junitparams/</a>, which is released under <a
059 * href="http://www.apache.org/licenses/LICENSE-2.0#in_browser">Apache License
060 * 2.0</a>.
061 * </p>
062 *
063 * @author Christopher Brooks
064 * @version $Id$
065 * @since Ptolemy II 10.0
066 * @Pt.ProposedRating Red (cxh)
067 * @Pt.AcceptedRating Red (cxh)
068 */
069@RunWith(JUnitParamsRunner.class)
070public class Auto32Tests extends ModelTests {
071
072    /**
073     * Return a two dimensional array of arrays of strings that name the model
074     * to be executed. If auto/ does not exist, or does not contain files that
075     * end with .xml or .moml, return a list with one element that contains
076     * the value of the THERE_ARE_NO_AUTO_TESTS variable.
077     *
078     * @return The List of model names in auto/
079     * @exception IOException If there is a problem accessing the auto/ directory.
080     */
081    @Override
082    public Object[] modelValues() throws IOException {
083        return modelValues("auto32/", THERE_ARE_NO_AUTO_TESTS);
084    }
085
086    /**
087     * Execute a model. Timeout after 140000 ms.
088     *
089     * <p>To check the time, go to the list of tests for AutoTest, for example
090     * <a href="http://sisyphus.eecs.berkeley.edu:8079/hudson/job/ptII/724/testReport/ptolemy.util.test.junit/AutoKnownFailedTests/"><code>http://sisyphus.eecs.berkeley.edu:8079/hudson/job/ptII/724/testReport/ptolemy.util.test.junit/AutoKnownFailedTests/</code></a>,
091     * where 724 is a recent run of the build.  Look for thhe longest
092     * passing test, typically OpenModelicaInteractive (15 sec.), and
093     * round up by a factor of 10 because 15 sec. is so small, then
094     * round down so that the time is different from
095     * AutoKnownFailedTests.</p>
096     *
097     * @param fullPath
098     *            The full path to the model file to be executed. If the
099     *            fullPath ends with the value of the
100     *            {@link #THERE_ARE_NO_AUTO_TESTS}, then the method returns
101     *            immediately.
102     * @exception Throwable
103     *                If thrown while executing the model.
104     */
105    @Test(timeout = 140000)
106    @Parameters(method = "modelValues")
107    public void RunModel(String fullPath) throws Throwable {
108        if (fullPath.endsWith(THERE_ARE_NO_AUTO_TESTS)) {
109            System.out.println("No auto32/*.xml tests in "
110                    + System.getProperty("user.dir"));
111            return;
112        }
113        if (modelFileIsOK(fullPath)) {
114            System.out.println("----------------- testing " + fullPath);
115            System.out.flush();
116            if (_applicationConstructor == null) {
117                // Delay instantiating MoMLSimpleApplication so that we
118                // can run the kernel tests without requiring moml
119                _applicationClass = Class
120                        .forName("ptolemy.moml.MoMLSimpleApplication");
121                _applicationConstructor = _applicationClass
122                        .getConstructor(String.class);
123
124            }
125            Object instance = _applicationConstructor.newInstance(fullPath);
126
127            System.out.println("----------------- testing again " + fullPath);
128            System.out.flush();
129            Method rerunMethod = _applicationClass.getMethod("rerun",
130                    (Class<?>[]) null);
131            rerunMethod.invoke(instance, (Object[]) null);
132        } else {
133            System.err.println(
134                    "----------------- *** Skipping testing of " + fullPath);
135            System.err.flush();
136
137        }
138    }
139
140    /** Return true if the model should be run.
141     *  This is a hack to avoid a problem where certain models
142     *  interact badly with the Cobertura code coverage tool.
143     *  @param fullPath The full path of the model to be executed
144     *  @return true if the model should be run.
145     */
146    public boolean modelFileIsOK(String fullPath) {
147        if (fullPath.endsWith("de/test/auto/ThreadedComposite.xml")
148                && !StringUtilities
149                        .getProperty("net.sourceforge.cobertura.datafile")
150                        .equals("")) {
151            System.err.println(
152                    "ModelTests: Skipping de/test/auto/ThreadedComposite.xml because it interacts badly with Cobertura.");
153            return false;
154        }
155        return true;
156    }
157}
