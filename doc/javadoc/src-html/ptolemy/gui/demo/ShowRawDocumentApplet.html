030import java.io.BufferedReader;
031import java.io.FileNotFoundException;
032import java.io.IOException;
033import java.io.InputStream;
034import java.io.InputStreamReader;
035import java.net.MalformedURLException;
036import java.net.URL;
037
038import javax.swing.JFrame;
039import javax.swing.JScrollPane;
040import javax.swing.JTextArea;
041
042import ptolemy.gui.BasicJApplet;
043
044///////////////////////////////////////////////////////////////////
045//// ShowRawDocumentApplet
046
047/**
048 Applet to display raw text from a URL in a TextArea.
049
050 This applet is useful for displaying raw xml source with out processing
051 it.  If you would like to display a document within a browser with regular
052 processing, see the documentation for appletContext.showDocument(sourceURL).
053
054 @author  Christopher Hylands
055 @version $Id$
056 @since Ptolemy II 0.4
057 @Pt.ProposedRating Red (eal)
058 @Pt.AcceptedRating Red (eal)
059 */
060@SuppressWarnings("serial")
061public class ShowRawDocumentApplet extends BasicJApplet {
062    /** Constructor.
063     */
064    public ShowRawDocumentApplet() {
065        super();
066    }
067
068    ///////////////////////////////////////////////////////////////////
069    ////                         public methods                    ////
070
071    /** Read applet parameters, open the URL and display the contents
072     *  in a TextArea.
073     */
074    @Override
075    public void init() {
076        super.init();
077
078        // Process the rows and columns applet parameters
079        int rows;
080
081        // Process the rows and columns applet parameters
082        int columns;
083        String rowsString = getParameter("rows");
084
085        if (rowsString != null) {
086            rows = Integer.parseInt(rowsString);
087        } else {
088            rows = 10;
089        }
090
091        String columnsString = getParameter("columns");
092
093        if (columnsString != null) {
094            columns = Integer.parseInt(columnsString);
095        } else {
096            columns = 40;
097        }
098
099        // Get the source applet parameter before we create the
100        // _jTextArea in case there are problems with the parameter.
101        String sourceURLString = getParameter("source");
102        URL sourceURL = null;
103
104        if (sourceURLString != null) {
105            try {
106                showStatus("Reading data . . .");
107                sourceURL = new URL(getDocumentBase(), sourceURLString);
108            } catch (MalformedURLException e) {
109                System.err.println(e.toString());
110            }
111        }
112
113        /*JFrame _frame = */new JFrame();
114        _jTextArea = new JTextArea(rows, columns);
115        _jTextArea.setEditable(false);
116        _scrollPane = new JScrollPane(_jTextArea);
117        getContentPane().add(_scrollPane);
118
119        String newline = System.getProperty("line.separator");
120
121        if (sourceURL == null) {
122            showStatus("Failed to open " + sourceURLString);
123            System.err.println("Failed to open " + sourceURLString);
124        } else {
125            // Read in the data one line at a time.
126            InputStream in = null;
127            try {
128                in = sourceURL.openStream();
129                _bufferedReader = new BufferedReader(new InputStreamReader(in,
130                        java.nio.charset.Charset.defaultCharset()));
131
132                String line = _bufferedReader.readLine();
133
134                while (line != null) {
135                    _jTextArea.append(line + newline);
136                    line = _bufferedReader.readLine();
137                }
138
139                showStatus("Done");
140            } catch (FileNotFoundException e) {
141                System.err.println("File not found: " + e);
142            } catch (IOException e) {
143                System.err.println("Error reading input file: " + e);
144            } finally {
145                if (in != null) {
146                    try {
147                        in.close();
148                    } catch (Exception ex) {
149                        System.err.println("Failed to close " + sourceURLString
150                                + ":" + ex);
151                    }
152                }
153            }
154        }
155
156        //getContentPane().add(_frame);
157        //getContentPane().add(_jTextArea);
158        validate();
159        repaint();
160    }
161
162    /** Describe the applet parameters.
163     *  @return An array describing the applet parameters.
164     */
165    @Override
166    public String[][] getParameterInfo() {
167        String[][] newinfo = { { "columns", "integer", "40" },
168                { "rows", "integer", "10" },
169                { "source", "", "URL of the file to be displayed" }, };
170        return _concatStringArrays(super.getParameterInfo(), newinfo);
171    }
172
173    ///////////////////////////////////////////////////////////////////
174    ////                         private variables                 ////
175    private BufferedReader _bufferedReader;
176
177    private JTextArea _jTextArea;
178
179    private JScrollPane _scrollPane;
180}
