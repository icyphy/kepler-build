029package ptolemy.gui;

031import java.lang.reflect.InvocationHandler;
032import java.lang.reflect.Method;
033import java.lang.reflect.Proxy;
034
035/**
036 * Support for Mac OS X specific features such as key bindings.
037 *
038 * There is no public constructor.  Instead, call stati set*Method() methods.
039 *
040 *  @author Christopher Brooks, Based on OSXAdapter.java, downloaded from: <a href="http://developer.apple.com/library/mac/#samplecode/OSXAdapter/Listings/src_OSXAdapter_java.html">http://developer.apple.com/library/mac/#samplecode/OSXAdapter/Listings/src_OSXAdapter_java.html</a> on July 26, 2011.
041 * @version $Id$
042 * @since Ptolemy II 10.0
043 * @Pt.ProposedRating Red (cxh)
044 * @Pt.AcceptedRating Red (cxh)
045 */
046public class MacOSXAdapter implements InvocationHandler {
047
048    ///////////////////////////////////////////////////////////////////
049    ////                         public methods                    ////
050
051    /** Invoke a method.  This method is part of the java.lang.reflect.InvocationHandler
052     *  interface.
053     *  @param proxy The object upon which the method is invoked.
054     *  @param method The method to be invoked.
055     *  @param args The arguments to the method, which must be non-null.
056     *  @exception Throwable If the method does not exist or is not accessible or if
057     *  thrown while invoking the method.
058     *  @return the result, which in this case is always null because
059     *  the com.apple.eawt.ApplicationListener methods are all void.
060     */
061    @Override
062    public Object invoke(Object proxy, Method method, Object[] args)
063            throws Throwable {
064        if (_topMethod == null || !_proxySignature.equals(method.getName())
065                || args.length != 1) {
066            return null;
067        }
068        boolean handled = true;
069        Object result = _topMethod.invoke(_top, (Object[]) null);
070        if (result != null) {
071            handled = Boolean.valueOf(result.toString()).booleanValue();
072        }
073
074        try {
075            Method setHandledMethod = args[0].getClass().getDeclaredMethod(
076                    "setHandled", new Class[] { boolean.class });
077            setHandledMethod.invoke(args[0],
078                    new Object[] { Boolean.valueOf(handled) });
079        } catch (Exception ex) {
080            _top.report("The Application event \"" + args[0]
081                    + "\" was not handled.", ex);
082        }
083        return null;
084    }
085
086    /** Set the about menu handler for a Top window.
087     *  Under Mac OS X, the About menu may be selected from the application
088     *  menu, which is in the upper left, next to the apple.
089     *  @param top the Top level window to perform the operation.
090     *  @param aboutMethod The method to invoke in Top, typically
091     *  {@link ptolemy.gui.Top#about()}.
092     */
093    public static void setAboutMethod(Top top, Method aboutMethod) {
094        _setHandler(top, new MacOSXAdapter("handleAbout", top, aboutMethod));
095        if (_macOSXApplication == null) {
096            // _setHandler should set _macOSXApplication, so perhaps
097            // we are running as Applet or under -sandbox.
098            return;
099        }
100        try {
101            Method enableAboutMethod = _macOSXApplication.getClass()
102                    .getDeclaredMethod("setEnabledAboutMenu",
103                            new Class[] { boolean.class });
104            enableAboutMethod.invoke(_macOSXApplication,
105                    new Object[] { Boolean.TRUE });
106        } catch (SecurityException ex) {
107            if (!_printedSecurityExceptionMessage) {
108                _printedSecurityExceptionMessage = true;
109                System.out.println(
110                        "Warning: Failed to enable " + "the about menu. "
111                                + "(applets and -sandbox always causes this)");
112            }
113        } catch (NoSuchMethodException ex2) {
114            if (!_printedNoSuchMethodExceptionMessageAboutMenu) {
115                _printedNoSuchMethodExceptionMessageAboutMenu = true;
116                System.out.println(
117                        "Warning: Failed to get the setEnabledAboutMenu method.  "
118                                + "This is a known limitation of Java 9 and later.");
119            }
120        } catch (Exception ex3) {
121            top.report("The about menu could not be set.", ex3);
122        }
123    }
124
125    /** Set the quit handler (Command-q) for a Top window.
126     *  @param top the Top level window to perform the operation.
127     *  @param quitMethod The method to invoke in Top, typically
128     *  {@link ptolemy.gui.Top#exit()}.
129     */
130    public static void setQuitMethod(Top top, Method quitMethod) {
131        _setHandler(top, new MacOSXAdapter("handleQuit", top, quitMethod));
132    }

137    /** Create an adapter that has the name of a method to be listened for,
138     *  the Top window that performs the task and the method in Top.
139     *  @param proxySignature  A method name from com.apple.eawt.ApplicationListener,
140     *  for example "handleQuit".
141     *  @param top the Top level window to perform the operation.
142     *  @param topMethod The Method in Top to be called.
143     */
144    private MacOSXAdapter(String proxySignature, Top top, Method topMethod) {
145        _proxySignature = proxySignature;
146        _top = top;
147        _topMethod = topMethod;
148    }
149
150    /** Create a proxy object from the adapter and add it as an ApplicationListener.
151     *  @param top the Top level window to perform the operation.
152     *  @param adapter The adapter.
153     */
154    private static void _setHandler(Top top, MacOSXAdapter adapter) {
155        try {
156            Class applicationClass = null;
157            String applicationClassName = "com.apple.eawt.Application";
158            try {
159                applicationClass = Class.forName(applicationClassName);
160            } catch (NoClassDefFoundError ex) {
161                if (!_printedNoClassDefFoundMessageApplication) {
162                    System.out.println("Warning: Failed to find the \""
163                            + applicationClassName + "\" class: " + ex
164                            + " (applets and -sandbox always causes this)");
165                    _printedNoClassDefFoundMessageApplication = true;
166                }
167                return;
168            } catch (ExceptionInInitializerError ex) {
169                if (ex.getCause() instanceof SecurityException) {
170                    if (!_printedSecurityExceptionMessage) {
171                        System.out.println("Warning: Failed to create new "
172                                + "instance of \"" + applicationClassName
173                                + "\": " + ex
174                                + "(applets and -sandbox always causes this)");
175                    }
176                    return;
177                }
178            }
179
180            if (applicationClass == null) {
181                throw new NullPointerException("Internal Error!  class "
182                        + applicationClassName + " was not found "
183                        + "and the exception was missed?");
184            } else {
185                if (_macOSXApplication == null) {
186                    try {
187                        _macOSXApplication = applicationClass
188                                .getConstructor((Class[]) null)
189                                .newInstance((Object[]) null);
190                    } catch (java.lang.reflect.InvocationTargetException ex) {
191                        if (ex.getCause() instanceof SecurityException) {
192                            if (!_printedSecurityExceptionMessage) {
193                                System.out.println("Warning: Failed to get the"
194                                        + "constructor of \""
195                                        + applicationClassName + "\" ("
196                                        + applicationClass + "): " + ex
197                                        + "(applets and -sandbox always causes this)");
198                                _printedSecurityExceptionMessage = true;
199                            }
200                        }
201                        return;
202                    } catch (java.lang.IllegalAccessException ex2) {
203                        if (!_printedIllegalAccessExceptionMessage) {
204                            System.out.println(
205                                    "Warning: Failed to access the Application class "
206                                            + applicationClassName + "\" ("
207                                            + applicationClass + "): " + ex2);
208                            _printedIllegalAccessExceptionMessage = true;
209                        }
210                        return;
211                    }
212                }
213                Class applicationListenerClass = Class
214                        .forName("com.apple.eawt.ApplicationListener");
215                Method addListenerMethod = applicationClass.getDeclaredMethod(
216                        "addApplicationListener",
217                        new Class[] { applicationListenerClass });
218
219                // Create a proxy object around this handler that can be
220                // reflectively added as an Apple ApplicationListener
221                Object osxAdapterProxy = Proxy.newProxyInstance(
222                        MacOSXAdapter.class.getClassLoader(),
223                        new Class[] { applicationListenerClass }, adapter);
224                addListenerMethod.invoke(_macOSXApplication,
225                        new Object[] { osxAdapterProxy });
226            }
227        } catch (ClassNotFoundException ex) {
228            if (!_printedNoClassDefFoundMessageApplicationListener) {
229                System.err.println(
230                        "Warning The com.apple.eawt.ApplicationListener class was not found.  "
231                                + "This is a known limitation of Java 9 and later.");
232                _printedNoClassDefFoundMessageApplicationListener = true;
233            }
234        } catch (Exception ex2) {
235            top.report(
236                    "There was a problem invoking the addApplicationListener method",
237                    ex2);
238        }
239    }

241    /** An instance of com.apple.eawt.Application, upon which methods are invoked.
242     *  We use Object here instead of com.apple.eawt.Application so as to avoid
243     *  compile-time dependencies on Apple-specific code.
244     *  The _setHandler() method sets macOSXApplication.
245     *  If we are running as an unsigned applet or using -sandbox, then
246     *  this variable will be null.
247     */
248    private static Object _macOSXApplication;
249
250    /** True if we have printed the IllegalAccess message. */
251    private static boolean _printedIllegalAccessExceptionMessage = false;
252
253    /** True if we have printed the NoClassDefFound message for com.apple.eawt.Application. */
254    private static boolean _printedNoClassDefFoundMessageApplication = false;
255
256    /** True if we have printed the NoClassDefFound message for com.apple.eawt.ApplicationListener. */
257    private static boolean _printedNoClassDefFoundMessageApplicationListener = false;
258
259    /** True if we can't find the setEnabledAboutMenu method and have printed the message. */
260    private static boolean _printedNoSuchMethodExceptionMessageAboutMenu = false;
261
262    /** True if we have printed the securityException message. */
263    private static boolean _printedSecurityExceptionMessage = false;
264
265    /**  The name of a method in com.apple.eawt.ApplicationListener,
266     *  for example "handleQuit".
267     */
268    private String _proxySignature;
269
270    /** The Top level window to perform the operation.  This window is also
271     *  used to report errors.
272     */
273    private Top _top;
274
275    /** The Method in Top to be called. */
276    private Method _topMethod;
